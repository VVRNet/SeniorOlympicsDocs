34namespace Spine.Unity.Examples {

36 // This is an example of how you could store animation transitions for use in your animation system.
37 // More ideally, this would be stored in a ScriptableObject in asset form rather than in a MonoBehaviour.
38 public sealed class TransitionDictionaryExample : MonoBehaviour {
39
40 [System.Serializable]
41 public struct SerializedEntry {
42 public AnimationReferenceAsset from;
43 public AnimationReferenceAsset to;
44 public AnimationReferenceAsset transition;
45 }
46
47 [SerializeField]
48 List<SerializedEntry> transitions = new List<SerializedEntry>();
49 readonly Dictionary<AnimationStateData.AnimationPair, Animation> dictionary = new Dictionary<AnimationStateData.AnimationPair, Animation>();
50
51 void Start () {
52 dictionary.Clear();
53 foreach (SerializedEntry e in transitions) {
54 dictionary.Add(new AnimationStateData.AnimationPair(e.from.Animation, e.to.Animation), e.transition.Animation);
55 }
56 }
57
58 public Animation GetTransition (Animation from, Animation to) {
59 Animation result;
60 dictionary.TryGetValue(new AnimationStateData.AnimationPair(from, to), out result);
61 return result;
62 }
63 }
64}
