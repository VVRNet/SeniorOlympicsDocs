113 public Skeleton (SkeletonData data) {
114 if (data == null) throw new ArgumentNullException("data", "data cannot be null.");
115 this.data = data;
116
117 bones = new ExposedList<Bone>(data.bones.Count);
118 Bone[] bonesItems = this.bones.Items;
119 foreach (BoneData boneData in data.bones) {
120 Bone bone;
121 if (boneData.parent == null) {
122 bone = new Bone(boneData, this, null);
123 } else {
124 Bone parent = bonesItems[boneData.parent.index];
125 bone = new Bone(boneData, this, parent);
126 parent.children.Add(bone);
127 }
128 this.bones.Add(bone);
129 }
130
131 slots = new ExposedList<Slot>(data.slots.Count);
132 drawOrder = new ExposedList<Slot>(data.slots.Count);
133 foreach (SlotData slotData in data.slots) {
134 Bone bone = bonesItems[slotData.boneData.index];
135 Slot slot = new Slot(slotData, bone);
136 slots.Add(slot);
137 drawOrder.Add(slot);
138 }
139
140 ikConstraints = new ExposedList<IkConstraint>(data.ikConstraints.Count);
141 foreach (IkConstraintData ikConstraintData in data.ikConstraints)
142 ikConstraints.Add(new IkConstraint(ikConstraintData, this));
143
144 transformConstraints = new ExposedList<TransformConstraint>(data.transformConstraints.Count);
145 foreach (TransformConstraintData transformConstraintData in data.transformConstraints)
146 transformConstraints.Add(new TransformConstraint(transformConstraintData, this));
147
148 pathConstraints = new ExposedList<PathConstraint>(data.pathConstraints.Count);
149 foreach (PathConstraintData pathConstraintData in data.pathConstraints)
150 pathConstraints.Add(new PathConstraint(pathConstraintData, this));
151
152 physicsConstraints = new ExposedList<PhysicsConstraint>(data.physicsConstraints.Count);
153 foreach (PhysicsConstraintData physicsConstraintData in data.physicsConstraints)
154 physicsConstraints.Add(new PhysicsConstraint(physicsConstraintData, this));
155
156 UpdateCache();
157 }

160 public Skeleton (Skeleton skeleton) {
161 if (skeleton == null) throw new ArgumentNullException("skeleton", "skeleton cannot be null.");
162 data = skeleton.data;
163
164 bones = new ExposedList<Bone>(skeleton.bones.Count);
165 foreach (Bone bone in skeleton.bones) {
166 Bone newBone;
167 if (bone.parent == null)
168 newBone = new Bone(bone, this, null);
169 else {
170 Bone parent = bones.Items[bone.parent.data.index];
171 newBone = new Bone(bone, this, parent);
172 parent.children.Add(newBone);
173 }
174 bones.Add(newBone);
175 }
176
177 slots = new ExposedList<Slot>(skeleton.slots.Count);
178 Bone[] bonesItems = bones.Items;
179 foreach (Slot slot in skeleton.slots) {
180 Bone bone = bonesItems[slot.bone.data.index];
181 slots.Add(new Slot(slot, bone));
182 }
183
184 drawOrder = new ExposedList<Slot>(slots.Count);
185 Slot[] slotsItems = slots.Items;
186 foreach (Slot slot in skeleton.drawOrder)
187 drawOrder.Add(slotsItems[slot.data.index]);
188
189 ikConstraints = new ExposedList<IkConstraint>(skeleton.ikConstraints.Count);
190 foreach (IkConstraint ikConstraint in skeleton.ikConstraints)
191 ikConstraints.Add(new IkConstraint(ikConstraint, skeleton));
192
193 transformConstraints = new ExposedList<TransformConstraint>(skeleton.transformConstraints.Count);
194 foreach (TransformConstraint transformConstraint in skeleton.transformConstraints)
195 transformConstraints.Add(new TransformConstraint(transformConstraint, skeleton));
196
197 pathConstraints = new ExposedList<PathConstraint>(skeleton.pathConstraints.Count);
198 foreach (PathConstraint pathConstraint in skeleton.pathConstraints)
199 pathConstraints.Add(new PathConstraint(pathConstraint, skeleton));
200
201 physicsConstraints = new ExposedList<PhysicsConstraint>(skeleton.physicsConstraints.Count);
202 foreach (PhysicsConstraint physicsConstraint in skeleton.physicsConstraints)
203 physicsConstraints.Add(new PhysicsConstraint(physicsConstraint, skeleton));
204
205 skin = skeleton.skin;
206 r = skeleton.r;
207 g = skeleton.g;
208 b = skeleton.b;
209 a = skeleton.a;
210 x = skeleton.x;
211 y = skeleton.y;
212 scaleX = skeleton.scaleX;
213 scaleY = skeleton.scaleY;
214 time = skeleton.time;
215
216 UpdateCache();
217 }

221 public void UpdateCache () {
222 ExposedList<IUpdatable> updateCache = this.updateCache;
223 updateCache.Clear();
224
225 int boneCount = this.bones.Count;
226 Bone[] bones = this.bones.Items;
227 for (int i = 0; i < boneCount; i++) {
228 Bone bone = bones[i];
229 bone.sorted = bone.data.skinRequired;
230 bone.active = !bone.sorted;
231 }
232 if (skin != null) {
233 BoneData[] skinBones = skin.bones.Items;
234 for (int i = 0, n = skin.bones.Count; i < n; i++) {
235 Bone bone = bones[skinBones[i].index];
236 do {
237 bone.sorted = false;
238 bone.active = true;
239 bone = bone.parent;
240 } while (bone != null);
241 }
242 }
243
244 int ikCount = this.ikConstraints.Count, transformCount = this.transformConstraints.Count, pathCount = this.pathConstraints.Count,
245 physicsCount = this.physicsConstraints.Count;
246 IkConstraint[] ikConstraints = this.ikConstraints.Items;
247 TransformConstraint[] transformConstraints = this.transformConstraints.Items;
248 PathConstraint[] pathConstraints = this.pathConstraints.Items;
249 PhysicsConstraint[] physicsConstraints = this.physicsConstraints.Items;
250 int constraintCount = ikCount + transformCount + pathCount + physicsCount;
251 for (int i = 0; i < constraintCount; i++) {
252 for (int ii = 0; ii < ikCount; ii++) {
253 IkConstraint constraint = ikConstraints[ii];
254 if (constraint.data.order == i) {
255 SortIkConstraint(constraint);
256 goto continue_outer;
257 }
258 }
259 for (int ii = 0; ii < transformCount; ii++) {
260 TransformConstraint constraint = transformConstraints[ii];
261 if (constraint.data.order == i) {
262 SortTransformConstraint(constraint);
263 goto continue_outer;
264 }
265 }
266 for (int ii = 0; ii < pathCount; ii++) {
267 PathConstraint constraint = pathConstraints[ii];
268 if (constraint.data.order == i) {
269 SortPathConstraint(constraint);
270 goto continue_outer;
271 }
272 }
273 for (int ii = 0; ii < physicsCount; ii++) {
274 PhysicsConstraint constraint = physicsConstraints[ii];
275 if (constraint.data.order == i) {
276 SortPhysicsConstraint(constraint);
277 goto continue_outer;
278 }
279 }
280 continue_outer: { }
281 }
282
283 for (int i = 0; i < boneCount; i++)
284 SortBone(bones[i]);
285 }

287 private void SortIkConstraint (IkConstraint constraint) {
288 constraint.active = constraint.target.active
289 && (!constraint.data.skinRequired || (skin != null && skin.constraints.Contains(constraint.data)));
290 if (!constraint.active) return;
291
292 Bone target = constraint.target;
293 SortBone(target);
294
295 ExposedList<Bone> constrained = constraint.bones;
296 Bone parent = constrained.Items[0];
297 SortBone(parent);
298
299 if (constrained.Count == 1) {
300 updateCache.Add(constraint);
301 SortReset(parent.children);
302 } else {
303 Bone child = constrained.Items[constrained.Count - 1];
304 SortBone(child);
305
306 updateCache.Add(constraint);
307
308 SortReset(parent.children);
309 child.sorted = true;
310 }
311 }

313 private void SortTransformConstraint (TransformConstraint constraint) {
314 constraint.active = constraint.target.active
315 && (!constraint.data.skinRequired || (skin != null && skin.constraints.Contains(constraint.data)));
316 if (!constraint.active) return;
317
318 SortBone(constraint.target);
319
320 Bone[] constrained = constraint.bones.Items;
321 int boneCount = constraint.bones.Count;
322 if (constraint.data.local) {
323 for (int i = 0; i < boneCount; i++) {
324 Bone child = constrained[i];
325 SortBone(child.parent);
326 SortBone(child);
327 }
328 } else {
329 for (int i = 0; i < boneCount; i++)
330 SortBone(constrained[i]);
331 }
332
333 updateCache.Add(constraint);
334
335 for (int i = 0; i < boneCount; i++)
336 SortReset(constrained[i].children);
337 for (int i = 0; i < boneCount; i++)
338 constrained[i].sorted = true;
339 }

341 private void SortPathConstraint (PathConstraint constraint) {
342 constraint.active = constraint.target.bone.active
343 && (!constraint.data.skinRequired || (skin != null && skin.constraints.Contains(constraint.data)));
344 if (!constraint.active) return;
345
346 Slot slot = constraint.target;
347 int slotIndex = slot.data.index;
348 Bone slotBone = slot.bone;
349 if (skin != null) SortPathConstraintAttachment(skin, slotIndex, slotBone);
350 if (data.defaultSkin != null && data.defaultSkin != skin)
351 SortPathConstraintAttachment(data.defaultSkin, slotIndex, slotBone);
352
353 Attachment attachment = slot.attachment;
354 if (attachment is PathAttachment) SortPathConstraintAttachment(attachment, slotBone);
355
356 Bone[] constrained = constraint.bones.Items;
357 int boneCount = constraint.bones.Count;
358 for (int i = 0; i < boneCount; i++)
359 SortBone(constrained[i]);
360
361 updateCache.Add(constraint);
362
363 for (int i = 0; i < boneCount; i++)
364 SortReset(constrained[i].children);
365 for (int i = 0; i < boneCount; i++)
366 constrained[i].sorted = true;
367 }

369 private void SortPathConstraintAttachment (Skin skin, int slotIndex, Bone slotBone) {
370 foreach (Skin.SkinEntry entry in skin.Attachments)
371 if (entry.SlotIndex == slotIndex) SortPathConstraintAttachment(entry.Attachment, slotBone);

374 private void SortPathConstraintAttachment (Attachment attachment, Bone slotBone) {
375 if (!(attachment is PathAttachment)) return;
376 int[] pathBones = ((PathAttachment)attachment).bones;
377 if (pathBones == null)
378 SortBone(slotBone);
379 else {
380 Bone[] bones = this.bones.Items;
381 for (int i = 0, n = pathBones.Length; i < n;) {
382 int nn = pathBones[i++];
383 nn += i;
384 while (i < nn)
385 SortBone(bones[pathBones[i++]]);
386 }
387 }
388 }

390 private void SortPhysicsConstraint (PhysicsConstraint constraint) {
391 Bone bone = constraint.bone;
392 constraint.active = bone.active
393 && (!constraint.data.skinRequired || (skin != null && skin.constraints.Contains(constraint.data)));
394 if (!constraint.active) return;
395
396 SortBone(bone);
397
398 updateCache.Add(constraint);
399
400 SortReset(bone.children);
401 bone.sorted = true;

404 private void SortBone (Bone bone) {
405 if (bone.sorted) return;
406 Bone parent = bone.parent;
407 if (parent != null) SortBone(parent);
408 bone.sorted = true;
409 updateCache.Add(bone);

450 public void UpdateWorldTransform (Physics physics, Bone parent) {
451 if (parent == null) throw new ArgumentNullException("parent", "parent cannot be null.");
452
453 // Apply the parent bone transform to the root bone. The root bone always inherits scale, rotation and reflection.
454 Bone rootBone = this.RootBone;
455 float pa = parent.a, pb = parent.b, pc = parent.c, pd = parent.d;
456 rootBone.worldX = pa * x + pb * y + parent.worldX;
457 rootBone.worldY = pc * x + pd * y + parent.worldY;
458
459 float rx = (rootBone.rotation + rootBone.shearX) * MathUtils.DegRad;
460 float ry = (rootBone.rotation + 90 + rootBone.shearY) * MathUtils.DegRad;
461 float la = (float)Math.Cos(rx) * rootBone.scaleX;
462 float lb = (float)Math.Cos(ry) * rootBone.scaleY;
463 float lc = (float)Math.Sin(rx) * rootBone.scaleX;
464 float ld = (float)Math.Sin(ry) * rootBone.scaleY;
465 rootBone.a = (pa * la + pb * lc) * scaleX;
466 rootBone.b = (pa * lb + pb * ld) * scaleX;
467 rootBone.c = (pc * la + pd * lc) * scaleY;
468 rootBone.d = (pc * lb + pd * ld) * scaleY;
469
470 // Update everything except root bone.
471 IUpdatable[] updateCache = this.updateCache.Items;
472 for (int i = 0, n = this.updateCache.Count; i < n; i++) {
473 IUpdatable updatable = updateCache[i];
474 if (updatable != rootBone) updatable.Update(physics);
475 }
476 }

700 public void GetBounds (out float x, out float y, out float width, out float height, ref float[] vertexBuffer,
701 SkeletonClipping clipper = null) {
702
703 float[] temp = vertexBuffer;
704 temp = temp ?? new float[8];
705 Slot[] drawOrder = this.drawOrder.Items;
706 float minX = int.MaxValue, minY = int.MaxValue, maxX = int.MinValue, maxY = int.MinValue;
707 for (int i = 0, n = this.drawOrder.Count; i < n; i++) {
708 Slot slot = drawOrder[i];
709 if (!slot.bone.active) continue;
710 int verticesLength = 0;
711 float[] vertices = null;
712 int[] triangles = null;
713 Attachment attachment = slot.attachment;
714 RegionAttachment region = attachment as RegionAttachment;
715 if (region != null) {
716 verticesLength = 8;
717 vertices = temp;
718 if (vertices.Length < 8) vertices = temp = new float[8];
719 region.ComputeWorldVertices(slot, temp, 0, 2);
720 triangles = quadTriangles;
721 } else {
722 MeshAttachment mesh = attachment as MeshAttachment;
723 if (mesh != null) {
724 verticesLength = mesh.WorldVerticesLength;
725 vertices = temp;
726 if (vertices.Length < verticesLength) vertices = temp = new float[verticesLength];
727 mesh.ComputeWorldVertices(slot, 0, verticesLength, temp, 0, 2);
728 triangles = mesh.Triangles;
729 } else if (clipper != null) {
730 ClippingAttachment clip = attachment as ClippingAttachment;
731 if (clip != null) {
732 clipper.ClipStart(slot, clip);
733 continue;
734 }
735 }
736 }
737
738 if (vertices != null) {
739 if (clipper != null && clipper.IsClipping) {
740 clipper.ClipTriangles(vertices, verticesLength, triangles, triangles.Length);
