57 public static void MatchAnimationTimelines (IEnumerable<Spine.Animation> animations, SkeletonData skeletonData) {
58 if (animations == null) return;
59 if (skeletonData == null) throw new System.ArgumentNullException("skeletonData", "Timelines can't be matched without a SkeletonData source.");
60
61 // Build a reference collection of timelines to match
62 // and a collection of dummy timelines that can be used to fill-in missing items.
63 Dictionary<string, Timeline> timelineDictionary = new Dictionary<string, Spine.Timeline>();
64 foreach (Animation animation in animations) {
65 foreach (Timeline timeline in animation.Timelines) {
66 if (timeline is EventTimeline) continue;
67
68 foreach (string propertyId in timeline.PropertyIds) {
69 if (!timelineDictionary.ContainsKey(propertyId)) {
70 timelineDictionary.Add(propertyId, GetFillerTimeline(timeline, skeletonData));
71 }
72 }
73 }
74 }
75 List<string> idsToMatch = new List<string>(timelineDictionary.Keys);
76
77 // For each animation in the list, check for and add missing timelines.
78 HashSet<string> currentAnimationIDs = new HashSet<string>();
79 foreach (Animation animation in animations) {
80 currentAnimationIDs.Clear();
81 foreach (Timeline timeline in animation.Timelines) {
82 if (timeline is EventTimeline) continue;
83 foreach (string propertyId in timeline.PropertyIds) {
84 currentAnimationIDs.Add(propertyId);
85 }
86 }
87
88 ExposedList<Timeline> animationTimelines = animation.Timelines;
89 foreach (string propertyId in idsToMatch) {
90 if (!currentAnimationIDs.Contains(propertyId))
91 animationTimelines.Add(timelineDictionary[propertyId]);
92 }
93 }
94
95 // These are locals, but sometimes Unity's GC does weird stuff. So let's clean up.
96 timelineDictionary.Clear();
97 timelineDictionary = null;
98 idsToMatch.Clear();
99 idsToMatch = null;
100 currentAnimationIDs.Clear();
101 currentAnimationIDs = null;
102 }

104 static Timeline GetFillerTimeline (Timeline timeline, SkeletonData skeletonData) {
105 if (timeline is RotateTimeline)
106 return GetFillerTimeline((RotateTimeline)timeline, skeletonData);
107 if (timeline is TranslateTimeline)
108 return GetFillerTimeline((TranslateTimeline)timeline, skeletonData);
109 if (timeline is ScaleTimeline)
110 return GetFillerTimeline((ScaleTimeline)timeline, skeletonData);
111 if (timeline is ShearTimeline)
112 return GetFillerTimeline((ShearTimeline)timeline, skeletonData);
113 if (timeline is AttachmentTimeline)
114 return GetFillerTimeline((AttachmentTimeline)timeline, skeletonData);
115 if (timeline is RGBATimeline)
116 return GetFillerTimeline((RGBATimeline)timeline, skeletonData);
117 if (timeline is RGBA2Timeline)
118 return GetFillerTimeline((RGBA2Timeline)timeline, skeletonData);
119 if (timeline is DeformTimeline)
120 return GetFillerTimeline((DeformTimeline)timeline, skeletonData);
121 if (timeline is DrawOrderTimeline)
122 return GetFillerTimeline((DrawOrderTimeline)timeline, skeletonData);
123 if (timeline is IkConstraintTimeline)
124 return GetFillerTimeline((IkConstraintTimeline)timeline, skeletonData);
125 if (timeline is TransformConstraintTimeline)
126 return GetFillerTimeline((TransformConstraintTimeline)timeline, skeletonData);
127 if (timeline is PathConstraintPositionTimeline)
128 return GetFillerTimeline((PathConstraintPositionTimeline)timeline, skeletonData);
129 if (timeline is PathConstraintSpacingTimeline)
130 return GetFillerTimeline((PathConstraintSpacingTimeline)timeline, skeletonData);
131 if (timeline is PathConstraintMixTimeline)
132 return GetFillerTimeline((PathConstraintMixTimeline)timeline, skeletonData);
133 return null;
134 }
