47#if UNITY_EDITOR
48 void OnValidate()
49 {
50 this.hideFlags = isDynamic ? HideFlags.None : HideFlags.DontSaveInBuild;
51
52 if (text == null || font == null)
53 {
54 return;
55 }
56
57 // parse text to only allow valid characters
58 List<char> allowed = new List<char>(font.CharSequence.ToCharArray());
59 allowed.Add(' ');
60
61 char[] chars;
62 switch (font.letterCase)
63 {
64 case CFXR_ParticleTextFontAsset.LetterCase.Lower: chars = text.ToLowerInvariant().ToCharArray(); break;
65 case CFXR_ParticleTextFontAsset.LetterCase.Upper: chars = text.ToUpperInvariant().ToCharArray(); break;
66 default:
67 case CFXR_ParticleTextFontAsset.LetterCase.Both: chars = text.ToCharArray(); break;
68 }
69
70 string newText = "";
71 foreach (var c in chars)
72 {
73 if (allowed.Contains(c))
74 {
75 newText += c;
76 }
77 }
78
79 text = newText;
80
81 // prevent negative or 0 size
82 size = Mathf.Max(0.001f, size);
83
84 // delay so that we are allowed to destroy GameObjects
85 if (autoUpdateEditor && !EditorApplication.isPlayingOrWillChangePlaymode)
86 {
87 EditorApplication.delayCall += () => { UpdateText(null); };
88 }
89 }

122 public void UpdateText(
123 string newText = null,
124 float? newSize = null,
125 Color? newColor1 = null, Color? newColor2 = null, Color? newBackgroundColor = null,
126 float? newLifetimeMultiplier = null
127 )
128 {
129#if UNITY_EDITOR
130 // Only allow updating text for GameObjects that aren't prefabs, since we are possibly destroying/adding GameObjects
131 if (this == null)
132 {
133 return;
134 }
135
136 var prefabInstanceStatus = PrefabUtility.GetPrefabInstanceStatus(this);
137 var prefabAssetType = PrefabUtility.GetPrefabAssetType(this);
138 if (!(prefabInstanceStatus == PrefabInstanceStatus.NotAPrefab && prefabAssetType == PrefabAssetType.NotAPrefab))
139 {
140 return;
141 }
142
143 if (!Application.isPlaying)
144 {
145 InitializeFirstParticle();
146 }
147#endif
148
149 if (Application.isPlaying && !isDynamic)
150 {
151 throw new System.Exception("[CFXR_ParticleText] You cannot update the text if it's not marked as dynamic.");
152 }
153
154 if (newText != null)
155 {
156 switch (font.letterCase)
157 {
158 case CFXR_ParticleTextFontAsset.LetterCase.Lower:
159 newText = newText.ToLowerInvariant();
160 break;
161 case CFXR_ParticleTextFontAsset.LetterCase.Upper:
162 newText = newText.ToUpperInvariant();
163 break;
164 }
165
166 // Verify that new text doesn't contain invalid characters
167 foreach (char c in newText)
168 {
169 if (char.IsWhiteSpace(c)) continue;
170 if (font.CharSequence.IndexOf(c) < 0)
171 {
172 throw new System.Exception("[CFXR_ParticleText] Invalid character supplied for the dynamic text: '" + c + "'\nThe allowed characters from the selected font are: " + font.CharSequence);
173 }
174 }
175
176 this.text = newText;
177 }
178
179 if (newSize != null) this.size = newSize.Value;
180 if (newColor1 != null) this.color1 = newColor1.Value;
181 if (newColor2 != null) this.color2 = newColor2.Value;
182 if (newBackgroundColor != null) this.backgroundColor = newBackgroundColor.Value;
183 if (newLifetimeMultiplier != null) this.lifetimeMultiplier = newLifetimeMultiplier.Value;
184
185 if (text == null || font == null || !font.IsValid())
186 {
187 return;
188 }
189
190 if (this.transform.childCount == 0)
191 {
192 throw new System.Exception("[CFXR_ParticleText] A disabled GameObject with a ParticleSystem component is required as the first child when 'isDyanmic' is enabled, so that its settings can be used as a base for the generated characters.");
193 }
194
195 // process text and calculate total width offset
196 float totalWidth = 0f;
197 int charCount = 0;
198 for (int i = 0; i < text.Length; i++)
199 {
200 if (char.IsWhiteSpace(text[i]))
201 {
202 if (i > 0)
203 {
204 totalWidth += letterSpacing * size;
205 }
206 }
207 else
208 {
209 charCount++;
210
211 if (i > 0)
212 {
213 int index = font.CharSequence.IndexOf(text[i]);
214 var sprite = font.CharSprites[index];
215 float charWidth = sprite.rect.width + font.CharKerningOffsets[index].post + font.CharKerningOffsets[index].pre;
216 totalWidth += (charWidth * 0.01f + letterSpacing) * size;
217 }
218 }
219 }
220
221#if UNITY_EDITOR
222 // delete all children in editor, to make sure we refresh the particle systems based on the first one
223 if (!Application.isPlaying)
224 {
225 int length = this.transform.childCount;
226 int overflow = 0;
227 while (this.transform.childCount > 1)
228 {
229 Object.DestroyImmediate(this.transform.GetChild(this.transform.childCount - 1).gameObject);
230 overflow++;
231 if (overflow > 1000)
232 {
233 // just in case...
234 Debug.LogError("Overflow!");
235 break;
236 }
237 }
238 }
239#endif
240
241 if (charCount > 0)
242 {
243 // calculate needed instances
244 int childCount = this.transform.childCount - (isDynamic ? 1 : 0); // first one is the particle source and always deactivated
245 if (childCount < charCount)
246 {
247 // instantiate new letter GameObjects if needed
248 GameObject model = isDynamic ? this.transform.GetChild(0).gameObject : null;
249 for (int i = childCount; i < charCount; i++)
250 {
251 var newLetter = isDynamic ? Instantiate(model, this.transform) : new GameObject();
252 if (!isDynamic)
253 {
254 newLetter.transform.SetParent(this.transform);
255 newLetter.AddComponent<ParticleSystem>();
256 }
257
258 newLetter.transform.localPosition = Vector3.zero;
259 newLetter.transform.localRotation = Quaternion.identity;
260 }
261 }
262
263 // update each letter
264 float offset = totalWidth / 2f;
265 totalWidth = 0f;
266 int currentChild = isDynamic ? 0 : -1;
267
268 // when not dynamic, we use CopySerialized to propagate the settings to the instances
269 var sourceParticle = isDynamic ? null : this.GetComponent<ParticleSystem>();
270 var sourceParticleRenderer = this.GetComponent<ParticleSystemRenderer>();
271
272 for (int i = 0; i < text.Length; i++)
273 {
274 var letter = text[i];
275 if (char.IsWhiteSpace(letter))
276 {
277 totalWidth += letterSpacing * size;
278 }
279 else
280 {
281 currentChild++;
282 int index = font.CharSequence.IndexOf(text[i]);
283 var sprite = font.CharSprites[index];
284
285 // calculate char particle size ratio
286 var ratio = size * sprite.rect.width / 50f;
287
288 // calculate char position
289 totalWidth += font.CharKerningOffsets[index].pre * 0.01f * size;
290 var position = (totalWidth - offset) / ratio;
291 float charWidth = sprite.rect.width + font.CharKerningOffsets[index].post;
292 totalWidth += (charWidth * 0.01f + letterSpacing) * size;
293
294 // update particle system for this letter
295 var letterObj = this.transform.GetChild(currentChild).gameObject;
296 letterObj.name = letter.ToString();
297 var ps = letterObj.GetComponent<ParticleSystem>();
298#if UNITY_EDITOR
299 if (!isDynamic)
300 {
301 EditorUtility.CopySerialized(sourceParticle, ps);
302 }
303#endif
304
305 var mainModule = ps.main;
306 mainModule.startSizeXMultiplier = baseScaleX * ratio;
307 mainModule.startSizeYMultiplier = baseScaleY * ratio;
308 mainModule.startSizeZMultiplier = baseScaleZ * ratio;
309
310 ps.textureSheetAnimation.SetSprite(0, sprite);
311
312 mainModule.startRotation = Mathf.Deg2Rad * rotation;
313 mainModule.startColor = backgroundColor;
314
315 var customData = ps.customData;
316 customData.enabled = true;
317 customData.SetColor(ParticleSystemCustomData.Custom1, color1);
318 customData.SetColor(ParticleSystemCustomData.Custom2, color2);
319
320 if (cumulativeDelay)
321 {
322 mainModule.startDelay = delay * i;
323 mainModule.startLifetime = Mathf.LerpUnclamped(baseLifetime, baseLifetime + (delay * (text.Length - i)), compensateLifetime / lifetimeMultiplier);
324 }
325 else
326 {
327 mainModule.startDelay = delay;
328 }
329
330 mainModule.startLifetime = mainModule.startLifetime.constant * lifetimeMultiplier;
331
332 // particle system renderer parameters
333 var particleRenderer = ps.GetComponent<ParticleSystemRenderer>();
334#if UNITY_EDITOR
335 if (!isDynamic)
336 {
337 EditorUtility.CopySerialized(sourceParticleRenderer, particleRenderer);
338 }
339#endif
340
341 particleRenderer.enabled = true;
342 particleRenderer.pivot = new Vector3(basePivot.x + position, basePivot.y, basePivot.z);
343 particleRenderer.sortingFudge += i * sortingFudgeOffset;
344 }
345 }
346 }
347
348 // set active state for needed letters only
349 for (int i = 1, l = this.transform.childCount; i < l; i++)
350 {
351 this.transform.GetChild(i).gameObject.SetActive(i <= charCount);
352 }
353
354#if UNITY_EDITOR
355 // automatically play the effect in Editor
356 if (!Application.isPlaying)
357 {
358 this.GetComponent<ParticleSystem>().Clear(true);
359 this.GetComponent<ParticleSystem>().Play(true);
360 }
361#endif
362 }
363 }
