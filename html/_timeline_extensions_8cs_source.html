41 public static Vector2 Evaluate (this TranslateTimeline timeline, float time, SkeletonData skeletonData = null) {
42 if (time < timeline.Frames[0]) return Vector2.zero;
43
44 float x, y;
45 timeline.GetCurveValue(out x, out y, time);
46
47 if (skeletonData == null) {
48 return new Vector2(x, y);
49 } else {
50 BoneData boneData = skeletonData.Bones.Items[timeline.BoneIndex];
51 return new Vector2(boneData.X + x, boneData.Y + y);
52 }
53 }

59 public static Vector2 Evaluate (TranslateXTimeline xTimeline, TranslateYTimeline yTimeline,
60 float time, SkeletonData skeletonData = null) {
61
62 float x = 0, y = 0;
63 if (xTimeline != null && time > xTimeline.Frames[0]) x = xTimeline.GetCurveValue(time);
64 if (yTimeline != null && time > yTimeline.Frames[0]) y = yTimeline.GetCurveValue(time);
65
66 if (skeletonData == null) {
67 return new Vector2(x, y);
68 } else {
69 BoneData[] bonesItems = skeletonData.Bones.Items;
70 BoneData boneDataX = bonesItems[xTimeline.BoneIndex];
71 BoneData boneDataY = bonesItems[yTimeline.BoneIndex];
72 return new Vector2(boneDataX.X + x, boneDataY.Y + y);
73 }
74 }

80 public static float Evaluate (this RotateTimeline timeline, float time, SkeletonData skeletonData = null) {
81 if (time < timeline.Frames[0]) return 0f;
82
83 float rotation = timeline.GetCurveValue(time);
84 if (skeletonData == null) {
85 return rotation;
86 } else {
87 BoneData boneData = skeletonData.Bones.Items[timeline.BoneIndex];
88 return (boneData.Rotation + rotation);

116 public static TranslateTimeline FindTranslateTimelineForBone (this Animation a, int boneIndex) {
117 foreach (Timeline timeline in a.Timelines) {
118 if (timeline.GetType().IsSubclassOf(typeof(TranslateTimeline)))
119 continue;
120
121 TranslateTimeline translateTimeline = timeline as TranslateTimeline;
122 if (translateTimeline != null && translateTimeline.BoneIndex == boneIndex)
123 return translateTimeline;
124 }
125 return null;
126 }
