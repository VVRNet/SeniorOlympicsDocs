455 public virtual void Initialize (bool overwrite, bool quiet = false) {
456 if (valid && !overwrite)
457 return;
458#if UNITY_EDITOR
459 if (BuildUtilities.IsInSkeletonAssetBuildPreProcessing)
460 return;
461#endif
462 // Clear
463 {
464 // Note: do not reset meshFilter.sharedMesh or meshRenderer.sharedMaterial to null,
465 // otherwise constant reloading will be triggered at prefabs.
466 currentInstructions.Clear();
467 rendererBuffers.Clear();
468 meshGenerator.Begin();
469 skeleton = null;
470 valid = false;
471 }
472
473 if (skeletonDataAsset == null)
474 return;
475
476 SkeletonData skeletonData = skeletonDataAsset.GetSkeletonData(quiet);
477 if (skeletonData == null) return;
478 valid = true;
479
480 meshFilter = GetComponent<MeshFilter>();
481 if (meshFilter == null)
482 meshFilter = gameObject.AddComponent<MeshFilter>();
483
484 meshRenderer = GetComponent<MeshRenderer>();
485 rendererBuffers.Initialize();
486
487 skeleton = new Skeleton(skeletonData) {
488 ScaleX = initialFlipX ? -1 : 1,
489 ScaleY = initialFlipY ? -1 : 1
490 };
491
492 ResetLastPositionAndRotation();
493
494 if (!string.IsNullOrEmpty(initialSkinName) && !string.Equals(initialSkinName, "default", System.StringComparison.Ordinal))
495 skeleton.SetSkin(initialSkinName);
496
497 separatorSlots.Clear();
498 for (int i = 0; i < separatorSlotNames.Length; i++)
499 separatorSlots.Add(skeleton.FindSlot(separatorSlotNames[i]));
500
501 // Generate mesh once, required to update mesh bounds for visibility
502 UpdateMode updateModeSaved = updateMode;
503 updateMode = UpdateMode.FullUpdate;
504 UpdateWorldTransform(Skeleton.Physics.Update);
505 LateUpdate();
506 updateMode = updateModeSaved;
507
508 if (OnRebuild != null)
509 OnRebuild(this);
510
511#if UNITY_EDITOR
512 if (!Application.isPlaying) {
513 string errorMessage = null;
514 if (!quiet && MaterialChecks.IsMaterialSetupProblematic(this, ref errorMessage))
515 Debug.LogWarningFormat(this, "Problematic material setup at {0}: {1}", this.name, errorMessage);
516 }
517#endif
518 }

520 public virtual void ApplyTransformMovementToPhysics () {
521 if (Application.isPlaying) {
522 if (physicsPositionInheritanceFactor != Vector2.zero) {
523 Vector2 position = GetPhysicsTransformPosition();
524 Vector2 positionDelta = position - lastPosition;
525
526 positionDelta = transform.InverseTransformVector(positionDelta);
527 if (physicsMovementRelativeTo != null) {
528 positionDelta = physicsMovementRelativeTo.TransformVector(positionDelta);
529 }
530 positionDelta.x *= physicsPositionInheritanceFactor.x;
531 positionDelta.y *= physicsPositionInheritanceFactor.y;
532
533 skeleton.PhysicsTranslate(positionDelta.x, positionDelta.y);
534 lastPosition = position;
535 }
536 if (physicsRotationInheritanceFactor != 0f) {
537 float rotation = GetPhysicsTransformRotation();
538 skeleton.PhysicsRotate(0, 0, physicsRotationInheritanceFactor * (rotation - lastRotation));
539 lastRotation = rotation;
540 }
541 }
542 }
543
544 protected Vector2 GetPhysicsTransformPosition () {
545 if (physicsMovementRelativeTo == null) {
546 return transform.position;
547 } else {
548 if (physicsMovementRelativeTo == transform.parent)
549 return transform.localPosition;
550 else
551 return physicsMovementRelativeTo.InverseTransformPoint(transform.position);
552 }
553 }
554
555 protected float GetPhysicsTransformRotation () {
556 if (physicsMovementRelativeTo == null) {
557 return this.transform.rotation.eulerAngles.z;
558 } else {
559 if (physicsMovementRelativeTo == this.transform.parent)
560 return this.transform.localRotation.eulerAngles.z;
561 else {
562 Quaternion relative = Quaternion.Inverse(physicsMovementRelativeTo.rotation) * this.transform.rotation;
563 return relative.eulerAngles.z;
564 }
565 }

574 public virtual void LateUpdate () {
575 if (!valid) return;
576
577#if UNITY_EDITOR && NEW_PREFAB_SYSTEM
578 // Don't store mesh or material at the prefab, otherwise it will permanently reload
579 UnityEditor.PrefabAssetType prefabType = UnityEditor.PrefabUtility.GetPrefabAssetType(this);
580 if (UnityEditor.PrefabUtility.IsPartOfPrefabAsset(this) &&
581 (prefabType == UnityEditor.PrefabAssetType.Regular || prefabType == UnityEditor.PrefabAssetType.Variant)) {
582 return;

585#endif

587 if (updateMode != UpdateMode.FullUpdate) return;

589 LateUpdateMesh();

592 public virtual void LateUpdateMesh () {
593#if SPINE_OPTIONAL_RENDEROVERRIDE
594 bool doMeshOverride = generateMeshOverride != null;
595 if ((!meshRenderer || !meshRenderer.enabled) && !doMeshOverride) return;
596#else
597 const bool doMeshOverride = false;
598 if (!meshRenderer.enabled) return;
599#endif
600 SkeletonRendererInstruction currentInstructions = this.currentInstructions;
601 ExposedList<SubmeshInstruction> workingSubmeshInstructions = currentInstructions.submeshInstructions;
602 MeshRendererBuffers.SmartMesh currentSmartMesh = rendererBuffers.GetNextMesh(); // Double-buffer for performance.
603
604 bool updateTriangles;
605
606 if (this.singleSubmesh) {
607 // STEP 1. Determine a SmartMesh.Instruction. Split up instructions into submeshes. =============================================
608 MeshGenerator.GenerateSingleSubmeshInstruction(currentInstructions, skeleton, skeletonDataAsset.atlasAssets[0].PrimaryMaterial);
609
610 // STEP 1.9. Post-process workingInstructions. ==================================================================================
611#if SPINE_OPTIONAL_MATERIALOVERRIDE
612 if (customMaterialOverride.Count > 0) // isCustomMaterialOverridePopulated
613 MeshGenerator.TryReplaceMaterials(workingSubmeshInstructions, customMaterialOverride);
614#endif
615
616 // STEP 2. Update vertex buffer based on verts from the attachments. ===========================================================
617 meshGenerator.settings = new MeshGenerator.Settings {
618 pmaVertexColors = this.pmaVertexColors,
619 zSpacing = this.zSpacing,
620 useClipping = this.useClipping,
621 tintBlack = this.tintBlack,
622 calculateTangents = this.calculateTangents,
623 addNormals = this.addNormals
624 };
625 meshGenerator.Begin();
626 updateTriangles = SkeletonRendererInstruction.GeometryNotEqual(currentInstructions, currentSmartMesh.instructionUsed);
627 if (currentInstructions.hasActiveClipping) {
628 meshGenerator.AddSubmesh(workingSubmeshInstructions.Items[0], updateTriangles);
629 } else {
630 meshGenerator.BuildMeshWithArrays(currentInstructions, updateTriangles);
631 }
632
633 } else {
634 // STEP 1. Determine a SmartMesh.Instruction. Split up instructions into submeshes. =============================================
635 MeshGenerator.GenerateSkeletonRendererInstruction(currentInstructions, skeleton, customSlotMaterials, separatorSlots, doMeshOverride, this.immutableTriangles);
636
637 // STEP 1.9. Post-process workingInstructions. ==================================================================================
638#if SPINE_OPTIONAL_MATERIALOVERRIDE
639 if (customMaterialOverride.Count > 0) // isCustomMaterialOverridePopulated
640 MeshGenerator.TryReplaceMaterials(workingSubmeshInstructions, customMaterialOverride);
641#endif
642
643#if SPINE_OPTIONAL_RENDEROVERRIDE
644 if (doMeshOverride) {
645 this.generateMeshOverride(currentInstructions);
646 if (disableRenderingOnOverride) return;
647 }
648#endif
649
650 updateTriangles = SkeletonRendererInstruction.GeometryNotEqual(currentInstructions, currentSmartMesh.instructionUsed);
651
652 // STEP 2. Update vertex buffer based on verts from the attachments. ===========================================================
653 meshGenerator.settings = new MeshGenerator.Settings {
654 pmaVertexColors = this.pmaVertexColors,
655 zSpacing = this.zSpacing,
656 useClipping = this.useClipping,
657 tintBlack = this.tintBlack,
658 calculateTangents = this.calculateTangents,
659 addNormals = this.addNormals
660 };
661 meshGenerator.Begin();
662 if (currentInstructions.hasActiveClipping)
663 meshGenerator.BuildMesh(currentInstructions, updateTriangles);
664 else
665 meshGenerator.BuildMeshWithArrays(currentInstructions, updateTriangles);
666 }
667
668 if (OnPostProcessVertices != null) OnPostProcessVertices.Invoke(this.meshGenerator.Buffers);
669
670 // STEP 3. Move the mesh data into a UnityEngine.Mesh ===========================================================================
671 Mesh currentMesh = currentSmartMesh.mesh;
672 meshGenerator.FillVertexData(currentMesh);
673
674 rendererBuffers.UpdateSharedMaterials(workingSubmeshInstructions);
675
676 bool materialsChanged = rendererBuffers.MaterialsChangedInLastUpdate();
677 if (updateTriangles) { // Check if the triangles should also be updated.
678 meshGenerator.FillTriangles(currentMesh);
679 meshRenderer.sharedMaterials = rendererBuffers.GetUpdatedSharedMaterialsArray();
680 } else if (materialsChanged) {
681 meshRenderer.sharedMaterials = rendererBuffers.GetUpdatedSharedMaterialsArray();
682 }
683 if (materialsChanged && (this.maskMaterials.AnyMaterialCreated)) {
684 this.maskMaterials = new SpriteMaskInteractionMaterials();
685 }
686
687 meshGenerator.FillLateVertexData(currentMesh);
688
689 // STEP 4. The UnityEngine.Mesh is ready. Set it as the MeshFilter's mesh. Store the instructions used for that mesh. ===========
690 if (meshFilter)
691 meshFilter.sharedMesh = currentMesh;
692 currentSmartMesh.instructionUsed.Set(currentInstructions);
693
694#if BUILT_IN_SPRITE_MASK_COMPONENT
695 if (meshRenderer != null) {
696 AssignSpriteMaskMaterials();
697 }
698#endif
699#if SPINE_OPTIONAL_ON_DEMAND_LOADING
700 if (Application.isPlaying)
701 HandleOnDemandLoading();
702#endif
703
704#if PER_MATERIAL_PROPERTY_BLOCKS
705 if (fixDrawOrder && meshRenderer.sharedMaterials.Length > 2) {
706 SetMaterialSettingsToFixDrawOrder();
707 }
708#endif
709
710 if (OnMeshAndMaterialsUpdated != null)
711 OnMeshAndMaterialsUpdated(this);
712 }

737 public void FindAndApplySeparatorSlots (System.Func<string, bool> slotNamePredicate, bool clearExistingSeparators = true, bool updateStringArray = false) {
738 if (slotNamePredicate == null) return;
739 if (!valid) return;
740
741 if (clearExistingSeparators)
742 separatorSlots.Clear();
743
744 ExposedList<Slot> slots = skeleton.Slots;
745 foreach (Slot slot in slots) {
746 if (slotNamePredicate.Invoke(slot.Data.Name))
747 separatorSlots.Add(slot);
748 }
749
750 if (updateStringArray) {
751 List<string> detectedSeparatorNames = new List<string>();
752 foreach (Slot slot in skeleton.Slots) {
753 string slotName = slot.Data.Name;
754 if (slotNamePredicate.Invoke(slotName))
755 detectedSeparatorNames.Add(slotName);
756 }
757 if (!clearExistingSeparators) {
758 string[] originalNames = this.separatorSlotNames;
759 foreach (string originalName in originalNames)
760 detectedSeparatorNames.Add(originalName);
761 }
762
763 this.separatorSlotNames = detectedSeparatorNames.ToArray();
764 }
765
766 }

780 Debug.LogWarning(separatorSlotNames[i] + " is not a slot in " + skeletonDataAsset.skeletonJSON.name);
781 }
782#endif

786#if BUILT_IN_SPRITE_MASK_COMPONENT
787 private void AssignSpriteMaskMaterials () {
788#if UNITY_EDITOR
789 if (!Application.isPlaying && !UnityEditor.EditorApplication.isUpdating) {
790 EditorFixStencilCompParameters();
791 }
792#endif
793
794 if (Application.isPlaying) {
795 if (maskInteraction != SpriteMaskInteraction.None && maskMaterials.materialsMaskDisabled.Length == 0)
796 maskMaterials.materialsMaskDisabled = meshRenderer.sharedMaterials;
797 }
798
799 if (maskMaterials.materialsMaskDisabled.Length > 0 && maskMaterials.materialsMaskDisabled[0] != null &&
800 maskInteraction == SpriteMaskInteraction.None) {
