131 public SkeletonData GetSkeletonData (bool quiet) {
132 if (skeletonJSON == null) {
133#if UNITY_EDITOR
134 if (!errorIfSkeletonFileNullGlobal) quiet = true;
135#endif
136 if (!quiet)
137 Debug.LogError("Skeleton JSON file not set for SkeletonData asset: " + name, this);
138 Clear();
139 return null;
140 }
141
142 // Disabled to support attachmentless/skinless SkeletonData.
143 // if (atlasAssets == null) {
144 // atlasAssets = new AtlasAsset[0];
145 // if (!quiet)
146 // Debug.LogError("Atlas not set for SkeletonData asset: " + name, this);
147 // Clear();
148 // return null;
149 // }
150 // #if !SPINE_TK2D
151 // if (atlasAssets.Length == 0) {
152 // Clear();
153 // return null;
154 // }
155 // #else
156 // if (atlasAssets.Length == 0 && spriteCollection == null) {
157 // Clear();
158 // return null;
159 // }
160 // #endif
161
162 if (skeletonData != null)
163 return skeletonData;
164
165 AttachmentLoader attachmentLoader;
166 float skeletonDataScale;
167 Atlas[] atlasArray = this.GetAtlasArray();
168
169#if !SPINE_TK2D
170 attachmentLoader = (atlasArray.Length == 0) ? (AttachmentLoader)new RegionlessAttachmentLoader() : (AttachmentLoader)new AtlasAttachmentLoader(atlasArray);
171 skeletonDataScale = scale;
172#else
173 if (spriteCollection != null) {
174 attachmentLoader = new Spine.Unity.TK2D.SpriteCollectionAttachmentLoader(spriteCollection);
175 skeletonDataScale = (1.0f / (spriteCollection.invOrthoSize * spriteCollection.halfTargetHeight) * scale);
176 } else {
177 if (atlasArray.Length == 0) {
178 Reset();
179 if (!quiet) Debug.LogError("Atlas not set for SkeletonData asset: " + name, this);
180 return null;
181 }
182 attachmentLoader = new AtlasAttachmentLoader(atlasArray);
183 skeletonDataScale = scale;
184 }
185#endif
186
187 bool hasBinaryExtension = skeletonJSON.name.ToLower().Contains(".skel");
188 SkeletonData loadedSkeletonData = null;
189
190 try {
191 if (hasBinaryExtension)
192 loadedSkeletonData = SkeletonDataAsset.ReadSkeletonData(skeletonJSON.bytes, attachmentLoader, skeletonDataScale);
193 else
194 loadedSkeletonData = SkeletonDataAsset.ReadSkeletonData(skeletonJSON.text, attachmentLoader, skeletonDataScale);
195 } catch (Exception ex) {
196 if (!quiet)
197 Debug.LogError("Error reading skeleton JSON file for SkeletonData asset: " + name + "\n" + ex.Message + "\n" + ex.StackTrace, skeletonJSON);
198 }
199
200#if UNITY_EDITOR
201 if (loadedSkeletonData == null && !quiet && skeletonJSON != null) {
202 string problemDescription = null;
203 bool isSpineSkeletonData;
204 SkeletonDataCompatibility.VersionInfo fileVersion = SkeletonDataCompatibility.GetVersionInfo(skeletonJSON, out isSpineSkeletonData, ref problemDescription);
205 if (problemDescription != null) {
206 if (!quiet)
207 Debug.LogError(problemDescription, skeletonJSON);
208 return null;
209 }
210 CompatibilityProblemInfo compatibilityProblemInfo = SkeletonDataCompatibility.GetCompatibilityProblemInfo(fileVersion);
211 if (compatibilityProblemInfo != null) {
212 SkeletonDataCompatibility.DisplayCompatibilityProblem(compatibilityProblemInfo.DescriptionString(), skeletonJSON);
213 return null;
214 }
215 }
216#endif
217 if (loadedSkeletonData == null)
218 return null;
219
220 if (skeletonDataModifiers != null) {
221 foreach (SkeletonDataModifierAsset modifier in skeletonDataModifiers) {
222 if (modifier != null && !(isUpgradingBlendModeMaterials && modifier is BlendModeMaterialsAsset)) {
223 modifier.Apply(loadedSkeletonData);
224 }
225 }
226 }
227 if (!isUpgradingBlendModeMaterials)
228 blendModeMaterials.ApplyMaterials(loadedSkeletonData);
229
230 this.InitializeWithData(loadedSkeletonData);
231
232 return skeletonData;
233 }

241 public void FillStateData (bool quiet = false) {
242 if (stateData != null) {
243 stateData.DefaultMix = defaultMix;
244
245 for (int i = 0, n = fromAnimation.Length; i < n; i++) {
246 string fromAnimationName = fromAnimation[i];
247 string toAnimationName = toAnimation[i];
248 if (fromAnimationName.Length == 0 || toAnimationName.Length == 0)
249 continue;
250#if UNITY_EDITOR
251 if (skeletonData.FindAnimation(fromAnimationName) == null) {
252 if (!quiet) Debug.LogError(
253 string.Format("Custom Mix Durations: Animation '{0}' not found, was it renamed?",
254 fromAnimationName), this);
255 continue;
256 }
257 if (skeletonData.FindAnimation(toAnimationName) == null) {
258 if (!quiet) Debug.LogError(
259 string.Format("Custom Mix Durations: Animation '{0}' not found, was it renamed?",
260 toAnimationName), this);
261 continue;
262 }
263#endif
264 stateData.SetMix(fromAnimationName, toAnimationName, duration[i]);
265 }
266 }
267 }

269 internal Atlas[] GetAtlasArray () {
270 List<Atlas> returnList = new System.Collections.Generic.List<Atlas>(atlasAssets.Length);
271 for (int i = 0; i < atlasAssets.Length; i++) {
272 AtlasAssetBase aa = atlasAssets[i];
273 if (aa == null) continue;
274 Atlas a = aa.GetAtlas();
275 if (a == null) continue;
276 returnList.Add(a);
277 }
278 return returnList.ToArray();
279 }
