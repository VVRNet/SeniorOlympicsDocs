34namespace Spine.Unity.Examples {
35 public class Goblins : MonoBehaviour {
36 SkeletonAnimation skeletonAnimation;
37 Bone headBone;
38 bool girlSkin;
39
40 [Range(-360, 360)]
41 public float extraRotation;
42
43 public void Start () {
44 skeletonAnimation = GetComponent<SkeletonAnimation>();
45 headBone = skeletonAnimation.Skeleton.FindBone("head");
46 skeletonAnimation.UpdateLocal += UpdateLocal;
47 }
48
49 // This is called after the animation is applied to the skeleton and can be used to adjust the bones dynamically.
50 public void UpdateLocal (ISkeletonAnimation skeletonRenderer) {
51 headBone.Rotation += extraRotation;
52 }
53
54 public void OnMouseDown () {
55 skeletonAnimation.Skeleton.SetSkin(girlSkin ? "goblin" : "goblingirl");
56 skeletonAnimation.Skeleton.SetSlotsToSetupPose();
57
58 girlSkin = !girlSkin;
59
60 if (girlSkin) {
61 skeletonAnimation.Skeleton.SetAttachment("right-hand-item", null);
62 skeletonAnimation.Skeleton.SetAttachment("left-hand-item", "spear");
63 } else
64 skeletonAnimation.Skeleton.SetAttachment("left-hand-item", "dagger");
65 }
66 }
67}
