183 protected virtual void PhysicsUpdate (bool skeletonAnimationUsesFixedUpdate) {
184 Vector2 callbackDisplacement = tempSkeletonDisplacement;
185 float callbackRotation = tempSkeletonRotation;
186
187 bool isApplyAtRigidbodyAllowed = PhysicsUpdateRootMotionOverride == null || !disableOnOverride;
188 if (isApplyAtRigidbodyAllowed) {
189 if (rigidBody2D != null) {
190 Vector2 gravityAndVelocityMovement = Vector2.zero;
191 if (applyRigidbody2DGravity) {
192 float deltaTime = Time.fixedDeltaTime;
193 float deltaTimeSquared = (deltaTime * deltaTime);
194
195 rigidBody2D.velocity += rigidBody2D.gravityScale * Physics2D.gravity * deltaTime;
196 gravityAndVelocityMovement = 0.5f * rigidBody2D.gravityScale * Physics2D.gravity * deltaTimeSquared +
197 rigidBody2D.velocity * deltaTime;
198 }
199
200 Vector2 rigidbodyDisplacement2D = new Vector2(rigidbodyDisplacement.x, rigidbodyDisplacement.y);
201 rigidBody2D.MovePosition(gravityAndVelocityMovement + new Vector2(rigidBody2D.position.x, rigidBody2D.position.y)
202 + rigidbodyDisplacement2D + additionalRigidbody2DMovement);
203 rigidBody2D.MoveRotation(rigidbody2DRotation + rigidBody2D.rotation);
204 } else if (rigidBody != null) {
205 rigidBody.MovePosition(rigidBody.position
206 + new Vector3(rigidbodyDisplacement.x, rigidbodyDisplacement.y, rigidbodyDisplacement.z));
207 rigidBody.MoveRotation(rigidBody.rotation * rigidbodyLocalRotation);
208 }
209 }
210
211 previousRigidbodyRootMotion = rigidbodyDisplacement;
212 if (accumulatedUntilFixedUpdate) {
213 Vector2 parentBoneScale;
214 GetScaleAffectingRootMotion(out parentBoneScale);
215 ClearEffectiveBoneOffsets(parentBoneScale);
216 skeletonComponent.Skeleton.UpdateWorldTransform(Skeleton.Physics.Pose);
217 }
218 ClearRigidbodyTempMovement();
219
220 if (PhysicsUpdateRootMotionOverride != null)
221 PhysicsUpdateRootMotionOverride(this, callbackDisplacement, callbackRotation);
222 }

280 public void AdjustRootMotionToDistance (Vector2 distanceToTarget, int trackIndex = 0, bool adjustX = true, bool adjustY = true,
281 float minX = 0, float maxX = float.MaxValue, float minY = 0, float maxY = float.MaxValue,
282 bool allowXTranslation = false, bool allowYTranslation = false) {
283
284 Vector2 distanceToTargetSkeletonSpace = (Vector2)transform.InverseTransformVector(distanceToTarget);
285 Vector2 scaleAffectingRootMotion = GetScaleAffectingRootMotion();
286 if (UsesRigidbody)
287 distanceToTargetSkeletonSpace -= tempSkeletonDisplacement;
288
289 Vector2 remainingRootMotionSkeletonSpace = GetRemainingRootMotion(trackIndex);
290 remainingRootMotionSkeletonSpace.Scale(scaleAffectingRootMotion);
291 if (remainingRootMotionSkeletonSpace.x == 0)
292 remainingRootMotionSkeletonSpace.x = 0.0001f;
293 if (remainingRootMotionSkeletonSpace.y == 0)
294 remainingRootMotionSkeletonSpace.y = 0.0001f;
295
296 if (adjustX)
297 rootMotionScaleX = Math.Min(maxX, Math.Max(minX, distanceToTargetSkeletonSpace.x / remainingRootMotionSkeletonSpace.x));
298 if (adjustY)
299 rootMotionScaleY = Math.Min(maxY, Math.Max(minY, distanceToTargetSkeletonSpace.y / remainingRootMotionSkeletonSpace.y));
300
301 if (allowXTranslation)
302 rootMotionTranslateXPerY = (distanceToTargetSkeletonSpace.x - remainingRootMotionSkeletonSpace.x * rootMotionScaleX) / remainingRootMotionSkeletonSpace.y;
303 if (allowYTranslation)
304 rootMotionTranslateYPerX = (distanceToTargetSkeletonSpace.y - remainingRootMotionSkeletonSpace.y * rootMotionScaleY) / remainingRootMotionSkeletonSpace.x;
305 }

311 public Vector2 GetAnimationRootMotion (float startTime, float endTime,
312 Animation animation) {
313
314 if (startTime == endTime)
315 return Vector2.zero;
316
317 TranslateTimeline translateTimeline = animation.FindTranslateTimelineForBone(rootMotionBoneIndex);
318 TranslateXTimeline xTimeline = animation.FindTimelineForBone<TranslateXTimeline>(rootMotionBoneIndex);
319 TranslateYTimeline yTimeline = animation.FindTimelineForBone<TranslateYTimeline>(rootMotionBoneIndex);
320
321 // Non-looped base
322 Vector2 endPos = Vector2.zero;
323 Vector2 startPos = Vector2.zero;
324 if (translateTimeline != null) {
325 endPos = translateTimeline.Evaluate(endTime);
326 startPos = translateTimeline.Evaluate(startTime);
327 } else if (xTimeline != null || yTimeline != null) {
328 endPos = TimelineExtensions.Evaluate(xTimeline, yTimeline, endTime);
329 startPos = TimelineExtensions.Evaluate(xTimeline, yTimeline, startTime);
330 }
331 TransformConstraint[] transformConstraintsItems = skeletonComponent.Skeleton.TransformConstraints.Items;
332 foreach (int constraintIndex in this.transformConstraintIndices) {
333 TransformConstraint constraint = transformConstraintsItems[constraintIndex];
334 ApplyConstraintToPos(animation, constraint, constraintIndex, endTime, false, ref endPos);
335 ApplyConstraintToPos(animation, constraint, constraintIndex, startTime, true, ref startPos);
336 }
337 Vector2 currentDelta = endPos - startPos;
338
339 // Looped additions
340 if (startTime > endTime) {
341 Vector2 loopPos = Vector2.zero;
342 Vector2 zeroPos = Vector2.zero;
343 if (translateTimeline != null) {
344 loopPos = translateTimeline.Evaluate(animation.Duration);
345 zeroPos = translateTimeline.Evaluate(0);
346 } else if (xTimeline != null || yTimeline != null) {
347 loopPos = TimelineExtensions.Evaluate(xTimeline, yTimeline, animation.Duration);
348 zeroPos = TimelineExtensions.Evaluate(xTimeline, yTimeline, 0);
349 }
350 foreach (int constraintIndex in this.transformConstraintIndices) {
351 TransformConstraint constraint = transformConstraintsItems[constraintIndex];
352 ApplyConstraintToPos(animation, constraint, constraintIndex, animation.Duration, false, ref loopPos);
353 ApplyConstraintToPos(animation, constraint, constraintIndex, 0, false, ref zeroPos);
354 }
355 currentDelta += loopPos - zeroPos;
356 }
357 UpdateLastConstraintPos(transformConstraintsItems);
358 return currentDelta;
359 }

365 public float GetAnimationRootMotionRotation (float startTime, float endTime,
366 Animation animation) {
367
368 if (startTime == endTime)
369 return 0;
370
371 RotateTimeline rotateTimeline = animation.FindTimelineForBone<RotateTimeline>(rootMotionBoneIndex);
372
373 // Non-looped base
374 float endRotation = 0;
375 float startRotation = 0;
376 if (rotateTimeline != null) {
377 endRotation = rotateTimeline.Evaluate(endTime);
378 startRotation = rotateTimeline.Evaluate(startTime);
379 }
380 TransformConstraint[] transformConstraintsItems = skeletonComponent.Skeleton.TransformConstraints.Items;
381 foreach (int constraintIndex in this.transformConstraintIndices) {
382 TransformConstraint constraint = transformConstraintsItems[constraintIndex];
383 ApplyConstraintToRotation(animation, constraint, constraintIndex, endTime, false, ref endRotation);
384 ApplyConstraintToRotation(animation, constraint, constraintIndex, startTime, true, ref startRotation);
385 }
386 float currentDelta = endRotation - startRotation;
387
388 // Looped additions
389 if (startTime > endTime) {
390 float loopRotation = 0;
391 float zeroPos = 0;
392 if (rotateTimeline != null) {
393 loopRotation = rotateTimeline.Evaluate(animation.Duration);
394 zeroPos = rotateTimeline.Evaluate(0);
395 }
396 foreach (int constraintIndex in this.transformConstraintIndices) {
397 TransformConstraint constraint = transformConstraintsItems[constraintIndex];
398 ApplyConstraintToRotation(animation, constraint, constraintIndex, animation.Duration, false, ref loopRotation);
399 ApplyConstraintToRotation(animation, constraint, constraintIndex, 0, false, ref zeroPos);
400 }
401 currentDelta += loopRotation - zeroPos;
402 }
403 UpdateLastConstraintRotation(transformConstraintsItems);
404 return currentDelta;
405 }

407 void ApplyConstraintToPos (Animation animation, TransformConstraint constraint,
408 int constraintIndex, float time, bool useLastConstraintPos, ref Vector2 pos) {
409 TransformConstraintTimeline timeline = animation.FindTransformConstraintTimeline(constraintIndex);
410 if (timeline == null)
411 return;
412 Vector2 mixXY = timeline.EvaluateTranslateXYMix(time);
413 Vector2 invMixXY = timeline.EvaluateTranslateXYMix(time);
414 Vector2 constraintPos;
415 if (useLastConstraintPos)
416 constraintPos = transformConstraintLastPos[GetConstraintLastPosIndex(constraintIndex)];
417 else {
418 Bone targetBone = constraint.Target;
419 constraintPos = new Vector2(targetBone.X, targetBone.Y);
420 }
421 pos = new Vector2(
422 pos.x * invMixXY.x + constraintPos.x * mixXY.x,
423 pos.y * invMixXY.y + constraintPos.y * mixXY.y);
424 }
425
426 void ApplyConstraintToRotation (Animation animation, TransformConstraint constraint,
427 int constraintIndex, float time, bool useLastConstraintRotation, ref float rotation) {
428 TransformConstraintTimeline timeline = animation.FindTransformConstraintTimeline(constraintIndex);
429 if (timeline == null)
430 return;
431 float mixRotate = timeline.EvaluateRotateMix(time);
432 float invMixRotate = timeline.EvaluateRotateMix(time);
433 float constraintRotation;
434 if (useLastConstraintRotation)
435 constraintRotation = transformConstraintLastRotation[GetConstraintLastPosIndex(constraintIndex)];
436 else {
437 Bone targetBone = constraint.Target;
438 constraintRotation = targetBone.Rotation;
439 }
440 rotation = rotation * invMixRotate + constraintRotation * mixRotate;
441 }

459 public RootMotionInfo GetAnimationRootMotionInfo (Animation animation, float currentTime) {
460 RootMotionInfo rootMotion = new RootMotionInfo();
461 float duration = animation.Duration;
462 float mid = duration * 0.5f;
463 rootMotion.timeIsPastMid = currentTime > mid;
464 TranslateTimeline timeline = animation.FindTranslateTimelineForBone(rootMotionBoneIndex);
465 if (timeline != null) {
466 rootMotion.start = timeline.Evaluate(0);
467 rootMotion.current = timeline.Evaluate(currentTime);
468 rootMotion.mid = timeline.Evaluate(mid);
469 rootMotion.end = timeline.Evaluate(duration);
470 return rootMotion;
471 }
472 TranslateXTimeline xTimeline = animation.FindTimelineForBone<TranslateXTimeline>(rootMotionBoneIndex);
473 TranslateYTimeline yTimeline = animation.FindTimelineForBone<TranslateYTimeline>(rootMotionBoneIndex);
474 if (xTimeline != null || yTimeline != null) {
475 rootMotion.start = TimelineExtensions.Evaluate(xTimeline, yTimeline, 0);
476 rootMotion.current = TimelineExtensions.Evaluate(xTimeline, yTimeline, currentTime);
477 rootMotion.mid = TimelineExtensions.Evaluate(xTimeline, yTimeline, mid);
478 rootMotion.end = TimelineExtensions.Evaluate(xTimeline, yTimeline, duration);
479 return rootMotion;
480 }
481 return rootMotion;
482 }

489 void FindTransformConstraintsAffectingBone () {
490 ExposedList<TransformConstraint> constraints = skeletonComponent.Skeleton.TransformConstraints;
491 TransformConstraint[] constraintsItems = constraints.Items;
492 for (int i = 0, n = constraints.Count; i < n; ++i) {
493 TransformConstraint constraint = constraintsItems[i];
494 if (constraint.Bones.Contains(rootMotionBone)) {
495 transformConstraintIndices.Add(i);
496 Bone targetBone = constraint.Target;
497 Vector2 constraintPos = new Vector2(targetBone.X, targetBone.Y);
498 transformConstraintLastPos.Add(constraintPos);
499 transformConstraintLastRotation.Add(targetBone.Rotation);
500 }
501 }

504 Vector2 GetTimelineMovementDelta (float startTime, float endTime,
505 TranslateXTimeline xTimeline, TranslateYTimeline yTimeline, Animation animation) {
506
507 Vector2 currentDelta;
508 if (startTime > endTime) // Looped
509 currentDelta =
510 (TimelineExtensions.Evaluate(xTimeline, yTimeline, animation.Duration)
511 - TimelineExtensions.Evaluate(xTimeline, yTimeline, startTime))
512 + (TimelineExtensions.Evaluate(xTimeline, yTimeline, endTime)
513 - TimelineExtensions.Evaluate(xTimeline, yTimeline, 0));
514 else if (startTime != endTime) // Non-looped
515 currentDelta = TimelineExtensions.Evaluate(xTimeline, yTimeline, endTime)
516 - TimelineExtensions.Evaluate(xTimeline, yTimeline, startTime);
517 else
518 currentDelta = Vector2.zero;
519 return currentDelta;
520 }

531 void HandleUpdateLocal (ISkeletonAnimation animatedSkeletonComponent) {
532 if (!this.isActiveAndEnabled)
533 return; // Root motion is only applied when component is enabled.
534
535 Vector2 boneLocalDelta = CalculateAnimationsMovementDelta();
536 Vector2 parentBoneScale;
537 Vector2 totalScale;
538 Vector2 skeletonTranslationDelta = GetSkeletonSpaceMovementDelta(boneLocalDelta, out parentBoneScale, out totalScale);
539 float skeletonRotationDelta = 0;
540 if (transformRotation) {
541 float boneLocalDeltaRotation = CalculateAnimationsRotationDelta();
542 boneLocalDeltaRotation *= rootMotionScaleRotation;
543 skeletonRotationDelta = GetSkeletonSpaceRotationDelta(boneLocalDeltaRotation, totalScale);
544 }
545
546 bool usesFixedUpdate = SkeletonAnimationUsesFixedUpdate;
547 ApplyRootMotion(skeletonTranslationDelta, skeletonRotationDelta, parentBoneScale, usesFixedUpdate);
548
549 if (usesFixedUpdate)
550 PhysicsUpdate(usesFixedUpdate);
551 }
552
553 void ApplyRootMotion (Vector2 skeletonTranslationDelta, float skeletonRotationDelta, Vector2 parentBoneScale,
554 bool skeletonAnimationUsesFixedUpdate) {
555
556 // Accumulated displacement is applied on the next Physics update in FixedUpdate.
557 // Until the next Physics update, tempSkeletonDisplacement and tempSkeletonRotation
558 // are offsetting bone locations to prevent stutter which would otherwise occur if
559 // we don't move every Update.
560 bool usesRigidbody = this.UsesRigidbody;
561 bool applyToTransform = !usesRigidbody && (ProcessRootMotionOverride == null || !disableOnOverride);
562 accumulatedUntilFixedUpdate = !applyToTransform && !skeletonAnimationUsesFixedUpdate;
563
564 if (ProcessRootMotionOverride != null)
565 ProcessRootMotionOverride(this, skeletonTranslationDelta, skeletonRotationDelta);
566
567 // Apply root motion to Transform or update values applied to RigidBody later (must happen in FixedUpdate).
568 if (usesRigidbody) {
569 rigidbodyDisplacement += transform.TransformVector(skeletonTranslationDelta);
570 if (skeletonRotationDelta != 0.0f) {
571 if (rigidBody != null) {
572 Quaternion addedWorldRotation = Quaternion.Euler(0, 0, skeletonRotationDelta);
573 rigidbodyLocalRotation = rigidbodyLocalRotation * addedWorldRotation;
574 } else if (rigidBody2D != null) {
575 Vector3 lossyScale = transform.lossyScale;
576 float rotationSign = lossyScale.x * lossyScale.y > 0 ? 1 : -1;
577 rigidbody2DRotation += rotationSign * skeletonRotationDelta;
578 }
579 }
580 } else if (applyToTransform) {
581 transform.position += transform.TransformVector(skeletonTranslationDelta);
582 if (skeletonRotationDelta != 0.0f) {
583 Vector3 lossyScale = transform.lossyScale;
584 float rotationSign = lossyScale.x * lossyScale.y > 0 ? 1 : -1;
585 transform.Rotate(0, 0, rotationSign * skeletonRotationDelta);
586 }
587 }
588
589 tempSkeletonDisplacement += skeletonTranslationDelta;
590 tempSkeletonRotation += skeletonRotationDelta;
591 if (accumulatedUntilFixedUpdate) {
592 SetEffectiveBoneOffsetsTo(tempSkeletonDisplacement, tempSkeletonRotation, parentBoneScale);
593 } else {
594 ClearEffectiveBoneOffsets(parentBoneScale);
595 }
596 }

598 void ApplyTransformConstraints () {
599 rootMotionBone.AX = rootMotionBone.X;
600 rootMotionBone.AY = rootMotionBone.Y;
601 rootMotionBone.AppliedRotation = rootMotionBone.Rotation;
602 TransformConstraint[] transformConstraintsItems = skeletonComponent.Skeleton.TransformConstraints.Items;
603 foreach (int constraintIndex in this.transformConstraintIndices) {
604 TransformConstraint constraint = transformConstraintsItems[constraintIndex];
605 // apply the constraint and sets Bone.ax, Bone.ay and Bone.arotation values.
608 constraint.Update(Skeleton.Physics.None);
609 }

617 Vector2 GetScaleAffectingRootMotion (out Vector2 parentBoneScale) {
618 Skeleton skeleton = skeletonComponent.Skeleton;
619 Vector2 totalScale = Vector2.one;
620 totalScale.x *= skeleton.ScaleX;
621 totalScale.y *= skeleton.ScaleY;
622
623 parentBoneScale = Vector2.one;
624 Bone scaleBone = rootMotionBone;
625 while ((scaleBone = scaleBone.Parent) != null) {
626 parentBoneScale.x *= scaleBone.ScaleX;
627 parentBoneScale.y *= scaleBone.ScaleY;
628 }
629 totalScale = Vector2.Scale(totalScale, parentBoneScale);
630 totalScale *= AdditionalScale;
631 return totalScale;
632 }
633
634 Vector2 GetSkeletonSpaceMovementDelta (Vector2 boneLocalDelta, out Vector2 parentBoneScale, out Vector2 totalScale) {
635 Vector2 skeletonDelta = boneLocalDelta;
636 totalScale = GetScaleAffectingRootMotion(out parentBoneScale);
637 skeletonDelta.Scale(totalScale);
638
639 Vector2 rootMotionTranslation = new Vector2(
640 rootMotionTranslateXPerY * skeletonDelta.y,
641 rootMotionTranslateYPerX * skeletonDelta.x);
642
643 skeletonDelta.x *= rootMotionScaleX;
644 skeletonDelta.y *= rootMotionScaleY;
645 skeletonDelta.x += rootMotionTranslation.x;
646 skeletonDelta.y += rootMotionTranslation.y;
647
648 if (!transformPositionX) skeletonDelta.x = 0f;
649 if (!transformPositionY) skeletonDelta.y = 0f;
650 return skeletonDelta;
651 }

653 float GetSkeletonSpaceRotationDelta (float boneLocalDelta, Vector2 totalScaleAffectingRootMotion) {
654 float rotationSign = totalScaleAffectingRootMotion.x * totalScaleAffectingRootMotion.y > 0 ? 1 : -1;
655 return rotationSign * boneLocalDelta;
656 }

658 void SetEffectiveBoneOffsetsTo (Vector2 displacementSkeletonSpace, float rotationSkeletonSpace, Vector2 parentBoneScale) {
659
660 ApplyTransformConstraints();
661
662 // Move top level bones in opposite direction of the root motion bone
663 Skeleton skeleton = skeletonComponent.Skeleton;
664 foreach (Bone topLevelBone in topLevelBones) {
665 if (topLevelBone == rootMotionBone) {
666 if (transformPositionX) topLevelBone.X = displacementSkeletonSpace.x / skeleton.ScaleX;
667 if (transformPositionY) topLevelBone.Y = displacementSkeletonSpace.y / skeleton.ScaleY;
668 if (transformRotation) {
669 float rotationSign = skeleton.ScaleX * skeleton.ScaleY > 0 ? 1 : -1;
670 topLevelBone.Rotation = rotationSign * rotationSkeletonSpace;
671 }
672 } else {
673 bool useAppliedTransform = transformConstraintIndices.Count > 0;
674 float rootMotionBoneX = useAppliedTransform ? rootMotionBone.AX : rootMotionBone.X;
675 float rootMotionBoneY = useAppliedTransform ? rootMotionBone.AY : rootMotionBone.Y;
676

678 float offsetY = (initialOffset.y - rootMotionBoneY) * parentBoneScale.y;
679
680 if (transformPositionX) topLevelBone.X = (displacementSkeletonSpace.x / skeleton.ScaleX) + offsetX;
681 if (transformPositionY) topLevelBone.Y = (displacementSkeletonSpace.y / skeleton.ScaleY) + offsetY;
682
683 if (transformRotation) {
684 float rootMotionBoneRotation = useAppliedTransform ? rootMotionBone.AppliedRotation : rootMotionBone.Rotation;
685

687 float offsetRotation = (initialOffsetRotation - rootMotionBoneRotation) * parentBoneRotationSign;
688
689 float skeletonRotationSign = skeleton.ScaleX * skeleton.ScaleY > 0 ? 1 : -1;
690 topLevelBone.Rotation = (rotationSkeletonSpace * skeletonRotationSign) + offsetRotation;
691 }
692 }
