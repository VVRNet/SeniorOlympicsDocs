39 [CustomEditor(typeof(PointFollower)), CanEditMultipleObjects]
40 public class PointFollowerInspector : Editor {

45 #region Context Menu Item
46 [MenuItem("CONTEXT/SkeletonRenderer/Add PointFollower GameObject")]
47 static void AddBoneFollowerGameObject (MenuCommand cmd) {
48 SkeletonRenderer skeletonRenderer = cmd.context as SkeletonRenderer;
49 GameObject go = EditorInstantiation.NewGameObject("PointFollower", true);
50 Transform t = go.transform;
51 t.SetParent(skeletonRenderer.transform);
52 t.localPosition = Vector3.zero;

54 PointFollower f = go.AddComponent<PointFollower>();
55 f.skeletonRenderer = skeletonRenderer;
56
57 EditorGUIUtility.PingObject(t);

59 Undo.RegisterCreatedObjectUndo(go, "Add PointFollower");
60 }

70 void OnEnable () {
71 skeletonRenderer = serializedObject.FindProperty("skeletonRenderer");
72 slotName = serializedObject.FindProperty("slotName");
73 pointAttachmentName = serializedObject.FindProperty("pointAttachmentName");
74
75 targetPointFollower = (PointFollower)target;
76 if (targetPointFollower.skeletonRenderer != null)
77 targetPointFollower.skeletonRenderer.Initialize(false);
78
79 if (!targetPointFollower.IsValid || needsReset) {
80 targetPointFollower.Initialize();
81 targetPointFollower.LateUpdate();
82 needsReset = false;
83 SceneView.RepaintAll();
84 }
85 }

87 public void OnSceneGUI () {
88 PointFollower tbf = target as PointFollower;
89 SkeletonRenderer skeletonRendererComponent = tbf.skeletonRenderer;
90 if (skeletonRendererComponent == null)
91 return;
92
93 Skeleton skeleton = skeletonRendererComponent.skeleton;
94 Transform skeletonTransform = skeletonRendererComponent.transform;
95
96 if (string.IsNullOrEmpty(pointAttachmentName.stringValue)) {
97 // Draw all active PointAttachments in the current skin
98 Skin currentSkin = skeleton.Skin;
99 if (currentSkin != skeleton.Data.DefaultSkin) DrawPointsInSkin(skeleton.Data.DefaultSkin, skeleton, skeletonTransform);
100 if (currentSkin != null) DrawPointsInSkin(currentSkin, skeleton, skeletonTransform);
101 } else {
102 Slot slot = skeleton.FindSlot(slotName.stringValue);
103 if (slot != null) {
104 int slotIndex = slot.Data.Index;
105 PointAttachment point = skeleton.GetAttachment(slotIndex, pointAttachmentName.stringValue) as PointAttachment;
106 if (point != null) {
107 DrawPointAttachmentWithLabel(point, slot.Bone, skeletonTransform);
108 }
109 }
110 }
111 }
112
113 static void DrawPointsInSkin (Skin skin, Skeleton skeleton, Transform transform) {
114 foreach (Skin.SkinEntry skinEntry in skin.Attachments) {
115 PointAttachment attachment = skinEntry.Attachment as PointAttachment;
116 if (attachment != null) {
117 Slot slot = skeleton.Slots.Items[skinEntry.SlotIndex];
118 DrawPointAttachmentWithLabel(attachment, slot.Bone, transform);
119 }
120 }
121 }
122
123 static void DrawPointAttachmentWithLabel (PointAttachment point, Bone bone, Transform transform) {
124 Vector3 labelOffset = new Vector3(0f, -0.2f, 0f);
125 SpineHandles.DrawPointAttachment(bone, point, transform);
126 Handles.Label(labelOffset + point.GetWorldPosition(bone, transform), point.Name, SpineHandles.PointNameStyle);
127 }
128
129 override public void OnInspectorGUI () {
130 if (serializedObject.isEditingMultipleObjects) {
131 if (needsReset) {
132 needsReset = false;
133 foreach (Object o in targets) {
134 BoneFollower bf = (BoneFollower)o;
135 bf.Initialize();
136 bf.LateUpdate();
137 }
138 SceneView.RepaintAll();
139 }
140
141 EditorGUI.BeginChangeCheck();
142 DrawDefaultInspector();
143 needsReset |= EditorGUI.EndChangeCheck();
144 return;
145 }
146
147 if (needsReset && Event.current.type == EventType.Layout) {
148 targetPointFollower.Initialize();
149 targetPointFollower.LateUpdate();
150 needsReset = false;
151 SceneView.RepaintAll();
152 }
153 serializedObject.Update();
154
155 DrawDefaultInspector();
156
157 // Find Renderer
158 if (skeletonRenderer.objectReferenceValue == null) {
159 SkeletonRenderer parentRenderer = targetPointFollower.GetComponentInParent<SkeletonRenderer>();
160 if (parentRenderer != null && parentRenderer.gameObject != targetPointFollower.gameObject) {
161 skeletonRenderer.objectReferenceValue = parentRenderer;
162 Debug.Log("Inspector automatically assigned PointFollower.SkeletonRenderer");
163 }
164 }
165
166 SkeletonRenderer skeletonRendererReference = skeletonRenderer.objectReferenceValue as SkeletonRenderer;
167 if (skeletonRendererReference != null) {
168 if (skeletonRendererReference.gameObject == targetPointFollower.gameObject) {
169 skeletonRenderer.objectReferenceValue = null;
170 EditorUtility.DisplayDialog("Invalid assignment.", "PointFollower can only follow a skeleton on a separate GameObject.\n\nCreate a new GameObject for your PointFollower, or choose a SkeletonRenderer from a different GameObject.", "Ok");
171 }
172 }
173
174 if (!targetPointFollower.IsValid) {
175 needsReset = true;
176 }
177
178 Event current = Event.current;
179 bool wasUndo = (current.type == EventType.ValidateCommand && current.commandName == "UndoRedoPerformed");
180 if (wasUndo)
181 targetPointFollower.Initialize();
182
183 serializedObject.ApplyModifiedProperties();
184 }
185 }
