37 [RequireComponent(typeof(SkeletonRenderer))]
38 public class SkeletonGhost : MonoBehaviour {
39 // Internal Settings
40 const HideFlags GhostHideFlags = HideFlags.HideInHierarchy;
41 const string GhostingShaderName = "Spine/Special/SkeletonGhost";
42
43 [Header("Animation")]
44 public bool ghostingEnabled = true;
45 [Tooltip("The time between invididual ghost pieces being spawned.")]
46 [UnityEngine.Serialization.FormerlySerializedAs("spawnRate")]
47 public float spawnInterval = 1f / 30f;
48 [Tooltip("Maximum number of ghosts that can exist at a time. If the fade speed is not fast enough, the oldest ghost will immediately disappear to enforce the maximum number.")]
49 public int maximumGhosts = 10;
50 public float fadeSpeed = 10;
51
52 [Header("Rendering")]
53 public Shader ghostShader;
54 public Color32 color = new Color32(0xFF, 0xFF, 0xFF, 0x00); // default for additive.
55 [Tooltip("Remember to set color alpha to 0 if Additive is true")]
56 public bool additive = true;
57 [Tooltip("0 is Color and Alpha, 1 is Alpha only.")]
58 [Range(0, 1)]
59 public float textureFade = 1;
60
61 [Header("Sorting")]
62 public bool sortWithDistanceOnly;
63 public float zOffset = 0f;
64
65 float nextSpawnTime;
66 SkeletonGhostRenderer[] pool;
67 int poolIndex = 0;
68 SkeletonRenderer skeletonRenderer;
69 MeshRenderer meshRenderer;
70 MeshFilter meshFilter;

72 readonly Dictionary<Material, Material> materialTable = new Dictionary<Material, Material>();
73
74 void Start () {
75 Initialize(false);
76 }
77
78 public void Initialize (bool overwrite) {
79 if (pool == null || overwrite) {
80 if (ghostShader == null)
81 ghostShader = Shader.Find(GhostingShaderName);
82
83 skeletonRenderer = GetComponent<SkeletonRenderer>();
84 meshFilter = GetComponent<MeshFilter>();
85 meshRenderer = GetComponent<MeshRenderer>();
86 nextSpawnTime = Time.time + spawnInterval;
87 pool = new SkeletonGhostRenderer[maximumGhosts];
88 for (int i = 0; i < maximumGhosts; i++) {
89 GameObject go = new GameObject(gameObject.name + " Ghost", typeof(SkeletonGhostRenderer));
90 pool[i] = go.GetComponent<SkeletonGhostRenderer>();
91 go.SetActive(false);
92 go.hideFlags = GhostHideFlags;
93 }
94
95 IAnimationStateComponent skeletonAnimation = skeletonRenderer as Spine.Unity.IAnimationStateComponent;
96 if (skeletonAnimation != null)
97 skeletonAnimation.AnimationState.Event += OnEvent;
98 }
99 }
100
101 //SkeletonAnimation
102 /*
103 * Int Value: 0 sets ghostingEnabled to false, 1 sets ghostingEnabled to true
104 * Float Value: Values greater than 0 set the spawnRate equal the float value
105 * String Value: Pass RGBA hex color values in to set the color property. IE: "A0FF8BFF"
106 */
107 void OnEvent (Spine.TrackEntry trackEntry, Spine.Event e) {
108 if (e.Data.Name.Equals("Ghosting", System.StringComparison.Ordinal)) {
109 ghostingEnabled = e.Int > 0;
110 if (e.Float > 0)
111 spawnInterval = e.Float;
112
113 if (!string.IsNullOrEmpty(e.String))
114 this.color = HexToColor(e.String);
115 }
116 }
117
118 //SkeletonAnimator
119 //SkeletonAnimator or Mecanim based animations only support toggling ghostingEnabled. Be sure not to set anything other than the Int param in Spine or String will take priority.
120 void Ghosting (float val) {
121 ghostingEnabled = val > 0;
122 }
123
124 void Update () {
125 if (!ghostingEnabled || poolIndex >= pool.Length)
126 return;
127
128 if (Time.time >= nextSpawnTime) {
129 GameObject go = pool[poolIndex].gameObject;
130
131 Material[] materials = meshRenderer.sharedMaterials;
132 for (int i = 0; i < materials.Length; i++) {
133 Material originalMat = materials[i];
134 Material ghostMat;
135 if (!materialTable.ContainsKey(originalMat)) {
136 ghostMat = new Material(originalMat) {
137 shader = ghostShader,
138 color = Color.white
139 };
140
141 if (ghostMat.HasProperty("_TextureFade"))
142 ghostMat.SetFloat("_TextureFade", textureFade);
143
144 materialTable.Add(originalMat, ghostMat);
145 } else {
146 ghostMat = materialTable[originalMat];
147 }
148
149 materials[i] = ghostMat;
150 }
151
152 Transform goTransform = go.transform;
153 goTransform.parent = transform;
154
155 pool[poolIndex].Initialize(meshFilter.sharedMesh, materials, color, additive, fadeSpeed, meshRenderer.sortingLayerID, (sortWithDistanceOnly) ? meshRenderer.sortingOrder : meshRenderer.sortingOrder - 1);
156
157 goTransform.localPosition = new Vector3(0f, 0f, zOffset);
158 goTransform.localRotation = Quaternion.identity;
159 goTransform.localScale = Vector3.one;
160
161 goTransform.parent = null;
162
163 poolIndex++;
164
165 if (poolIndex == pool.Length)
166 poolIndex = 0;
167
168 nextSpawnTime = Time.time + spawnInterval;
169 }
170 }
171
172 void OnDestroy () {
173 if (pool != null) {
174 for (int i = 0; i < maximumGhosts; i++)
175 if (pool[i] != null) pool[i].Cleanup();
176 }
177
178 foreach (Material mat in materialTable.Values)
179 Destroy(mat);
180 }

182 // based on UnifyWiki http://wiki.unity3d.com/index.php?title=HexConverter
183 static Color32 HexToColor (string hex) {
184 const System.Globalization.NumberStyles HexNumber = System.Globalization.NumberStyles.HexNumber;
185
186 if (hex.Length < 6)
187 return Color.magenta;
188
189 hex = hex.Replace("#", "");
190 byte r = byte.Parse(hex.Substring(0, 2), HexNumber);
191 byte g = byte.Parse(hex.Substring(2, 2), HexNumber);
192 byte b = byte.Parse(hex.Substring(4, 2), HexNumber);
193 byte a = 0xFF;
194 if (hex.Length == 8)
195 a = byte.Parse(hex.Substring(6, 2), HexNumber);
196
197 return new Color32(r, g, b, a);
198 }
199 }
