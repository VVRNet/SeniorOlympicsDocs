135 public void Update (Physics physics) {
136 float mix = this.mix;
137 if (mix == 0) return;
138
139 bool x = data.x > 0, y = data.y > 0, rotateOrShearX = data.rotate > 0 || data.shearX > 0, scaleX = data.scaleX > 0;
140 Bone bone = this.bone;
141 float l = bone.data.length;
142
143 switch (physics) {
144 case Physics.None:
145 return;
146 case Physics.Reset:
147 Reset();
148 goto case Physics.Update; // Fall through.
149 case Physics.Update:
150 float delta = Math.Max(skeleton.time - lastTime, 0);
151 remaining += delta;
152 lastTime = skeleton.time;
153
154 float bx = bone.worldX, by = bone.worldY;
155 if (reset) {
156 reset = false;
157 ux = bx;
158 uy = by;
159 } else {
160 float a = this.remaining, i = inertia, q = data.limit * delta, t = data.step, f = skeleton.data.referenceScale;
161 if (x || y) {
162 if (x) {
163 float u = (ux - bx) * i;
164 xOffset += u > q ? q : u < -q ? -q : u;
165 ux = bx;
166 }
167 if (y) {
168 float u = (uy - by) * i;
169 yOffset += u > q ? q : u < -q ? -q : u;
170 uy = by;
171 }
172 if (a >= t) {
173 float m = massInverse * t, e = strength, w = wind * f, g = (Bone.yDown ? -gravity : gravity) * f;
174 float d = (float)Math.Pow(damping, 60 * t);
175 do {
176 if (x) {
177 xVelocity += (w - xOffset * e) * m;
178 xOffset += xVelocity * t;
179 xVelocity *= d;
180 }
181 if (y) {
182 yVelocity -= (g + yOffset * e) * m;
183 yOffset += yVelocity * t;
184 yVelocity *= d;
185 }
186 a -= t;
187 } while (a >= t);
188 }
189 if (x) bone.worldX += xOffset * mix * data.x;
190 if (y) bone.worldY += yOffset * mix * data.y;
191 }
192 if (rotateOrShearX || scaleX) {
193 float ca = (float)Math.Atan2(bone.c, bone.a), c, s, mr = 0;
194 float dx = cx - bone.worldX, dy = cy - bone.worldY;
195 if (dx > q)
196 dx = q;
197 else if (dx < -q)
198 dx = -q;
199 if (dy > q)
200 dy = q;
201 else if (dy < -q)
202 dy = -q;
203 if (rotateOrShearX) {
204 mr = (data.rotate + data.shearX) * mix;
205 float r = (float)Math.Atan2(dy + ty, dx + tx) - ca - rotateOffset * mr;
206 rotateOffset += (r - (float)Math.Ceiling(r * MathUtils.InvPI2 - 0.5f) * MathUtils.PI2) * i;
207 r = rotateOffset * mr + ca;
208 c = (float)Math.Cos(r);
209 s = (float)Math.Sin(r);
210 if (scaleX) {
211 r = l * bone.WorldScaleX;
212 if (r > 0) scaleOffset += (dx * c + dy * s) * i / r;
213 }
214 } else {
215 c = (float)Math.Cos(ca);
216 s = (float)Math.Sin(ca);
217 float r = l * bone.WorldScaleX;
218 if (r > 0) scaleOffset += (dx * c + dy * s) * i / r;
219 }
220 a = this.remaining;
221 if (a >= t) {
222 float m = massInverse * t, e = strength, w = wind, g = (Bone.yDown ? -gravity : gravity);
223 float d = (float)Math.Pow(damping, 60 * t), h = l / f;
224 while (true) {
225 a -= t;
226 if (scaleX) {
227 scaleVelocity += (w * c - g * s - scaleOffset * e) * m;
228 scaleOffset += scaleVelocity * t;
229 scaleVelocity *= d;
230 }
231 if (rotateOrShearX) {
232 rotateVelocity -= ((w * s + g * c) * h + rotateOffset * e) * m;
233 rotateOffset += rotateVelocity * t;
234 rotateVelocity *= d;
235 if (a < t) break;
236 float r = rotateOffset * mr + ca;
237 c = (float)Math.Cos(r);
238 s = (float)Math.Sin(r);
239 } else if (a < t) //
240 break;
241 }
242 }
243 }
244 this.remaining = a;
245 }
246 cx = bone.worldX;
247 cy = bone.worldY;
248 break;
249 case Physics.Pose:
250 if (x) bone.worldX += xOffset * mix * data.x;
251 if (y) bone.worldY += yOffset * mix * data.y;
252 break;
253 }
254
255 if (rotateOrShearX) {
256 float o = rotateOffset * mix, s, c, a;
257 if (data.shearX > 0) {
258 float r = 0;
259 if (data.rotate > 0) {
260 r = o * data.rotate;
261 s = (float)Math.Sin(r);
262 c = (float)Math.Cos(r);
263 a = bone.b;
264 bone.b = c * a - s * bone.d;
265 bone.d = s * a + c * bone.d;
266 }
267 r += o * data.shearX;
268 s = (float)Math.Sin(r);
269 c = (float)Math.Cos(r);
270 a = bone.a;
271 bone.a = c * a - s * bone.c;
272 bone.c = s * a + c * bone.c;
273 } else {
274 o *= data.rotate;
275 s = (float)Math.Sin(o);
276 c = (float)Math.Cos(o);
277 a = bone.a;
278 bone.a = c * a - s * bone.c;
279 bone.c = s * a + c * bone.c;
280 a = bone.b;
281 bone.b = c * a - s * bone.d;
282 bone.d = s * a + c * bone.d;
283 }
284 }
285 if (scaleX) {
286 float s = 1 + scaleOffset * mix * data.scaleX;
287 bone.a *= s;
288 bone.c *= s;
289 }
290 if (physics != Physics.Pose) {
291 tx = l * bone.a;
292 ty = l * bone.c;
293 }
294 bone.UpdateAppliedTransform();
295 }
