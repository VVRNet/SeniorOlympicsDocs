135 public void Apply () {
136 isActive = true;
137 mix = 1;
138
139 Bone startingBone = this.StartingBone = skeleton.FindBone(startingBoneName);
140 RecursivelyCreateBoneProxies(startingBone);
141
142 RootRigidbody = boneTable[startingBone].GetComponent<Rigidbody2D>();
143 RootRigidbody.isKinematic = pinStartBone;
144 RootRigidbody.mass = rootMass;
145 List<Collider2D> boneColliders = new List<Collider2D>();
146 foreach (KeyValuePair<Bone, Transform> pair in boneTable) {
147 Bone b = pair.Key;
148 Transform t = pair.Value;
149 Transform parentTransform;
150 boneColliders.Add(t.GetComponent<Collider2D>());
151 if (b == startingBone) {
152 ragdollRoot = new GameObject("RagdollRoot").transform;
153 ragdollRoot.SetParent(transform, false);
154 if (b == skeleton.RootBone) { // RagdollRoot is skeleton root's parent, thus the skeleton's scale and position.
155 ragdollRoot.localPosition = new Vector3(skeleton.X, skeleton.Y, 0);
156 ragdollRoot.localRotation = (skeleton.ScaleX < 0) ? Quaternion.Euler(0, 0, 180.0f) : Quaternion.identity;
157 } else {
158 ragdollRoot.localPosition = new Vector3(b.Parent.WorldX, b.Parent.WorldY, 0);
159 ragdollRoot.localRotation = Quaternion.Euler(0, 0, b.Parent.WorldRotationX - b.Parent.ShearX);
160 }
161 parentTransform = ragdollRoot;
162 rootOffset = t.position - transform.position;
163 } else {
164 parentTransform = boneTable[b.Parent];
165 }
166
167 // Add joint and attach to parent.
168 Rigidbody2D rbParent = parentTransform.GetComponent<Rigidbody2D>();
169 if (rbParent != null) {
170 HingeJoint2D joint = t.gameObject.AddComponent<HingeJoint2D>();
171 joint.connectedBody = rbParent;
172 Vector3 localPos = parentTransform.InverseTransformPoint(t.position);
173 joint.connectedAnchor = localPos;
174
175 joint.GetComponent<Rigidbody2D>().mass = joint.connectedBody.mass * massFalloffFactor;
176
177#if HINGE_JOINT_2019_BEHAVIOUR
178 float referenceAngle = (rbParent.transform.eulerAngles.z - t.eulerAngles.z + 360f) % 360f;
179 float minAngle = referenceAngle - rotationLimit;
180 float maxAngle = referenceAngle + rotationLimit;
181 if (maxAngle > 180f) {
182 minAngle -= 360f;
183 maxAngle -= 360f;
184 }
185#else
186 float minAngle = -rotationLimit;
187 float maxAngle = rotationLimit;
188#endif
189 joint.limits = new JointAngleLimits2D {
190 min = minAngle,
191 max = maxAngle
192 };
193 joint.useLimits = true;
194 }
195 }
196
197 // Ignore collisions among bones.
198 for (int x = 0; x < boneColliders.Count; x++) {
199 for (int y = 0; y < boneColliders.Count; y++) {
200 if (x == y) continue;
201 Physics2D.IgnoreCollision(boneColliders[x], boneColliders[y]);
202 }
203 }
204
205 // Destroy existing override-mode SkeletonUtility bones.
206 SkeletonUtilityBone[] utilityBones = GetComponentsInChildren<SkeletonUtilityBone>();
207 if (utilityBones.Length > 0) {
208 List<string> destroyedUtilityBoneNames = new List<string>();
209 foreach (SkeletonUtilityBone ub in utilityBones) {
210 if (ub.mode == SkeletonUtilityBone.Mode.Override) {
211 destroyedUtilityBoneNames.Add(ub.gameObject.name);
212 Destroy(ub.gameObject);
213 }
214 }
215 if (destroyedUtilityBoneNames.Count > 0) {
216 string msg = "Destroyed Utility Bones: ";
217 for (int i = 0; i < destroyedUtilityBoneNames.Count; i++) {
218 msg += destroyedUtilityBoneNames[i];
219 if (i != destroyedUtilityBoneNames.Count - 1) {
220 msg += ",";
221 }
222 }
223 Debug.LogWarning(msg);
224 }
225 }
226
227 // Disable skeleton constraints.
228 if (disableIK) {
229 ExposedList<IkConstraint> ikConstraints = skeleton.IkConstraints;
230 for (int i = 0, n = ikConstraints.Count; i < n; i++)
231 ikConstraints.Items[i].Mix = 0;
232 }
233
234 if (disableOtherConstraints) {
235 ExposedList<TransformConstraint> transformConstraints = skeleton.TransformConstraints;
236 for (int i = 0, n = transformConstraints.Count; i < n; i++) {
237 transformConstraints.Items[i].MixRotate = 0;
238 transformConstraints.Items[i].MixScaleX = 0;
239 transformConstraints.Items[i].MixScaleY = 0;
240 transformConstraints.Items[i].MixShearY = 0;
241 transformConstraints.Items[i].MixX = 0;
242 transformConstraints.Items[i].MixY = 0;
243 }
244
245 ExposedList<PathConstraint> pathConstraints = skeleton.PathConstraints;
246 for (int i = 0, n = pathConstraints.Count; i < n; i++) {
247 pathConstraints.Items[i].MixRotate = 0;
248 pathConstraints.Items[i].MixX = 0;
249 pathConstraints.Items[i].MixY = 0;
250 }
251 }
252
253 targetSkeletonComponent.UpdateWorld += UpdateSpineSkeleton;
254 }

257 public Coroutine SmoothMix (float target, float duration) {
258 return StartCoroutine(SmoothMixCoroutine(target, duration));
259 }
260
261 IEnumerator SmoothMixCoroutine (float target, float duration) {
262 float startTime = Time.time;
263 float startMix = mix;
264 while (mix > 0) {
265 skeleton.SetBonesToSetupPose();
266 mix = Mathf.SmoothStep(startMix, target, (Time.time - startTime) / duration);
267 yield return null;
268 }

305 void RecursivelyCreateBoneProxies (Bone b) {
306 string boneName = b.Data.Name;
307 if (stopBoneNames.Contains(boneName))
308 return;
309
310 GameObject boneGameObject = new GameObject(boneName);
311 boneGameObject.layer = this.colliderLayer;
312 Transform t = boneGameObject.transform;
313 boneTable.Add(b, t);
314
315 t.parent = transform;
316 t.localPosition = new Vector3(b.WorldX, b.WorldY, 0);
317 t.localRotation = Quaternion.Euler(0, 0, b.WorldRotationX - b.ShearX);
318 t.localScale = new Vector3(b.WorldScaleX, b.WorldScaleY, 1);
319
320 List<Collider2D> colliders = AttachBoundingBoxRagdollColliders(b, boneGameObject, skeleton, this.gravityScale);
321 if (colliders.Count == 0) {
322 float length = b.Data.Length;
323 if (length == 0) {
324 CircleCollider2D circle = boneGameObject.AddComponent<CircleCollider2D>();
325 circle.radius = thickness * 0.5f;
326 } else {
327 BoxCollider2D box = boneGameObject.AddComponent<BoxCollider2D>();
328 box.size = new Vector2(length, thickness);
329 box.offset = new Vector2(length * 0.5f, 0); // box.center in UNITY_4
330 }
331 }
332
333 Rigidbody2D rb = boneGameObject.GetComponent<Rigidbody2D>();
334 if (rb == null) rb = boneGameObject.AddComponent<Rigidbody2D>();
335 rb.gravityScale = this.gravityScale;
336
337 foreach (Bone child in b.Children)
338 RecursivelyCreateBoneProxies(child);
339 }

342 void UpdateSpineSkeleton (ISkeletonAnimation animatedSkeleton) {
343 bool parentFlipX;
344 bool parentFlipY;
345 Bone startingBone = this.StartingBone;
346 GetStartBoneParentFlipState(out parentFlipX, out parentFlipY);
347
348 foreach (KeyValuePair<Bone, Transform> pair in boneTable) {
349 Bone b = pair.Key;
350 Transform t = pair.Value;
351 bool isStartingBone = (b == startingBone);
352 Bone parentBone = b.Parent;
353 Transform parentTransform = isStartingBone ? ragdollRoot : boneTable[parentBone];
354 if (!isStartingBone) {
355 BoneFlipEntry parentBoneFlip = boneFlipTable[parentBone];
356 parentFlipX = parentBoneFlip.flipX;
357 parentFlipY = parentBoneFlip.flipY;
358 }
359 bool flipX = parentFlipX ^ (b.ScaleX < 0);
360 bool flipY = parentFlipY ^ (b.ScaleY < 0);
361
362 BoneFlipEntry boneFlip;
363 boneFlipTable.TryGetValue(b, out boneFlip);
364 boneFlip.flipX = flipX;
365 boneFlip.flipY = flipY;
366 boneFlipTable[b] = boneFlip;
367
368 bool flipXOR = flipX ^ flipY;
369 bool parentFlipXOR = parentFlipX ^ parentFlipY;
370
371 if (!oldRagdollBehaviour && isStartingBone) {
372 if (b != skeleton.RootBone) { // RagdollRoot is not skeleton root.
373 ragdollRoot.localPosition = new Vector3(parentBone.WorldX, parentBone.WorldY, 0);
374 ragdollRoot.localRotation = Quaternion.Euler(0, 0, parentBone.WorldRotationX - parentBone.ShearX);
375 ragdollRoot.localScale = new Vector3(parentBone.WorldScaleX, parentBone.WorldScaleY, 1);
376 }
377 }
378
379 Vector3 parentTransformWorldPosition = parentTransform.position;
380 Quaternion parentTransformWorldRotation = parentTransform.rotation;
381
382 parentSpaceHelper.position = parentTransformWorldPosition;
383 parentSpaceHelper.rotation = parentTransformWorldRotation;
384 parentSpaceHelper.localScale = parentTransform.lossyScale;
385
386 if (oldRagdollBehaviour) {
387 if (isStartingBone && b != skeleton.RootBone) {
388 Vector3 localPosition = new Vector3(b.Parent.WorldX, b.Parent.WorldY, 0);
389 parentSpaceHelper.position = ragdollRoot.TransformPoint(localPosition);
390 parentSpaceHelper.localRotation = Quaternion.Euler(0, 0, parentBone.WorldRotationX - parentBone.ShearX);
391 parentSpaceHelper.localScale = new Vector3(parentBone.WorldScaleX, parentBone.WorldScaleY, 1);
392 }
393 }
394
395 Vector3 boneWorldPosition = t.position;
396 Vector3 right = parentSpaceHelper.InverseTransformDirection(t.right);
397
398 Vector3 boneLocalPosition = parentSpaceHelper.InverseTransformPoint(boneWorldPosition);
399 float boneLocalRotation = Mathf.Atan2(right.y, right.x) * Mathf.Rad2Deg;
400
401 if (flipXOR) boneLocalPosition.y *= -1f;
402 if (parentFlipXOR != flipXOR) boneLocalPosition.y *= -1f;
403
404 if (parentFlipXOR) boneLocalRotation *= -1f;
405 if (parentFlipX != flipX) boneLocalRotation += 180;
406
407 b.X = Mathf.Lerp(b.X, boneLocalPosition.x, mix);
408 b.Y = Mathf.Lerp(b.Y, boneLocalPosition.y, mix);
409 b.Rotation = Mathf.Lerp(b.Rotation, boneLocalRotation, mix);
410 //b.AppliedRotation = Mathf.Lerp(b.AppliedRotation, boneLocalRotation, mix);
411 }
412 }

414 void GetStartBoneParentFlipState (out bool parentFlipX, out bool parentFlipY) {
415 parentFlipX = skeleton.ScaleX < 0;
416 parentFlipY = skeleton.ScaleY < 0;
417 Bone parent = this.StartingBone == null ? null : this.StartingBone.Parent;
418 while (parent != null) {
419 parentFlipX ^= parent.ScaleX < 0;
420 parentFlipY ^= parent.ScaleY < 0;
421 parent = parent.Parent;
422 }
423 }

425 static List<Collider2D> AttachBoundingBoxRagdollColliders (Bone b, GameObject go, Skeleton skeleton, float gravityScale) {
426 const string AttachmentNameMarker = "ragdoll";
427 List<Collider2D> colliders = new List<Collider2D>();
428 Skin skin = skeleton.Skin ?? skeleton.Data.DefaultSkin;
429
430 List<Skin.SkinEntry> skinEntries = new List<Skin.SkinEntry>();
431 foreach (Slot slot in skeleton.Slots) {
432 if (slot.Bone == b) {
433 skin.GetAttachments(skeleton.Slots.IndexOf(slot), skinEntries);
434
435 bool bbAttachmentAdded = false;
436 foreach (Skin.SkinEntry entry in skinEntries) {
437 BoundingBoxAttachment bbAttachment = entry.Attachment as BoundingBoxAttachment;
438 if (bbAttachment != null) {
439 if (!entry.Name.ToLower().Contains(AttachmentNameMarker))
440 continue;
441
442 bbAttachmentAdded = true;
443 PolygonCollider2D bbCollider = SkeletonUtility.AddBoundingBoxAsComponent(bbAttachment, slot, go, isTrigger: false);
444 colliders.Add(bbCollider);
445 }
446 }
447
448 if (bbAttachmentAdded)
449 SkeletonUtility.AddBoneRigidbody2D(go, isKinematic: false, gravityScale: gravityScale);
450 }
451 }
452
453 return colliders;
454 }
