24using UnityEngine;
25using UnityEditor;
26
27namespace Kino
28{
29 // Class used for drawing the brightness response curve
30 public class BloomGraphDrawer
31 {
32 #region Public Methods
33
34 // Update internal state with a given bloom instance.
35 public void Prepare(Bloom bloom)
36 {
37 #if UNITY_5_6_OR_NEWER
38 if (bloom.GetComponent<Camera>().allowHDR)
39 #else
40 if (bloom.GetComponent<Camera>().hdr)
41 #endif
42 {
43 _rangeX = 6;
44 _rangeY = 1.5f;
45 }
46 else
47 {
48 _rangeX = 1;
49 _rangeY = 1;
50 }
51
52 _threshold = bloom.thresholdLinear;
53 _knee = bloom.softKnee * _threshold + 1e-5f;
54
55 // Intensity is capped to prevent sampling errors.
56 _intensity = Mathf.Min(bloom.intensity, 10);
57 }
58
59 // Draw the graph at the current position.
60 public void DrawGraph()
61 {
62 _rectGraph = GUILayoutUtility.GetRect(128, 80);
63
64 // Background
65 DrawRect(0, 0, _rangeX, _rangeY, 0.1f, 0.4f);
66
67 // Soft-knee range
68 DrawRect(_threshold - _knee, 0, _threshold + _knee, _rangeY, 0.25f, -1);
69
70 // Horizontal lines
71 for (var i = 1; i < _rangeY; i++)
72 DrawLine(0, i, _rangeX, i, 0.4f);
73
74 // Vertical lines
75 for (var i = 1; i < _rangeX; i++)
76 DrawLine(i, 0, i, _rangeY, 0.4f);
77
78 // Label
79 Handles.Label(
80 PointInRect(0, _rangeY) + Vector3.right,
81 "Brightness Response (linear)", EditorStyles.miniLabel
82 );
83
84 // Threshold line
85 DrawLine(_threshold, 0, _threshold, _rangeY, 0.6f);
86
87 // Response curve
88 var vcount = 0;
89 while (vcount < _curveResolution)
90 {
91 var x = _rangeX * vcount / (_curveResolution - 1);
92 var y = ResponseFunction(x);
93 if (y < _rangeY)
94 {
95 _curveVertices[vcount++] = PointInRect(x, y);
96 }
97 else
98 {
99 if (vcount > 1)
100 {
101 // Extend the last segment to the top edge of the rect.
102 var v1 = _curveVertices[vcount - 2];
103 var v2 = _curveVertices[vcount - 1];
104 var clip = (_rectGraph.y - v1.y) / (v2.y - v1.y);
105 _curveVertices[vcount - 1] = v1 + (v2 - v1) * clip;
106 }
107 break;
108 }
109 }
110
111 if (vcount > 1)
112 {
113 Handles.color = Color.white * 0.9f;
114 Handles.DrawAAPolyLine(2.0f, vcount, _curveVertices);
115 }
116 }
117
118 #endregion
119
120 #region Response Function
121
122 float _threshold;
123 float _knee;
124 float _intensity;
125
126 float ResponseFunction(float x)
127 {
128 var rq = Mathf.Clamp(x - _threshold + _knee, 0, _knee * 2);
129 rq = rq * rq * 0.25f / _knee;
130 return Mathf.Max(rq, x - _threshold) * _intensity;
131 }
132
133 #endregion
134
135 #region Graph Functions
136
137 // Number of vertices in curve
138 const int _curveResolution = 96;
139
140 // Vertex buffers
141 Vector3[] _rectVertices = new Vector3[4];
142 Vector3[] _lineVertices = new Vector3[2];
143 Vector3[] _curveVertices = new Vector3[_curveResolution];
144
145 Rect _rectGraph;
146 float _rangeX;
147 float _rangeY;
148
149 // Transform a point into the graph rect.
150 Vector3 PointInRect(float x, float y)
151 {
152 x = Mathf.Lerp(_rectGraph.x, _rectGraph.xMax, x / _rangeX);
153 y = Mathf.Lerp(_rectGraph.yMax, _rectGraph.y, y / _rangeY);
154 return new Vector3(x, y, 0);
155 }
156
157 // Draw a line in the graph rect.
158 void DrawLine(float x1, float y1, float x2, float y2, float grayscale)
159 {
160 _lineVertices[0] = PointInRect(x1, y1);
161 _lineVertices[1] = PointInRect(x2, y2);
162 Handles.color = Color.white * grayscale;
163 Handles.DrawAAPolyLine(2.0f, _lineVertices);
164 }
165
166 // Draw a rect in the graph rect.
167 void DrawRect(float x1, float y1, float x2, float y2, float fill, float line)
168 {
169 _rectVertices[0] = PointInRect(x1, y1);
170 _rectVertices[1] = PointInRect(x2, y1);
171 _rectVertices[2] = PointInRect(x2, y2);
172 _rectVertices[3] = PointInRect(x1, y2);
173
174 Handles.DrawSolidRectangleWithOutline(
175 _rectVertices,
176 fill < 0 ? Color.clear : Color.white * fill,
177 line < 0 ? Color.clear : Color.white * line
178 );
179 }
180
181 #endregion
182 }
183}
