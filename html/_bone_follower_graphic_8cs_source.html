122 public virtual void LateUpdate () {
123 if (!valid) {
124 Initialize();
125 return;
126 }
127
128#if UNITY_EDITOR
129 if (!Application.isPlaying)
130 skeletonTransformIsParent = Transform.ReferenceEquals(skeletonTransform, transform.parent);
131#endif
132
133 if (bone == null) {
134 if (string.IsNullOrEmpty(boneName)) return;
135 bone = skeletonGraphic.Skeleton.FindBone(boneName);
136 if (!SetBone(boneName)) return;
137 }
138
139 RectTransform thisTransform = this.transform as RectTransform;
140 if (thisTransform == null) return;
141
142 float scale = skeletonGraphic.MeshScale;
143
144 float additionalFlipScale = 1;
145 if (skeletonTransformIsParent) {
146 // Recommended setup: Use local transform properties if Spine GameObject is the immediate parent
147 thisTransform.localPosition = new Vector3(followXYPosition ? bone.WorldX * scale : thisTransform.localPosition.x,
148 followXYPosition ? bone.WorldY * scale : thisTransform.localPosition.y,
149 followZPosition ? 0f : thisTransform.localPosition.z);
150 if (followBoneRotation) thisTransform.localRotation = bone.GetQuaternion();
151 } else {
152 // For special cases: Use transform world properties if transform relationship is complicated
153 Vector3 targetWorldPosition = skeletonTransform.TransformPoint(new Vector3(bone.WorldX * scale, bone.WorldY * scale, 0f));
154 if (!followZPosition) targetWorldPosition.z = thisTransform.position.z;
155 if (!followXYPosition) {
156 targetWorldPosition.x = thisTransform.position.x;
157 targetWorldPosition.y = thisTransform.position.y;
158 }
159
160 Vector3 skeletonLossyScale = skeletonTransform.lossyScale;
161 Transform transformParent = thisTransform.parent;
162 Vector3 parentLossyScale = transformParent != null ? transformParent.lossyScale : Vector3.one;
163 if (followBoneRotation) {
164 float boneWorldRotation = bone.WorldRotationX;
165
166 if ((skeletonLossyScale.x * skeletonLossyScale.y) < 0)
167 boneWorldRotation = -boneWorldRotation;
168
169 if (followSkeletonFlip || maintainedAxisOrientation == AxisOrientation.XAxis) {
170 if ((skeletonLossyScale.x * parentLossyScale.x < 0))
171 boneWorldRotation += 180f;
172 } else {
173 if ((skeletonLossyScale.y * parentLossyScale.y < 0))
174 boneWorldRotation += 180f;
175 }
176
177 Vector3 worldRotation = skeletonTransform.rotation.eulerAngles;
178 if (followLocalScale && bone.ScaleX < 0) boneWorldRotation += 180f;
179 thisTransform.SetPositionAndRotation(targetWorldPosition, Quaternion.Euler(worldRotation.x, worldRotation.y, worldRotation.z + boneWorldRotation));
180 } else {
181 thisTransform.position = targetWorldPosition;
182 }
183
184 additionalFlipScale = Mathf.Sign(skeletonLossyScale.x * parentLossyScale.x
185 * skeletonLossyScale.y * parentLossyScale.y);
186 }
187
188 Bone parentBone = bone.Parent;
189 if (followParentWorldScale || followLocalScale || followSkeletonFlip) {
190 Vector3 localScale = new Vector3(1f, 1f, 1f);
191 if (followParentWorldScale && parentBone != null)
192 localScale = new Vector3(parentBone.WorldScaleX, parentBone.WorldScaleY, 1f);
193 if (followLocalScale)
194 localScale.Scale(new Vector3(bone.ScaleX, bone.ScaleY, 1f));
195 if (followSkeletonFlip)
196 localScale.y *= Mathf.Sign(bone.Skeleton.ScaleX * bone.Skeleton.ScaleY) * additionalFlipScale;
197 thisTransform.localScale = localScale;
198 }
199 }
