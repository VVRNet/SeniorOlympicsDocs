76 public void Reset () {

107 public void DoUpdate (UpdatePhase phase) {
108 if (!valid) {
109 Reset();
110 return;
111 }
112
113 Skeleton skeleton = hierarchy.Skeleton;
114
115 if (bone == null) {
116 if (string.IsNullOrEmpty(boneName)) return;
117 bone = skeleton.FindBone(boneName);
118 if (bone == null) {
119 Debug.LogError("Bone not found: " + boneName, this);
120 return;
121 }
122 }
123 if (!bone.Active) return;
124
125 float positionScale = hierarchy.PositionScale;
126
127 Transform thisTransform = cachedTransform;
128 float skeletonFlipRotation = Mathf.Sign(skeleton.ScaleX * skeleton.ScaleY);
129 if (mode == Mode.Follow) {
130 switch (phase) {
131 case UpdatePhase.Local:
132 if (position)
133 thisTransform.localPosition = new Vector3(bone.X * positionScale, bone.Y * positionScale,
134 zPosition ? 0 : thisTransform.localPosition.z);
135
136 if (rotation) {
137 if (bone.Data.Inherit.InheritsRotation()) {
138 thisTransform.localRotation = Quaternion.Euler(0, 0, bone.Rotation);
139 } else {
140 Vector3 euler = skeletonTransform.rotation.eulerAngles;
141 thisTransform.rotation = Quaternion.Euler(euler.x, euler.y, euler.z + (bone.WorldRotationX * skeletonFlipRotation));
142 }
143 }
144
145 if (scale) {
146 thisTransform.localScale = new Vector3(bone.ScaleX, bone.ScaleY, 1f);
147 incompatibleTransformMode = BoneTransformModeIncompatible(bone);
148 }
149 break;
150 case UpdatePhase.World:
151 case UpdatePhase.Complete:
152 if (position)
153 thisTransform.localPosition = new Vector3(bone.AX * positionScale, bone.AY * positionScale,
154 zPosition ? 0 : thisTransform.localPosition.z);
155
156 if (rotation) {
157 if (bone.Data.Inherit.InheritsRotation()) {
158 thisTransform.localRotation = Quaternion.Euler(0, 0, bone.AppliedRotation);
159 } else {
160 Vector3 euler = skeletonTransform.rotation.eulerAngles;
161 thisTransform.rotation = Quaternion.Euler(euler.x, euler.y, euler.z + (bone.WorldRotationX * skeletonFlipRotation));
162 }
163 }
164
165 if (scale) {
166 thisTransform.localScale = new Vector3(bone.AScaleX, bone.AScaleY, 1f);
167 incompatibleTransformMode = BoneTransformModeIncompatible(bone);
168 }
169 break;
170 }
171
172 } else if (mode == Mode.Override) {
173 if (transformLerpComplete)
174 return;
175
176 if (parentReference == null) {
177 if (position) {
178 Vector3 clp = thisTransform.localPosition / positionScale;
179 bone.X = Mathf.Lerp(bone.X, clp.x, overrideAlpha);
180 bone.Y = Mathf.Lerp(bone.Y, clp.y, overrideAlpha);
181 }
182
183 if (rotation) {
184 float angle = Mathf.LerpAngle(bone.Rotation, thisTransform.localRotation.eulerAngles.z, overrideAlpha);
185 bone.Rotation = angle;
186 bone.AppliedRotation = angle;
187 }
188
189 if (scale) {
190 Vector3 cls = thisTransform.localScale;
191 bone.ScaleX = Mathf.Lerp(bone.ScaleX, cls.x, overrideAlpha);
192 bone.ScaleY = Mathf.Lerp(bone.ScaleY, cls.y, overrideAlpha);
193 }
194
195 } else {
196 if (transformLerpComplete)
197 return;
198
199 if (position) {
200 Vector3 pos = parentReference.InverseTransformPoint(thisTransform.position) / positionScale;
201 bone.X = Mathf.Lerp(bone.X, pos.x, overrideAlpha);
202 bone.Y = Mathf.Lerp(bone.Y, pos.y, overrideAlpha);
203 }
204
205 if (rotation) {
206 float angle = Mathf.LerpAngle(bone.Rotation, Quaternion.LookRotation(Vector3.forward, parentReference.InverseTransformDirection(thisTransform.up)).eulerAngles.z, overrideAlpha);
207 bone.Rotation = angle;
208 bone.AppliedRotation = angle;
209 }
210
211 if (scale) {
212 Vector3 cls = thisTransform.localScale;
213 bone.ScaleX = Mathf.Lerp(bone.ScaleX, cls.x, overrideAlpha);
214 bone.ScaleY = Mathf.Lerp(bone.ScaleY, cls.y, overrideAlpha);
215 }
216
217 incompatibleTransformMode = BoneTransformModeIncompatible(bone);
218 }
219
220 transformLerpComplete = true;
221 }
222 }

224 public static bool BoneTransformModeIncompatible (Bone bone) {
225 return !bone.Data.Inherit.InheritsScale();
226 }
