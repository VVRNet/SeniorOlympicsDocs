67 override public void OnInspectorGUI () {
68 if (serializedObject.isEditingMultipleObjects) {
69 DrawDefaultInspector();
70 return;
71 }
72
73 serializedObject.Update();
74 atlasAsset = (atlasAsset == null) ? (SpineSpriteAtlasAsset)target : atlasAsset;
75
76 if (atlasAsset.RegionsNeedLoading) {
77 if (GUILayout.Button(SpineInspectorUtility.TempContent("Load regions by entering Play mode"), GUILayout.Height(20))) {
78 EditorApplication.isPlaying = true;
79 }
80 }
81
82 EditorGUI.BeginChangeCheck();
83 EditorGUILayout.PropertyField(atlasFile);
84 EditorGUILayout.PropertyField(materials, true);
85 if (EditorGUI.EndChangeCheck()) {
86 serializedObject.ApplyModifiedProperties();
87 atlasAsset.Clear();
88 atlasAsset.GetAtlas();
89 atlasAsset.updateRegionsInPlayMode = true;
90 }
91
92 if (materials.arraySize == 0) {
93 EditorGUILayout.HelpBox("No materials", MessageType.Error);
94 return;
95 }
96
97 for (int i = 0; i < materials.arraySize; i++) {
98 SerializedProperty prop = materials.GetArrayElementAtIndex(i);
99 Material material = (Material)prop.objectReferenceValue;
100 if (material == null) {
101 EditorGUILayout.HelpBox("Materials cannot be null.", MessageType.Error);
102 return;
103 }
104 }
105
106 if (atlasFile.objectReferenceValue != null) {
107 int baseIndent = EditorGUI.indentLevel;
108
109 List<AtlasRegion> regions = SpineSpriteAtlasAssetInspector.GetRegions(atlasAsset.GetAtlas());
110 int regionsCount = regions.Count;
111 using (new EditorGUILayout.HorizontalScope()) {
112 EditorGUILayout.LabelField("Atlas Regions", EditorStyles.boldLabel);
113 EditorGUILayout.LabelField(string.Format("{0} regions total", regionsCount));
114 }
115 AtlasPage lastPage = null;
116 for (int i = 0; i < regionsCount; i++) {
117 if (lastPage != regions[i].page) {
118 if (lastPage != null) {
119 EditorGUILayout.Separator();
120 EditorGUILayout.Separator();
121 }
122 lastPage = regions[i].page;
123 Material mat = ((Material)lastPage.rendererObject);
124 if (mat != null) {
125 EditorGUI.indentLevel = baseIndent;
126 using (new GUILayout.HorizontalScope())
127 using (new EditorGUI.DisabledGroupScope(true))
128 EditorGUILayout.ObjectField(mat, typeof(Material), false, GUILayout.Width(250));
129 EditorGUI.indentLevel = baseIndent + 1;
130 } else {
131 EditorGUILayout.HelpBox("Page missing material!", MessageType.Warning);
132 }
133 }
134
135 string regionName = regions[i].name;
136 Texture2D icon = SpineEditorUtilities.Icons.image;
137 if (regionName.EndsWith(" ")) {
138 regionName = string.Format("'{0}'", regions[i].name);
139 icon = SpineEditorUtilities.Icons.warning;
140 EditorGUILayout.LabelField(SpineInspectorUtility.TempContent(regionName, icon, "Region name ends with whitespace. This may cause errors. Please check your source image filenames."));
141 } else {
142 EditorGUILayout.LabelField(SpineInspectorUtility.TempContent(regionName, icon));
143 }
144 }
145 EditorGUI.indentLevel = baseIndent;
146 }
147
148 if (serializedObject.ApplyModifiedProperties() || SpineInspectorUtility.UndoRedoPerformed(Event.current))
149 atlasAsset.Clear();
150 }
151 }
