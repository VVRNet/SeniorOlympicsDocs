73 public override void OnInspectorGUI () {
74 component = (SkeletonGraphicCustomMaterials)target;
75 SkeletonGraphic skeletonGraphic = component.skeletonGraphic;
76
77 // Draw the default inspector
78 DrawDefaultInspector();
79
80 if (serializedObject.isEditingMultipleObjects)
81 return;
82
83 if (componentCustomMaterialOverrides == null) {
84 Type cm = typeof(SkeletonGraphicCustomMaterials);
85 componentCustomMaterialOverrides = cm.GetField("customMaterialOverrides", PrivateInstance).GetValue(component) as List<SkeletonGraphicCustomMaterials.AtlasMaterialOverride>;
86 componentCustomTextureOverrides = cm.GetField("customTextureOverrides", PrivateInstance).GetValue(component) as List<SkeletonGraphicCustomMaterials.AtlasTextureOverride>;
87 if (componentCustomMaterialOverrides == null) {
88 Debug.Log("Reflection failed.");
89 return;
90 }
91 }
92
93 // Fill with current values at start
94 if (_customMaterialOverridesPrev == null || _customTextureOverridesPrev == null) {
95 _customMaterialOverridesPrev = CopyList(componentCustomMaterialOverrides);
96 _customTextureOverridesPrev = CopyList(componentCustomTextureOverrides);
97 }
98
99 // Compare new values with saved. If change is detected:
100 // store new values, restore old values, remove overrides, restore new values, restore overrides.
101
102 // 1. Store new values
103 var customMaterialOverridesNew = CopyList(componentCustomMaterialOverrides);
104 var customTextureOverridesNew = CopyList(componentCustomTextureOverrides);
105
106 // Detect changes
107 if (!_customMaterialOverridesPrev.SequenceEqual(customMaterialOverridesNew) ||
108 !_customTextureOverridesPrev.SequenceEqual(customTextureOverridesNew)) {
109 // 2. Restore old values
110 componentCustomMaterialOverrides.Clear();
111 componentCustomTextureOverrides.Clear();
112 componentCustomMaterialOverrides.AddRange(_customMaterialOverridesPrev);
113 componentCustomTextureOverrides.AddRange(_customTextureOverridesPrev);
114
115 // 3. Remove overrides
116 RemoveCustomMaterials();
117
118 // 4. Restore new values
119 componentCustomMaterialOverrides.Clear();
120 componentCustomTextureOverrides.Clear();
121 componentCustomMaterialOverrides.AddRange(customMaterialOverridesNew);
122 componentCustomTextureOverrides.AddRange(customTextureOverridesNew);
123
124 // 5. Restore overrides
125 SetCustomMaterials();
126
127 if (skeletonGraphic != null)
128 skeletonGraphic.LateUpdate();
129 }
130
131 _customMaterialOverridesPrev = CopyList(componentCustomMaterialOverrides);
132 _customTextureOverridesPrev = CopyList(componentCustomTextureOverrides);
133
134 if (SpineInspectorUtility.LargeCenteredButton(SpineInspectorUtility.TempContent("Clear and Reapply Changes", tooltip: "Removes all non-serialized overrides in the SkeletonGraphic and reapplies the overrides on this component."))) {
135 if (skeletonGraphic != null) {
136 skeletonGraphic.CustomMaterialOverride.Clear();
137 skeletonGraphic.CustomTextureOverride.Clear();
138 RemoveCustomMaterials();
139 SetCustomMaterials();
140 skeletonGraphic.LateUpdate();
141 }
142 }
143 }
144
145 void RemoveCustomMaterials () {
146 RemoveCustomMaterialOverrides.Invoke(component, null);
147 RemoveCustomTextureOverrides.Invoke(component, null);
148 }
149
150 void SetCustomMaterials () {
151 SetCustomMaterialOverrides.Invoke(component, null);
152 SetCustomTextureOverrides.Invoke(component, null);
153 }

155 static List<T> CopyList<T> (List<T> list) {
156 return list.GetRange(0, list.Count);
157 }
158 }
