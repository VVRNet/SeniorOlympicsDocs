47 #region Inspector
48 public SkeletonGraphic skeletonGraphic;
49 [SerializeField] protected List<AtlasMaterialOverride> customMaterialOverrides = new List<AtlasMaterialOverride>();
50 [SerializeField] protected List<AtlasTextureOverride> customTextureOverrides = new List<AtlasTextureOverride>();

52#if UNITY_EDITOR
53 void Reset () {
54 skeletonGraphic = GetComponent<SkeletonGraphic>();
55
56 // Populate material list
57 if (skeletonGraphic != null && skeletonGraphic.skeletonDataAsset != null) {
58 AtlasAssetBase[] atlasAssets = skeletonGraphic.skeletonDataAsset.atlasAssets;
59
60 List<AtlasMaterialOverride> initialAtlasMaterialOverrides = new List<AtlasMaterialOverride>();
61 foreach (AtlasAssetBase atlasAsset in atlasAssets) {
62 foreach (Material atlasMaterial in atlasAsset.Materials) {
63 AtlasMaterialOverride atlasMaterialOverride = new AtlasMaterialOverride {
64 overrideEnabled = false,
65 originalTexture = atlasMaterial.mainTexture
66 };
67
68 initialAtlasMaterialOverrides.Add(atlasMaterialOverride);
69 }
70 }
71 customMaterialOverrides = initialAtlasMaterialOverrides;
72 }
73
74 // Populate texture list
75 if (skeletonGraphic != null && skeletonGraphic.skeletonDataAsset != null) {
76 AtlasAssetBase[] atlasAssets = skeletonGraphic.skeletonDataAsset.atlasAssets;
77
78 List<AtlasTextureOverride> initialAtlasTextureOverrides = new List<AtlasTextureOverride>();
79 foreach (AtlasAssetBase atlasAsset in atlasAssets) {
80 foreach (Material atlasMaterial in atlasAsset.Materials) {
81 AtlasTextureOverride atlasTextureOverride = new AtlasTextureOverride {
82 overrideEnabled = false,
83 originalTexture = atlasMaterial.mainTexture
84 };
85
86 initialAtlasTextureOverrides.Add(atlasTextureOverride);
87 }
88 }
89 customTextureOverrides = initialAtlasTextureOverrides;
90 }
91 }
92#endif
93 #endregion
94
95 void SetCustomMaterialOverrides () {
96 if (skeletonGraphic == null) {
97 Debug.LogError("skeletonGraphic == null");
98 return;
99 }
100
101 for (int i = 0; i < customMaterialOverrides.Count; i++) {
102 AtlasMaterialOverride atlasMaterialOverride = customMaterialOverrides[i];
103 if (atlasMaterialOverride.overrideEnabled)
104 skeletonGraphic.CustomMaterialOverride[atlasMaterialOverride.originalTexture] = atlasMaterialOverride.replacementMaterial;
105 }
106 }
107
108 void RemoveCustomMaterialOverrides () {
109 if (skeletonGraphic == null) {
110 Debug.LogError("skeletonGraphic == null");
111 return;
112 }
113
114 for (int i = 0; i < customMaterialOverrides.Count; i++) {
115 AtlasMaterialOverride atlasMaterialOverride = customMaterialOverrides[i];
116 Material currentMaterial;
117
118 if (!skeletonGraphic.CustomMaterialOverride.TryGetValue(atlasMaterialOverride.originalTexture, out currentMaterial))
119 continue;
120
121 // Do not revert the material if it was changed by something else
122 if (currentMaterial != atlasMaterialOverride.replacementMaterial)
123 continue;
124
125 skeletonGraphic.CustomMaterialOverride.Remove(atlasMaterialOverride.originalTexture);
126 }
127 }
128
129 void SetCustomTextureOverrides () {
130 if (skeletonGraphic == null) {
131 Debug.LogError("skeletonGraphic == null");
132 return;
133 }
134
135 for (int i = 0; i < customTextureOverrides.Count; i++) {
136 AtlasTextureOverride atlasTextureOverride = customTextureOverrides[i];
137 if (atlasTextureOverride.overrideEnabled)
138 skeletonGraphic.CustomTextureOverride[atlasTextureOverride.originalTexture] = atlasTextureOverride.replacementTexture;
139 }
140 }
141
142 void RemoveCustomTextureOverrides () {
143 if (skeletonGraphic == null) {
144 Debug.LogError("skeletonGraphic == null");
145 return;
146 }
147
148 for (int i = 0; i < customTextureOverrides.Count; i++) {
149 AtlasTextureOverride atlasTextureOverride = customTextureOverrides[i];
150 Texture currentTexture;
151
152 if (!skeletonGraphic.CustomTextureOverride.TryGetValue(atlasTextureOverride.originalTexture, out currentTexture))
153 continue;
154
155 // Do not revert the material if it was changed by something else
156 if (currentTexture != atlasTextureOverride.replacementTexture)
157 continue;
158
159 skeletonGraphic.CustomTextureOverride.Remove(atlasTextureOverride.originalTexture);
160 }
161 }
162
163 // OnEnable applies the overrides at runtime, and when the editor loads.
164 void OnEnable () {
165 if (skeletonGraphic == null)
166 skeletonGraphic = GetComponent<SkeletonGraphic>();
167
168 if (skeletonGraphic == null) {
169 Debug.LogError("skeletonGraphic == null");
170 return;
171 }
172
173 skeletonGraphic.Initialize(false);
174 SetCustomMaterialOverrides();
175 SetCustomTextureOverrides();
176 }
177
178 // OnDisable removes the overrides at runtime, and in the editor when the component is disabled or destroyed.
179 void OnDisable () {
180 if (skeletonGraphic == null) {
181 Debug.LogError("skeletonGraphic == null");
182 return;
183 }
184
185 RemoveCustomMaterialOverrides();
186 RemoveCustomTextureOverrides();
187 }

189 [Serializable]
190 public struct AtlasMaterialOverride : IEquatable<AtlasMaterialOverride> {
191 public bool overrideEnabled;
192 public Texture originalTexture;
193 public Material replacementMaterial;
194
195 public bool Equals (AtlasMaterialOverride other) {
196 return overrideEnabled == other.overrideEnabled && originalTexture == other.originalTexture && replacementMaterial == other.replacementMaterial;
197 }
198 }

205

208 }
209 }
