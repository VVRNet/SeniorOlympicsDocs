195 public void OnSceneGUI () {
196 SkeletonRenderer skeletonRenderer = (SkeletonRenderer)target;
197 if (loadingFailed)
198 return;
199
200 Skeleton skeleton = skeletonRenderer.Skeleton;
201 if (skeleton == null) {
202 loadingFailed = true;
203 return;
204 }
205 Transform transform = skeletonRenderer.transform;
206 if (skeleton == null) return;
207
208 SpineHandles.DrawBones(transform, skeleton);

495 public static void SkeletonRootMotionParameter (Object[] targets) {
496 int rootMotionComponentCount = 0;
497 foreach (UnityEngine.Object t in targets) {
498 Component component = t as Component;
499 if (component.GetComponent<SkeletonRootMotion>() != null) {
500 ++rootMotionComponentCount;
501 }
502 }
503 bool allHaveRootMotion = rootMotionComponentCount == targets.Length;
504 bool anyHaveRootMotion = rootMotionComponentCount > 0;
505
506 using (new GUILayout.HorizontalScope()) {
507 EditorGUILayout.PrefixLabel("Root Motion");
508
509 if (!allHaveRootMotion) {
510 if (GUILayout.Button(SpineInspectorUtility.TempContent("Add Component", Icons.constraintTransform), GUILayout.MaxWidth(130), GUILayout.Height(18))) {
511 foreach (UnityEngine.Object t in targets) {
512 Component component = t as Component;
513 if (component.GetComponent<SkeletonRootMotion>() == null) {
514 component.gameObject.AddComponent<SkeletonRootMotion>();
515 }
516 }
517 }
518 }
519 if (anyHaveRootMotion) {
520 if (GUILayout.Button(SpineInspectorUtility.TempContent("Remove Component", Icons.constraintTransform), GUILayout.MaxWidth(140), GUILayout.Height(18))) {
521 foreach (UnityEngine.Object t in targets) {
522 Component component = t as Component;
523 SkeletonRootMotion rootMotionComponent = component.GetComponent<SkeletonRootMotion>();
524 if (rootMotionComponent != null) {
525 DestroyImmediate(rootMotionComponent);
526 }
527 }
528 }
529 }

543 public static void SeparatorsField (SerializedProperty separatorSlotNames) {
544 bool multi = separatorSlotNames.serializedObject.isEditingMultipleObjects;
545 bool hasTerminalSlot = false;
546 if (!multi) {
547 ISkeletonComponent sr = separatorSlotNames.serializedObject.targetObject as ISkeletonComponent;
548 Skeleton skeleton = sr.Skeleton;

615 bool mismatchDetected = false;
616 string newSkinName = initialSkinName.stringValue;
617 foreach (UnityEngine.Object o in targets) {

619 }

622 mismatchDetected = false;

624 }

628 static bool UpdateIfSkinMismatch (SkeletonRenderer skeletonRenderer, string componentSkinName) {
629 if (!skeletonRenderer.valid || skeletonRenderer.EditorSkipSkinSync) return false;
630
631 Skin skin = skeletonRenderer.Skeleton.Skin;
632 string skeletonSkinName = skin != null ? skin.Name : null;
633 bool defaultCase = skin == null && string.IsNullOrEmpty(componentSkinName);
634 bool fieldMatchesSkin = defaultCase || string.Equals(componentSkinName, skeletonSkinName, System.StringComparison.Ordinal);
635
636 if (!fieldMatchesSkin) {
637 Skin skinToSet = string.IsNullOrEmpty(componentSkinName) ? null : skeletonRenderer.Skeleton.Data.FindSkin(componentSkinName);
638 skeletonRenderer.Skeleton.SetSkin(skinToSet);
639 skeletonRenderer.Skeleton.SetSlotsToSetupPose();
640
641 // Note: the UpdateIfSkinMismatch concept shall be replaced with e.g. an OnValidate based
642 // solution or in a separate commit. The current solution does not repaint the Game view because
643 // it is first applying values and in the next editor pass is calling this skin-changing method.
644 if (skeletonRenderer is SkeletonAnimation)
645 ((SkeletonAnimation)skeletonRenderer).Update(0f);
646 else if (skeletonRenderer is SkeletonMecanim)
647 ((SkeletonMecanim)skeletonRenderer).Update();
648
649 skeletonRenderer.LateUpdate();
650 return true;
651 }
652 return false;
