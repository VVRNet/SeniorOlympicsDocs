35 public class RootMotionDeltaCompensation : MonoBehaviour {
36
37 [SerializeField] protected SkeletonRootMotionBase rootMotion;
38 public Transform targetPosition;
39 public int trackIndex = 0;
40 public bool adjustX = true;
41 public bool adjustY = true;
42 public float minScaleX = -999;
43 public float minScaleY = -999;
44 public float maxScaleX = 999;
45 public float maxScaleY = 999;
46
47 public bool allowXTranslation = false;
48 public bool allowYTranslation = true;
49
50 void Start () {
51 if (rootMotion == null)
52 rootMotion = this.GetComponent<SkeletonRootMotionBase>();
53 }
54
55 void Update () {
56 AdjustDelta();
57 }
58
59 void OnDisable () {
60 if (adjustX)
61 rootMotion.rootMotionScaleX = 1;
62 if (adjustY)
63 rootMotion.rootMotionScaleY = 1;
64 if (allowXTranslation)
65 rootMotion.rootMotionTranslateXPerY = 0;
66 if (allowYTranslation)
67 rootMotion.rootMotionTranslateYPerX = 0;
68 }
69
70 void AdjustDelta () {
71 Vector3 toTarget = targetPosition.position - this.transform.position;
72 rootMotion.AdjustRootMotionToDistance(toTarget, trackIndex, adjustX, adjustY,
73 minScaleX, maxScaleX, minScaleY, maxScaleY,
74 allowXTranslation, allowYTranslation);
75 }
76 }
