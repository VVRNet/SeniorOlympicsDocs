37 public SkeletonRenderer source;

40 SkeletonGraphic skeletonGraphic;

46 private void Awake () {

50 void Start () {
51 if (mirrorOnStart)

56 skeletonGraphic.UpdateMesh();

60 if (restoreOnDisable)
61 RestoreIndependentSkeleton();

65 public void StartMirroring () {
66 if (source == null)
67 return;
68 if (skeletonGraphic == null)
69 return;
70
71 skeletonGraphic.startingAnimation = string.Empty;
72
73 if (originalSkeleton == null) {
74 originalSkeleton = skeletonGraphic.Skeleton;
75 originalFreeze = skeletonGraphic.freeze;
76 }
77
78 skeletonGraphic.Skeleton = source.skeleton;
79 skeletonGraphic.freeze = true;
80 if (overrideTexture != null)
81 skeletonGraphic.OverrideTexture = overrideTexture;
82 }
83
85 public void UpdateTexture (Texture2D newOverrideTexture) {
86 overrideTexture = newOverrideTexture;
87 if (newOverrideTexture != null)
88 skeletonGraphic.OverrideTexture = overrideTexture;
89 }
90
92 public void RestoreIndependentSkeleton () {
93 if (originalSkeleton == null)
94 return;
95
96 skeletonGraphic.Skeleton = originalSkeleton;
97 skeletonGraphic.freeze = originalFreeze;
98 skeletonGraphic.OverrideTexture = null;
99
100 originalSkeleton = null;
101 }
102 }
