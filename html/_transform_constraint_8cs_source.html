91 public void Update (Physics physics) {
92 if (mixRotate == 0 && mixX == 0 && mixY == 0 && mixScaleX == 0 && mixScaleY == 0 && mixShearY == 0) return;
93 if (data.local) {
94 if (data.relative)
95 ApplyRelativeLocal();

97 ApplyAbsoluteLocal();

100 ApplyRelativeWorld();
101 else
102 ApplyAbsoluteWorld();

106 void ApplyAbsoluteWorld () {
107 float mixRotate = this.mixRotate, mixX = this.mixX, mixY = this.mixY, mixScaleX = this.mixScaleX,
108 mixScaleY = this.mixScaleY, mixShearY = this.mixShearY;
109 bool translate = mixX != 0 || mixY != 0;
110
111 Bone target = this.target;
112 float ta = target.a, tb = target.b, tc = target.c, td = target.d;
113 float degRadReflect = ta * td - tb * tc > 0 ? MathUtils.DegRad : -MathUtils.DegRad;
114 float offsetRotation = data.offsetRotation * degRadReflect, offsetShearY = data.offsetShearY * degRadReflect;
115
116 Bone[] bones = this.bones.Items;
117 for (int i = 0, n = this.bones.Count; i < n; i++) {
118 Bone bone = bones[i];
119
120 if (mixRotate != 0) {
121 float a = bone.a, b = bone.b, c = bone.c, d = bone.d;
122 float r = MathUtils.Atan2(tc, ta) - MathUtils.Atan2(c, a) + offsetRotation;
123 if (r > MathUtils.PI)
124 r -= MathUtils.PI2;
125 else if (r < -MathUtils.PI) //
126 r += MathUtils.PI2;
127 r *= mixRotate;
128 float cos = MathUtils.Cos(r), sin = MathUtils.Sin(r);
129 bone.a = cos * a - sin * c;
130 bone.b = cos * b - sin * d;
131 bone.c = sin * a + cos * c;
132 bone.d = sin * b + cos * d;
133 }
134
135 if (translate) {
136 float tx, ty; //Vector2 temp = this.temp;
137 target.LocalToWorld(data.offsetX, data.offsetY, out tx, out ty); //target.localToWorld(temp.set(data.offsetX, data.offsetY));
138 bone.worldX += (tx - bone.worldX) * mixX;
139 bone.worldY += (ty - bone.worldY) * mixY;
140 }
141
142 if (mixScaleX != 0) {
143 float s = (float)Math.Sqrt(bone.a * bone.a + bone.c * bone.c);
144 if (s != 0) s = (s + ((float)Math.Sqrt(ta * ta + tc * tc) - s + data.offsetScaleX) * mixScaleX) / s;
145 bone.a *= s;
146 bone.c *= s;
147 }
148 if (mixScaleY != 0) {
149 float s = (float)Math.Sqrt(bone.b * bone.b + bone.d * bone.d);
150 if (s != 0) s = (s + ((float)Math.Sqrt(tb * tb + td * td) - s + data.offsetScaleY) * mixScaleY) / s;
151 bone.b *= s;
152 bone.d *= s;
153 }
154
155 if (mixShearY > 0) {
156 float b = bone.b, d = bone.d;
157 float by = MathUtils.Atan2(d, b);
158 float r = MathUtils.Atan2(td, tb) - MathUtils.Atan2(tc, ta) - (by - MathUtils.Atan2(bone.c, bone.a));
159 if (r > MathUtils.PI)
160 r -= MathUtils.PI2;
161 else if (r < -MathUtils.PI) //
162 r += MathUtils.PI2;
163 r = by + (r + offsetShearY) * mixShearY;
164 float s = (float)Math.Sqrt(b * b + d * d);
165 bone.b = MathUtils.Cos(r) * s;
166 bone.d = MathUtils.Sin(r) * s;
167 }
168
169 bone.UpdateAppliedTransform();
170 }
171 }

173 void ApplyRelativeWorld () {
174 float mixRotate = this.mixRotate, mixX = this.mixX, mixY = this.mixY, mixScaleX = this.mixScaleX,
175 mixScaleY = this.mixScaleY, mixShearY = this.mixShearY;

177

180 float degRadReflect = ta * td - tb * tc > 0 ? MathUtils.DegRad : -MathUtils.DegRad;
181 float offsetRotation = data.offsetRotation * degRadReflect, offsetShearY = data.offsetShearY * degRadReflect;
182
183 Bone[] bones = this.bones.Items;
184 for (int i = 0, n = this.bones.Count; i < n; i++) {
185 Bone bone = bones[i];
186
187 if (mixRotate != 0) {
188 float a = bone.a, b = bone.b, c = bone.c, d = bone.d;
189 float r = MathUtils.Atan2(tc, ta) + offsetRotation;
190 if (r > MathUtils.PI)
191 r -= MathUtils.PI2;
192 else if (r < -MathUtils.PI) //
193 r += MathUtils.PI2;
194 r *= mixRotate;
195 float cos = MathUtils.Cos(r), sin = MathUtils.Sin(r);
196 bone.a = cos * a - sin * c;
197 bone.b = cos * b - sin * d;
198 bone.c = sin * a + cos * c;
199 bone.d = sin * b + cos * d;
200 }
201
202 if (translate) {
203 float tx, ty; //Vector2 temp = this.temp;
204 target.LocalToWorld(data.offsetX, data.offsetY, out tx, out ty); //target.localToWorld(temp.set(data.offsetX, data.offsetY));
205 bone.worldX += tx * mixX;
206 bone.worldY += ty * mixY;
207 }
208
209 if (mixScaleX != 0) {
210 float s = ((float)Math.Sqrt(ta * ta + tc * tc) - 1 + data.offsetScaleX) * mixScaleX + 1;
211 bone.a *= s;
212 bone.c *= s;
213 }
214 if (mixScaleY != 0) {
215 float s = ((float)Math.Sqrt(tb * tb + td * td) - 1 + data.offsetScaleY) * mixScaleY + 1;
216 bone.b *= s;
217 bone.d *= s;
218 }
219
220 if (mixShearY > 0) {
221 float r = MathUtils.Atan2(td, tb) - MathUtils.Atan2(tc, ta);
222 if (r > MathUtils.PI)
223 r -= MathUtils.PI2;
224 else if (r < -MathUtils.PI) //
225 r += MathUtils.PI2;
226 float b = bone.b, d = bone.d;
227 r = MathUtils.Atan2(d, b) + (r - MathUtils.PI / 2 + offsetShearY) * mixShearY;
228 float s = (float)Math.Sqrt(b * b + d * d);
229 bone.b = MathUtils.Cos(r) * s;
230 bone.d = MathUtils.Sin(r) * s;
231 }
232
233 bone.UpdateAppliedTransform();
234 }

237 void ApplyAbsoluteLocal () {
238 float mixRotate = this.mixRotate, mixX = this.mixX, mixY = this.mixY, mixScaleX = this.mixScaleX,
239 mixScaleY = this.mixScaleY, mixShearY = this.mixShearY;
240
241 Bone target = this.target;
242
243 Bone[] bones = this.bones.Items;
244 for (int i = 0, n = this.bones.Count; i < n; i++) {
245 Bone bone = bones[i];
246
247 float rotation = bone.arotation;
248 if (mixRotate != 0) rotation += (target.arotation - rotation + data.offsetRotation) * mixRotate;
249
250 float x = bone.ax, y = bone.ay;
251 x += (target.ax - x + data.offsetX) * mixX;
252 y += (target.ay - y + data.offsetY) * mixY;
253
254 float scaleX = bone.ascaleX, scaleY = bone.ascaleY;
255 if (mixScaleX != 0 && scaleX != 0)
256 scaleX = (scaleX + (target.ascaleX - scaleX + data.offsetScaleX) * mixScaleX) / scaleX;
257 if (mixScaleY != 0 && scaleY != 0)
258 scaleY = (scaleY + (target.ascaleY - scaleY + data.offsetScaleY) * mixScaleY) / scaleY;
259
260 float shearY = bone.ashearY;
261 if (mixShearY != 0) shearY += (target.ashearY - shearY + data.offsetShearY) * mixShearY;
262
263 bone.UpdateWorldTransform(x, y, rotation, scaleX, scaleY, bone.ashearX, shearY);
264 }
265 }

267 void ApplyRelativeLocal () {
268 float mixRotate = this.mixRotate, mixX = this.mixX, mixY = this.mixY, mixScaleX = this.mixScaleX,
269 mixScaleY = this.mixScaleY, mixShearY = this.mixShearY;
270
271 Bone target = this.target;
272
273 Bone[] bones = this.bones.Items;
274 for (int i = 0, n = this.bones.Count; i < n; i++) {
275 Bone bone = bones[i];
276
277 float rotation = bone.arotation + (target.arotation + data.offsetRotation) * mixRotate;
278 float x = bone.ax + (target.ax + data.offsetX) * mixX;
279 float y = bone.ay + (target.ay + data.offsetY) * mixY;
280 float scaleX = bone.ascaleX * (((target.ascaleX - 1 + data.offsetScaleX) * mixScaleX) + 1);
281 float scaleY = bone.ascaleY * (((target.ascaleY - 1 + data.offsetScaleY) * mixScaleY) + 1);
282 float shearY = bone.ashearY + (target.ashearY + data.offsetShearY) * mixShearY;
283
284 bone.UpdateWorldTransform(x, y, rotation, scaleX, scaleY, bone.ashearX, shearY);
285 }
