52 void Update () {

54 Vector2 localRectPosition;
55 RectTransformUtility.ScreenPointToLocalPointInRectangle(
56 skeletonGraphic.rectTransform, mousePosition, null, out localRectPosition);
57 Vector3 skeletonSpacePoint = localRectPosition / skeletonGraphic.MeshScale;
58 skeletonSpacePoint.x *= skeletonGraphic.Skeleton.ScaleX;
59 skeletonSpacePoint.y *= skeletonGraphic.Skeleton.ScaleY;
60 bone.SetLocalPosition(skeletonSpacePoint);
