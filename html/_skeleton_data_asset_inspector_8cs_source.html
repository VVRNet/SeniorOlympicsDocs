253 void CreateAnimationReferenceAssets () {
254 const string AssetFolderName = "ReferenceAssets";
255 string parentFolder = System.IO.Path.GetDirectoryName(AssetDatabase.GetAssetPath(targetSkeletonDataAsset));
256 string dataPath = parentFolder + "/" + AssetFolderName;
257 if (!AssetDatabase.IsValidFolder(dataPath)) {
258 AssetDatabase.CreateFolder(parentFolder, AssetFolderName);
259 }
260
261 FieldInfo nameField = typeof(AnimationReferenceAsset).GetField("animationName", BindingFlags.NonPublic | BindingFlags.Instance);
262 FieldInfo skeletonDataAssetField = typeof(AnimationReferenceAsset).GetField("skeletonDataAsset", BindingFlags.NonPublic | BindingFlags.Instance);
263 foreach (Animation animation in targetSkeletonData.Animations) {
264 string assetPath = string.Format("{0}/{1}.asset", dataPath, AssetUtility.GetPathSafeName(animation.Name));
265 AnimationReferenceAsset existingAsset = AssetDatabase.LoadAssetAtPath<AnimationReferenceAsset>(assetPath);
266 if (existingAsset == null) {
267 AnimationReferenceAsset newAsset = ScriptableObject.CreateInstance<AnimationReferenceAsset>();
268 skeletonDataAssetField.SetValue(newAsset, targetSkeletonDataAsset);
269 nameField.SetValue(newAsset, animation.Name);
270 AssetDatabase.CreateAsset(newAsset, assetPath);
271 }
272 }
273
274 UnityEngine.Object folderObject = AssetDatabase.LoadAssetAtPath(dataPath, typeof(UnityEngine.Object));
275 if (folderObject != null) {
276 Selection.activeObject = folderObject;
277 EditorGUIUtility.PingObject(folderObject);
278 }

384 void HandleAtlasAssetsNulls () {
385 bool hasNulls = false;
386 foreach (AtlasAssetBase a in targetSkeletonDataAsset.atlasAssets) {
387 if (a == null) {
388 hasNulls = true;
389 break;
390 }
391 }
392 if (hasNulls) {
393 if (targetSkeletonDataAsset.atlasAssets.Length == 1) {
394 EditorGUILayout.HelpBox("Atlas array cannot have null entries!", MessageType.None);
395 } else {
396 EditorGUILayout.HelpBox("Atlas array should not have null entries!", MessageType.Error);
397 if (SpineInspectorUtility.CenteredButton(SpineInspectorUtility.TempContent("Remove null entries"))) {
398 List<AtlasAssetBase> trimmedAtlasAssets = new List<AtlasAssetBase>();
399 foreach (AtlasAssetBase a in targetSkeletonDataAsset.atlasAssets) {
400 if (a != null)
401 trimmedAtlasAssets.Add(a);
402 }
403 targetSkeletonDataAsset.atlasAssets = trimmedAtlasAssets.ToArray();
404 serializedObject.Update();
405 }
406 }

507 void DrawSlotList () {
508 showSlotList = EditorGUILayout.Foldout(showSlotList, SpineInspectorUtility.TempContent("Slots", Icons.slotRoot));
509
510 if (!showSlotList) return;
511 if (!preview.IsValid) return;
512
513 Skin defaultSkin = targetSkeletonData.DefaultSkin;
514 Skin skin = preview.Skeleton.Skin ?? defaultSkin;
515
516 using (new SpineInspectorUtility.IndentScope()) {
517
518 using (new EditorGUILayout.HorizontalScope()) {
519 showAttachments = EditorGUILayout.ToggleLeft("Show Attachments", showAttachments, GUILayout.MaxWidth(150f));
520 if (showAttachments) {
521 if (skin != null) {
522 int attachmentCount = skin.Attachments.Count;
523 EditorGUILayout.LabelField(SpineInspectorUtility.TempContent(string.Format("{0} ({1} attachment{2})", skin.Name, attachmentCount, SpineInspectorUtility.PluralThenS(attachmentCount)), Icons.skin));
524 }
525
526 }
527 }
528
529 List<Skin.SkinEntry> slotAttachments = new List<Skin.SkinEntry>();
530 List<Skin.SkinEntry> defaultSkinAttachments = new List<Skin.SkinEntry>();
531 Slot[] slotsItems = preview.Skeleton.Slots.Items;
532 for (int i = preview.Skeleton.Slots.Count - 1; i >= 0; i--) {
533 Slot slot = slotsItems[i];
534 EditorGUILayout.LabelField(SpineInspectorUtility.TempContent(slot.Data.Name, Icons.slot));
535 if (showAttachments) {
536 slotAttachments.Clear();
537 defaultSkinAttachments.Clear();
538
539 using (new SpineInspectorUtility.IndentScope()) {
540 {
541 skin.GetAttachments(i, slotAttachments);
542 if (defaultSkin != null) {
543 if (skin != defaultSkin) {
544 defaultSkin.GetAttachments(i, slotAttachments);
545 defaultSkin.GetAttachments(i, defaultSkinAttachments);
546 } else {
547 defaultSkin.GetAttachments(i, defaultSkinAttachments);
548 }
549 }
550 }
551
552 for (int a = 0; a < slotAttachments.Count; a++) {
553 Skin.SkinEntry skinEntry = slotAttachments[a];
554 Attachment attachment = skinEntry.Attachment;
555 string attachmentName = skinEntry.Name;
556 bool attachmentIsFromSkin = !defaultSkinAttachments.Contains(skinEntry);
557
558 Texture2D attachmentTypeIcon = Icons.GetAttachmentIcon(attachment);
559 bool initialState = slot.Attachment == attachment;
560
561 Texture2D iconToUse = attachmentIsFromSkin ? Icons.skinPlaceholder : attachmentTypeIcon;
562 bool toggled = EditorGUILayout.ToggleLeft(SpineInspectorUtility.TempContent(attachmentName, iconToUse), slot.Attachment == attachment, GUILayout.MinWidth(150f));
563
564 if (attachmentIsFromSkin) {
565 Rect extraIconRect = GUILayoutUtility.GetLastRect();
566 extraIconRect.x += extraIconRect.width - (attachmentTypeIcon.width * 2f);
567 extraIconRect.width = attachmentTypeIcon.width;
568 extraIconRect.height = attachmentTypeIcon.height;
569 GUI.DrawTexture(extraIconRect, attachmentTypeIcon);
570 }
571
572 if (toggled != initialState) {
573 slot.Attachment = toggled ? attachment : null;
574 preview.RefreshOnNextUpdate();
575 }
576 }
577 }
578
579 }

623 void PopulateWarnings () {
624 warnings.Clear();
625 compatibilityProblemInfo = null;
626
627 if (skeletonJSON.objectReferenceValue == null) {
628 warnings.Add("Missing Skeleton JSON");
629 } else {
630 TextAsset fieldValue = (TextAsset)skeletonJSON.objectReferenceValue;
631 string problemDescription = null;
632 if (!AssetUtility.IsSpineData(fieldValue, out compatibilityProblemInfo, ref problemDescription)) {
633 if (problemDescription != null)
634 warnings.Add(problemDescription);
635 else
636 warnings.Add("Skeleton data file is not a valid Spine JSON or binary file.");
637 } else {
638#if SPINE_TK2D
639 bool searchForSpineAtlasAssets = (compatibilityProblemInfo == null);
640 bool isSpriteCollectionNull = spriteCollection.objectReferenceValue == null;
641 if (!isSpriteCollectionNull) searchForSpineAtlasAssets = false;
642#else
643 // Analysis disable once ConvertToConstant.Local
644 bool searchForSpineAtlasAssets = (compatibilityProblemInfo == null);
645#endif
646
647 if (searchForSpineAtlasAssets) {
648 bool detectedNullAtlasEntry = false;
649 List<Atlas> atlasList = new List<Atlas>();
650 AtlasAssetBase[] actualAtlasAssets = targetSkeletonDataAsset.atlasAssets;
651
652 for (int i = 0; i < actualAtlasAssets.Length; i++) {
653 if (actualAtlasAssets[i] == null) {
654 detectedNullAtlasEntry = true;
655 break;
656 } else {
657 if (actualAtlasAssets[i].MaterialCount > 0)
658 atlasList.Add(actualAtlasAssets[i].GetAtlas());
659 }
660 }
661
662 if (detectedNullAtlasEntry) {
663 warnings.Add("AtlasAsset elements should not be null.");
664 } else {
665 List<string> missingPaths = null;
666 if (atlasAssets.arraySize > 0) {
667 missingPaths = AssetUtility.GetRequiredAtlasRegions(AssetDatabase.GetAssetPath(skeletonJSON.objectReferenceValue));
668 foreach (Atlas atlas in atlasList) {
669 if (atlas == null)
670 continue;
671 for (int i = 0; i < missingPaths.Count; i++) {
672 if (atlas.FindRegionIgnoringNumberSuffix(missingPaths[i]) != null) {
673 missingPaths.RemoveAt(i);
674 i--;
675 }
676 }
677 }
678
679#if SPINE_TK2D
680 if (missingPaths.Count > 0)
681 warnings.Add("Missing regions. SkeletonData asset requires tk2DSpriteCollectionData or Spine AtlasAssets.");
682#endif
683 }
684
685 if (missingPaths != null) {
686 foreach (string missingRegion in missingPaths)
687 warnings.Add(string.Format("Missing Region: '{0}'", missingRegion));
688 }
689
690 }
