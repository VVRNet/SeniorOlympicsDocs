51 public EquipType type;

61 Gun,
62 Goggles

65 public void Equip (EquipAssetExample asset) {
66 EquipType equipType = asset.equipType;
67 EquipHook howToEquip = equippables.Find(x => x.type == equipType);
68
69 SkeletonData skeletonData = skeletonDataAsset.GetSkeletonData(true);
70 int slotIndex = skeletonData.FindSlot(howToEquip.slot).Index;
71 Attachment attachment = GenerateAttachmentFromEquipAsset(asset, slotIndex, howToEquip.templateSkin, howToEquip.templateAttachment);
72 target.Equip(slotIndex, howToEquip.templateAttachment, attachment);
73 }
74
75 Attachment GenerateAttachmentFromEquipAsset (EquipAssetExample asset, int slotIndex, string templateSkinName, string templateAttachmentName) {
76 Attachment attachment;
77 cachedAttachments.TryGetValue(asset, out attachment);
78
79 if (attachment == null) {
80 SkeletonData skeletonData = skeletonDataAsset.GetSkeletonData(true);
81 Skin templateSkin = skeletonData.FindSkin(templateSkinName);
82 Attachment templateAttachment = templateSkin.GetAttachment(slotIndex, templateAttachmentName);
83 attachment = templateAttachment.GetRemappedClone(asset.sprite, sourceMaterial, premultiplyAlpha: this.applyPMA);
84 // Note: Each call to `GetRemappedClone()` with parameter `premultiplyAlpha` set to `true` creates
85 // a cached Texture copy which can be cleared by calling AtlasUtilities.ClearCache() as shown in the method below.
86
87 cachedAttachments.Add(asset, attachment); // Cache this value for next time this asset is used.
88 }
89
90 return attachment;
91 }
92
93 public void Done () {
94 target.OptimizeSkin();
95 // `GetRepackedSkin()` and each call to `GetRemappedClone()` with parameter `premultiplyAlpha` set to `true`
96 // creates cached Texture copies which can be cleared by calling AtlasUtilities.ClearCache().
97 // You can optionally clear the textures cache after multiple repack operations.
98 // Just be aware that while this cleanup frees up memory, it is also a costly operation
99 // and will likely cause a spike in the framerate.
100
101 //AtlasUtilities.ClearCache();

103 }
104
105 }
