119 private int[] RandomInt(int length, int min, int max)
120 {
121 // length 길이의 정수 배열 생성
122 int[] randomArray = new int[length];
123 //중복 여부 확인
124 bool isSame;
125
126 for (int i = 0; i < length; ++i)
127 {
128 //중복되지 않는 랜덤 숫자가 생성될 때까지 반복
129 while(true)
130 {
131 //min과 max 사이의 랜덤 숫자 생성하여 인덱스 i에 할당
132 randomArray[i] = Random.Range(min, max);
133 //중복 여부 초기화, 현재 생성된 숫자가 중복이 아니라고 가정
134 isSame = false;
135
136 // 현자 인덱스 i 이전의 모든 인덱스를 순회하는 루프
137 for (int j = 0; j < i; ++j)
138 {
139 // 현재 생성된 숫자 randomArray[i]가 이전에 생성된 숫자 randomArray[j]와 같은지 확인
140 if (randomArray[j].Equals(randomArray[i]))
141 {
142 //중복되면 isSame 변수 true로 설정
143 isSame = true;
144 //중복이 발견되면 내부 for 루프 종료
145 break;
146 }
147 }
148 // 중복이 없으면 while(true) 루프 종료하고 다음인덱스로
149 if (!isSame)
150 break;
151 }
152 }
153 return randomArray;
154 }
