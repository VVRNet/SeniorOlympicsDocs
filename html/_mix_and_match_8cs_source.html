30using Spine.Unity.AttachmentTools;
31using System.Collections;
32using UnityEngine;
33
34namespace Spine.Unity.Examples {
35
36 // This is an example script that shows you how to change images on your skeleton using UnityEngine.Sprites.
37 public class MixAndMatch : MonoBehaviour {
38
39 #region Inspector
40 [SpineSkin]
41 public string templateAttachmentsSkin = "base";
42 public Material sourceMaterial; // This will be used as the basis for shader and material property settings.
43
44 [Header("Visor")]
45 public Sprite visorSprite;
46 [SpineSlot] public string visorSlot;
47 [SpineAttachment(slotField: "visorSlot", skinField: "baseSkinName")] public string visorKey = "goggles";
48
49 [Header("Gun")]
50 public Sprite gunSprite;
51 [SpineSlot] public string gunSlot;
52 [SpineAttachment(slotField: "gunSlot", skinField: "baseSkinName")] public string gunKey = "gun";
53
54 [Header("Runtime Repack")]
55 public bool repack = true;
56 public BoundingBoxFollower bbFollower;
57
58 [Header("Do not assign")]
59 public Texture2D runtimeAtlas;
60 public Material runtimeMaterial;
61 #endregion
62
63 Skin customSkin;
64
65 void OnValidate () {
66 if (sourceMaterial == null) {
67 SkeletonAnimation skeletonAnimation = GetComponent<SkeletonAnimation>();
68 if (skeletonAnimation != null)
69 sourceMaterial = skeletonAnimation.SkeletonDataAsset.atlasAssets[0].PrimaryMaterial;
70 }
71 }
72
73 IEnumerator Start () {
74 yield return new WaitForSeconds(1f); // Delay for one second before applying. For testing.
75 Apply();
76 }
77
78 void Apply () {
79 SkeletonAnimation skeletonAnimation = GetComponent<SkeletonAnimation>();
80 Skeleton skeleton = skeletonAnimation.Skeleton;
81
82 // STEP 0: PREPARE SKINS
83 // Let's prepare a new skin to be our custom skin with equips/customizations. We get a clone so our original skins are unaffected.
84 customSkin = customSkin ?? new Skin("custom skin"); // This requires that all customizations are done with skin placeholders defined in Spine.
85 Skin templateSkin = skeleton.Data.FindSkin(templateAttachmentsSkin);
86
87 // STEP 1: "EQUIP" ITEMS USING SPRITES
88 // STEP 1.1 Find the original/template attachment.
89 // Step 1.2 Get a clone of the original/template attachment.
90 // Step 1.3 Apply the Sprite image to the clone.
91 // Step 1.4 Add the remapped clone to the new custom skin.
92
93 // Let's do this for the visor.
94 int visorSlotIndex = skeleton.Data.FindSlot(visorSlot).Index; // You can access GetAttachment and SetAttachment via string, but caching the slotIndex is faster.
95 Attachment templateAttachment = templateSkin.GetAttachment(visorSlotIndex, visorKey); // STEP 1.1
96
97 // Note: Each call to `GetRemappedClone()` with parameter `premultiplyAlpha` set to `true` creates
98 // a cached Texture copy which can be cleared by calling AtlasUtilities.ClearCache() as done in the method below.
99 Attachment newAttachment = templateAttachment.GetRemappedClone(visorSprite, sourceMaterial, pivotShiftsMeshUVCoords: false); // STEP 1.2 - 1.3
100 customSkin.SetAttachment(visorSlotIndex, visorKey, newAttachment); // STEP 1.4
101
102 // And now for the gun.
103 int gunSlotIndex = skeleton.Data.FindSlot(gunSlot).Index;
104 Attachment templateGun = templateSkin.GetAttachment(gunSlotIndex, gunKey); // STEP 1.1
105 Attachment newGun = templateGun.GetRemappedClone(gunSprite, sourceMaterial, pivotShiftsMeshUVCoords: false); // STEP 1.2 - 1.3
106 if (newGun != null) customSkin.SetAttachment(gunSlotIndex, gunKey, newGun); // STEP 1.4
107
108 // customSkin.RemoveAttachment(gunSlotIndex, gunKey); // To remove an item.
109 // customSkin.Clear()
110 // Use skin.Clear() To remove all customizations.
111 // Customizations will fall back to the value in the default skin if it was defined there.
112 // To prevent fallback from happening, make sure the key is not defined in the default skin.
113
114 // STEP 3: APPLY AND CLEAN UP.
115 // Recommended, preferably at level-load-time: REPACK THE CUSTOM SKIN TO MINIMIZE DRAW CALLS
116 // IMPORTANT NOTE: the GetRepackedSkin() operation is expensive - if multiple characters
117 // need to call it every few seconds the overhead will outweigh the draw call benefits.
118 //
119 // Repacking requires that you set all source textures/sprites/atlases to be Read/Write enabled in the inspector.
120 // Combine all the attachment sources into one skin. Usually this means the default skin and the custom skin.
121 // call Skin.GetRepackedSkin to get a cloned skin with cloned attachments that all use one texture.
122 if (repack) {
123 Skin repackedSkin = new Skin("repacked skin");
124 repackedSkin.AddSkin(skeleton.Data.DefaultSkin); // Include the "default" skin. (everything outside of skin placeholders)
125 repackedSkin.AddSkin(customSkin); // Include your new custom skin.
126
127 // Note: materials and textures returned by GetRepackedSkin() behave like 'new Texture2D()' and need to be destroyed
128 if (runtimeMaterial)
129 Destroy(runtimeMaterial);
130 if (runtimeAtlas)
131 Destroy(runtimeAtlas);
132 repackedSkin = repackedSkin.GetRepackedSkin("repacked skin", sourceMaterial, out runtimeMaterial, out runtimeAtlas); // Pack all the items in the skin.
133 skeleton.SetSkin(repackedSkin); // Assign the repacked skin to your Skeleton.
134 if (bbFollower != null) bbFollower.Initialize(true);
135 } else {
136 skeleton.SetSkin(customSkin); // Just use the custom skin directly.
137 }
138
139 skeleton.SetSlotsToSetupPose(); // Use the pose from setup pose.
140 skeletonAnimation.Update(0); // Use the pose in the currently active animation.
141
142 // `GetRepackedSkin()` and each call to `GetRemappedClone()` with parameter `premultiplyAlpha` set to `true`
143 // cache necessarily created Texture copies which can be cleared by calling AtlasUtilities.ClearCache().
144 // You can optionally clear the textures cache after multiple repack operations.
145 // Just be aware that while this cleanup frees up memory, it is also a costly operation
146 // and will likely cause a spike in the framerate.
147 AtlasUtilities.ClearCache();
148 Resources.UnloadUnusedAssets();
149 }
150 }
