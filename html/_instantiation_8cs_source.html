57 public static void SceneViewDragAndDrop (SceneView sceneview) {
58 UnityEngine.Event current = UnityEngine.Event.current;
59 UnityEngine.Object[] references = DragAndDrop.objectReferences;
60 if (current.type == EventType.Layout)
61 return;
62
63 // Allow drag and drop of one SkeletonDataAsset.
64 if (references.Length == 1) {
65 SkeletonDataAsset skeletonDataAsset = references[0] as SkeletonDataAsset;
66 if (skeletonDataAsset != null) {
67 Vector2 mousePos = current.mousePosition;
68
69 bool invalidSkeletonData = skeletonDataAsset.GetSkeletonData(true) == null;
70 if (invalidSkeletonData) {
71 DragAndDrop.visualMode = DragAndDropVisualMode.Rejected;
72 Handles.BeginGUI();
73 GUI.Label(new Rect(mousePos + new Vector2(20f, 20f), new Vector2(400f, 40f)), new GUIContent(string.Format("{0} is invalid.\nCannot create new Spine GameObject.", skeletonDataAsset.name), SpineEditorUtilities.Icons.warning));
74 Handles.EndGUI();
75 return;
76 } else {
77 DragAndDrop.visualMode = DragAndDropVisualMode.Copy;
78 Handles.BeginGUI();
79 GUI.Label(new Rect(mousePos + new Vector2(20f, 20f), new Vector2(400f, 20f)), new GUIContent(string.Format("Create Spine GameObject ({0})", skeletonDataAsset.skeletonJSON.name), SpineEditorUtilities.Icons.skeletonDataAssetIcon));
80 Handles.EndGUI();
81
82 if (current.type == EventType.DragPerform) {
83 RectTransform rectTransform = (Selection.activeGameObject == null) ? null : Selection.activeGameObject.GetComponent<RectTransform>();
84 Plane plane = (rectTransform == null) ? new Plane(Vector3.back, Vector3.zero) : new Plane(-rectTransform.forward, rectTransform.position);
85 Vector3 spawnPoint = MousePointToWorldPoint2D(mousePos, sceneview.camera, plane);
86 ShowInstantiateContextMenu(skeletonDataAsset, spawnPoint, null, 0);
87 DragAndDrop.AcceptDrag();
88 current.Use();
89 }
90 }
91 }
92 }

104 siblingIndex = siblingIndex,

106 isUI = false
107 });
108
109 // SkeletonGraphic
110 System.Type skeletonGraphicInspectorType = System.Type.GetType("Spine.Unity.Editor.SkeletonGraphicInspector");

113 if (graphicInstantiateDelegate != null)

118 siblingIndex = siblingIndex,

121 });
122 }

140 public static void HandleSkeletonComponentDrop (object spawnMenuData) {
141 SpawnMenuData data = (SpawnMenuData)spawnMenuData;
142
143 if (data.skeletonDataAsset.GetSkeletonData(true) == null) {
144 EditorUtility.DisplayDialog("Invalid SkeletonDataAsset", "Unable to create Spine GameObject.\n\nPlease check your SkeletonDataAsset.", "Ok");
145 return;
146 }
147
148 bool isUI = data.isUI;
149
150 Component newSkeletonComponent = data.instantiateDelegate.Invoke(data.skeletonDataAsset);
151 GameObject newGameObject = newSkeletonComponent.gameObject;
152 Transform newTransform = newGameObject.transform;
153
154 GameObject usedParent = data.parent != null ? data.parent.gameObject : isUI ? Selection.activeGameObject : null;
155 if (usedParent)
156 newTransform.SetParent(usedParent.transform, false);
157 if (data.siblingIndex != 0)
158 newTransform.SetSiblingIndex(data.siblingIndex);
159
160 newTransform.position = isUI ? data.spawnPoint : RoundVector(data.spawnPoint, 2);
161
162 if (isUI) {
163 SkeletonGraphic skeletonGraphic = ((SkeletonGraphic)newSkeletonComponent);
164 if (usedParent != null && usedParent.GetComponent<RectTransform>() != null) {
165 skeletonGraphic.MatchRectTransformWithBounds();
166 } else
167 Debug.Log("Created a UI Skeleton GameObject not under a RectTransform. It may not be visible until you parent it to a canvas.");
168 if (skeletonGraphic.HasMultipleSubmeshInstructions() && !skeletonGraphic.allowMultipleCanvasRenderers)
169 Debug.Log("This mesh uses multiple atlas pages or blend modes. " +
170 "You need to enable 'Multiple Canvas Renderers for correct rendering. " +
171 "Consider packing attachments to a single atlas page if possible.", skeletonGraphic);
172 }
173
174 if (!isUI && usedParent != null && usedParent.transform.localScale != Vector3.one)
175 Debug.Log("New Spine GameObject was parented to a scaled Transform. It may not be the intended size.");
176
177 Selection.activeGameObject = newGameObject;
178 //EditorGUIUtility.PingObject(newGameObject); // Doesn't work when setting activeGameObject.
179 Undo.RegisterCreatedObjectUndo(newGameObject, "Create Spine GameObject");
180 }

185 public static Vector3 RoundVector (Vector3 vector, int digits) {
186 vector.x = (float)System.Math.Round(vector.x, digits);
187 vector.y = (float)System.Math.Round(vector.y, digits);
188 vector.z = (float)System.Math.Round(vector.z, digits);
189 return vector;
190 }

195 static Vector3 MousePointToWorldPoint2D (Vector2 mousePosition, Camera camera, Plane plane) {
196 Vector3 screenPos = new Vector3(mousePosition.x, camera.pixelHeight - mousePosition.y, 0f);
197 Ray ray = camera.ScreenPointToRay(screenPos);
198 float distance;
199 bool hit = plane.Raycast(ray, out distance);
200 return ray.GetPoint(distance);
201 }
202 }
