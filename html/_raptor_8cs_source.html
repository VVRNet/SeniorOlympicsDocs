34namespace Spine.Unity.Examples {
35 public class Raptor : MonoBehaviour {
36
37 #region Inspector
38 public AnimationReferenceAsset walk;

40 public AnimationReferenceAsset gunkeep;
41 #endregion
42
43 SkeletonAnimation skeletonAnimation;
44
45 void Start () {
46 skeletonAnimation = GetComponent<SkeletonAnimation>();
47 StartCoroutine(GunGrabRoutine());
48 }
49
50 IEnumerator GunGrabRoutine () {
51 // Play the walk animation on track 0.
52 skeletonAnimation.AnimationState.SetAnimation(0, walk, true);
53
54 // Repeatedly play the gungrab and gunkeep animation on track 1.
55 while (true) {
56 yield return new WaitForSeconds(Random.Range(0.5f, 3f));
57 skeletonAnimation.AnimationState.SetAnimation(1, gungrab, false);
58
59 yield return new WaitForSeconds(Random.Range(0.5f, 3f));
60 skeletonAnimation.AnimationState.SetAnimation(1, gunkeep, false);
61 }
62
63 }
64
65 }
