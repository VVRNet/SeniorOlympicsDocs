31using UnityEngine;

33namespace Spine.Unity.Examples {
34 public class SpineboyBeginnerInput : MonoBehaviour {
35 #region Inspector
36 public string horizontalAxis = "Horizontal";
37 public string attackButton = "Fire1";
38 public string aimButton = "Fire2";
39 public string jumpButton = "Jump";
40
41 public SpineboyBeginnerModel model;
42
43 void OnValidate () {
44 if (model == null)
45 model = GetComponent<SpineboyBeginnerModel>();
46 }
47 #endregion
48
49 void Update () {
50 if (model == null) return;
51
52 float currentHorizontal = Input.GetAxisRaw(horizontalAxis);
53 model.TryMove(currentHorizontal);
54
55 if (Input.GetButton(attackButton))
56 model.TryShoot();
57
58 if (Input.GetButtonDown(aimButton))
59 model.StartAim();
60 if (Input.GetButtonUp(aimButton))
61 model.StopAim();
62
63 if (Input.GetButtonDown(jumpButton))
64 model.TryJump();
65 }
66 }

68}
