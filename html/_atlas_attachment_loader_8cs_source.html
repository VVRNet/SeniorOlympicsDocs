46 private void LoadSequence (string name, string basePath, Sequence sequence) {
47 TextureRegion[] regions = sequence.Regions;
48 for (int i = 0, n = regions.Length; i < n; i++) {
49 string path = sequence.GetPath(basePath, i);
50 regions[i] = FindRegion(path);
51 if (regions[i] == null) throw new ArgumentException(string.Format("Region not found in atlas: {0} (region attachment: {1})", path, name));
52 }
53 }

55 public RegionAttachment NewRegionAttachment (Skin skin, string name, string path, Sequence sequence) {
56 RegionAttachment attachment = new RegionAttachment(name);
57 if (sequence != null)
58 LoadSequence(name, path, sequence);
59 else {
60 AtlasRegion region = FindRegion(path);
61 if (region == null)
62 throw new ArgumentException(string.Format("Region not found in atlas: {0} (region attachment: {1})", path, name));
63 attachment.Region = region;
64 }
65 return attachment;
66 }

97 public AtlasRegion FindRegion (string name) {
98 AtlasRegion region;
99
100 for (int i = 0; i < atlasArray.Length; i++) {
101 region = atlasArray[i].FindRegion(name);
