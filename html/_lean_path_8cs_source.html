298 {

300
301 // If the line doesn't point anywhere, return origin
302 if (denom == 0.0f)

304 return origin;

312 private Vector3 GetClosestPoint(Ray ray, Vector3 origin, Vector3 direction)
313 {
314 var crossA = Vector3.Cross(ray.direction, direction);
315 var denom = Vector3.Dot(crossA, crossA);
316
317 // If lines are parallel, we can return any point on line
318 if (denom == 0.0f)
319 {
320 return origin;
321 }
322
323 var crossB = Vector3.Cross(ray.direction, ray.origin - origin);
324 var dist01 = Vector3.Dot(crossA, crossB) / denom;
325
