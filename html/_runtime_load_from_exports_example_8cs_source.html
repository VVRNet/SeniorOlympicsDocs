57 void CreateRuntimeAssetsAndGameObject () {
58 // 1. Create the AtlasAsset (needs atlas text asset and textures, and materials/shader);
59 // 2. Create SkeletonDataAsset (needs json or binary asset file, and an AtlasAsset)
60 // 2.1 Optional: Setup blend mode materials at SkeletonDataAsset. Only required if the skeleton
61 // uses blend modes which require blend mode materials.
62 // 3.a Create SkeletonAnimation (needs a valid SkeletonDataAsset)
63 // 3.b Create SkeletonGraphic (needs a valid SkeletonDataAsset)
64
65 runtimeAtlasAsset = SpineAtlasAsset.CreateRuntimeInstance(atlasText, textures, materialPropertySource, true, null, true);
66 runtimeSkeletonDataAsset = SkeletonDataAsset.CreateRuntimeInstance(skeletonJson, runtimeAtlasAsset, true);
67 if (blendModeMaterials)
68 runtimeSkeletonDataAsset.SetupRuntimeBlendModeMaterials(
69 applyAdditiveMaterial, blendModeTemplateMaterials);
70 }
71
72 IEnumerator Start () {
73 CreateRuntimeAssetsAndGameObject();
74 if (delay > 0) {
75 runtimeSkeletonDataAsset.GetSkeletonData(false); // preload
76 yield return new WaitForSeconds(delay);
77 }
78 InstantiateSkeletonAnimation();
79
80 InstantiateSkeletonGraphic();
81 }
82
83 void InstantiateSkeletonAnimation () {
84 runtimeSkeletonAnimation = SkeletonAnimation.NewSkeletonAnimationGameObject(runtimeSkeletonDataAsset);
85 runtimeSkeletonAnimation.transform.parent = transform;
86 runtimeSkeletonAnimation.name = "SkeletonAnimation Instance";
87
88 // additional initialization
89 runtimeSkeletonAnimation.Initialize(false);
90 if (skinName != "")
91 runtimeSkeletonAnimation.Skeleton.SetSkin(skinName);
92 runtimeSkeletonAnimation.Skeleton.SetSlotsToSetupPose();
93 if (animationName != "")
94 runtimeSkeletonAnimation.AnimationState.SetAnimation(0, animationName, true);
95 }
96
97 void InstantiateSkeletonGraphic () {
98 Canvas canvas = this.GetComponentInChildren<Canvas>();
99 Transform parent = canvas.transform;
100
101 runtimeSkeletonGraphic = SkeletonGraphic.NewSkeletonGraphicGameObject(runtimeSkeletonDataAsset, parent, skeletonGraphicMaterial);
102 runtimeSkeletonGraphic.name = "SkeletonGraphic Instance";
103
104 if (blendModeMaterials) {
105 runtimeSkeletonGraphic.allowMultipleCanvasRenderers = true;
106 runtimeSkeletonGraphic.additiveMaterial = graphicBlendModeMaterials.additiveTemplate;
107 runtimeSkeletonGraphic.multiplyMaterial = graphicBlendModeMaterials.multiplyTemplate;
108 runtimeSkeletonGraphic.screenMaterial = graphicBlendModeMaterials.screenTemplate;
109 }
110
111 // additional initialization
112 runtimeSkeletonGraphic.Initialize(false);
113 if (skinName != "")
114 runtimeSkeletonGraphic.Skeleton.SetSkin(skinName);
115 runtimeSkeletonGraphic.Skeleton.SetSlotsToSetupPose();
116 if (animationName != "")
117 runtimeSkeletonGraphic.AnimationState.SetAnimation(0, animationName, true);
118 }
119 }
