90 public void Initialize (bool overwrite = false) {
91 if (skeletonRenderer == null)
92 return;
93
94 skeletonRenderer.Initialize(false);
95
96 if (string.IsNullOrEmpty(slotName))
97 return;
98
99 // Don't reinitialize if the setup did not change.
100 if (!overwrite &&
101 colliderTable.Count > 0 && slot != null && // Slot is set and colliders already populated.
102 skeletonRenderer.skeleton == slot.Skeleton && // Skeleton object did not change.
103 slotName == slot.Data.Name // Slot object did not change.
104 )
105 return;
106
107 slot = null;
108 currentAttachment = null;
109 currentAttachmentName = null;
110 currentCollider = null;
111 colliderTable.Clear();
112 nameTable.Clear();
113
114 Skeleton skeleton = skeletonRenderer.skeleton;
115 if (skeleton == null)
116 return;
117 slot = skeleton.FindSlot(slotName);
118 if (slot == null) {
119 if (BoundingBoxFollower.DebugMessages)
120 Debug.LogWarning(string.Format("Slot '{0}' not found for BoundingBoxFollower on '{1}'. (Previous colliders were disposed.)", slotName, this.gameObject.name));
121 return;
122 }
123 int slotIndex = slot.Data.Index;
124
125 int requiredCollidersCount = 0;
126 PolygonCollider2D[] colliders = GetComponents<PolygonCollider2D>();
127 if (this.gameObject.activeInHierarchy) {
128 foreach (Skin skin in skeleton.Data.Skins)
129 AddCollidersForSkin(skin, slotIndex, colliders, ref requiredCollidersCount);
130
131 if (skeleton.Skin != null)
132 AddCollidersForSkin(skeleton.Skin, slotIndex, colliders, ref requiredCollidersCount);
133 }
134 DisposeExcessCollidersAfter(requiredCollidersCount);
135
136 if (BoundingBoxFollower.DebugMessages) {
137 bool valid = colliderTable.Count != 0;
138 if (!valid) {
139 if (this.gameObject.activeInHierarchy)
140 Debug.LogWarning("Bounding Box Follower not valid! Slot [" + slotName + "] does not contain any Bounding Box Attachments!");
141 else
142 Debug.LogWarning("Bounding Box Follower tried to rebuild as a prefab.");
143 }
144 }
145 }
146
147 void AddCollidersForSkin (Skin skin, int slotIndex, PolygonCollider2D[] previousColliders, ref int collidersCount) {
148 if (skin == null) return;
149 List<Skin.SkinEntry> skinEntries = new List<Skin.SkinEntry>();
150 skin.GetAttachments(slotIndex, skinEntries);
151
152 foreach (Skin.SkinEntry entry in skinEntries) {
153 Attachment attachment = skin.GetAttachment(slotIndex, entry.Name);
154 BoundingBoxAttachment boundingBoxAttachment = attachment as BoundingBoxAttachment;
155
156 if (BoundingBoxFollower.DebugMessages && attachment != null && boundingBoxAttachment == null)
157 Debug.Log("BoundingBoxFollower tried to follow a slot that contains non-boundingbox attachments: " + slotName);
158
159 if (boundingBoxAttachment != null) {
160 if (!colliderTable.ContainsKey(boundingBoxAttachment)) {
161 PolygonCollider2D bbCollider = collidersCount < previousColliders.Length ?
162 previousColliders[collidersCount] : gameObject.AddComponent<PolygonCollider2D>();
163 ++collidersCount;
164 SkeletonUtility.SetColliderPointsLocal(bbCollider, slot, boundingBoxAttachment);
165 bbCollider.isTrigger = isTrigger;
166 bbCollider.usedByEffector = usedByEffector;
167 bbCollider.usedByComposite = usedByComposite;
168 bbCollider.enabled = false;
169 bbCollider.hideFlags = HideFlags.NotEditable;
170 colliderTable.Add(boundingBoxAttachment, bbCollider);
171 nameTable.Add(boundingBoxAttachment, entry.Name);
172 }
173 }
174 }

185 public void ClearState () {

191 currentAttachmentName = null;

195 void DisposeExcessCollidersAfter (int requiredCount) {
196 PolygonCollider2D[] colliders = GetComponents<PolygonCollider2D>();
197 if (colliders.Length == 0) return;
198
199 for (int i = requiredCount; i < colliders.Length; ++i) {
200 PolygonCollider2D collider = colliders[i];
201 if (collider != null) {
202#if UNITY_EDITOR
203 if (Application.isEditor && !Application.isPlaying)
204 DestroyImmediate(collider);
205 else
206#endif
207 Destroy(collider);
208 }
209 }
210 }

212 void LateUpdate () {
213 if (slot != null && slot.Attachment != currentAttachment)
214 MatchAttachment(slot.Attachment);
215 }
216
219 void MatchAttachment (Attachment attachment) {
220 BoundingBoxAttachment bbAttachment = attachment as BoundingBoxAttachment;
221
222 if (BoundingBoxFollower.DebugMessages && attachment != null && bbAttachment == null)
223 Debug.LogWarning("BoundingBoxFollower tried to match a non-boundingbox attachment. It will treat it as null.");
224
225 if (currentCollider != null)
226 currentCollider.enabled = false;
227
228 if (bbAttachment == null) {
229 currentCollider = null;
230 currentAttachment = null;
231 currentAttachmentName = null;
232 } else {
233 PolygonCollider2D foundCollider;
234 colliderTable.TryGetValue(bbAttachment, out foundCollider);
235 if (foundCollider != null) {
236 currentCollider = foundCollider;
237 currentCollider.enabled = true;
238 currentAttachment = bbAttachment;
239 currentAttachmentName = nameTable[bbAttachment];
240 } else {
241 currentCollider = null;
242 currentAttachment = bbAttachment;
243 currentAttachmentName = null;
244 if (BoundingBoxFollower.DebugMessages) Debug.LogFormat("Collider for BoundingBoxAttachment named '{0}' was not initialized. It is possibly from a new skin. currentAttachmentName will be null. You may need to call BoundingBoxFollower.Initialize(overwrite: true);", bbAttachment.Name);
245 }
246 }
