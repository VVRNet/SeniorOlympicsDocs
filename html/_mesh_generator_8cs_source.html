264#endif

271 }

307 public static void GenerateSkeletonRendererInstruction (SkeletonRendererInstruction instructionOutput, Skeleton skeleton, Dictionary<Slot, Material> customSlotMaterials, List<Slot> separatorSlots, bool generateMeshOverride, bool immutableTriangles = false) {
308 // if (skeleton == null) throw new ArgumentNullException("skeleton");
309 // if (instructionOutput == null) throw new ArgumentNullException("instructionOutput");
310
311 ExposedList<Slot> drawOrder = skeleton.DrawOrder;
312 int drawOrderCount = drawOrder.Count;
313
314 // Clear last state of attachments and submeshes
315 instructionOutput.Clear(); // submeshInstructions.Clear(); attachments.Clear();
316 ExposedList<SubmeshInstruction> workingSubmeshInstructions = instructionOutput.submeshInstructions;
317#if SPINE_TRIANGLECHECK
318 instructionOutput.attachments.Resize(drawOrderCount);
319 Attachment[] workingAttachmentsItems = instructionOutput.attachments.Items;
320 int totalRawVertexCount = 0;
321 bool skeletonHasClipping = false;
322#endif
323
324 SubmeshInstruction current = new SubmeshInstruction {
325 skeleton = skeleton,
326 preActiveClippingSlotSource = -1
327 };
328
329#if !SPINE_TK2D
330 bool isCustomSlotMaterialsPopulated = customSlotMaterials != null && customSlotMaterials.Count > 0;
331#endif
332
333 int separatorCount = separatorSlots == null ? 0 : separatorSlots.Count;
334 bool hasSeparators = separatorCount > 0;
335
336 int clippingAttachmentSource = -1;
337 int lastPreActiveClipping = -1; // The index of the last slot that had an active ClippingAttachment.
338 SlotData clippingEndSlot = null;
339 int submeshIndex = 0;
340 Slot[] drawOrderItems = drawOrder.Items;
341 for (int i = 0; i < drawOrderCount; i++) {
342 Slot slot = drawOrderItems[i];
343 if (!slot.Bone.Active
344#if SLOT_ALPHA_DISABLES_ATTACHMENT
345 || slot.A == 0f
346#endif
347 ) {
348 workingAttachmentsItems[i] = null;
349 continue;
350 }
351 if (slot.Data.BlendMode == BlendMode.Additive) current.hasPMAAdditiveSlot = true;
352 Attachment attachment = slot.Attachment;
353#if SPINE_TRIANGLECHECK
354 workingAttachmentsItems[i] = attachment;
355 int attachmentVertexCount = 0, attachmentTriangleCount = 0;
356#endif
357
358 object region = null;
359 bool noRender = false; // Using this allows empty slots as separators, and keeps separated parts more stable despite slots being reordered
360
361 RegionAttachment regionAttachment = attachment as RegionAttachment;
362 if (regionAttachment != null) {
363 if (regionAttachment.Sequence != null) regionAttachment.Sequence.Apply(slot, regionAttachment);
364 region = regionAttachment.Region;
365#if SPINE_TRIANGLECHECK
366 attachmentVertexCount = 4;
367 attachmentTriangleCount = 6;
368#endif
369 } else {
370 MeshAttachment meshAttachment = attachment as MeshAttachment;
371 if (meshAttachment != null) {
372 if (meshAttachment.Sequence != null) meshAttachment.Sequence.Apply(slot, meshAttachment);
373 region = meshAttachment.Region;
374#if SPINE_TRIANGLECHECK
375 attachmentVertexCount = meshAttachment.WorldVerticesLength >> 1;
376 attachmentTriangleCount = meshAttachment.Triangles.Length;
377#endif
378 } else {
379#if SPINE_TRIANGLECHECK
380 ClippingAttachment clippingAttachment = attachment as ClippingAttachment;
381 if (clippingAttachment != null) {
382 clippingEndSlot = clippingAttachment.EndSlot;
383 clippingAttachmentSource = i;
384 current.hasClipping = true;
385 skeletonHasClipping = true;
386 }
387#endif
388 noRender = true;
389 }
390 }
391
392 // Create a new SubmeshInstruction when material changes. (or when forced to separate by a submeshSeparator)
393 // Slot with a separator/new material will become the starting slot of the next new instruction.
394 if (hasSeparators) { //current.forceSeparate = hasSeparators && separatorSlots.Contains(slot);
395 current.forceSeparate = false;
396 for (int s = 0; s < separatorCount; s++) {
397 if (Slot.ReferenceEquals(slot, separatorSlots[s])) {
398 current.forceSeparate = true;
399 break;
400 }
401 }
402 }
403
404 if (noRender) {
405 if (current.forceSeparate && generateMeshOverride) { // && current.rawVertexCount > 0) {
406 { // Add
407 current.endSlot = i;
408 current.preActiveClippingSlotSource = lastPreActiveClipping;
409
410 workingSubmeshInstructions.Resize(submeshIndex + 1);
411 workingSubmeshInstructions.Items[submeshIndex] = current;
412
413 submeshIndex++;
414 }
415
416 current.startSlot = i;
417 lastPreActiveClipping = clippingAttachmentSource;
418#if SPINE_TRIANGLECHECK
419 current.rawTriangleCount = 0;
420 current.rawVertexCount = 0;
421 current.rawFirstVertexIndex = totalRawVertexCount;
422 current.hasClipping = clippingAttachmentSource >= 0;
423#endif
424 }
425 } else {
426#if !SPINE_TK2D
427 Material material;
428 if (isCustomSlotMaterialsPopulated) {
429 if (!customSlotMaterials.TryGetValue(slot, out material))
430 material = (Material)((AtlasRegion)region).page.rendererObject;
431 } else {
432 material = (Material)((AtlasRegion)region).page.rendererObject;
433 }
434#else
435 // An AtlasRegion in plain spine-unity, spine-TK2D hooks into TK2D's system. eventual source of Material object.
436 Material material = (region is Material) ? (Material)region : (Material)((AtlasRegion)region).page.rendererObject;
437#endif
438
439 if (current.forceSeparate || (current.rawVertexCount > 0 && !System.Object.ReferenceEquals(current.material, material))) { // Material changed. Add the previous submesh.
440 { // Add
441 current.endSlot = i;
442 current.preActiveClippingSlotSource = lastPreActiveClipping;
443
444 workingSubmeshInstructions.Resize(submeshIndex + 1);
445 workingSubmeshInstructions.Items[submeshIndex] = current;
446 submeshIndex++;
447 }
448 current.startSlot = i;
449 lastPreActiveClipping = clippingAttachmentSource;
450#if SPINE_TRIANGLECHECK
451 current.rawTriangleCount = 0;
452 current.rawVertexCount = 0;
453 current.rawFirstVertexIndex = totalRawVertexCount;
454 current.hasClipping = clippingAttachmentSource >= 0;
455#endif
456 }
457
458 // Update state for the next Attachment.
459 current.material = material;
460#if SPINE_TRIANGLECHECK
461 current.rawTriangleCount += attachmentTriangleCount;
462 current.rawVertexCount += attachmentVertexCount;
463 current.rawFirstVertexIndex = totalRawVertexCount;
464 totalRawVertexCount += attachmentVertexCount;
465#endif
466 }
467
468 if (clippingEndSlot != null && slot.Data == clippingEndSlot && i != clippingAttachmentSource) {
469 clippingEndSlot = null;
470 clippingAttachmentSource = -1;
471 }
472 }
473
474 if (current.rawVertexCount > 0) {
475 { // Add last or only submesh.
476 current.endSlot = drawOrderCount;
477 current.preActiveClippingSlotSource = lastPreActiveClipping;
478 current.forceSeparate = false;
479
480 workingSubmeshInstructions.Resize(submeshIndex + 1);
481 workingSubmeshInstructions.Items[submeshIndex] = current;
482 //submeshIndex++;
483 }
484 }
485
486#if SPINE_TRIANGLECHECK
487 instructionOutput.hasActiveClipping = skeletonHasClipping;
488 instructionOutput.rawVertexCount = totalRawVertexCount;
489#endif
490 instructionOutput.immutableTriangles = immutableTriangles;
491 }

494 // Material overrides are done here so they can be applied per submesh instead of per slot

501

503 if (customMaterialOverride.TryGetValue(material, out overrideMaterial))
504 wsii[i].material = overrideMaterial;
505 }

529 public void AddSubmesh (SubmeshInstruction instruction, bool updateTriangles = true) {
530 Settings settings = this.settings;
531
532 int newSubmeshCount = submeshIndex + 1;
533 if (submeshes.Items.Length < newSubmeshCount)
534 submeshes.Resize(newSubmeshCount);
535 submeshes.Count = newSubmeshCount;
536 ExposedList<int> submesh = submeshes.Items[submeshIndex];
537 if (submesh == null)
538 submeshes.Items[submeshIndex] = submesh = new ExposedList<int>();
539 submesh.Clear(false);
540
541 Skeleton skeleton = instruction.skeleton;
542 Slot[] drawOrderItems = skeleton.DrawOrder.Items;
543
544 Color32 color = default(Color32);
545 float skeletonA = skeleton.A, skeletonR = skeleton.R, skeletonG = skeleton.G, skeletonB = skeleton.B;
546 Vector2 meshBoundsMin = this.meshBoundsMin, meshBoundsMax = this.meshBoundsMax;
547
548 // Settings
549 float zSpacing = settings.zSpacing;
550 bool pmaVertexColors = settings.pmaVertexColors;
551 bool tintBlack = settings.tintBlack;
552#if SPINE_TRIANGLECHECK
553 bool useClipping = settings.useClipping && instruction.hasClipping;
554#else
555 bool useClipping = settings.useClipping;
556#endif
557 bool canvasGroupTintBlack = settings.tintBlack && settings.canvasGroupCompatible;
558
559 if (useClipping) {
560 if (instruction.preActiveClippingSlotSource >= 0) {
561 Slot slot = drawOrderItems[instruction.preActiveClippingSlotSource];
562 clipper.ClipStart(slot, slot.Attachment as ClippingAttachment);
563 }
564 }
565
566 for (int slotIndex = instruction.startSlot; slotIndex < instruction.endSlot; slotIndex++) {
567 Slot slot = drawOrderItems[slotIndex];
568 if (!slot.Bone.Active) {
569 clipper.ClipEnd(slot);
570 continue;
571 }
572 Attachment attachment = slot.Attachment;
573 float z = zSpacing * slotIndex;
574
575 float[] workingVerts = this.tempVerts;
576 float[] uvs;
577 int[] attachmentTriangleIndices;
578 int attachmentVertexCount;
579 int attachmentIndexCount;
580
581 Color c = default(Color);
582
583 // Identify and prepare values.
584 RegionAttachment region = attachment as RegionAttachment;
585 if (region != null) {
586 region.ComputeWorldVertices(slot, workingVerts, 0);
587 uvs = region.UVs;
588 attachmentTriangleIndices = regionTriangles;
589 c.r = region.R; c.g = region.G; c.b = region.B; c.a = region.A;
590 attachmentVertexCount = 4;
591 attachmentIndexCount = 6;
592 } else {
593 MeshAttachment mesh = attachment as MeshAttachment;
594 if (mesh != null) {
595 int meshVerticesLength = mesh.WorldVerticesLength;
596 if (workingVerts.Length < meshVerticesLength) {
597 workingVerts = new float[meshVerticesLength];
598 this.tempVerts = workingVerts;
599 }
600 mesh.ComputeWorldVertices(slot, 0, meshVerticesLength, workingVerts, 0); //meshAttachment.ComputeWorldVertices(slot, tempVerts);
601 uvs = mesh.UVs;
602 attachmentTriangleIndices = mesh.Triangles;
603 c.r = mesh.R; c.g = mesh.G; c.b = mesh.B; c.a = mesh.A;
604 attachmentVertexCount = meshVerticesLength >> 1; // meshVertexCount / 2;
605 attachmentIndexCount = mesh.Triangles.Length;
606 } else {
607 if (useClipping) {
608 ClippingAttachment clippingAttachment = attachment as ClippingAttachment;
609 if (clippingAttachment != null) {
610 clipper.ClipStart(slot, clippingAttachment);
611 continue;
612 }
613 }
614
615 // If not any renderable attachment.
616 clipper.ClipEnd(slot);
617 continue;
618 }
619 }
620
621 float tintBlackAlpha = 1.0f;
622 if (pmaVertexColors) {
623 float colorA = skeletonA * slot.A * c.a;
624 color.a = (byte)(colorA * 255);
625 color.r = (byte)(skeletonR * slot.R * c.r * color.a);
626 color.g = (byte)(skeletonG * slot.G * c.g * color.a);
627 color.b = (byte)(skeletonB * slot.B * c.b * color.a);
628 if (canvasGroupTintBlack) {
629 tintBlackAlpha = (slot.Data.BlendMode == BlendMode.Additive) ? 0 : colorA;
630 color.a = 255;
631 } else {
632 if (slot.Data.BlendMode == BlendMode.Additive)
633 color.a = 0;
634 }
635 } else {
636 color.a = (byte)(skeletonA * slot.A * c.a * 255);
637 color.r = (byte)(skeletonR * slot.R * c.r * 255);
638 color.g = (byte)(skeletonG * slot.G * c.g * 255);
639 color.b = (byte)(skeletonB * slot.B * c.b * 255);
640 }
641
642 if (useClipping && clipper.IsClipping) {
643 clipper.ClipTriangles(workingVerts, attachmentVertexCount << 1, attachmentTriangleIndices, attachmentIndexCount, uvs);
644 workingVerts = clipper.ClippedVertices.Items;
645 attachmentVertexCount = clipper.ClippedVertices.Count >> 1;
646 attachmentTriangleIndices = clipper.ClippedTriangles.Items;
647 attachmentIndexCount = clipper.ClippedTriangles.Count;
648 uvs = clipper.ClippedUVs.Items;
649 }
650
651 // Actually add slot/attachment data into buffers.
652 if (attachmentVertexCount != 0 && attachmentIndexCount != 0) {
653 if (tintBlack) {
654 float r2 = slot.R2;
655 float g2 = slot.G2;
656 float b2 = slot.B2;
657 if (pmaVertexColors) {
658 float alpha = skeletonA * slot.A * c.a;
659 r2 *= alpha;
660 g2 *= alpha;
661 b2 *= alpha;
662 }
663 AddAttachmentTintBlack(r2, g2, b2, tintBlackAlpha, attachmentVertexCount);
664 }
665
666 //AddAttachment(workingVerts, uvs, color, attachmentTriangleIndices, attachmentVertexCount, attachmentIndexCount, ref meshBoundsMin, ref meshBoundsMax, z);
667 int ovc = vertexBuffer.Count;
668 // Add data to vertex buffers
669 {
670 int newVertexCount = ovc + attachmentVertexCount;
671 int oldArraySize = vertexBuffer.Items.Length;
672 if (newVertexCount > oldArraySize) {
673 int newArraySize = (int)(oldArraySize * 1.3f);
674 if (newArraySize < newVertexCount) newArraySize = newVertexCount;
675 Array.Resize(ref vertexBuffer.Items, newArraySize);
676 Array.Resize(ref uvBuffer.Items, newArraySize);
677 Array.Resize(ref colorBuffer.Items, newArraySize);
678 }
679 vertexBuffer.Count = uvBuffer.Count = colorBuffer.Count = newVertexCount;
680 }
681
682 Vector3[] vbi = vertexBuffer.Items;
683 Vector2[] ubi = uvBuffer.Items;
684 Color32[] cbi = colorBuffer.Items;
685 if (ovc == 0) {
686 for (int i = 0; i < attachmentVertexCount; i++) {
