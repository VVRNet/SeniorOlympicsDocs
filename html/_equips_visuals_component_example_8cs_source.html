49 void Start () {
50 equipsSkin = new Skin("Equips");
51
52 // OPTIONAL: Add all the attachments from the template skin.
53 Skin templateSkin = skeletonAnimation.Skeleton.Data.FindSkin(templateSkinName);
54 if (templateSkin != null)
55 equipsSkin.AddSkin(templateSkin);
56
57 skeletonAnimation.Skeleton.Skin = equipsSkin;
58 RefreshSkeletonAttachments();
59 }
60
61 public void Equip (int slotIndex, string attachmentName, Attachment attachment) {
62 equipsSkin.SetAttachment(slotIndex, attachmentName, attachment);
63 skeletonAnimation.Skeleton.SetSkin(equipsSkin);
64 RefreshSkeletonAttachments();
65 }
66
67 public void OptimizeSkin () {
68 // 1. Collect all the attachments of all active skins.
69 collectedSkin = collectedSkin ?? new Skin("Collected skin");
70 collectedSkin.Clear();
71 collectedSkin.AddSkin(skeletonAnimation.Skeleton.Data.DefaultSkin);
72 collectedSkin.AddSkin(equipsSkin);
73
74 // 2. Create a repacked skin.
75 // Note: materials and textures returned by GetRepackedSkin() behave like 'new Texture2D()' and need to be destroyed
76 if (runtimeMaterial)
77 Destroy(runtimeMaterial);
78 if (runtimeAtlas)
79 Destroy(runtimeAtlas);
80 Skin repackedSkin = collectedSkin.GetRepackedSkin("Repacked skin", skeletonAnimation.SkeletonDataAsset.atlasAssets[0].PrimaryMaterial,
81 out runtimeMaterial, out runtimeAtlas, maxAtlasSize: 1024, clearCache: false);
82 collectedSkin.Clear();
83
84 // You can optionally clear the textures cache after each ore multiple repack operations are done.
85 //AtlasUtilities.ClearCache();
86 //Resources.UnloadUnusedAssets();
87
88 // 3. Use the repacked skin.
89 skeletonAnimation.Skeleton.Skin = repackedSkin;
90 RefreshSkeletonAttachments();
91 }
92
93 void RefreshSkeletonAttachments () {
94 skeletonAnimation.Skeleton.SetSlotsToSetupPose();
95 skeletonAnimation.AnimationState.Apply(skeletonAnimation.Skeleton); //skeletonAnimation.Update(0);
96 }
97
98 }
