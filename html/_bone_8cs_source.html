172 public void UpdateWorldTransform (float x, float y, float rotation, float scaleX, float scaleY, float shearX, float shearY) {
173 ax = x;
174 ay = y;
175 arotation = rotation;
176 ascaleX = scaleX;
177 ascaleY = scaleY;
178 ashearX = shearX;
179 ashearY = shearY;
180
181 Bone parent = this.parent;
182 if (parent == null) { // Root bone.
183 Skeleton skeleton = this.skeleton;
184 float sx = skeleton.scaleX, sy = skeleton.ScaleY;
185 float rx = (rotation + shearX) * MathUtils.DegRad;
186 float ry = (rotation + 90 + shearY) * MathUtils.DegRad;
187 a = (float)Math.Cos(rx) * scaleX * sx;
188 b = (float)Math.Cos(ry) * scaleY * sx;
189 c = (float)Math.Sin(rx) * scaleX * sy;
190 d = (float)Math.Sin(ry) * scaleY * sy;
191 worldX = x * sx + skeleton.x;
192 worldY = y * sy + skeleton.y;
193 return;
194 }
195
196 float pa = parent.a, pb = parent.b, pc = parent.c, pd = parent.d;
197 worldX = pa * x + pb * y + parent.worldX;
198 worldY = pc * x + pd * y + parent.worldY;
199
200 switch (inherit) {
201 case Inherit.Normal: {
202 float rx = (rotation + shearX) * MathUtils.DegRad;
203 float ry = (rotation + 90 + shearY) * MathUtils.DegRad;
204 float la = (float)Math.Cos(rx) * scaleX;
205 float lb = (float)Math.Cos(ry) * scaleY;
206 float lc = (float)Math.Sin(rx) * scaleX;
207 float ld = (float)Math.Sin(ry) * scaleY;
208 a = pa * la + pb * lc;
209 b = pa * lb + pb * ld;
210 c = pc * la + pd * lc;
211 d = pc * lb + pd * ld;
212 return;
213 }
214 case Inherit.OnlyTranslation: {
215 float rx = (rotation + shearX) * MathUtils.DegRad;
216 float ry = (rotation + 90 + shearY) * MathUtils.DegRad;
217 a = (float)Math.Cos(rx) * scaleX;
218 b = (float)Math.Cos(ry) * scaleY;
219 c = (float)Math.Sin(rx) * scaleX;
220 d = (float)Math.Sin(ry) * scaleY;
221 break;
222 }
223 case Inherit.NoRotationOrReflection: {
224 float s = pa * pa + pc * pc, prx;
225 if (s > 0.0001f) {
226 s = Math.Abs(pa * pd - pb * pc) / s;
227 pa /= skeleton.scaleX;
228 pc /= skeleton.ScaleY;
229 pb = pc * s;
230 pd = pa * s;
231 prx = MathUtils.Atan2Deg(pc, pa);
232 } else {
233 pa = 0;
234 pc = 0;
235 prx = 90 - MathUtils.Atan2Deg(pd, pb);
236 }
237 float rx = (rotation + shearX - prx) * MathUtils.DegRad;
238 float ry = (rotation + shearY - prx + 90) * MathUtils.DegRad;
239 float la = (float)Math.Cos(rx) * scaleX;
240 float lb = (float)Math.Cos(ry) * scaleY;
241 float lc = (float)Math.Sin(rx) * scaleX;
242 float ld = (float)Math.Sin(ry) * scaleY;
243 a = pa * la - pb * lc;
244 b = pa * lb - pb * ld;
245 c = pc * la + pd * lc;
246 d = pc * lb + pd * ld;
247 break;
248 }
249 case Inherit.NoScale:
250 case Inherit.NoScaleOrReflection: {
251 rotation *= MathUtils.DegRad;
252 float cos = (float)Math.Cos(rotation), sin = (float)Math.Sin(rotation);
253 float za = (pa * cos + pb * sin) / skeleton.scaleX;
254 float zc = (pc * cos + pd * sin) / skeleton.ScaleY;
255 float s = (float)Math.Sqrt(za * za + zc * zc);
256 if (s > 0.00001f) s = 1 / s;
257 za *= s;
258 zc *= s;
259 s = (float)Math.Sqrt(za * za + zc * zc);
260 if (inherit == Inherit.NoScale && (pa * pd - pb * pc < 0) != (skeleton.scaleX < 0 != skeleton.ScaleY < 0)) s = -s;
261 rotation = MathUtils.PI / 2 + MathUtils.Atan2(zc, za);
262 float zb = (float)Math.Cos(rotation) * s;
263 float zd = (float)Math.Sin(rotation) * s;
264 shearX *= MathUtils.DegRad;
265 shearY = (90 + shearY) * MathUtils.DegRad;
266 float la = (float)Math.Cos(shearX) * scaleX;
267 float lb = (float)Math.Cos(shearY) * scaleY;
268 float lc = (float)Math.Sin(shearX) * scaleX;
269 float ld = (float)Math.Sin(shearY) * scaleY;
270 a = za * la + zb * lc;
271 b = za * lb + zb * ld;
272 c = zc * la + zd * lc;
273 d = zc * lb + zd * ld;
274 break;
275 }
276 }
277 a *= skeleton.scaleX;
278 b *= skeleton.scaleX;
279 c *= skeleton.ScaleY;
280 d *= skeleton.ScaleY;
281 }

306 public void UpdateAppliedTransform () {
307 Bone parent = this.parent;
308 if (parent == null) {
309 ax = worldX - skeleton.x;
310 ay = worldY - skeleton.y;
311 float a = this.a, b = this.b, c = this.c, d = this.d;
312 arotation = MathUtils.Atan2Deg(c, a);
313 ascaleX = (float)Math.Sqrt(a * a + c * c);
314 ascaleY = (float)Math.Sqrt(b * b + d * d);
315 ashearX = 0;
316 ashearY = MathUtils.Atan2Deg(a * b + c * d, a * d - b * c);
317 return;
318 }
319
320 float pa = parent.a, pb = parent.b, pc = parent.c, pd = parent.d;
321 float pid = 1 / (pa * pd - pb * pc);
322 float ia = pd * pid, ib = pb * pid, ic = pc * pid, id = pa * pid;
323 float dx = worldX - parent.worldX, dy = worldY - parent.worldY;
324 ax = (dx * ia - dy * ib);
325 ay = (dy * id - dx * ic);
326
327 float ra, rb, rc, rd;
328 if (inherit == Inherit.OnlyTranslation) {
329 ra = a;
330 rb = b;
331 rc = c;
332 rd = d;
333 } else {
334 switch (inherit) {
335 case Inherit.NoRotationOrReflection: {
336 float s = Math.Abs(pa * pd - pb * pc) / (pa * pa + pc * pc);
337 float sa = pa / skeleton.scaleX;
338 float sc = pc / skeleton.ScaleY;
339 pb = -sc * s * skeleton.scaleX;
340 pd = sa * s * skeleton.ScaleY;
341 pid = 1 / (pa * pd - pb * pc);
342 ia = pd * pid;
343 ib = pb * pid;
344 break;
345 }
346 case Inherit.NoScale:
347 case Inherit.NoScaleOrReflection: {
348 float r = rotation * MathUtils.DegRad, cos = (float)Math.Cos(r), sin = (float)Math.Sin(r);
349 pa = (pa * cos + pb * sin) / skeleton.scaleX;
350 pc = (pc * cos + pd * sin) / skeleton.ScaleY;
351 float s = (float)Math.Sqrt(pa * pa + pc * pc);
352 if (s > 0.00001f) s = 1 / s;
353 pa *= s;
354 pc *= s;
355 s = (float)Math.Sqrt(pa * pa + pc * pc);
356 if (inherit == Inherit.NoScale && pid < 0 != (skeleton.scaleX < 0 != skeleton.ScaleY < 0)) s = -s;
357 r = MathUtils.PI / 2 + MathUtils.Atan2(pc, pa);
358 pb = (float)Math.Cos(r) * s;
359 pd = (float)Math.Sin(r) * s;
360 pid = 1 / (pa * pd - pb * pc);
361 ia = pd * pid;
362 ib = pb * pid;
363 ic = pc * pid;
364 id = pa * pid;
365 break;
366 }
367 }
368 ra = ia * a - ib * c;
369 rb = ia * b - ib * d;
370 rc = id * c - ic * a;
371 rd = id * d - ic * b;
372 }
373
374 ashearX = 0;
375 ascaleX = (float)Math.Sqrt(ra * ra + rc * rc);
376 if (ascaleX > 0.0001f) {
377 float det = ra * rd - rb * rc;
378 ascaleY = det / ascaleX;
379 ashearY = -MathUtils.Atan2Deg(ra * rb + rc * rd, det);
380 arotation = MathUtils.Atan2Deg(rc, ra);
381 } else {
382 ascaleX = 0;
383 ascaleY = (float)Math.Sqrt(rb * rb + rd * rd);
384 ashearY = 0;
385 arotation = 90 - MathUtils.Atan2Deg(rd, rb);
386 }
387 }

390 public void WorldToLocal (float worldX, float worldY, out float localX, out float localY) {
391 float a = this.a, b = this.b, c = this.c, d = this.d;
392 float det = a * d - b * c;
393 float x = worldX - this.worldX, y = worldY - this.worldY;
394 localX = (x * d - y * b) / det;
395 localY = (y * a - x * c) / det;
396 }

444 public void RotateWorld (float degrees) {
445 degrees *= MathUtils.DegRad;
446 float sin = (float)Math.Sin(degrees), cos = (float)Math.Cos(degrees);
447 float ra = a, rb = b;
448 a = cos * ra - sin * c;
449 b = cos * rb - sin * d;
450 c = sin * ra + cos * c;
451 d = sin * rb + cos * d;
