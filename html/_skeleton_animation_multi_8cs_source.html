41 public class SkeletonAnimationMulti : MonoBehaviour {

44 public bool initialFlipX, initialFlipY;

54 readonly Dictionary<Animation, SkeletonAnimation> animationSkeletonTable = new Dictionary<Animation, SkeletonAnimation>();
55 //Stateful
56 SkeletonAnimation currentSkeletonAnimation;
57
58 void Clear () {
59 foreach (SkeletonAnimation skeletonAnimation in skeletonAnimations)
60 Destroy(skeletonAnimation.gameObject);
61
62 skeletonAnimations.Clear();
63 animationNameTable.Clear();
64 animationSkeletonTable.Clear();
65 }

67 void SetActiveSkeleton (int index) {
68 if (index < 0 || index >= skeletonAnimations.Count)
69 SetActiveSkeleton(null);
70 else
71 SetActiveSkeleton(skeletonAnimations[index]);
72 }
73
74 void SetActiveSkeleton (SkeletonAnimation skeletonAnimation) {
75 foreach (SkeletonAnimation iter in skeletonAnimations)
76 iter.gameObject.SetActive(iter == skeletonAnimation);
77
78 currentSkeletonAnimation = skeletonAnimation;
79 }

81 #region Lifecycle

83 Initialize(false);
84 }
85 #endregion

87 #region API
88 public Dictionary<Animation, SkeletonAnimation> AnimationSkeletonTable { get { return this.animationSkeletonTable; } }
89 public Dictionary<string, Animation> AnimationNameTable { get { return this.animationNameTable; } }
90 public SkeletonAnimation CurrentSkeletonAnimation { get { return this.currentSkeletonAnimation; } }
91 public List<SkeletonAnimation> SkeletonAnimations { get { return skeletonAnimations; } }
92
93 public void Initialize (bool overwrite) {
94 if (skeletonAnimations.Count != 0 && !overwrite) return;
95
96 Clear();
97
98 MeshGenerator.Settings settings = this.meshGeneratorSettings;
99 Transform thisTransform = this.transform;
100 foreach (SkeletonDataAsset dataAsset in skeletonDataAssets) {
101 SkeletonAnimation newSkeletonAnimation = SkeletonAnimation.NewSkeletonAnimationGameObject(dataAsset);
102 newSkeletonAnimation.transform.SetParent(thisTransform, false);
103
104 newSkeletonAnimation.SetMeshSettings(settings);
105 newSkeletonAnimation.initialFlipX = this.initialFlipX;
106 newSkeletonAnimation.initialFlipY = this.initialFlipY;
107 Skeleton skeleton = newSkeletonAnimation.skeleton;
108 skeleton.ScaleX = this.initialFlipX ? -1 : 1;
109 skeleton.ScaleY = this.initialFlipY ? -1 : 1;
110
111 newSkeletonAnimation.Initialize(false);
112 skeletonAnimations.Add(newSkeletonAnimation);
113 }
114
115 // Build cache
116 Dictionary<string, Animation> animationNameTable = this.animationNameTable;
117 Dictionary<Animation, SkeletonAnimation> animationSkeletonTable = this.animationSkeletonTable;
118 foreach (SkeletonAnimation skeletonAnimation in skeletonAnimations) {
119 foreach (Animation animationObject in skeletonAnimation.Skeleton.Data.Animations) {
120 animationNameTable[animationObject.Name] = animationObject;
121 animationSkeletonTable[animationObject] = skeletonAnimation;
122 }
123 }
124
125 SetActiveSkeleton(skeletonAnimations[0]);
126 SetAnimation(initialAnimation, initialLoop);
127 }
128
129 public Animation FindAnimation (string animationName) {
130 Animation animation;
131 animationNameTable.TryGetValue(animationName, out animation);
132 return animation;
133 }
134
135 public TrackEntry SetAnimation (string animationName, bool loop) {
136 return SetAnimation(FindAnimation(animationName), loop);
137 }
138
139 public TrackEntry SetAnimation (Animation animation, bool loop) {
140 if (animation == null) return null;
141
142 SkeletonAnimation skeletonAnimation;
143 animationSkeletonTable.TryGetValue(animation, out skeletonAnimation);
144
145 if (skeletonAnimation != null) {
146 SetActiveSkeleton(skeletonAnimation);
147 skeletonAnimation.skeleton.SetToSetupPose();
148 TrackEntry trackEntry = skeletonAnimation.state.SetAnimation(MainTrackIndex, animation, loop);
149 skeletonAnimation.Update(0);
150 return trackEntry;
151 }
152 return null;
153 }
154
155 public void SetEmptyAnimation (float mixDuration) {
156 currentSkeletonAnimation.state.SetEmptyAnimation(MainTrackIndex, mixDuration);
157 }
158
159 public void ClearAnimation () {
160 currentSkeletonAnimation.state.ClearTrack(MainTrackIndex);
161 }
162
163 public TrackEntry GetCurrent () {
164 return currentSkeletonAnimation.state.GetCurrent(MainTrackIndex);
165 }
166 #endregion
167 }
