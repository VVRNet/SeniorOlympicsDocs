138 void Awake()
139 {
140 camInitialPosition = Camera.main.transform.position;

142
143 var list = new List<GameObject>();
144 for (int i = 0; i < this.transform.childCount; i++)

154 void Update()
155 {
156 if (rotateCamera)
157 {
158 demoCamera.RotateAround(Vector3.zero, Vector3.up, rotationSpeed * Time.deltaTime);
159 }
160
161 if (Input.GetKeyDown(KeyCode.Space))
162 {
163 if (currentEffect != null)
164 {
165 var ps = currentEffect.GetComponent<ParticleSystem>();
166 if (ps == null)
167 {
168 return;
169 }
170
171 if (ps.isEmitting)
172 {
173 ps.Stop(true);
174 }
175 else
176 {
177 if (!currentEffect.gameObject.activeSelf)
178 {
179 currentEffect.SetActive(true);
180 }
181 else
182 {
183 ps.Play(true);
184 var cfxrEffects = currentEffect.GetComponentsInChildren<CFXR_Effect>();
185 foreach (var cfxr in cfxrEffects)
186 {
187 cfxr.ResetState();
188 }
189 }
190 }
191 }
192 }
193
194 if (Input.GetKeyDown(KeyCode.Delete) || Input.GetKeyDown(KeyCode.Backspace))
195 {
196 if (currentEffect != null)
197 {
198 currentEffect.SetActive(false);
199 currentEffect.SetActive(true);
200 }
201 }
202
203 if (Input.GetKeyDown(KeyCode.LeftArrow))
204 {
205 PreviousEffect();
206 }
207
208 if (Input.GetKeyDown(KeyCode.RightArrow))
209 {
210 NextEffect();
211 }
212
213 if (Input.GetMouseButtonDown(0))
214 {
215 var ray = demoCamera.GetComponent<Camera>().ScreenPointToRay(Input.mousePosition);
216 if (Physics.Raycast(ray))
217 {
218 if (currentEffect != null)
219 {
220 currentEffect.SetActive(false);
221 currentEffect.SetActive(true);
222 }
223 }
224 }
225
226 if (Input.GetMouseButtonDown(1) || Input.GetMouseButtonDown(2))
227 {
228 ResetCam();
229 }
230
231 float scroll = Input.GetAxis("Mouse ScrollWheel");
232 if (scroll != 0f)
233 {
234 Camera.main.transform.Translate(Vector3.forward * (scroll < 0f ? -1f : 1f), Space.Self);
235 }
236 }
