98 public override Atlas GetAtlas (bool onlyMetaData = false) {
99 if (spriteAtlasFile == null) {
100 Debug.LogError("SpriteAtlas file not set for SpineSpriteAtlasAsset: " + name, this);
101 Clear();
102 return null;
103 }
104
105 if (!onlyMetaData && (materials == null || materials.Length == 0)) {
106 Debug.LogError("Materials not set for SpineSpriteAtlasAsset: " + name, this);
107 Clear();
108 return null;
109 }
110
111 if (atlas != null) return atlas;
112
113 try {
114 atlas = LoadAtlas(spriteAtlasFile);
115 return atlas;
116 } catch (Exception ex) {
117 Debug.LogError("Error analyzing SpriteAtlas for SpineSpriteAtlasAsset: " + name + "\n" + ex.Message + "\n" + ex.StackTrace, this);
118 return null;
119 }

122 protected void AssignRegionsFromSavedRegions (Sprite[] sprites, Atlas usedAtlas) {
123
124 if (savedRegions == null || savedRegions.Length != sprites.Length)
125 return;
126
127 int i = 0;
128 foreach (AtlasRegion region in usedAtlas) {
129 SavedRegionInfo savedRegion = savedRegions[i];
130 AtlasPage page = region.page;
131
132 region.degrees = savedRegion.packingRotation == SpritePackingRotation.None ? 0 : 90;
133
134 float x = savedRegion.x;
135 float y = savedRegion.y;
136 float width = savedRegion.width;
137 float height = savedRegion.height;
138
139 region.u = x / (float)page.width;
140 region.v = y / (float)page.height;
141 if (region.degrees == 90) {
142 region.u2 = (x + height) / (float)page.width;
143 region.v2 = (y + width) / (float)page.height;
144 } else {
145 region.u2 = (x + width) / (float)page.width;
146 region.v2 = (y + height) / (float)page.height;
147 }
148 region.x = (int)x;
149 region.y = (int)y;
150 region.width = Math.Abs((int)width);
151 region.height = Math.Abs((int)height);
152
153 // flip upside down
154 float temp = region.v;
155 region.v = region.v2;
156 region.v2 = temp;
157
158 region.originalWidth = (int)width;
159 region.originalHeight = (int)height;
160
161 // note: currently sprite pivot offsets are ignored.
162 // Sprite sprite = sprites[i];
163 region.offsetX = 0;//sprite.pivot.x;
164 region.offsetY = 0;//sprite.pivot.y;
165
166 ++i;
167 }
168 }

170 private Atlas LoadAtlas (UnityEngine.U2D.SpriteAtlas spriteAtlas) {
171
172 List<AtlasPage> pages = new List<AtlasPage>();
173 List<AtlasRegion> regions = new List<AtlasRegion>();
174
175 Sprite[] sprites = new UnityEngine.Sprite[spriteAtlas.spriteCount];
176 spriteAtlas.GetSprites(sprites);
177 if (sprites.Length == 0)
178 return new Atlas(pages, regions);
179
180 Texture2D texture = null;
181#if UNITY_EDITOR
182 if (!Application.isPlaying)
183 texture = AccessPackedTextureEditor(spriteAtlas);
184 else
185#endif
186 texture = AccessPackedTexture(sprites);
187
188 Material material = materials[0];
189#if !UNITY_EDITOR
190 material.mainTexture = texture;
191#endif
192
193 Spine.AtlasPage page = new AtlasPage();
194 page.name = spriteAtlas.name;
195 page.width = texture.width;
196 page.height = texture.height;
197 page.format = Spine.Format.RGBA8888;
198
199 page.minFilter = TextureFilter.Linear;
200 page.magFilter = TextureFilter.Linear;
201 page.uWrap = TextureWrap.ClampToEdge;
202 page.vWrap = TextureWrap.ClampToEdge;
203 page.rendererObject = material;
204 pages.Add(page);
205
206 sprites = AccessPackedSprites(spriteAtlas);
207
208 int i = 0;
209 for (; i < sprites.Length; ++i) {
210 Sprite sprite = sprites[i];
211 AtlasRegion region = new AtlasRegion();
212 region.name = sprite.name.Replace("(Clone)", "");
213 region.page = page;
214 region.degrees = sprite.packingRotation == SpritePackingRotation.None ? 0 : 90;
215
216 region.u2 = 1;
217 region.v2 = 1;
218 region.width = page.width;
219 region.height = page.height;
220 region.originalWidth = page.width;
221 region.originalHeight = page.height;
222
223 region.index = i;
224 regions.Add(region);
225 }
226
227 Atlas atlas = new Atlas(pages, regions);
228 AssignRegionsFromSavedRegions(sprites, atlas);
229
230 return atlas;
231 }

282 public void LoadRegionsInEditorPlayMode () {
283
284 Sprite[] sprites = null;
285 System.Type T = Type.GetType("UnityEditor.U2D.SpriteAtlasExtensions,UnityEditor");
286 MethodInfo method = T.GetMethod("GetPackedSprites", BindingFlags.NonPublic | BindingFlags.Static);
287 if (method != null) {
288 object retval = method.Invoke(null, new object[] { spriteAtlasFile });
289 Sprite[] spritesArray = retval as Sprite[];
290 if (spritesArray != null && spritesArray.Length > 0) {
291 sprites = spritesArray;
292 }
293 }
294 if (sprites == null) {
295 sprites = new UnityEngine.Sprite[spriteAtlasFile.spriteCount];
296 spriteAtlasFile.GetSprites(sprites);
297 }
298 if (sprites.Length == 0) {
299 Debug.LogWarning(string.Format("SpriteAtlas '{0}' contains no sprites. Please make sure all assigned images are set to import type 'Sprite'.", spriteAtlasFile.name), spriteAtlasFile);
300 return;
301 } else if (sprites[0].packingMode == SpritePackingMode.Tight) {
302 Debug.LogError(string.Format("SpriteAtlas '{0}': Tight packing is not supported. Please disable 'Tight Packing' in the SpriteAtlas Inspector.", spriteAtlasFile.name), spriteAtlasFile);
303 return;
304 }
305
306 if (savedRegions == null || savedRegions.Length != sprites.Length)
307 savedRegions = new SavedRegionInfo[sprites.Length];
308
309 int i = 0;
310 foreach (AtlasRegion region in atlas) {
311 Sprite sprite = sprites[i];
312 Rect rect = sprite.textureRect;
313 float x = rect.min.x;
314 float y = rect.min.y;
315 float width = rect.width;
316 float height = rect.height;
317
318 SavedRegionInfo savedRegion = new SavedRegionInfo();
319 savedRegion.x = x;
320 savedRegion.y = y;
321 savedRegion.width = width;
322 savedRegion.height = height;
323 savedRegion.packingRotation = sprite.packingRotation;
324 savedRegions[i] = savedRegion;
325
326 ++i;
327 }
328 updateRegionsInPlayMode = false;
329 AssignRegionsFromSavedRegions(sprites, atlas);
330 EditorUtility.SetDirty(this);
331 AssetDatabase.SaveAssets();
332 }
