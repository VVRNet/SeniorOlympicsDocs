130 void RenderOntoQuad (SkeletonRenderer skeletonRenderer) {

141 protected void PrepareForMesh () {
142 // We need to get the min/max of all four corners, rotation of the skeleton
143 // in combination with perspective projection otherwise might lead to incorrect
144 // screen space min/max.
145 Bounds boundsLocalSpace = meshFilter.sharedMesh.bounds;
146 Vector3 localCorner0 = boundsLocalSpace.min;
147 Vector3 localCorner3 = boundsLocalSpace.max;
148 Vector3 localCorner1 = new Vector3(localCorner0.x, localCorner3.y, localCorner0.z);
149 Vector3 localCorner2 = new Vector3(localCorner3.x, localCorner0.y, localCorner3.z);
150
151 Vector3 worldCorner0 = transform.TransformPoint(localCorner0);
152 Vector3 worldCorner1 = transform.TransformPoint(localCorner1);
153 Vector3 worldCorner2 = transform.TransformPoint(localCorner2);
154 Vector3 worldCorner3 = transform.TransformPoint(localCorner3);
155
156 Vector3 screenCorner0 = targetCamera.WorldToScreenPoint(worldCorner0);
157 Vector3 screenCorner1 = targetCamera.WorldToScreenPoint(worldCorner1);
158 Vector3 screenCorner2 = targetCamera.WorldToScreenPoint(worldCorner2);
159 Vector3 screenCorner3 = targetCamera.WorldToScreenPoint(worldCorner3);
160
161 // To avoid perspective distortion when rotated, we project all vertices
162 // onto a plane parallel to the view frustum near plane.
163 // Avoids the requirement of 'noperspective' vertex attribute interpolation modifier in shaders.
164 float averageScreenDepth = (screenCorner0.z + screenCorner1.z + screenCorner2.z + screenCorner3.z) / 4.0f;
165 screenCorner0.z = screenCorner1.z = screenCorner2.z = screenCorner3.z = averageScreenDepth;
166 worldCornerNoDistortion0 = targetCamera.ScreenToWorldPoint(screenCorner0);
167 worldCornerNoDistortion1 = targetCamera.ScreenToWorldPoint(screenCorner1);
168 worldCornerNoDistortion2 = targetCamera.ScreenToWorldPoint(screenCorner2);
169 worldCornerNoDistortion3 = targetCamera.ScreenToWorldPoint(screenCorner3);
170
171 Vector3 screenSpaceMin, screenSpaceMax;
172 PrepareTextureMapping(out screenSpaceMin, out screenSpaceMax,
173 screenCorner0, screenCorner1, screenCorner2, screenCorner3);
174 PrepareCommandBuffer(targetCamera, screenSpaceMin, screenSpaceMax);
175 }
176
177 protected void PrepareCommandBuffer (Camera targetCamera, Vector3 screenSpaceMin, Vector3 screenSpaceMax) {
178 commandBuffer.Clear();
179 commandBuffer.SetRenderTarget(renderTexture);
180 commandBuffer.ClearRenderTarget(true, true, Color.clear);
181
182 commandBuffer.SetViewMatrix(targetCamera.worldToCameraMatrix);
183
184 Matrix4x4 projectionMatrix = CalculateProjectionMatrix(targetCamera,
185 screenSpaceMin, screenSpaceMax, targetCamera.pixelRect.size);
186 commandBuffer.SetProjectionMatrix(projectionMatrix);
187
188 Vector2 targetViewportSize = new Vector2(
189 screenSpaceMax.x - screenSpaceMin.x,
190 screenSpaceMax.y - screenSpaceMin.y);
191 Rect viewportRect = new Rect(Vector2.zero, targetViewportSize * downScaleFactor);
192 commandBuffer.SetViewport(viewportRect);
193 }
194
195 protected void RenderToRenderTexture () {
196 meshRenderer.GetPropertyBlock(propertyBlock);
197 meshRenderer.GetSharedMaterials(materials);
198
199 for (int i = 0; i < materials.Count; i++) {
200 foreach (int shaderPass in shaderPasses)
201 commandBuffer.DrawMesh(meshFilter.sharedMesh, transform.localToWorldMatrix,
202 materials[i], meshRenderer.subMeshStartIndex + i, shaderPass, propertyBlock);
203 }
204 Graphics.ExecuteCommandBuffer(commandBuffer);
205 }

211 }
