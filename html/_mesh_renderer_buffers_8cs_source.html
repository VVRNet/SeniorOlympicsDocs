57 public Material[] GetUpdatedSharedMaterialsArray () {
58 if (submeshMaterials.Count == sharedMaterials.Length)
59 submeshMaterials.CopyTo(sharedMaterials);
60 else
61 sharedMaterials = submeshMaterials.ToArray();
62
63 return sharedMaterials;
64 }

67 public bool MaterialsChangedInLastUpdate () {
68 int newSubmeshMaterials = submeshMaterials.Count;
69 Material[] sharedMaterials = this.sharedMaterials;
70 if (newSubmeshMaterials != sharedMaterials.Length) return true;
71
72 Material[] submeshMaterialsItems = submeshMaterials.Items;
73 for (int i = 0; i < newSubmeshMaterials; i++)
74 if (!Material.ReferenceEquals(submeshMaterialsItems[i], sharedMaterials[i])) return true; //if (submeshMaterialsItems[i].GetInstanceID() != sharedMaterials[i].GetInstanceID()) return true;
75
76 return false;
77 }

80 public void UpdateSharedMaterials (ExposedList<SubmeshInstruction> instructions) {
81 int newSize = instructions.Count;
82 { //submeshMaterials.Resize(instructions.Count);
83 if (newSize > submeshMaterials.Items.Length)
84 Array.Resize(ref submeshMaterials.Items, newSize);
85 submeshMaterials.Count = newSize;

87
88 Material[] submeshMaterialsItems = submeshMaterials.Items;
89 SubmeshInstruction[] instructionsItems = instructions.Items;

92 }
