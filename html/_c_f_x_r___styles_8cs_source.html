32 if(gs != null)
33 {
34 _closeCrossButton = System.Array.Find<GUIStyle>(gs.customStyles, x => x.name == "CloseCrossButton");

36

334 static Texture2D warnIcon;
335 static Texture2D infoIcon;
336 static Texture2D errorIcon;
337
338 static public void HelpBoxRichText(Rect position, string message, MessageType msgType)
339 {
340 Texture2D icon = null;
341 switch(msgType)
342 {
343 case MessageType.Warning: icon = warnIcon ?? (warnIcon = EditorGUIUtility.Load("console.warnicon") as Texture2D); break;
344 case MessageType.Info: icon = infoIcon ?? (infoIcon = EditorGUIUtility.Load("console.infoicon") as Texture2D); break;
345 case MessageType.Error: icon = errorIcon ?? (errorIcon = EditorGUIUtility.Load("console.erroricon") as Texture2D); break;
346 }
347 EditorGUI.LabelField(position, GUIContent.none, new GUIContent(message, icon), HelpBoxRichTextStyle);
348 }
