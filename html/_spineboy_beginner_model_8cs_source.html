30using System.Collections;
31using UnityEngine;

33namespace Spine.Unity.Examples {
34 [SelectionBase]
35 public class SpineboyBeginnerModel : MonoBehaviour {
36
37 #region Inspector
38 [Header("Current State")]
39 public SpineBeginnerBodyState state;
40 public bool facingLeft;
41 [Range(-1f, 1f)]
42 public float currentSpeed;
43
44 [Header("Balance")]
45 public float shootInterval = 0.12f;
46 #endregion
47
48 float lastShootTime;
49 public event System.Action ShootEvent; // Lets other scripts know when Spineboy is shooting. Check C# Documentation to learn more about events and delegates.
50 public event System.Action StartAimEvent; // Lets other scripts know when Spineboy is aiming.
51 public event System.Action StopAimEvent; // Lets other scripts know when Spineboy is no longer aiming.
52
53 #region API
54 public void TryJump () {
55 StartCoroutine(JumpRoutine());
56 }
57
58 public void TryShoot () {
59 float currentTime = Time.time;
60
61 if (currentTime - lastShootTime > shootInterval) {
62 lastShootTime = currentTime;
63 if (ShootEvent != null) ShootEvent(); // Fire the "ShootEvent" event.
64 }
65 }
66
67 public void StartAim () {
68 if (StartAimEvent != null) StartAimEvent(); // Fire the "StartAimEvent" event.
69 }
70
71 public void StopAim () {
72 if (StopAimEvent != null) StopAimEvent(); // Fire the "StopAimEvent" event.
73 }
74
75 public void TryMove (float speed) {
76 currentSpeed = speed; // show the "speed" in the Inspector.
77
78 if (speed != 0) {
79 bool speedIsNegative = (speed < 0f);
80 facingLeft = speedIsNegative; // Change facing direction whenever speed is not 0.
81 }
82
83 if (state != SpineBeginnerBodyState.Jumping) {
84 state = (speed == 0) ? SpineBeginnerBodyState.Idle : SpineBeginnerBodyState.Running;
85 }
86
87 }
88 #endregion
89
90 IEnumerator JumpRoutine () {
91 if (state == SpineBeginnerBodyState.Jumping) yield break; // Don't jump when already jumping.
92
93 state = SpineBeginnerBodyState.Jumping;
94
95 // Fake jumping.
96 {
97 Vector3 pos = transform.localPosition;
98 const float jumpTime = 1.2f;
99 const float half = jumpTime * 0.5f;
100 const float jumpPower = 20f;
101 for (float t = 0; t < half; t += Time.deltaTime) {
102 float d = jumpPower * (half - t);
103 transform.Translate((d * Time.deltaTime) * Vector3.up);
104 yield return null;
105 }
106 for (float t = 0; t < half; t += Time.deltaTime) {
107 float d = jumpPower * t;
108 transform.Translate((d * Time.deltaTime) * Vector3.down);
109 yield return null;
110 }
111 transform.localPosition = pos;
112 }
113
114 state = SpineBeginnerBodyState.Idle;
115 }
116
117 }
118
119 public enum SpineBeginnerBodyState {
120 Idle,
121 Running,
122 Jumping
123 }
