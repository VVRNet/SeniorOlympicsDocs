54 public void SetGaugePercent (float percent) {

56 Skeleton skeleton = skeletonRenderer.skeleton; if (skeleton == null) return;
57
58 fillAnimation.Animation.Apply(skeleton, 0, percent, false, null, 1f, MixBlend.Setup, MixDirection.In);
