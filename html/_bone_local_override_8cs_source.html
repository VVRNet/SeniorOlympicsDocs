34namespace Spine.Unity.Examples {
35 public class BoneLocalOverride : MonoBehaviour {
36 [SpineBone]
37 public string boneName;
38
39 [Space]
40 [Range(0, 1)] public float alpha = 1;
41
42 [Space]
43 public bool overridePosition = true;
44 public Vector2 localPosition;
45
46 [Space]
47 public bool overrideRotation = true;
48 [Range(0, 360)] public float rotation = 0;
49
50 ISkeletonAnimation spineComponent;
51 Bone bone;
52
53#if UNITY_EDITOR
54 void OnValidate () {
55 if (Application.isPlaying) return;
56 if (spineComponent == null) spineComponent = GetComponent<ISkeletonAnimation>();
57 if (spineComponent.IsNullOrDestroyed()) return;
58 if (bone != null) bone.SetToSetupPose();
59 OverrideLocal(spineComponent);
60 }
61#endif
62
63 void Awake () {
64 spineComponent = GetComponent<ISkeletonAnimation>();
65 if (spineComponent == null) { this.enabled = false; return; }
66 spineComponent.UpdateLocal += OverrideLocal;
67
68 if (bone == null) { this.enabled = false; return; }
69 }
70
71 void OverrideLocal (ISkeletonAnimation animated) {
72 if (bone == null || bone.Data.Name != boneName) {
73 if (string.IsNullOrEmpty(boneName)) return;
74 bone = spineComponent.Skeleton.FindBone(boneName);
75 if (bone == null) {
76 Debug.LogFormat("Cannot find bone: '{0}'", boneName);
77 return;
78 }
79 }
80
81 if (overridePosition) {
82 bone.X = Mathf.Lerp(bone.X, localPosition.x, alpha);
83 bone.Y = Mathf.Lerp(bone.Y, localPosition.y, alpha);
84 }
85
86 if (overrideRotation)
87 bone.Rotation = Mathf.Lerp(bone.Rotation, rotation, alpha);
88 }
89
90 }
