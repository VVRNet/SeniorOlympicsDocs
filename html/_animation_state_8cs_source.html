139 public void Update (float delta) {
140 delta *= timeScale;
141 TrackEntry[] tracksItems = tracks.Items;
142 for (int i = 0, n = tracks.Count; i < n; i++) {
143 TrackEntry current = tracksItems[i];
144 if (current == null) continue;
145
146 current.animationLast = current.nextAnimationLast;
147 current.trackLast = current.nextTrackLast;
148
149 float currentDelta = delta * current.timeScale;
150
151 if (current.delay > 0) {
152 current.delay -= currentDelta;
153 if (current.delay > 0) continue;
154 currentDelta = -current.delay;
155 current.delay = 0;
156 }
157
158 TrackEntry next = current.next;
159 if (next != null) {
160 // When the next entry's delay is passed, change to the next entry, preserving leftover time.
161 float nextTime = current.trackLast - next.delay;
162 if (nextTime >= 0) {
163 next.delay = 0;
164 next.trackTime += current.timeScale == 0 ? 0 : (nextTime / current.timeScale + delta) * next.timeScale;
165 current.trackTime += currentDelta;
166 SetCurrent(i, next, true);
167 while (next.mixingFrom != null) {
168 next.mixTime += delta;
169 next = next.mixingFrom;
170 }
171 continue;
172 }
173 } else if (current.trackLast >= current.trackEnd && current.mixingFrom == null) {
174 // Clear the track when there is no next entry, the track end time is reached, and there is no mixingFrom.
175 tracksItems[i] = null;
176 queue.End(current);
177 ClearNext(current);
178 continue;
179 }
180 if (current.mixingFrom != null && UpdateMixingFrom(current, delta)) {
181 // End mixing from entries once all have completed.
182 TrackEntry from = current.mixingFrom;
183 current.mixingFrom = null;
184 if (from != null) from.mixingTo = null;
185 while (from != null) {
186 queue.End(from);
187 from = from.mixingFrom;
188 }
189 }
190
191 current.trackTime += currentDelta;
192 }
193
194 queue.Drain();
195 }

198 private bool UpdateMixingFrom (TrackEntry to, float delta) {
199 TrackEntry from = to.mixingFrom;
200 if (from == null) return true;
201
202 bool finished = UpdateMixingFrom(from, delta);
203
204 from.animationLast = from.nextAnimationLast;
205 from.trackLast = from.nextTrackLast;
206
207 // Require mixTime > 0 to ensure the mixing from entry was applied at least once.
208 if (to.mixTime > 0 && to.mixTime >= to.mixDuration) {
209 // Require totalAlpha == 0 to ensure mixing is complete, unless mixDuration == 0 (the transition is a single frame).
210 if (from.totalAlpha == 0 || to.mixDuration == 0) {
211 to.mixingFrom = from.mixingFrom;
212 if (from.mixingFrom != null) from.mixingFrom.mixingTo = to;
213 to.interruptAlpha = from.interruptAlpha;
214 queue.End(from);
215 }
216 return finished;
217 }
218
219 from.trackTime += delta * from.timeScale;
220 to.mixTime += delta;
221 return false;
222 }

228 public bool Apply (Skeleton skeleton) {
229 if (skeleton == null) throw new ArgumentNullException("skeleton", "skeleton cannot be null.");
230 if (animationsChanged) AnimationsChanged();
231
232 ExposedList<Event> events = this.events;
233 bool applied = false;
234 TrackEntry[] tracksItems = tracks.Items;
235 for (int i = 0, n = tracks.Count; i < n; i++) {
236 TrackEntry current = tracksItems[i];
237 if (current == null || current.delay > 0) continue;
238 applied = true;
239
240 // Track 0 animations aren't for layering, so do not show the previously applied animations before the first key.
241 MixBlend blend = i == 0 ? MixBlend.First : current.mixBlend;
242
243 // Apply mixing from entries first.
244 float alpha = current.alpha;
245 if (current.mixingFrom != null)
246 alpha *= ApplyMixingFrom(current, skeleton, blend);
247 else if (current.trackTime >= current.trackEnd && current.next == null) //
248 alpha = 0; // Set to setup pose the last time the entry will be applied.
249 bool attachments = alpha >= current.alphaAttachmentThreshold;
250
251 // Apply current entry.
252 float animationLast = current.animationLast, animationTime = current.AnimationTime, applyTime = animationTime;
253 ExposedList<Event> applyEvents = events;
254 if (current.reverse) {
255 applyTime = current.animation.duration - applyTime;
256 applyEvents = null;
257 }
258
259 int timelineCount = current.animation.timelines.Count;
260 Timeline[] timelines = current.animation.timelines.Items;
261 if ((i == 0 && alpha == 1) || blend == MixBlend.Add) {
262 if (i == 0) attachments = true;
263 for (int ii = 0; ii < timelineCount; ii++) {
264 Timeline timeline = timelines[ii];
265 if (timeline is AttachmentTimeline)
266 ApplyAttachmentTimeline((AttachmentTimeline)timeline, skeleton, applyTime, blend, attachments);
267 else
268 timeline.Apply(skeleton, animationLast, applyTime, applyEvents, alpha, blend, MixDirection.In);
269 }
270 } else {
271 int[] timelineMode = current.timelineMode.Items;
272
273 bool shortestRotation = current.shortestRotation;
274 bool firstFrame = !shortestRotation && current.timelinesRotation.Count != timelineCount << 1;
275 if (firstFrame) current.timelinesRotation.Resize(timelineCount << 1);
276 float[] timelinesRotation = current.timelinesRotation.Items;
277
278 for (int ii = 0; ii < timelineCount; ii++) {
279 Timeline timeline = timelines[ii];
280 MixBlend timelineBlend = timelineMode[ii] == AnimationState.Subsequent ? blend : MixBlend.Setup;
281 RotateTimeline rotateTimeline = timeline as RotateTimeline;
282 if (!shortestRotation && rotateTimeline != null)
283 ApplyRotateTimeline(rotateTimeline, skeleton, applyTime, alpha, timelineBlend, timelinesRotation,
284 ii << 1, firstFrame);
285 else if (timeline is AttachmentTimeline)
286 ApplyAttachmentTimeline((AttachmentTimeline)timeline, skeleton, applyTime, blend, attachments);
287 else
288 timeline.Apply(skeleton, animationLast, applyTime, applyEvents, alpha, timelineBlend, MixDirection.In);
289 }
290 }
291 QueueEvents(current, animationTime);
292 events.Clear(false);
293 current.nextAnimationLast = animationTime;
294 current.nextTrackLast = current.trackTime;
295 }
296
297 // Set slots attachments to the setup pose, if needed. This occurs if an animation that is mixing out sets attachments so
298 // subsequent timelines see any deform, but the subsequent timelines don't set an attachment (eg they are also mixing out or
299 // the time is before the first key).
300 int setupState = unkeyedState + Setup;
301 Slot[] slots = skeleton.slots.Items;
302 for (int i = 0, n = skeleton.slots.Count; i < n; i++) {
303 Slot slot = slots[i];
304 if (slot.attachmentState == setupState) {
305 string attachmentName = slot.data.attachmentName;
306 slot.Attachment = (attachmentName == null ? null : skeleton.GetAttachment(slot.data.index, attachmentName));
307 }
308 }
309 unkeyedState += 2; // Increasing after each use avoids the need to reset attachmentState for every slot.
310
311 queue.Drain();
312 return applied;
313 }

319 public bool ApplyEventTimelinesOnly (Skeleton skeleton, bool issueEvents = true) {
320 if (skeleton == null) throw new ArgumentNullException("skeleton", "skeleton cannot be null.");
321
322 ExposedList<Event> events = this.events;
323 bool applied = false;
324 TrackEntry[] tracksItems = tracks.Items;
325 for (int i = 0, n = tracks.Count; i < n; i++) {
326 TrackEntry current = tracksItems[i];
327 if (current == null || current.delay > 0) continue;
328 applied = true;
329
330 // Apply mixing from entries first.
331 if (current.mixingFrom != null) ApplyMixingFromEventTimelinesOnly(current, skeleton, issueEvents);
332
333 // Apply current entry.
334 float animationLast = current.animationLast, animationTime = current.AnimationTime;
335
336 if (issueEvents) {
337 int timelineCount = current.animation.timelines.Count;
338 Timeline[] timelines = current.animation.timelines.Items;
339 for (int ii = 0; ii < timelineCount; ii++) {
340 Timeline timeline = timelines[ii];
341 if (timeline is EventTimeline)
342 timeline.Apply(skeleton, animationLast, animationTime, events, 1.0f, MixBlend.Setup, MixDirection.In);
343 }
344 QueueEvents(current, animationTime);
345 events.Clear(false);
346 }
347 current.nextAnimationLast = animationTime;
348 current.nextTrackLast = current.trackTime;
349 }
350
351 if (issueEvents)
352 queue.Drain();
353 return applied;
354 }

356 private float ApplyMixingFrom (TrackEntry to, Skeleton skeleton, MixBlend blend) {
357 TrackEntry from = to.mixingFrom;
358 if (from.mixingFrom != null) ApplyMixingFrom(from, skeleton, blend);
359
360 float mix;
361 if (to.mixDuration == 0) { // Single frame mix to undo mixingFrom changes.
362 mix = 1;
363 if (blend == MixBlend.First) blend = MixBlend.Setup; // Tracks > 0 are transparent and can't reset to setup pose.
364 } else {
365 mix = to.mixTime / to.mixDuration;
366 if (mix > 1) mix = 1;
367 if (blend != MixBlend.First) blend = from.mixBlend; // Track 0 ignores track mix blend.
368 }
369
370 bool attachments = mix < from.mixAttachmentThreshold, drawOrder = mix < from.mixDrawOrderThreshold;
371 int timelineCount = from.animation.timelines.Count;
372 Timeline[] timelines = from.animation.timelines.Items;
373 float alphaHold = from.alpha * to.interruptAlpha, alphaMix = alphaHold * (1 - mix);
374 float animationLast = from.animationLast, animationTime = from.AnimationTime, applyTime = animationTime;
375 ExposedList<Event> events = null;
376 if (from.reverse)
377 applyTime = from.animation.duration - applyTime;
378 else {
379 if (mix < from.eventThreshold) events = this.events;
380 }
381
382 if (blend == MixBlend.Add) {
383 for (int i = 0; i < timelineCount; i++)
384 timelines[i].Apply(skeleton, animationLast, applyTime, events, alphaMix, blend, MixDirection.Out);
385 } else {
386 int[] timelineMode = from.timelineMode.Items;
387 TrackEntry[] timelineHoldMix = from.timelineHoldMix.Items;
388
389 bool shortestRotation = from.shortestRotation;
390 bool firstFrame = !shortestRotation && from.timelinesRotation.Count != timelineCount << 1;
391 if (firstFrame) from.timelinesRotation.Resize(timelineCount << 1);
392 float[] timelinesRotation = from.timelinesRotation.Items;
393
394 from.totalAlpha = 0;
395 for (int i = 0; i < timelineCount; i++) {
396 Timeline timeline = timelines[i];
397 MixDirection direction = MixDirection.Out;
398 MixBlend timelineBlend;
399 float alpha;
400 switch (timelineMode[i]) {
401 case AnimationState.Subsequent:
402 if (!drawOrder && timeline is DrawOrderTimeline) continue;
403 timelineBlend = blend;
404 alpha = alphaMix;
405 break;
406 case AnimationState.First:
407 timelineBlend = MixBlend.Setup;
408 alpha = alphaMix;
409 break;
410 case AnimationState.HoldSubsequent:
411 timelineBlend = blend;
412 alpha = alphaHold;
413 break;
414 case AnimationState.HoldFirst:
415 timelineBlend = MixBlend.Setup;
416 alpha = alphaHold;
417 break;
418 default: // HoldMix
419 timelineBlend = MixBlend.Setup;
420 TrackEntry holdMix = timelineHoldMix[i];
421 alpha = alphaHold * Math.Max(0, 1 - holdMix.mixTime / holdMix.mixDuration);
422 break;
423 }
424 from.totalAlpha += alpha;
425 RotateTimeline rotateTimeline = timeline as RotateTimeline;
426 if (!shortestRotation && rotateTimeline != null) {
427 ApplyRotateTimeline(rotateTimeline, skeleton, applyTime, alpha, timelineBlend, timelinesRotation, i << 1,
428 firstFrame);
429 } else if (timeline is AttachmentTimeline) {
430 ApplyAttachmentTimeline((AttachmentTimeline)timeline, skeleton, applyTime, timelineBlend,
431 attachments && alpha >= from.alphaAttachmentThreshold);
432 } else {
433 if (drawOrder && timeline is DrawOrderTimeline && timelineBlend == MixBlend.Setup)
434 direction = MixDirection.In;
435 timeline.Apply(skeleton, animationLast, applyTime, events, alpha, timelineBlend, direction);
436 }
437 }
438 }
439
440 if (to.mixDuration > 0) QueueEvents(from, animationTime);
441 this.events.Clear(false);
442 from.nextAnimationLast = animationTime;
443 from.nextTrackLast = from.trackTime;
444
445 return mix;
446 }

453 TrackEntry from = to.mixingFrom;

478 if (to.mixDuration > 0) QueueEvents(from, animationTime);
479 this.events.Clear(false);
480 }
481 from.nextAnimationLast = animationTime;
482 from.nextTrackLast = from.trackTime;

484 return mix;

491 private void ApplyAttachmentTimeline (AttachmentTimeline timeline, Skeleton skeleton, float time, MixBlend blend,
492 bool attachments) {
493
494 Slot slot = skeleton.slots.Items[timeline.SlotIndex];
495 if (!slot.bone.active) return;
496
497 float[] frames = timeline.frames;
498 if (time < frames[0]) { // Time is before first frame.
499 if (blend == MixBlend.Setup || blend == MixBlend.First)
500 SetAttachment(skeleton, slot, slot.data.attachmentName, attachments);
501 } else
502 SetAttachment(skeleton, slot, timeline.AttachmentNames[Timeline.Search(frames, time)], attachments);
503
504 // If an attachment wasn't set (ie before the first frame or attachments is false), set the setup attachment later.
505 if (slot.attachmentState <= unkeyedState) slot.attachmentState = unkeyedState + Setup;
506 }

508 private void SetAttachment (Skeleton skeleton, Slot slot, String attachmentName, bool attachments) {
509 slot.Attachment = attachmentName == null ? null : skeleton.GetAttachment(slot.data.index, attachmentName);
510 if (attachments) slot.attachmentState = unkeyedState + Current;
511 }

516 static private void ApplyRotateTimeline (RotateTimeline timeline, Skeleton skeleton, float time, float alpha, MixBlend blend,
517 float[] timelinesRotation, int i, bool firstFrame) {
518
519 if (firstFrame) timelinesRotation[i] = 0;
520
521 if (alpha == 1) {
522 timeline.Apply(skeleton, 0, time, null, 1, blend, MixDirection.In);
523 return;
524 }
525
526 Bone bone = skeleton.bones.Items[timeline.BoneIndex];
527 if (!bone.active) return;
528
529 float[] frames = timeline.frames;
530 float r1, r2;
531 if (time < frames[0]) { // Time is before first frame.
532 switch (blend) {
533 case MixBlend.Setup:
534 bone.rotation = bone.data.rotation;
535 goto default; // Fall through.
536 default:
537 return;
538 case MixBlend.First:
539 r1 = bone.rotation;
540 r2 = bone.data.rotation;
541 break;
542 }
543 } else {
544 r1 = blend == MixBlend.Setup ? bone.data.rotation : bone.rotation;
545 r2 = bone.data.rotation + timeline.GetCurveValue(time);
546 }
547
548 // Mix between rotations using the direction of the shortest route on the first frame.
549 float total, diff = r2 - r1;
550 diff -= (float)Math.Ceiling(diff / 360 - 0.5f) * 360;
551 if (diff == 0) {
552 total = timelinesRotation[i];
553 } else {
554 float lastTotal, lastDiff;
555 if (firstFrame) {
556 lastTotal = 0;
557 lastDiff = diff;
558 } else {
559 lastTotal = timelinesRotation[i];
560 lastDiff = timelinesRotation[i + 1];
561 }
562 float loops = lastTotal - lastTotal % 360;
563 total = diff + loops;
564 bool current = diff >= 0, dir = lastTotal >= 0;
565 if (Math.Abs(lastDiff) <= 90 && Math.Sign(lastDiff) != Math.Sign(diff)) {
566 if (Math.Abs(lastTotal - loops) > 180) {
567 total += 360 * Math.Sign(lastTotal);
568 dir = current;
569 } else if (loops != 0)
570 total -= 360 * Math.Sign(lastTotal);
571 else
572 dir = current;
573 }
574 if (dir != current) total += 360 * Math.Sign(lastTotal);
575 timelinesRotation[i] = total;
576 }
577 timelinesRotation[i + 1] = diff;
578 bone.rotation = r1 + total * alpha;
579 }

581 private void QueueEvents (TrackEntry entry, float animationTime) {
582 float animationStart = entry.animationStart, animationEnd = entry.animationEnd;
583 float duration = animationEnd - animationStart;
584 float trackLastWrapped = entry.trackLast % duration;
585
586 // Queue events before complete.
587 Event[] eventsItems = this.events.Items;
588 int i = 0, n = events.Count;
589 for (; i < n; i++) {
590 Event e = eventsItems[i];
591 if (e.time < trackLastWrapped) break;
592 if (e.time > animationEnd) continue; // Discard events outside animation start/end.
593 queue.Event(entry, e);
594 }
595
596 // Queue complete if completed a loop iteration or the animation.
597 bool complete = false;
598 if (entry.loop) {
599 if (duration == 0)
600 complete = true;
601 else {
602 int cycles = (int)(entry.trackTime / duration);
603 complete = cycles > 0 && cycles > (int)(entry.trackLast / duration);
604 }
605 } else
606 complete = animationTime >= animationEnd && entry.animationLast < animationEnd;
607 if (complete) queue.Complete(entry);
608
609 // Queue events after complete.
610 for (; i < n; i++) {
611 Event e = eventsItems[i];
612 if (e.time < animationStart) continue; // Discard events outside animation start/end.
613 queue.Event(entry, eventsItems[i]);
614 }
615 }

640 public void ClearTrack (int trackIndex) {
641 if (trackIndex >= tracks.Count) return;
642 TrackEntry current = tracks.Items[trackIndex];
643 if (current == null) return;
644
645 queue.End(current);
646
647 ClearNext(current);
648
649 TrackEntry entry = current;
650 while (true) {
651 TrackEntry from = entry.mixingFrom;
652 if (from == null) break;
653 queue.End(from);
654 entry.mixingFrom = null;
655 entry.mixingTo = null;
656 entry = from;
657 }
658
659 tracks.Items[current.trackIndex] = null;
660
661 queue.Drain();
662 }

665 private void SetCurrent (int index, TrackEntry current, bool interrupt) {
666 TrackEntry from = ExpandToIndex(index);
667 tracks.Items[index] = current;
668 current.previous = null;
669
670 if (from != null) {
671 if (interrupt) queue.Interrupt(from);
672 current.mixingFrom = from;
673 from.mixingTo = current;
674 current.mixTime = 0;
675
676 // Store the interrupted mix percentage.
677 if (from.mixingFrom != null && from.mixDuration > 0)
678 current.interruptAlpha *= Math.Min(1, from.mixTime / from.mixDuration);
679
680 from.timelinesRotation.Clear(); // Reset rotation for mixing out, in case entry was mixed in.
681 }
682
683 queue.Start(current); // triggers AnimationsChanged
684 }

699 public TrackEntry SetAnimation (int trackIndex, Animation animation, bool loop) {
700 if (animation == null) throw new ArgumentNullException("animation", "animation cannot be null.");
701 bool interrupt = true;
702 TrackEntry current = ExpandToIndex(trackIndex);
703 if (current != null) {
704 if (current.nextTrackLast == -1) {
705 // Don't mix from an entry that was never applied.
706 tracks.Items[trackIndex] = current.mixingFrom;
707 queue.Interrupt(current);
708 queue.End(current);
709 ClearNext(current);
710 current = current.mixingFrom;
711 interrupt = false; // mixingFrom is current again, but don't interrupt it twice.
712 } else
713 ClearNext(current);
714 }
715 TrackEntry entry = NewTrackEntry(trackIndex, animation, loop, current);
716 SetCurrent(trackIndex, entry, interrupt);
717 queue.Drain();
718 return entry;
719 }

739 public TrackEntry AddAnimation (int trackIndex, Animation animation, bool loop, float delay) {
740 if (animation == null) throw new ArgumentNullException("animation", "animation cannot be null.");
741
742 TrackEntry last = ExpandToIndex(trackIndex);
743 if (last != null) {
744 while (last.next != null)
745 last = last.next;
746 }
747
748 TrackEntry entry = NewTrackEntry(trackIndex, animation, loop, last);
749
750 if (last == null) {
751 SetCurrent(trackIndex, entry, true);
752 queue.Drain();
753 } else {
754 last.next = entry;
755 entry.previous = last;
756 if (delay <= 0) delay += last.TrackComplete - entry.mixDuration;
757 }
758
759 entry.delay = delay;
760 return entry;

873 private void AnimationsChanged () {
874 animationsChanged = false;
875
876 // Process in the order that animations are applied.
877 propertyIds.Clear();
878 int n = tracks.Count;
879 TrackEntry[] tracksItems = tracks.Items;
880 for (int i = 0; i < n; i++) {
881 TrackEntry entry = tracksItems[i];
882 if (entry == null) continue;
883 while (entry.mixingFrom != null) // Move to last entry, then iterate in reverse.
884 entry = entry.mixingFrom;
885 do {
886 if (entry.mixingTo == null || entry.mixBlend != MixBlend.Add) ComputeHold(entry);
887 entry = entry.mixingTo;
888 } while (entry != null);
889 }
890 }

892 private void ComputeHold (TrackEntry entry) {
893 TrackEntry to = entry.mixingTo;
894 Timeline[] timelines = entry.animation.timelines.Items;
895 int timelinesCount = entry.animation.timelines.Count;
896 int[] timelineMode = entry.timelineMode.Resize(timelinesCount).Items;
897 entry.timelineHoldMix.Clear();
898 TrackEntry[] timelineHoldMix = entry.timelineHoldMix.Resize(timelinesCount).Items;
899 HashSet<string> propertyIds = this.propertyIds;
900
901 if (to != null && to.holdPrevious) {
902 for (int i = 0; i < timelinesCount; i++)
903 timelineMode[i] = propertyIds.AddAll(timelines[i].PropertyIds) ? AnimationState.HoldFirst : AnimationState.HoldSubsequent;
904
905 return;
906 }
907
908 // outer:
909 for (int i = 0; i < timelinesCount; i++) {
910 Timeline timeline = timelines[i];
911 String[] ids = timeline.PropertyIds;
912 if (!propertyIds.AddAll(ids))
913 timelineMode[i] = AnimationState.Subsequent;
914 else if (to == null || timeline is AttachmentTimeline || timeline is DrawOrderTimeline
915 || timeline is EventTimeline || !to.animation.HasTimeline(ids)) {
916 timelineMode[i] = AnimationState.First;
917 } else {
918 for (TrackEntry next = to.mixingTo; next != null; next = next.mixingTo) {
919 if (next.animation.HasTimeline(ids)) continue;
920 if (next.mixDuration > 0) {
921 timelineMode[i] = AnimationState.HoldMix;
922 timelineHoldMix[i] = next;
923 goto continue_outer; // continue outer;
924 }
925 break;
926 }
927 timelineMode[i] = AnimationState.HoldFirst;
928 }
929 continue_outer: { }
930 }
931 }

1346 class EventQueue {
1347 private readonly List<EventQueueEntry> eventQueueEntries = new List<EventQueueEntry>();
1348 internal bool drainDisabled;
1349
1350 private readonly AnimationState state;
1351 private readonly Pool<TrackEntry> trackEntryPool;
1352 internal event Action AnimationsChanged;
1353
1354 internal EventQueue (AnimationState state, Action HandleAnimationsChanged, Pool<TrackEntry> trackEntryPool) {
1355 this.state = state;
1356 this.AnimationsChanged += HandleAnimationsChanged;
1357 this.trackEntryPool = trackEntryPool;
1358 }
1359
1360 internal void Start (TrackEntry entry) {
1361 eventQueueEntries.Add(new EventQueueEntry(EventType.Start, entry));
1362 if (AnimationsChanged != null) AnimationsChanged();
1363 }
1364
1365 internal void Interrupt (TrackEntry entry) {
1366 eventQueueEntries.Add(new EventQueueEntry(EventType.Interrupt, entry));
1367 }
1368
1369 internal void End (TrackEntry entry) {
1370 eventQueueEntries.Add(new EventQueueEntry(EventType.End, entry));
1371 if (AnimationsChanged != null) AnimationsChanged();
1372 }
1373
1374 internal void Dispose (TrackEntry entry) {
1375 eventQueueEntries.Add(new EventQueueEntry(EventType.Dispose, entry));
1376 }
1377
1378 internal void Complete (TrackEntry entry) {
1379 eventQueueEntries.Add(new EventQueueEntry(EventType.Complete, entry));
1380 }
1381
1382 internal void Event (TrackEntry entry, Event e) {
1383 eventQueueEntries.Add(new EventQueueEntry(EventType.Event, entry, e));
1384 }
1385
1387 internal void Drain () {
1388 if (drainDisabled) return;
1389 drainDisabled = true;
1390
1391 List<EventQueueEntry> eventQueueEntries = this.eventQueueEntries;
1392 AnimationState state = this.state;
1393
1394 // Don't cache eventQueueEntries.Count so callbacks can queue their own events (eg, call SetAnimation in AnimationState_Complete).
1395 for (int i = 0; i < eventQueueEntries.Count; i++) {
1396 EventQueueEntry queueEntry = eventQueueEntries[i];
1397 TrackEntry trackEntry = queueEntry.entry;
1398
1399 switch (queueEntry.type) {
1400 case EventType.Start:
1401 trackEntry.OnStart();
1402 state.OnStart(trackEntry);
1403 break;
1404 case EventType.Interrupt:
1405 trackEntry.OnInterrupt();
1406 state.OnInterrupt(trackEntry);
1407 break;
1408 case EventType.End:
1409 trackEntry.OnEnd();
1410 state.OnEnd(trackEntry);
1411 goto case EventType.Dispose; // Fall through. (C#)
1412 case EventType.Dispose:
1413 trackEntry.OnDispose();
1414 state.OnDispose(trackEntry);
1415 trackEntryPool.Free(trackEntry);
1416 break;
1417 case EventType.Complete:
1418 trackEntry.OnComplete();
1419 state.OnComplete(trackEntry);
1420 break;
1421 case EventType.Event:
1422 trackEntry.OnEvent(queueEntry.e);
1423 state.OnEvent(trackEntry, queueEntry.e);
1424 break;
1425 }
1426 }
1427 eventQueueEntries.Clear();
1428
1429 drainDisabled = false;
1430 }
1431
1432 internal void Clear () {
1433 eventQueueEntries.Clear();
1434 }
