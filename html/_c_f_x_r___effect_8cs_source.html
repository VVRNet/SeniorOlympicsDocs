72 public float rangeEnd = 0f;

75 public bool perlinRange;
76 public float perlinRangeSpeed = 1f;
77
78 public bool animateColor;
79 public Gradient colorGradient;
80 public float colorDuration = 0.5f;

82 public bool perlinColor;
83 public float perlinColorSpeed = 1f;

85 public void animate(float time)
86 {
87#if UNITY_EDITOR
88 if (!editorPreview && !EditorApplication.isPlaying)
89 {
90 return;
91 }
92#endif
93
94 if (light != null)
95 {
96 if (animateIntensity)
97 {
98 float delta = loop ? Mathf.Clamp01((time % intensityDuration)/intensityDuration) : Mathf.Clamp01(time/intensityDuration);
99 delta = perlinIntensity ? Mathf.PerlinNoise(Time.time * perlinIntensitySpeed, 0f) : intensityCurve.Evaluate(delta);
100 light.intensity = Mathf.LerpUnclamped(intensityEnd, intensityStart, delta);
101
102 if (fadeIn && time < fadeInDuration)
103 {
104 light.intensity *= Mathf.Clamp01(time / fadeInDuration);
105 }
106 }
107
108 if (animateRange)
109 {
110 float delta = loop ? Mathf.Clamp01((time % rangeDuration)/rangeDuration) : Mathf.Clamp01(time/rangeDuration);
111 delta = perlinRange ? Mathf.PerlinNoise(Time.time * perlinRangeSpeed, 10f) : rangeCurve.Evaluate(delta);
112 light.range = Mathf.LerpUnclamped(rangeEnd, rangeStart, delta);
113 }
114
115 if (animateColor)
116 {
117 float delta = loop ? Mathf.Clamp01((time % colorDuration)/colorDuration) : Mathf.Clamp01(time/colorDuration);
118 delta = perlinColor ? Mathf.PerlinNoise(Time.time * perlinColorSpeed, 0f) : colorCurve.Evaluate(delta);
119 light.color = colorGradient.Evaluate(delta);
120 }
121 }
122 }
123
124 public void animateFadeOut(float time)
125 {
126 if (fadeOut && light != null)
127 {
128 light.intensity *= 1.0f - Mathf.Clamp01(time / fadeOutDuration);
129 }
130 }
131
132 public void reset()
133 {
134 if (light != null)
135 {
136 if (animateIntensity)
137 {
138 light.intensity = (fadeIn || fadeOut) ? 0 : intensityEnd;
139 }
140
141 if (animateRange)
142 {
143 light.range = rangeEnd;
144 }
145
146 if (animateColor)
147 {
148 light.color = colorGradient.Evaluate(1f);
149 }
150 }
151 }

531 void Update()
532 {
533#if !DISABLE_LIGHTS || !DISABLE_CAMERA_SHAKE
534 time += Time.deltaTime;
535
536 Animate(time);
537
538 if (fadeOutReference != null && !fadeOutReference.isEmitting && (fadeOutReference.isPlaying || isFadingOut))
539 {
540 FadeOut(time);
541 }
542#endif
543#if !DISABLE_CLEAR_BEHAVIOR
544 if (clearBehavior != ClearBehavior.None)
545 {
546 if (rootParticleSystem == null)
547 {
548 rootParticleSystem = this.GetComponent<ParticleSystem>();
549 }
550
551 // Check isAlive every N frame, with an offset so that all active effects aren't checked at once
552 if ((Time.renderedFrameCount + startFrameOffset) % CHECK_EVERY_N_FRAME == 0)
553 {
554 if (!rootParticleSystem.IsAlive(true))
555 {
556 if (clearBehavior == ClearBehavior.Destroy)
557 {
558 GameObject.Destroy(this.gameObject);
559 }
560 else
561 {
562 this.gameObject.SetActive(false);
563 }
564 }
565 }
566 }
567#endif
568 if (materialPropertyBlock != null)
569 {
570 particleRenderer.GetPropertyBlock(materialPropertyBlock);
571 materialPropertyBlock.SetVector("_GameObjectWorldPosition", this.transform.position);
572 particleRenderer.SetPropertyBlock(materialPropertyBlock);
573 }
574 }
575#endif

655 public void RegisterEditorUpdate()
656 {
657 var type = PrefabUtility.GetPrefabAssetType(this.gameObject);
658 var status = PrefabUtility.GetPrefabInstanceStatus(this.gameObject);
659
660 // Prefab in Project window
661 if ((type == PrefabAssetType.Regular || type == PrefabAssetType.Variant) && status == PrefabInstanceStatus.NotAPrefab)
662 {
663 return;
664 }
665
666 if (!editorUpdateRegistered)
667 {
668 EditorApplication.update += onEditorUpdate;
669 editorUpdateRegistered = true;
670 }
671 }
672
673 public void UnregisterEditorUpdate()
674 {
675 if (editorUpdateRegistered)
676 {
677 editorUpdateRegistered = false;
678 EditorApplication.update -= onEditorUpdate;
679 }
680 ResetState();
681 }
682
683 void onEditorUpdate()
684 {
685 if (EditorApplication.isPlayingOrWillChangePlaymode)
686 {
687 return;
688 }
689
690 if (this == null)
691 {
692 return;
693 }
694
695 var renderer = this.GetComponent<ParticleSystemRenderer>();
696 if (renderer.sharedMaterial != null && renderer.sharedMaterial.IsKeywordEnabled("_CFXR_LIGHTING_WPOS_OFFSET"))
697 {
698 if (materialPropertyBlock == null)
699 {
700 materialPropertyBlock = new MaterialPropertyBlock();
701 }
702
703 renderer.GetPropertyBlock(materialPropertyBlock);
704 materialPropertyBlock.SetVector("_GameObjectWorldPosition", this.transform.position);
705 renderer.SetPropertyBlock(materialPropertyBlock);
706 }
707
708 // Need to track unwrapped time when playing back from Editor
709 // because the parentParticle.time will be reset at each loop
710 float delta = parentParticle.time - particleTime;
711
712 if (delta < 0 && parentParticle.isPlaying)
713 {
714 delta = parentParticle.main.duration + delta;
715 if (delta > 0.1 || delta < 0)
716 {
717 // try to detect when "Restart" is pressed
718 ResetState();
719 particleTimeUnwrapped = 0;
720 delta = 0;
721 }
722 }
723 particleTimeUnwrapped += delta;
724
725 if (particleTime != parentParticle.time)
726 {
727#if !DISABLE_CAMERA_SHAKE
728 if (cameraShake != null && cameraShake.enabled && parentParticle.time < particleTime && parentParticle.time < 0.05f)
729 {
730 cameraShake.StartShake();
731 }
732#endif
733#if !DISABLE_LIGHTS || !DISABLE_CAMERA_SHAKES
734 Animate(particleTimeUnwrapped);
735
736 if (!parentParticle.isEmitting)
737 {
738 FadeOut(particleTimeUnwrapped);
739 }
740#endif
741 }
742
743 if (particleWasStopped != parentParticle.isStopped)
744 {
745 if (parentParticle.isStopped)
746 {
747 ResetState();
748 }
749 particleTimeUnwrapped = 0;
750 }
751
752 particleWasStopped = parentParticle.isStopped;
753 particleTime = parentParticle.time;
754 }
755#endif
