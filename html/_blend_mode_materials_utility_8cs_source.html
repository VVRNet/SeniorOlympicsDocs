67 public static void UpdateBlendModeMaterials (SkeletonDataAsset skeletonDataAsset, ref SkeletonData skeletonData,
68 bool upgradeFromModifierAssets = ShallUpgradeBlendModeMaterials) {
69
70 TemplateMaterials templateMaterials = new TemplateMaterials();
71 bool anyMaterialsChanged = ClearUndesiredMaterialEntries(skeletonDataAsset);
72
73 BlendModeMaterialsAsset blendModesModifierAsset = FindBlendModeMaterialsModifierAsset(skeletonDataAsset);
74 if (blendModesModifierAsset) {
75 if (upgradeFromModifierAssets) {
76 TransferSettingsFromModifierAsset(blendModesModifierAsset,
77 skeletonDataAsset, templateMaterials);
78 UpdateBlendmodeMaterialsRequiredState(skeletonDataAsset, skeletonData);
79 } else
80 return;
81 } else {
82 if (!UpdateBlendmodeMaterialsRequiredState(skeletonDataAsset, skeletonData))
83 return;
84 AssignPreferencesTemplateMaterials(templateMaterials);
85 }
86 bool success = CreateAndAssignMaterials(skeletonDataAsset, templateMaterials, ref anyMaterialsChanged);
87 if (success) {
88 if (blendModesModifierAsset != null) {
89 RemoveObsoleteModifierAsset(blendModesModifierAsset, skeletonDataAsset);
90 }
91 }
92
93 SpineEditorUtilities.ClearSkeletonDataAsset(skeletonDataAsset);
94 skeletonData = skeletonDataAsset.GetSkeletonData(true);
95 if (anyMaterialsChanged)
96 ReloadSceneSkeletons(skeletonDataAsset);
97 AssetDatabase.SaveAssets();
98 }

100 protected static bool ClearUndesiredMaterialEntries (SkeletonDataAsset skeletonDataAsset) {
101 Predicate<BlendModeMaterials.ReplacementMaterial> ifMaterialMissing = r => r.material == null;
102
103 bool anyMaterialsChanged = false;
104 if (!skeletonDataAsset.blendModeMaterials.applyAdditiveMaterial) {
105 anyMaterialsChanged |= skeletonDataAsset.blendModeMaterials.additiveMaterials.Count > 0;
106 skeletonDataAsset.blendModeMaterials.additiveMaterials.Clear();
107 } else
108 anyMaterialsChanged |= skeletonDataAsset.blendModeMaterials.additiveMaterials.RemoveAll(ifMaterialMissing) != 0;
109 anyMaterialsChanged |= skeletonDataAsset.blendModeMaterials.multiplyMaterials.RemoveAll(ifMaterialMissing) != 0;
110 anyMaterialsChanged |= skeletonDataAsset.blendModeMaterials.screenMaterials.RemoveAll(ifMaterialMissing) != 0;
111 return anyMaterialsChanged;
112 }

162 protected static bool CreateForRegion (ref List<BlendModeMaterials.ReplacementMaterial> replacementMaterials,
163 ref bool anyReplacementMaterialsChanged,
164 AtlasRegion originalRegion, Material materialTemplate, string materialSuffix,
165 SkeletonDataAsset skeletonDataAsset) {
166
167 bool anyCreationFailed = false;
168 bool replacementExists = replacementMaterials.Exists(
169 replacement => replacement.pageName == originalRegion.page.name);
170 if (!replacementExists) {
171 bool createdNewMaterial;
172 BlendModeMaterials.ReplacementMaterial replacement = CreateOrLoadReplacementMaterial(originalRegion, materialTemplate, materialSuffix, out createdNewMaterial);
173 if (replacement != null) {
174 replacementMaterials.Add(replacement);
175 anyReplacementMaterialsChanged = true;
176 if (createdNewMaterial) {
177 Debug.Log(string.Format("Created blend mode Material '{0}' for SkeletonData asset '{1}'.",
178 replacement.material.name, skeletonDataAsset), replacement.material);
179 }
180 } else {
181 Debug.LogError(string.Format("Failed creating blend mode Material for SkeletonData asset '{0}'," +
182 " atlas page '{1}', template '{2}'.",
183 skeletonDataAsset.name, originalRegion.page.name, materialTemplate.name),
184 skeletonDataAsset);
185 anyCreationFailed = true;
186 }
187 }
188 return anyCreationFailed;
189 }

191 protected static string GetBlendModeMaterialPath (AtlasPage originalPage, string materialSuffix) {

197 protected static BlendModeMaterials.ReplacementMaterial CreateOrLoadReplacementMaterial (
198 AtlasRegion originalRegion, Material materialTemplate, string materialSuffix, out bool createdNewMaterial) {
199
200 createdNewMaterial = false;
201 BlendModeMaterials.ReplacementMaterial newReplacement = new BlendModeMaterials.ReplacementMaterial();
202 AtlasPage originalPage = originalRegion.page;
203 Material originalMaterial = originalPage.rendererObject as Material;
204 string blendMaterialPath = GetBlendModeMaterialPath(originalPage, materialSuffix);
205
206 newReplacement.pageName = originalPage.name;
207 if (File.Exists(blendMaterialPath)) {
208 newReplacement.material = AssetDatabase.LoadAssetAtPath<Material>(blendMaterialPath);
209 } else {
210 Material blendModeMaterial = new Material(materialTemplate) {
211 name = originalMaterial.name + " " + materialTemplate.name,
212 mainTexture = originalMaterial.mainTexture
213 };
214 newReplacement.material = blendModeMaterial;
215
216 AssetDatabase.CreateAsset(blendModeMaterial, blendMaterialPath);
217 EditorUtility.SetDirty(blendModeMaterial);
218 createdNewMaterial = true;
219 }
220
221 if (newReplacement.material)
222 return newReplacement;
223 else
224 return null;
225 }
