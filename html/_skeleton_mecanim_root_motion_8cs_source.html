49 public class SkeletonMecanimRootMotion : SkeletonRootMotionBase {

59 public SkeletonMecanim SkeletonMecanim {
60 get {
61 return skeletonMecanim ? skeletonMecanim : skeletonMecanim = GetComponent<SkeletonMecanim>();

65 public override Vector2 GetRemainingRootMotion (int layerIndex) {
66 KeyValuePair<Animation, float> pair = skeletonMecanim.Translator.GetActiveAnimationAndTime(layerIndex);
67 Animation animation = pair.Key;
68 float time = pair.Value;
69 if (animation == null)
70 return Vector2.zero;
71
72 float start = time;
73 float end = animation.Duration;
74 return GetAnimationRootMotion(start, end, animation);
75 }
76
77 public override RootMotionInfo GetRootMotionInfo (int layerIndex) {
78 KeyValuePair<Animation, float> pair = skeletonMecanim.Translator.GetActiveAnimationAndTime(layerIndex);
79 Animation animation = pair.Key;
80 float time = pair.Value;
81 if (animation == null)
82 return new RootMotionInfo();
83 return GetAnimationRootMotionInfo(animation, time);
84 }
85
86 protected override void Reset () {
87 base.Reset();
88 mecanimLayerFlags = DefaultMecanimLayerFlags;
89 }
90
91 protected override void Start () {
92 base.Start();
93 skeletonMecanim = GetComponent<SkeletonMecanim>();
94 if (skeletonMecanim) {
95 skeletonMecanim.Translator.OnClipApplied -= OnClipApplied;
96 skeletonMecanim.Translator.OnClipApplied += OnClipApplied;
97 }
98 }
99
100 void OnClipApplied (Spine.Animation animation, int layerIndex, float weight,
101 float time, float lastTime, bool playsBackward) {
102
103 if (((mecanimLayerFlags & 1 << layerIndex) == 0) || weight == 0)
104 return;
105
106 if (!playsBackward) {
107 movementDelta += weight * GetAnimationRootMotion(lastTime, time, animation);
108 } else {
109 movementDelta -= weight * GetAnimationRootMotion(time, lastTime, animation);
110 }
111 if (transformRotation) {
112 if (!playsBackward) {
113 rotationDelta += weight * GetAnimationRootMotionRotation(lastTime, time, animation);
114 } else {
115 rotationDelta -= weight * GetAnimationRootMotionRotation(time, lastTime, animation);
116 }
117 }
118 }
119
120 protected override Vector2 CalculateAnimationsMovementDelta () {
121 // Note: movement delta is not gathered after animation but
122 // in OnClipApplied after every applied animation.
123 Vector2 result = movementDelta;
124 movementDelta = Vector2.zero;
125 return result;
126 }
127
128 protected override float CalculateAnimationsRotationDelta () {
129 // Note: movement delta is not gathered after animation but
130 // in OnClipApplied after every applied animation.
131 float result = rotationDelta;
132 rotationDelta = 0;
133 return result;
134 }
135 }
