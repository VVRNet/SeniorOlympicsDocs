49 public static PolygonCollider2D AddBoundingBoxGameObject (Skeleton skeleton, string skinName, string slotName, string attachmentName, Transform parent, bool isTrigger = true) {
50 Skin skin = string.IsNullOrEmpty(skinName) ? skeleton.Data.DefaultSkin : skeleton.Data.FindSkin(skinName);
51 if (skin == null) {
52 Debug.LogError("Skin " + skinName + " not found!");
53 return null;
54 }
55
56 Slot slot = skeleton.FindSlot(slotName);
57 Attachment attachment = slot != null ? skin.GetAttachment(slot.Data.Index, attachmentName) : null;
58 if (attachment == null) {
59 Debug.LogFormat("Attachment in slot '{0}' named '{1}' not found in skin '{2}'.", slotName, attachmentName, skin.Name);
60 return null;
61 }
62
63 BoundingBoxAttachment box = attachment as BoundingBoxAttachment;
64 if (box != null) {
65 return AddBoundingBoxGameObject(box.Name, box, slot, parent, isTrigger);
66 } else {
67 Debug.LogFormat("Attachment '{0}' was not a Bounding Box.", attachmentName);
68 return null;
69 }

94 public static void SetColliderPointsLocal (PolygonCollider2D collider, Slot slot, BoundingBoxAttachment box, float scale = 1.0f) {
95 if (box == null) return;
96 if (box.IsWeighted()) Debug.LogWarning("UnityEngine.PolygonCollider2D does not support weighted or animated points. Collider points will not be animated and may have incorrect orientation. If you want to use it as a collider, please remove weights and animations from the bounding box in Spine editor.");
97 Vector2[] verts = box.GetLocalVertices(slot, null);
98 if (scale != 1.0f) {
99 for (int i = 0, n = verts.Length; i < n; ++i)
100 verts[i] *= scale;
101 }
102 collider.SetPath(0, verts);

105 public static Bounds GetBoundingBoxBounds (BoundingBoxAttachment boundingBox, float depth = 0) {
106 float[] floats = boundingBox.Vertices;
107 int floatCount = floats.Length;
108
109 Bounds bounds = new Bounds();
110
111 bounds.center = new Vector3(floats[0], floats[1], 0);
112 for (int i = 2; i < floatCount; i += 2)
113 bounds.Encapsulate(new Vector3(floats[i], floats[i + 1], 0));
114
115 Vector3 size = bounds.size;
116 size.z = depth;
117 bounds.size = size;
118
119 return bounds;
120 }

299 public void CollectBones () {
300 Skeleton skeleton = skeletonComponent.Skeleton;
301 if (skeleton == null) return;
302
303 if (boneRoot != null) {
304 List<object> constraintTargets = new List<System.Object>();
305 ExposedList<IkConstraint> ikConstraints = skeleton.IkConstraints;
306 for (int i = 0, n = ikConstraints.Count; i < n; i++)
307 constraintTargets.Add(ikConstraints.Items[i].Target);
308
309 ExposedList<TransformConstraint> transformConstraints = skeleton.TransformConstraints;
310 for (int i = 0, n = transformConstraints.Count; i < n; i++)
311 constraintTargets.Add(transformConstraints.Items[i].Target);
312
313 List<SkeletonUtilityBone> boneComponents = this.boneComponents;
314 for (int i = 0, n = boneComponents.Count; i < n; i++) {
315 SkeletonUtilityBone b = boneComponents[i];
316 if (b.bone == null) {
317 b.DoUpdate(SkeletonUtilityBone.UpdatePhase.Local);
318 if (b.bone == null) continue;
319 }
320 hasOverrideBones |= (b.mode == SkeletonUtilityBone.Mode.Override);
321 hasConstraints |= constraintTargets.Contains(b.bone);
322 }
323
324 hasConstraints |= constraintComponents.Count > 0;
325
326 if (skeletonAnimation != null) {
327 skeletonAnimation.UpdateWorld -= UpdateWorld;
328 skeletonAnimation.UpdateComplete -= UpdateComplete;
329
330 if (hasOverrideBones || hasConstraints)
331 skeletonAnimation.UpdateWorld += UpdateWorld;
332
333 if (hasConstraints)
334 skeletonAnimation.UpdateComplete += UpdateComplete;
335 }
336
337 needToReprocessBones = false;
338 } else {
339 boneComponents.Clear();
340 constraintComponents.Clear();
341 }
342 }

344 void UpdateLocal (ISkeletonAnimation anim) {
345 if (needToReprocessBones)
346 CollectBones();
347
348 List<SkeletonUtilityBone> boneComponents = this.boneComponents;
349 if (boneComponents == null) return;
350 for (int i = 0, n = boneComponents.Count; i < n; i++)
351 boneComponents[i].transformLerpComplete = false;
352
353 UpdateAllBones(SkeletonUtilityBone.UpdatePhase.Local);
354 }
355
356 void UpdateWorld (ISkeletonAnimation anim) {
357 UpdateAllBones(SkeletonUtilityBone.UpdatePhase.World);
358 for (int i = 0, n = constraintComponents.Count; i < n; i++)
359 constraintComponents[i].DoUpdate();
360 }
361
362 void UpdateComplete (ISkeletonAnimation anim) {
363 UpdateAllBones(SkeletonUtilityBone.UpdatePhase.Complete);
364 }

371 if (boneComponents == null) return;
372 for (int i = 0, n = boneComponents.Count; i < n; i++)
373 boneComponents[i].DoUpdate(phase);
374 }

414 public GameObject SpawnBoneRecursively (Bone bone, Transform parent, SkeletonUtilityBone.Mode mode, bool pos, bool rot, bool sca) {
415 GameObject go = SpawnBone(bone, parent, mode, pos, rot, sca);
416
417 ExposedList<Bone> childrenBones = bone.Children;
418 for (int i = 0, n = childrenBones.Count; i < n; i++) {

426 public GameObject SpawnBone (Bone bone, Transform parent, SkeletonUtilityBone.Mode mode, bool pos, bool rot, bool sca) {
427 GameObject go = new GameObject(bone.Data.Name);
428#if UNITY_EDITOR
429 if (!Application.isPlaying)
430 UnityEditor.Undo.RegisterCreatedObjectUndo(go, "Spawn Bone");
431#endif
432 if (skeletonGraphic != null)
433 go.AddComponent<RectTransform>();
434
435 Transform goTransform = go.transform;
436 goTransform.SetParent(parent);
437
438 SkeletonUtilityBone b = go.AddComponent<SkeletonUtilityBone>();
439 b.hierarchy = this;
440 b.position = pos;
441 b.rotation = rot;
442 b.scale = sca;
443 b.mode = mode;
444 b.zPosition = true;
445 b.Reset();
446 b.bone = bone;
447 b.boneName = bone.Data.Name;
448 b.valid = true;
449
450 if (mode == SkeletonUtilityBone.Mode.Override) {
451 if (rot) goTransform.localRotation = Quaternion.Euler(0, 0, b.bone.AppliedRotation);
452 if (pos) goTransform.localPosition = new Vector3(b.bone.X * positionScale, b.bone.Y * positionScale, 0);
453 goTransform.localScale = new Vector3(b.bone.ScaleX, b.bone.ScaleY, 0);
454 }
455
456 return go;
457 }
