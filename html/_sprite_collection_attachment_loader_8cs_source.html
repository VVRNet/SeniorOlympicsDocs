52 private AtlasRegion ProcessSpriteDefinition (String name) {
53 // Strip folder names.
54 int index = name.LastIndexOfAny(new char[] { '/', '\\' });
55 if (index != -1)
56 name = name.Substring(index + 1);
57
58 tk2dSpriteDefinition def = sprites.inst.GetSpriteDefinition(name);
59
60 if (def == null) {
61 Debug.Log("Sprite not found in atlas: " + name, sprites);
62 throw new Exception("Sprite not found in atlas: " + name);
63 }
64 if (def.complexGeometry)
65 throw new NotImplementedException("Complex geometry is not supported: " + name);
66 if (def.flipped == tk2dSpriteDefinition.FlipMode.TPackerCW)
67 throw new NotImplementedException("Only 2D Toolkit atlases are supported: " + name);
68
69 Vector2 minTexCoords = Vector2.one, maxTexCoords = Vector2.zero;
70 for (int i = 0; i < def.uvs.Length; ++i) {
71 Vector2 uv = def.uvs[i];
72 minTexCoords = Vector2.Min(minTexCoords, uv);
73 maxTexCoords = Vector2.Max(maxTexCoords, uv);
74 }
75 regionRotated = def.flipped == tk2dSpriteDefinition.FlipMode.Tk2d;
76 if (regionRotated) {
77 float temp = minTexCoords.x;
78 minTexCoords.x = maxTexCoords.x;
79 maxTexCoords.x = temp;
80 }
81 u = minTexCoords.x;
82 v = maxTexCoords.y;
83 u2 = maxTexCoords.x;
84 v2 = minTexCoords.y;
85
86 regionOriginalWidth = (int)(def.untrimmedBoundsData[1].x / def.texelSize.x);
87 regionOriginalHeight = (int)(def.untrimmedBoundsData[1].y / def.texelSize.y);
88
89 regionWidth = (int)(def.boundsData[1].x / def.texelSize.x);
90 regionHeight = (int)(def.boundsData[1].y / def.texelSize.y);
91 if (regionRotated) {
92 float tempSwap = regionWidth;
93 regionWidth = regionHeight;
94 regionHeight = tempSwap;
95 }
96
97 float x0 = def.untrimmedBoundsData[0].x - def.untrimmedBoundsData[1].x / 2;
98 float x1 = def.boundsData[0].x - def.boundsData[1].x / 2;
99 regionOffsetX = (int)((x1 - x0) / def.texelSize.x);
100
101 float y0 = def.untrimmedBoundsData[0].y - def.untrimmedBoundsData[1].y / 2;
102 float y1 = def.boundsData[0].y - def.boundsData[1].y / 2;
103 regionOffsetY = (int)((y1 - y0) / def.texelSize.y);
104
105 material = def.materialInst;
106
107 AtlasRegion region = new AtlasRegion();
108 region.name = name;
109 AtlasPage page = new AtlasPage();
110 page.rendererObject = material;
111 region.page = page;
112 region.u = u;
113 region.v = v;
114 region.u2 = u2;
115 region.v2 = v2;
116 region.rotate = regionRotated;
117 region.degrees = regionRotated ? 90 : 0;
118 region.originalWidth = (int)regionOriginalWidth;
119 region.originalHeight = (int)regionOriginalHeight;
120 region.width = (int)regionWidth;
121 region.height = (int)regionHeight;
122 region.offsetX = regionOffsetX;
123 region.offsetY = regionOffsetY;
124 return region;
125 }

127 private void LoadSequence (string name, string basePath, Sequence sequence) {
128 TextureRegion[] regions = sequence.Regions;
129 for (int i = 0, n = regions.Length; i < n; i++) {
130 string path = sequence.GetPath(basePath, i);
131 regions[i] = ProcessSpriteDefinition(path);
132 if (regions[i] == null) throw new ArgumentException(string.Format("Region not found in atlas: {0} (region attachment: {1})", path, name));
133 }
134 }

136 public RegionAttachment NewRegionAttachment (Skin skin, String name, String path, Sequence sequence) {
137 RegionAttachment attachment = new RegionAttachment(name);
138 if (sequence != null)
139 LoadSequence(name, path, sequence);
140 else {
141 AtlasRegion region = ProcessSpriteDefinition(path);
142 if (region == null)
143 throw new ArgumentException(string.Format("Region not found in atlas: {0} (region attachment: {1})", path, name));
144 attachment.Region = region;
145 attachment.Path = path;
146 }
147 return attachment;
148 }
