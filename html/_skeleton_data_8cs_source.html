118 public BoneData FindBone (string boneName) {
119 if (boneName == null) throw new ArgumentNullException("boneName", "boneName cannot be null.");
120 BoneData[] bones = this.bones.Items;
121 for (int i = 0, n = this.bones.Count; i < n; i++) {
122 BoneData bone = bones[i];
123 if (bone.name == boneName) return bone;
124 }
125 return null;
126 }
