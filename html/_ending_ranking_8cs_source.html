482 IEnumerator GetTexture(string _url, Action<Sprite> callback)
483 {
484 yield return new WaitUntil(() => _url != null);
485 Debug.Log("Requesting URL: " + _url);
486 UnityWebRequest www = UnityWebRequestTexture.GetTexture(_url);
487 yield return www.SendWebRequest();
488
489 Debug.Log("Request completed with result: " + www.result.ToString());
490 if (www.result != UnityWebRequest.Result.Success)
491 {
492 Debug.LogError("Error: " + www.error);
493 callback?.Invoke(null);
494 }
495 else
496 {
497 Texture2D myTexture = ((DownloadHandlerTexture)www.downloadHandler).texture;
498 Sprite mySprite = TextureToSprite(myTexture);
499 callback?.Invoke(mySprite);
500 }
501 }

508 public string GetNumberSuffix(int number)
509 {
510 if (number % 100 >= 11 && number % 100 <= 13)
511 {
512 return "th";
513 }
514
515 switch (number % 10)
516 {
517 case 1:
518 return "<size=11>st</size>";
519 case 2:
520 return "<size=11>nd</size>";
521 case 3:
522 return "<size=11>rd</size>";
523 default:
524 return "<size=11>th</size>";
525 }
