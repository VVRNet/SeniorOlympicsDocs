34namespace Spine.Unity.Examples {
35 public class SpineboyFacialExpression : MonoBehaviour {
36
37 public SpineboyFootplanter footPlanter;
38
39 [SpineSlot]
40 public string eyeSlotName, mouthSlotName;
41
42 [SpineAttachment(slotField: "eyeSlotName")]
43 public string shockEyeName, normalEyeName;
44
45 [SpineAttachment(slotField: "mouthSlotName")]
46 public string shockMouthName, normalMouthName;
47
48 public Slot eyeSlot, mouthSlot;
49 public Attachment shockEye, normalEye, shockMouth, normalMouth;
50
51 public float balanceThreshold = 2.5f;
52 public float shockDuration = 1f;
53
54 [Header("Debug")]
55 public float shockTimer = 0f;
56
57 void Start () {
58 SkeletonAnimation skeletonAnimation = GetComponent<SkeletonAnimation>();
59 Skeleton skeleton = skeletonAnimation.Skeleton;
60 eyeSlot = skeleton.FindSlot(eyeSlotName);
61 mouthSlot = skeleton.FindSlot(mouthSlotName);
62
63 int eyeSlotIndex = skeleton.Data.FindSlot(eyeSlotName).Index;
64 shockEye = skeleton.GetAttachment(eyeSlotIndex, shockEyeName);
65 normalEye = skeleton.GetAttachment(eyeSlotIndex, normalEyeName);
66
67 int mouthSlotIndex = skeleton.Data.FindSlot(mouthSlotName).Index;
68 shockMouth = skeleton.GetAttachment(mouthSlotIndex, shockMouthName);
69 normalMouth = skeleton.GetAttachment(mouthSlotIndex, normalMouthName);
70 }
71
72 void Update () {
73 if (Mathf.Abs(footPlanter.Balance) > balanceThreshold)
74 shockTimer = shockDuration;
75
76 if (shockTimer > 0)
77 shockTimer -= Time.deltaTime;
78
79 if (shockTimer > 0) {
80 eyeSlot.Attachment = shockEye;
81 mouthSlot.Attachment = shockMouth;
82 } else {
83 eyeSlot.Attachment = normalEye;
84 mouthSlot.Attachment = normalMouth;
85 }
86 }
87 }
