30using System.Collections;

32using UnityEngine;

34namespace Spine.Unity.Examples {
35 public class SpawnSkeletonGraphicExample : MonoBehaviour {
36
37 public SkeletonDataAsset skeletonDataAsset;
38
39 [SpineAnimation(dataField: "skeletonDataAsset")]
40 public string startingAnimation;
41
42 [SpineSkin(dataField: "skeletonDataAsset")]
43 public string startingSkin = "base";
44 public Material skeletonGraphicMaterial;
45
46 IEnumerator Start () {
47 if (skeletonDataAsset == null) yield break;
48 skeletonDataAsset.GetSkeletonData(false); // Preload SkeletonDataAsset.
49 yield return new WaitForSeconds(1f); // Pretend stuff is happening.
50
51 SkeletonGraphic sg = SkeletonGraphic.NewSkeletonGraphicGameObject(skeletonDataAsset, this.transform, skeletonGraphicMaterial); // Spawn a new SkeletonGraphic GameObject.
52 sg.gameObject.name = "SkeletonGraphic Instance";
53
54 // Extra Stuff
55 sg.Initialize(false);
56 sg.Skeleton.SetSkin(startingSkin);
57 sg.Skeleton.SetSlotsToSetupPose();
58 sg.AnimationState.SetAnimation(0, startingAnimation, true);
59 }
60 }
