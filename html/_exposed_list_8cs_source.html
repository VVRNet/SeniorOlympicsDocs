91 public ExposedList<T> Resize (int newSize) {
92 int itemsLength = Items.Length;
93 T[] oldItems = Items;
94 if (newSize > itemsLength) {
95 Array.Resize(ref Items, newSize);
96 } else if (newSize < itemsLength) {
97 // Allow nulling of T reference type to allow GC.
98 for (int i = newSize; i < itemsLength; i++)
99 oldItems[i] = default(T);
100 }
101 Count = newSize;
102 return this;
103 }

441 public int RemoveAll (Predicate<T> match) {
442 CheckMatch(match);
443 int i = 0;
444 int j = 0;
445
446 // Find the first item to remove
447 for (i = 0; i < Count; i++)
448 if (match(Items[i]))
449 break;
450
451 if (i == Count)
452 return 0;
453
454 version++;
455
456 // Remove any additional items
457 for (j = i + 1; j < Count; j++) {
458 if (!match(Items[j]))
459 Items[i++] = Items[j];
460 }
461 if (j - i > 0)
462 Array.Clear(Items, i, j - i);
463
464 Count = i;
465 return (j - i);
466 }

577 public struct Enumerator : IEnumerator<T>, IDisposable {
578 private ExposedList<T> l;
579 private int next;
580 private int ver;
581 private T current;
582
583 internal Enumerator (ExposedList<T> l)
584 : this() {
585 this.l = l;
586 ver = l.version;
587 }
588
589 public void Dispose () {
590 l = null;
591 }
592
593 private void VerifyState () {
594 if (l == null)
595 throw new ObjectDisposedException(GetType().FullName);
596 if (ver != l.version)
597 throw new InvalidOperationException(
598 "Collection was modified; enumeration operation may not execute.");
599 }
600
601 public bool MoveNext () {
602 VerifyState();
603
604 if (next < 0)
605 return false;
606
607 if (next < l.Count) {
608 current = l.Items[next++];
609 return true;
610 }
611
612 next = -1;
613 return false;
614 }
615
616 public T Current {
617 get {
618 return current;
619 }
620 }

622 void IEnumerator.Reset () {
623 VerifyState();
624 next = 0;
625 }

628 get {

630 if (next <= 0)
631 throw new InvalidOperationException();
632 return current;
633 }
