178 public static void GetWorldToLocalMatrix (this Bone bone, out float ia, out float ib, out float ic, out float id) {
179 float a = bone.A, b = bone.B, c = bone.C, d = bone.D;
180 float invDet = 1 / (a * d - b * c);
181 ia = invDet * d;
182 ib = invDet * -b;
183 ic = invDet * -c;
184 id = invDet * a;

230 public static Vector2[] GetLocalVertices (this VertexAttachment va, Slot slot, Vector2[] buffer) {
231 int floatsCount = va.WorldVerticesLength;
232 int bufferTargetSize = floatsCount >> 1;
233 buffer = buffer ?? new Vector2[bufferTargetSize];
234 if (buffer.Length < bufferTargetSize) throw new System.ArgumentException(string.Format("Vector2 buffer too small. {0} requires an array of size {1}. Use the attachment's .WorldVerticesLength to get the correct size.", va.Name, floatsCount), "buffer");
235
236 if (va.Bones == null && slot.Deform.Count == 0) {
237 float[] localVerts = va.Vertices;
238 for (int i = 0; i < bufferTargetSize; i++) {
239 int j = i * 2;
240 buffer[i] = new Vector2(localVerts[j], localVerts[j + 1]);
241 }
242 } else {
243 float[] floats = new float[floatsCount];
244 va.ComputeWorldVertices(slot, floats);
245
246 Bone sb = slot.Bone;
247 float ia, ib, ic, id, bwx = sb.WorldX, bwy = sb.WorldY;
248 sb.GetWorldToLocalMatrix(out ia, out ib, out ic, out id);
249
250 for (int i = 0; i < bufferTargetSize; i++) {
251 int j = i * 2;
252 float x = floats[j] - bwx, y = floats[j + 1] - bwy;
253 buffer[i] = new Vector2(x * ia + y * ib, x * ic + y * id);
254 }
255 }
256
257 return buffer;
258 }

307 public static BoneMatrix CalculateSetupWorld (BoneData boneData) {
308 if (boneData == null)
309 return default(BoneMatrix);
310
311 // End condition: isRootBone
312 if (boneData.Parent == null)
313 return GetInheritedInternal(boneData, default(BoneMatrix));
314
315 BoneMatrix result = CalculateSetupWorld(boneData.Parent);
316 return GetInheritedInternal(boneData, result);
317 }

319 static BoneMatrix GetInheritedInternal (BoneData boneData, BoneMatrix parentMatrix) {
320 BoneData parent = boneData.Parent;
321 if (parent == null) return new BoneMatrix(boneData); // isRootBone
322
323 float pa = parentMatrix.a, pb = parentMatrix.b, pc = parentMatrix.c, pd = parentMatrix.d;
324 BoneMatrix result = default(BoneMatrix);
325 result.x = pa * boneData.X + pb * boneData.Y + parentMatrix.x;
326 result.y = pc * boneData.X + pd * boneData.Y + parentMatrix.y;
327
328 switch (boneData.Inherit) {
329 case Inherit.Normal: {
330 float rotationY = boneData.Rotation + 90 + boneData.ShearY;
331 float la = MathUtils.CosDeg(boneData.Rotation + boneData.ShearX) * boneData.ScaleX;
332 float lb = MathUtils.CosDeg(rotationY) * boneData.ScaleY;
333 float lc = MathUtils.SinDeg(boneData.Rotation + boneData.ShearX) * boneData.ScaleX;
334 float ld = MathUtils.SinDeg(rotationY) * boneData.ScaleY;
335 result.a = pa * la + pb * lc;
336 result.b = pa * lb + pb * ld;
337 result.c = pc * la + pd * lc;
338 result.d = pc * lb + pd * ld;
339 break;
340 }
341 case Inherit.OnlyTranslation: {
342 float rotationY = boneData.Rotation + 90 + boneData.ShearY;
343 result.a = MathUtils.CosDeg(boneData.Rotation + boneData.ShearX) * boneData.ScaleX;
344 result.b = MathUtils.CosDeg(rotationY) * boneData.ScaleY;
345 result.c = MathUtils.SinDeg(boneData.Rotation + boneData.ShearX) * boneData.ScaleX;
346 result.d = MathUtils.SinDeg(rotationY) * boneData.ScaleY;
347 break;
348 }
349 case Inherit.NoRotationOrReflection: {
350 float s = pa * pa + pc * pc, prx;
351 if (s > 0.0001f) {
352 s = Math.Abs(pa * pd - pb * pc) / s;
353 pb = pc * s;
354 pd = pa * s;
355 prx = MathUtils.Atan2(pc, pa) * MathUtils.RadDeg;
356 } else {
357 pa = 0;
358 pc = 0;
359 prx = 90 - MathUtils.Atan2(pd, pb) * MathUtils.RadDeg;
360 }
361 float rx = boneData.Rotation + boneData.ShearX - prx;
362 float ry = boneData.Rotation + boneData.ShearY - prx + 90;
363 float la = MathUtils.CosDeg(rx) * boneData.ScaleX;
364 float lb = MathUtils.CosDeg(ry) * boneData.ScaleY;
365 float lc = MathUtils.SinDeg(rx) * boneData.ScaleX;
366 float ld = MathUtils.SinDeg(ry) * boneData.ScaleY;
367 result.a = pa * la - pb * lc;
368 result.b = pa * lb - pb * ld;
369 result.c = pc * la + pd * lc;
370 result.d = pc * lb + pd * ld;
371 break;
372 }
373 case Inherit.NoScale:
374 case Inherit.NoScaleOrReflection: {
375 float cos = MathUtils.CosDeg(boneData.Rotation), sin = MathUtils.SinDeg(boneData.Rotation);
376 float za = pa * cos + pb * sin;
377 float zc = pc * cos + pd * sin;
378 float s = (float)Math.Sqrt(za * za + zc * zc);
379 if (s > 0.00001f)
380 s = 1 / s;
381 za *= s;
382 zc *= s;
383 s = (float)Math.Sqrt(za * za + zc * zc);
384 float r = MathUtils.PI / 2 + MathUtils.Atan2(zc, za);
385 float zb = MathUtils.Cos(r) * s;
386 float zd = MathUtils.Sin(r) * s;
387 float la = MathUtils.CosDeg(boneData.ShearX) * boneData.ScaleX;
388 float lb = MathUtils.CosDeg(90 + boneData.ShearY) * boneData.ScaleY;
389 float lc = MathUtils.SinDeg(boneData.ShearX) * boneData.ScaleX;
390 float ld = MathUtils.SinDeg(90 + boneData.ShearY) * boneData.ScaleY;
391 if (boneData.Inherit != Inherit.NoScaleOrReflection ? pa * pd - pb * pc < 0 : false) {
392 zb = -zb;
393 zd = -zd;
394 }
395 result.a = za * la + zb * lc;
396 result.b = za * lb + zb * ld;
397 result.c = zc * la + zd * lc;
398 result.d = zc * lb + zd * ld;
399 break;
400 }
401 }
402
403 return result;
404 }

432 public BoneMatrix TransformMatrix (BoneMatrix local) {
433 return new BoneMatrix {
434 a = this.a * local.a + this.b * local.c,
435 b = this.a * local.b + this.b * local.d,
436 c = this.c * local.a + this.d * local.c,
437 d = this.c * local.b + this.d * local.d,
438 x = this.a * local.x + this.b * local.y + this.x,
439 y = this.c * local.x + this.d * local.y + this.y
440 };
