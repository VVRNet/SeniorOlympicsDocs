30using System.Collections.Generic;
31using UnityEngine;
32
33namespace Spine.Unity.Examples {
34
35 // SkeletonUtilityKinematicShadow allows hinge chains to inherit a velocity interpreted from changes in parent transform position or from unrelated rigidbodies.
36 // Note: Uncheck "useRootTransformIfNull
37 public class SkeletonUtilityKinematicShadow : MonoBehaviour {
38 #region Inspector
39 [Tooltip("If checked, the hinge chain can inherit your root transform's velocity or position/rotation changes.")]
40 public bool detachedShadow = false;
41 public Transform parent;
42 public bool hideShadow = true;
43 public PhysicsSystem physicsSystem = PhysicsSystem.Physics3D;
44 #endregion
45
46 GameObject shadowRoot;
47 readonly List<TransformPair> shadowTable = new List<TransformPair>();
48 struct TransformPair {
49 public Transform dest, src;
50 }
51
52 public enum PhysicsSystem {
53 Physics2D,
54 Physics3D
55 };
56
57 void Start () {
58 // Duplicate this gameObject as the "shadow" with a different parent.
59 shadowRoot = Instantiate<GameObject>(this.gameObject);
60 Destroy(shadowRoot.GetComponent<SkeletonUtilityKinematicShadow>());
61
62 // Prepare shadow gameObject's properties.
63 Transform shadowRootTransform = shadowRoot.transform;
64 shadowRootTransform.position = transform.position;
65 shadowRootTransform.rotation = transform.rotation;
66
67 Vector3 scaleRef = transform.TransformPoint(Vector3.right);
68 float scale = Vector3.Distance(transform.position, scaleRef);
69 shadowRootTransform.localScale = Vector3.one;
70
71 if (!detachedShadow) {
72 // Do not change to null coalescing operator (??). Unity overloads null checks for UnityEngine.Objects but not the ?? operator.
73 if (parent == null)
74 shadowRootTransform.parent = transform.root;
75 else
76 shadowRootTransform.parent = parent;
77 }
78
79 if (hideShadow)
80 shadowRoot.hideFlags = HideFlags.HideInHierarchy;
81
82 Joint[] shadowJoints = shadowRoot.GetComponentsInChildren<Joint>();
83 foreach (Joint j in shadowJoints)
84 j.connectedAnchor *= scale;
85
86 // Build list of bone pairs (matches shadow transforms with bone transforms)
87 SkeletonUtilityBone[] bones = GetComponentsInChildren<SkeletonUtilityBone>();
88 SkeletonUtilityBone[] shadowBones = shadowRoot.GetComponentsInChildren<SkeletonUtilityBone>();
89 foreach (SkeletonUtilityBone b in bones) {
90 if (b.gameObject == this.gameObject)
91 continue;
92
93 System.Type checkType = (physicsSystem == PhysicsSystem.Physics2D) ? typeof(Rigidbody2D) : typeof(Rigidbody);
94 foreach (SkeletonUtilityBone sb in shadowBones) {
95 if (sb.GetComponent(checkType) != null && sb.boneName == b.boneName) {
96 shadowTable.Add(new TransformPair {
97 dest = b.transform,
98 src = sb.transform
99 });
100 break;
101 }
102 }
103
104 }
105
106 // Destroy conflicting and unneeded components
107 DestroyComponents(shadowBones);
108
109 DestroyComponents(GetComponentsInChildren<Joint>());
110 DestroyComponents(GetComponentsInChildren<Rigidbody>());
111 DestroyComponents(GetComponentsInChildren<Collider>());
112 }
113
114 static void DestroyComponents (Component[] components) {
115 for (int i = 0, n = components.Length; i < n; i++)
116 Destroy(components[i]);
117 }
118
119 void FixedUpdate () {
120 if (physicsSystem == PhysicsSystem.Physics2D) {
121 Rigidbody2D shadowRootRigidbody = shadowRoot.GetComponent<Rigidbody2D>();
122 shadowRootRigidbody.MovePosition(transform.position);
123 shadowRootRigidbody.MoveRotation(transform.rotation.eulerAngles.z);
124 } else {
125 Rigidbody shadowRootRigidbody = shadowRoot.GetComponent<Rigidbody>();
126 shadowRootRigidbody.MovePosition(transform.position);
127 shadowRootRigidbody.MoveRotation(transform.rotation);
128 }
129
130 for (int i = 0, n = shadowTable.Count; i < n; i++) {
131 TransformPair pair = shadowTable[i];
132 pair.dest.localPosition = pair.src.localPosition;
133 pair.dest.localRotation = pair.src.localRotation;
134 }
135 }
136 }
137}
