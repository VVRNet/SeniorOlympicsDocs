56 public void SetTimelines (ExposedList<Timeline> timelines) {
57 if (timelines == null) throw new ArgumentNullException("timelines", "timelines cannot be null.");
58 this.timelines = timelines;
59 // Note: avoiding reallocations by adding all hash set entries at
60 // once (EnsureCapacity() is only available in newer .Net versions).
61 int idCount = 0;
62 int timelinesCount = timelines.Count;
63 Timeline[] timelinesItems = timelines.Items;
64 for (int t = 0; t < timelinesCount; ++t)
65 idCount += timelinesItems[t].PropertyIds.Length;
66 string[] propertyIds = new string[idCount];
67 int currentId = 0;
68 for (int t = 0; t < timelinesCount; ++t) {
69 string[] ids = timelinesItems[t].PropertyIds;
70 for (int i = 0, idsLength = ids.Length; i < idsLength; ++i)
71 propertyIds[currentId++] = ids[i];
72 }
73 this.timelineIds = new HashSet<string>(propertyIds);
74 }

344 public void SetBezier (int bezier, int frame, int value, float time1, float value1, float cx1, float cy1, float cx2,
345 float cy2, float time2, float value2) {
346
347 float[] curves = this.curves;
348 int i = FrameCount + bezier * BEZIER_SIZE;
349 if (value == 0) curves[frame] = BEZIER + i;
350 float tmpx = (time1 - cx1 * 2 + cx2) * 0.03f, tmpy = (value1 - cy1 * 2 + cy2) * 0.03f;
351 float dddx = ((cx1 - cx2) * 3 - time1 + time2) * 0.006f, dddy = ((cy1 - cy2) * 3 - value1 + value2) * 0.006f;
352 float ddx = tmpx * 2 + dddx, ddy = tmpy * 2 + dddy;
353 float dx = (cx1 - time1) * 0.3f + tmpx + dddx * 0.16666667f, dy = (cy1 - value1) * 0.3f + tmpy + dddy * 0.16666667f;
354 float x = time1 + dx, y = value1 + dy;
355 for (int n = i + BEZIER_SIZE; i < n; i += 2) {
356 curves[i] = x;
357 curves[i + 1] = y;
358 dx += ddx;
359 dy += ddy;
360 ddx += dddx;
361 ddy += dddy;
362 x += dx;
363 y += dy;
364 }
365 }

372 public float GetBezierValue (float time, int frameIndex, int valueOffset, int i) {
373 float[] curves = this.curves;
374 if (curves[i] > time) {
375 float x = frames[frameIndex], y = frames[frameIndex + valueOffset];
376 return y + (time - x) / (curves[i] - x) * (curves[i + 1] - y);
377 }
378 int n = i + BEZIER_SIZE;
379 for (i += 2; i < n; i += 2) {
380 if (curves[i] >= time) {
381 float x = curves[i - 2], y = curves[i - 1];
382 return y + (time - x) / (curves[i] - x) * (curves[i + 1] - y);
383 }
384 }
385 frameIndex += FrameEntries;
386 { // scope added to prevent compile error "float x and y declared in enclosing scope"
387 float x = curves[n - 2], y = curves[n - 1];
388 return y + (time - x) / (frames[frameIndex] - x) * (frames[frameIndex + valueOffset] - y);
389 }
390 }

394 public abstract class CurveTimeline1 : CurveTimeline {
395 public const int ENTRIES = 2;
396 internal const int VALUE = 1;
397
400 public CurveTimeline1 (int frameCount, int bezierCount, string propertyId)
401 : base(frameCount, bezierCount, propertyId) {
402 }
403
404 public override int FrameEntries {
405 get { return ENTRIES; }
406 }
407
411 public void SetFrame (int frame, float time, float value) {
412 frame <<= 1;
413 frames[frame] = time;
414 frames[frame + VALUE] = value;
415 }
416
418 public float GetCurveValue (float time) {
419 float[] frames = this.frames;
420 int i = frames.Length - 2;
421 for (int ii = 2; ii <= i; ii += 2) {
422 if (frames[ii] > time) {
423 i = ii - 2;
424 break;
425 }
426 }
427
428 int curveType = (int)curves[i >> 1];
429 switch (curveType) {
430 case LINEAR:
431 float before = frames[i], value = frames[i + VALUE];
432 return value + (time - before) / (frames[i + ENTRIES] - before) * (frames[i + ENTRIES + VALUE] - value);
433 case STEPPED:
434 return frames[i + VALUE];
435 }
436 return GetBezierValue(time, i, VALUE, curveType - BEZIER);
437 }
438
439 public float GetRelativeValue (float time, float alpha, MixBlend blend, float current, float setup) {
440 if (time < frames[0]) {
441 switch (blend) {
442 case MixBlend.Setup:
443 return setup;
444 case MixBlend.First:
445 return current + (setup - current) * alpha;
446 }
447 return current;
448 }
449 float value = GetCurveValue(time);
450 switch (blend) {
451 case MixBlend.Setup:
452 return setup + value * alpha;
453 case MixBlend.First:
454 case MixBlend.Replace:
455 value += setup - current;
456 break;
457 }
458 return current + value * alpha;
459 }
460
461 public float GetAbsoluteValue (float time, float alpha, MixBlend blend, float current, float setup) {
462 if (time < frames[0]) {
463 switch (blend) {
464 case MixBlend.Setup:
465 return setup;
466 case MixBlend.First:
467 return current + (setup - current) * alpha;
468 }
469 return current;
470 }
471 float value = GetCurveValue(time);
472 if (blend == MixBlend.Setup) return setup + (value - setup) * alpha;
473 return current + (value - current) * alpha;
474 }
475
476 public float GetAbsoluteValue (float time, float alpha, MixBlend blend, float current, float setup, float value) {
477 if (time < frames[0]) {
478 switch (blend) {
479 case MixBlend.Setup:
480 return setup;
481 case MixBlend.First:
482 return current + (setup - current) * alpha;
483 }
484 return current;
485 }
486 if (blend == MixBlend.Setup) return setup + (value - setup) * alpha;
487 return current + (value - current) * alpha;
488 }
489
490 public float GetScaleValue (float time, float alpha, MixBlend blend, MixDirection direction, float current, float setup) {
491 float[] frames = this.frames;
492 if (time < frames[0]) {
493 switch (blend) {
494 case MixBlend.Setup:
495 return setup;
496 case MixBlend.First:
497 return current + (setup - current) * alpha;
498 }
499 return current;
500 }
501 float value = GetCurveValue(time) * setup;
502 if (alpha == 1) {
503 if (blend == MixBlend.Add) return current + value - setup;
504 return value;
505 }
506 // Mixing out uses sign of setup or current pose, else use sign of key.
507 if (direction == MixDirection.Out) {
508 switch (blend) {
509 case MixBlend.Setup:
510 return setup + (Math.Abs(value) * Math.Sign(setup) - setup) * alpha;
511 case MixBlend.First:
512 case MixBlend.Replace:
513 return current + (Math.Abs(value) * Math.Sign(current) - current) * alpha;
514 }
515 } else {
516 float s;
517 switch (blend) {
518 case MixBlend.Setup:
519 s = Math.Abs(setup) * Math.Sign(value);
520 return s + (value - s) * alpha;
521 case MixBlend.First:
522 case MixBlend.Replace:
523 s = Math.Abs(current) * Math.Sign(value);
524 return s + (value - s) * alpha;
525 }
526 }
527 return current + (value - setup) * alpha;
528 }
529 }

580 public class TranslateTimeline : CurveTimeline2, IBoneTimeline {
581 readonly int boneIndex;
582
583 public TranslateTimeline (int frameCount, int bezierCount, int boneIndex)
584 : base(frameCount, bezierCount, //
585 (int)Property.X + "|" + boneIndex, //
586 (int)Property.Y + "|" + boneIndex) {

588 }

590 public int BoneIndex {
591 get {
592 return boneIndex;
593 }
594 }

596 override public void Apply (Skeleton skeleton, float lastTime, float time, ExposedList<Event> firedEvents, float alpha, MixBlend blend,
597 MixDirection direction) {
598 Bone bone = skeleton.bones.Items[boneIndex];
599 if (!bone.active) return;
600
601 float[] frames = this.frames;
602 if (time < frames[0]) {
603 switch (blend) {
604 case MixBlend.Setup:
605 bone.x = bone.data.x;
606 bone.y = bone.data.y;
607 return;
608 case MixBlend.First:
609 bone.x += (bone.data.x - bone.x) * alpha;
610 bone.y += (bone.data.y - bone.y) * alpha;
611 return;
612 }
613 return;
614 }
615
616 float x, y;
617 GetCurveValue(out x, out y, time); // note: reference implementation has code inlined
618
619 switch (blend) {
620 case MixBlend.Setup:
621 bone.x = bone.data.x + x * alpha;
622 bone.y = bone.data.y + y * alpha;
623 break;
624 case MixBlend.First:
625 case MixBlend.Replace:
626 bone.x += (bone.data.x + x - bone.x) * alpha;
627 bone.y += (bone.data.y + y - bone.y) * alpha;
628 break;
629 case MixBlend.Add:
630 bone.x += x * alpha;
631 bone.y += y * alpha;
632 break;
633 }
634 }
635
636 public void GetCurveValue (out float x, out float y, float time) {
637 int i = Search(frames, time, ENTRIES), curveType = (int)curves[i / ENTRIES];
638 switch (curveType) {
639 case LINEAR:
640 float before = frames[i];
641 x = frames[i + VALUE1];
642 y = frames[i + VALUE2];
643 float t = (time - before) / (frames[i + ENTRIES] - before);
644 x += (frames[i + ENTRIES + VALUE1] - x) * t;
645 y += (frames[i + ENTRIES + VALUE2] - y) * t;
646 break;
647 case STEPPED:
648 x = frames[i + VALUE1];
649 y = frames[i + VALUE2];
650 break;
651 default:
652 x = GetBezierValue(time, i, VALUE1, curveType - BEZIER);
653 y = GetBezierValue(time, i, VALUE2, curveType + BEZIER_SIZE - BEZIER);
654 break;
655 }
656 }

704 public class ScaleTimeline : CurveTimeline2, IBoneTimeline {
705 readonly int boneIndex;
706
707 public ScaleTimeline (int frameCount, int bezierCount, int boneIndex)
708 : base(frameCount, bezierCount, //
709 (int)Property.ScaleX + "|" + boneIndex, //
710 (int)Property.ScaleY + "|" + boneIndex) {
711 this.boneIndex = boneIndex;
712 }
713
714 public int BoneIndex {
715 get {
716 return boneIndex;
717 }
718 }
719
720 override public void Apply (Skeleton skeleton, float lastTime, float time, ExposedList<Event> firedEvents, float alpha, MixBlend blend,
721 MixDirection direction) {
722 Bone bone = skeleton.bones.Items[boneIndex];
723 if (!bone.active) return;
724
725 float[] frames = this.frames;
726 if (time < frames[0]) {
727 switch (blend) {
728 case MixBlend.Setup:
729 bone.scaleX = bone.data.scaleX;
730 bone.scaleY = bone.data.scaleY;
731 return;
732 case MixBlend.First:
733 bone.scaleX += (bone.data.scaleX - bone.scaleX) * alpha;
734 bone.scaleY += (bone.data.scaleY - bone.scaleY) * alpha;
735 return;
736 }
737 return;
738 }
739
740 float x, y;
741 int i = Search(frames, time, ENTRIES), curveType = (int)curves[i / ENTRIES];
742 switch (curveType) {
743 case LINEAR:
744 float before = frames[i];
745 x = frames[i + VALUE1];
746 y = frames[i + VALUE2];
747 float t = (time - before) / (frames[i + ENTRIES] - before);
748 x += (frames[i + ENTRIES + VALUE1] - x) * t;
749 y += (frames[i + ENTRIES + VALUE2] - y) * t;
750 break;
751 case STEPPED:
752 x = frames[i + VALUE1];
753 y = frames[i + VALUE2];
754 break;
755 default:
756 x = GetBezierValue(time, i, VALUE1, curveType - BEZIER);
757 y = GetBezierValue(time, i, VALUE2, curveType + BEZIER_SIZE - BEZIER);
758 break;
759 }
760 x *= bone.data.scaleX;
761 y *= bone.data.scaleY;
762
763 if (alpha == 1) {
764 if (blend == MixBlend.Add) {
765 bone.scaleX += x - bone.data.scaleX;
766 bone.scaleY += y - bone.data.scaleY;
767 } else {
768 bone.scaleX = x;
769 bone.scaleY = y;
770 }
771 } else {
772 // Mixing out uses sign of setup or current pose, else use sign of key.
773 float bx, by;
774 if (direction == MixDirection.Out) {
775 switch (blend) {
776 case MixBlend.Setup:
777 bx = bone.data.scaleX;
778 by = bone.data.scaleY;
779 bone.scaleX = bx + (Math.Abs(x) * Math.Sign(bx) - bx) * alpha;
780 bone.scaleY = by + (Math.Abs(y) * Math.Sign(by) - by) * alpha;
781 break;
782 case MixBlend.First:
783 case MixBlend.Replace:
784 bx = bone.scaleX;
785 by = bone.scaleY;
786 bone.scaleX = bx + (Math.Abs(x) * Math.Sign(bx) - bx) * alpha;
787 bone.scaleY = by + (Math.Abs(y) * Math.Sign(by) - by) * alpha;
788 break;
789 case MixBlend.Add:
790 bone.scaleX += (x - bone.data.scaleX) * alpha;
791 bone.scaleY += (y - bone.data.scaleY) * alpha;
792 break;
793 }
794 } else {
795 switch (blend) {
796 case MixBlend.Setup:
797 bx = Math.Abs(bone.data.scaleX) * Math.Sign(x);
798 by = Math.Abs(bone.data.scaleY) * Math.Sign(y);
799 bone.scaleX = bx + (x - bx) * alpha;
800 bone.scaleY = by + (y - by) * alpha;
801 break;
802 case MixBlend.First:
803 case MixBlend.Replace:
804 bx = Math.Abs(bone.scaleX) * Math.Sign(x);
805 by = Math.Abs(bone.scaleY) * Math.Sign(y);
806 bone.scaleX = bx + (x - bx) * alpha;
807 bone.scaleY = by + (y - by) * alpha;
808 break;
809 case MixBlend.Add:
810 bone.scaleX += (x - bone.data.scaleX) * alpha;
811 bone.scaleY += (y - bone.data.scaleY) * alpha;
812 break;
813 }
814 }
815 }
