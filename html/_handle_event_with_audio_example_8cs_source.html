50 Spine.EventData eventData;
51
52 void OnValidate () {
53 if (skeletonAnimation == null) GetComponent<SkeletonAnimation>();
54 if (audioSource == null) GetComponent<AudioSource>();
55 }
56
57 void Start () {
58 if (audioSource == null) return;
59 if (skeletonAnimation == null) return;
60 skeletonAnimation.Initialize(false);
61 if (!skeletonAnimation.valid) return;
62
63 eventData = skeletonAnimation.Skeleton.Data.FindEvent(eventName);
64 skeletonAnimation.AnimationState.Event += HandleAnimationStateEvent;
65 }
66
67 private void HandleAnimationStateEvent (TrackEntry trackEntry, Event e) {
68 if (logDebugMessage) Debug.Log("Event fired! " + e.Data.Name);
69 //bool eventMatch = string.Equals(e.Data.Name, eventName, System.StringComparison.Ordinal); // Testing recommendation: String compare.
70 bool eventMatch = (eventData == e.Data); // Performance recommendation: Match cached reference instead of string.
71 if (eventMatch) {
72 Play();
73 }
74 }
75
76 public void Play () {
77 audioSource.pitch = basePitch + Random.Range(-randomPitchOffset, randomPitchOffset);
78 audioSource.clip = audioClip;
79 audioSource.Play();
80 }
81 }
