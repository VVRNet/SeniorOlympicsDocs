44 public ExposedList<int> Triangulate (ExposedList<float> verticesArray) {
45 float[] vertices = verticesArray.Items;
46 int vertexCount = verticesArray.Count >> 1;
47
48 ExposedList<int> indicesArray = this.indicesArray;
49 indicesArray.Clear();
50 int[] indices = indicesArray.Resize(vertexCount).Items;
51 for (int i = 0; i < vertexCount; i++)
52 indices[i] = i;
53
54 ExposedList<bool> isConcaveArray = this.isConcaveArray;
55 bool[] isConcave = isConcaveArray.Resize(vertexCount).Items;
56 for (int i = 0, n = vertexCount; i < n; ++i)
57 isConcave[i] = IsConcave(i, vertexCount, vertices, indices);
58
59 ExposedList<int> triangles = this.triangles;
60 triangles.Clear();
61 triangles.EnsureCapacity(Math.Max(0, vertexCount - 2) << 2);
62
63 while (vertexCount > 3) {
64 // Find ear tip.
65 int previous = vertexCount - 1, i = 0, next = 1;
66
67 // outer:
68 while (true) {
69 if (!isConcave[i]) {
70 int p1 = indices[previous] << 1, p2 = indices[i] << 1, p3 = indices[next] << 1;
71 float p1x = vertices[p1], p1y = vertices[p1 + 1];
72 float p2x = vertices[p2], p2y = vertices[p2 + 1];
73 float p3x = vertices[p3], p3y = vertices[p3 + 1];
74 for (int ii = (next + 1) % vertexCount; ii != previous; ii = (ii + 1) % vertexCount) {
75 if (!isConcave[ii]) continue;
76 int v = indices[ii] << 1;
77 float vx = vertices[v], vy = vertices[v + 1];
78 if (PositiveArea(p3x, p3y, p1x, p1y, vx, vy)) {
79 if (PositiveArea(p1x, p1y, p2x, p2y, vx, vy)) {
80 if (PositiveArea(p2x, p2y, p3x, p3y, vx, vy)) goto break_outer; // break outer;
81 }
82 }
83 }
84 break;
85 }
86 break_outer:
87
88 if (next == 0) {
89 do {
90 if (!isConcave[i]) break;
91 i--;
92 } while (i > 0);
93 break;
94 }
95
96 previous = i;
97 i = next;
98 next = (next + 1) % vertexCount;
99 }
100
101 // Cut ear tip.
102 triangles.Add(indices[(vertexCount + i - 1) % vertexCount]);
103 triangles.Add(indices[i]);
104 triangles.Add(indices[(i + 1) % vertexCount]);
105 indicesArray.RemoveAt(i);
106 isConcaveArray.RemoveAt(i);
107 vertexCount--;
108
109 int previousIndex = (vertexCount + i - 1) % vertexCount;
110 int nextIndex = i == vertexCount ? 0 : i;
111 isConcave[previousIndex] = IsConcave(previousIndex, vertexCount, vertices, indices);
112 isConcave[nextIndex] = IsConcave(nextIndex, vertexCount, vertices, indices);
113 }
114
115 if (vertexCount == 3) {
116 triangles.Add(indices[2]);
117 triangles.Add(indices[0]);
118 triangles.Add(indices[1]);
119 }
120
121 return triangles;
122 }

124 public ExposedList<ExposedList<float>> Decompose (ExposedList<float> verticesArray, ExposedList<int> triangles) {
125 float[] vertices = verticesArray.Items;
126 ExposedList<ExposedList<float>> convexPolygons = this.convexPolygons;
127 for (int i = 0, n = convexPolygons.Count; i < n; i++)
128 polygonPool.Free(convexPolygons.Items[i]);
129 convexPolygons.Clear();
130
131 ExposedList<ExposedList<int>> convexPolygonsIndices = this.convexPolygonsIndices;
132 for (int i = 0, n = convexPolygonsIndices.Count; i < n; i++)
133 polygonIndicesPool.Free(convexPolygonsIndices.Items[i]);
134 convexPolygonsIndices.Clear();
135
136 ExposedList<int> polygonIndices = polygonIndicesPool.Obtain();
137 polygonIndices.Clear();
138
139 ExposedList<float> polygon = polygonPool.Obtain();
140 polygon.Clear();
141
142 // Merge subsequent triangles if they form a triangle fan.
143 int fanBaseIndex = -1, lastWinding = 0;
144 int[] trianglesItems = triangles.Items;
145 for (int i = 0, n = triangles.Count; i < n; i += 3) {
146 int t1 = trianglesItems[i] << 1, t2 = trianglesItems[i + 1] << 1, t3 = trianglesItems[i + 2] << 1;
147 float x1 = vertices[t1], y1 = vertices[t1 + 1];
148 float x2 = vertices[t2], y2 = vertices[t2 + 1];
149 float x3 = vertices[t3], y3 = vertices[t3 + 1];
150
151 // If the base of the last triangle is the same as this triangle, check if they form a convex polygon (triangle fan).
152 bool merged = false;
153 if (fanBaseIndex == t1) {
154 int o = polygon.Count - 4;
155 float[] p = polygon.Items;
156 int winding1 = Winding(p[o], p[o + 1], p[o + 2], p[o + 3], x3, y3);
157 int winding2 = Winding(x3, y3, p[0], p[1], p[2], p[3]);
158 if (winding1 == lastWinding && winding2 == lastWinding) {
159 polygon.Add(x3);
160 polygon.Add(y3);
161 polygonIndices.Add(t3);
162 merged = true;
163 }
164 }
165
166 // Otherwise make this triangle the new base.
167 if (!merged) {
168 if (polygon.Count > 0) {
169 convexPolygons.Add(polygon);
170 convexPolygonsIndices.Add(polygonIndices);
171 } else {
172 polygonPool.Free(polygon);
173 polygonIndicesPool.Free(polygonIndices);
174 }
175 polygon = polygonPool.Obtain();
176 polygon.Clear();
177 polygon.Add(x1);
178 polygon.Add(y1);
179 polygon.Add(x2);
180 polygon.Add(y2);
181 polygon.Add(x3);
182 polygon.Add(y3);
183 polygonIndices = polygonIndicesPool.Obtain();
184 polygonIndices.Clear();
185 polygonIndices.Add(t1);
186 polygonIndices.Add(t2);
187 polygonIndices.Add(t3);
188 lastWinding = Winding(x1, y1, x2, y2, x3, y3);
189 fanBaseIndex = t1;
190 }
191 }
192
193 if (polygon.Count > 0) {
194 convexPolygons.Add(polygon);
195 convexPolygonsIndices.Add(polygonIndices);
196 }
197
198 // Go through the list of polygons and try to merge the remaining triangles with the found triangle fans.
199 for (int i = 0, n = convexPolygons.Count; i < n; i++) {
200 polygonIndices = convexPolygonsIndices.Items[i];
201 if (polygonIndices.Count == 0) continue;
202 int firstIndex = polygonIndices.Items[0];
203 int lastIndex = polygonIndices.Items[polygonIndices.Count - 1];
204
205 polygon = convexPolygons.Items[i];
206 int o = polygon.Count - 4;
207 float[] p = polygon.Items;
208 float prevPrevX = p[o], prevPrevY = p[o + 1];
209 float prevX = p[o + 2], prevY = p[o + 3];
210 float firstX = p[0], firstY = p[1];
211 float secondX = p[2], secondY = p[3];
212 int winding = Winding(prevPrevX, prevPrevY, prevX, prevY, firstX, firstY);
213
214 for (int ii = 0; ii < n; ii++) {
215 if (ii == i) continue;
216 ExposedList<int> otherIndices = convexPolygonsIndices.Items[ii];
217 if (otherIndices.Count != 3) continue;
218 int otherFirstIndex = otherIndices.Items[0];
219 int otherSecondIndex = otherIndices.Items[1];
220 int otherLastIndex = otherIndices.Items[2];
221
222 ExposedList<float> otherPoly = convexPolygons.Items[ii];
223 float x3 = otherPoly.Items[otherPoly.Count - 2], y3 = otherPoly.Items[otherPoly.Count - 1];
224
225 if (otherFirstIndex != firstIndex || otherSecondIndex != lastIndex) continue;
226 int winding1 = Winding(prevPrevX, prevPrevY, prevX, prevY, x3, y3);
227 int winding2 = Winding(x3, y3, firstX, firstY, secondX, secondY);
228 if (winding1 == winding && winding2 == winding) {
229 otherPoly.Clear();
230 otherIndices.Clear();
231 polygon.Add(x3);
232 polygon.Add(y3);
233 polygonIndices.Add(otherLastIndex);
234 prevPrevX = prevX;
235 prevPrevY = prevY;
236 prevX = x3;
237 prevY = y3;
238 ii = 0;
239 }
240 }
241 }
242
243 // Remove empty polygons that resulted from the merge step above.
244 for (int i = convexPolygons.Count - 1; i >= 0; i--) {
245 polygon = convexPolygons.Items[i];
246 if (polygon.Count == 0) {
247 convexPolygons.RemoveAt(i);
248 polygonPool.Free(polygon);
249 polygonIndices = convexPolygonsIndices.Items[i];
250 convexPolygonsIndices.RemoveAt(i);
251 polygonIndicesPool.Free(polygonIndices);
252 }
253 }
254
255 return convexPolygons;
256 }

258 static private bool IsConcave (int index, int vertexCount, float[] vertices, int[] indices) {
259 int previous = indices[(vertexCount + index - 1) % vertexCount] << 1;
260 int current = indices[index] << 1;
261 int next = indices[(index + 1) % vertexCount] << 1;
262 return !PositiveArea(vertices[previous], vertices[previous + 1], vertices[current], vertices[current + 1], vertices[next],
263 vertices[next + 1]);
264 }
265
266 static private bool PositiveArea (float p1x, float p1y, float p2x, float p2y, float p3x, float p3y) {
267 return p1x * (p3y - p2y) + p2x * (p1y - p3y) + p3x * (p2y - p1y) >= 0;
268 }
269
270 static private int Winding (float p1x, float p1y, float p2x, float p2y, float p3x, float p3y) {
271 float px = p2x - p1x, py = p2y - p1y;
272 return p3x * py - p3y * px + px * p1y - p1x * py >= 0 ? 1 : -1;
273 }
274 }
