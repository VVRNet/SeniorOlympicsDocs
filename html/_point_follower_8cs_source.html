87 skeletonRenderer.OnRebuild += HandleRebuildRenderer;
88 skeletonTransformIsParent = Transform.ReferenceEquals(skeletonTransform, transform.parent);

109 public void LateUpdate () {
110#if UNITY_EDITOR
111 if (!Application.isPlaying) skeletonTransformIsParent = Transform.ReferenceEquals(skeletonTransform, transform.parent);
112#endif
113
114 if (point == null) {
115 if (string.IsNullOrEmpty(pointAttachmentName)) return;
116 UpdateReferences();
117 if (point == null) return;
118 }
119
120 Vector2 worldPos;
121 point.ComputeWorldPosition(bone, out worldPos.x, out worldPos.y);
122 float rotation = point.ComputeWorldRotation(bone);
123
124 Transform thisTransform = this.transform;
125 if (skeletonTransformIsParent) {
126 // Recommended setup: Use local transform properties if Spine GameObject is the immediate parent
127 thisTransform.localPosition = new Vector3(worldPos.x, worldPos.y, followSkeletonZPosition ? 0f : thisTransform.localPosition.z);
128 if (followRotation) {
129 float halfRotation = rotation * 0.5f * Mathf.Deg2Rad;
130
131 Quaternion q = default(Quaternion);
132 q.z = Mathf.Sin(halfRotation);
133 q.w = Mathf.Cos(halfRotation);
134 thisTransform.localRotation = q;
135 }
136 } else {
137 // For special cases: Use transform world properties if transform relationship is complicated
138 Vector3 targetWorldPosition = skeletonTransform.TransformPoint(new Vector3(worldPos.x, worldPos.y, 0f));
139 if (!followSkeletonZPosition)
140 targetWorldPosition.z = thisTransform.position.z;
141
142 Transform transformParent = thisTransform.parent;
143 if (transformParent != null) {
144 Matrix4x4 m = transformParent.localToWorldMatrix;
145 if (m.m00 * m.m11 - m.m01 * m.m10 < 0) // Determinant2D is negative
146 rotation = -rotation;
147 }
148
149 if (followRotation) {
150 Vector3 transformWorldRotation = skeletonTransform.rotation.eulerAngles;
151 thisTransform.SetPositionAndRotation(targetWorldPosition, Quaternion.Euler(transformWorldRotation.x, transformWorldRotation.y, transformWorldRotation.z + rotation));
152 } else {
153 thisTransform.position = targetWorldPosition;
154 }
155 }
156
157 if (followSkeletonFlip) {
158 Vector3 localScale = thisTransform.localScale;
159 localScale.y = Mathf.Abs(localScale.y) * Mathf.Sign(bone.Skeleton.ScaleX * bone.Skeleton.ScaleY);
160 thisTransform.localScale = localScale;
161 }
162 }
163 }
