77 public override void OnInspectorGUI () {
78 component = (SkeletonRendererCustomMaterials)target;
79 SkeletonRenderer skeletonRenderer = component.skeletonRenderer;
80
81 // Draw the default inspector
82 DrawDefaultInspector();
83
84 if (serializedObject.isEditingMultipleObjects)
85 return;
86
87 if (componentCustomMaterialOverrides == null) {
88 Type cm = typeof(SkeletonRendererCustomMaterials);
89 componentCustomMaterialOverrides = cm.GetField("customMaterialOverrides", PrivateInstance).GetValue(component) as List<SkeletonRendererCustomMaterials.AtlasMaterialOverride>;
90 componentCustomSlotMaterials = cm.GetField("customSlotMaterials", PrivateInstance).GetValue(component) as List<SkeletonRendererCustomMaterials.SlotMaterialOverride>;
91 if (componentCustomMaterialOverrides == null) {
92 Debug.Log("Reflection failed.");
93 return;
94 }
95 }
96
97 // Fill with current values at start
98 if (_customMaterialOverridesPrev == null || _customSlotMaterialsPrev == null) {
99 _customMaterialOverridesPrev = CopyList(componentCustomMaterialOverrides);
100 _customSlotMaterialsPrev = CopyList(componentCustomSlotMaterials);
101 }
102
103 // Compare new values with saved. If change is detected:
104 // store new values, restore old values, remove overrides, restore new values, restore overrides.
105
106 // 1. Store new values
107 var customMaterialOverridesNew = CopyList(componentCustomMaterialOverrides);
108 var customSlotMaterialsNew = CopyList(componentCustomSlotMaterials);
109
110 // Detect changes
111 if (!_customMaterialOverridesPrev.SequenceEqual(customMaterialOverridesNew) ||
112 !_customSlotMaterialsPrev.SequenceEqual(customSlotMaterialsNew)) {
113 // 2. Restore old values
114 componentCustomMaterialOverrides.Clear();
115 componentCustomSlotMaterials.Clear();
116 componentCustomMaterialOverrides.AddRange(_customMaterialOverridesPrev);
117 componentCustomSlotMaterials.AddRange(_customSlotMaterialsPrev);
118
119 // 3. Remove overrides
120 RemoveCustomMaterials();
121
122 // 4. Restore new values
123 componentCustomMaterialOverrides.Clear();
124 componentCustomSlotMaterials.Clear();
125 componentCustomMaterialOverrides.AddRange(customMaterialOverridesNew);
126 componentCustomSlotMaterials.AddRange(customSlotMaterialsNew);
127
128 // 5. Restore overrides
129 SetCustomMaterials();
130
131 if (skeletonRenderer != null)
132 skeletonRenderer.LateUpdate();
133 }
134
135 _customMaterialOverridesPrev = CopyList(componentCustomMaterialOverrides);
136 _customSlotMaterialsPrev = CopyList(componentCustomSlotMaterials);
137
138 if (SpineInspectorUtility.LargeCenteredButton(SpineInspectorUtility.TempContent("Clear and Reapply Changes", tooltip: "Removes all non-serialized overrides in the SkeletonRenderer and reapplies the overrides on this component."))) {
139 if (skeletonRenderer != null) {
140#if SPINE_OPTIONAL_MATERIALOVERRIDE
141 skeletonRenderer.CustomMaterialOverride.Clear();
142#endif
143 skeletonRenderer.CustomSlotMaterials.Clear();
144 RemoveCustomMaterials();
145 SetCustomMaterials();
146 skeletonRenderer.LateUpdate();
147 }
148 }
149 }

154 }

156 void SetCustomMaterials () {
157 SetCustomMaterialOverrides.Invoke(component, null);

159 }

163 }
