84 public static SpineAtlasAsset CreateRuntimeInstance (TextAsset atlasText, Texture2D[] textures,
85 Material materialPropertySource, bool initialize,
86 Func<SpineAtlasAsset, TextureLoader> newCustomTextureLoader = null,
87 bool renameMaterial = false) {
88
89 // Get atlas page names.
90 string atlasString = atlasText.text;
91 atlasString = atlasString.Replace("\r", "");
92 string[] atlasLines = atlasString.Split('\n');
93 List<string> pages = new List<string>();
94 for (int i = 0; i < atlasLines.Length - 1; i++) {
95 string line = atlasLines[i].Trim();
96 if (line.EndsWith(".png"))
97 pages.Add(line.Replace(".png", ""));
98 }
99
100 // Populate Materials[] by matching texture names with page names.
101 Material[] materials = new Material[pages.Count];
102 for (int i = 0, n = pages.Count; i < n; i++) {
103 Material mat = null;
104
105 // Search for a match.
106 string pageName = pages[i];
107 for (int j = 0, m = textures.Length; j < m; j++) {
108 if (string.Equals(pageName, textures[j].name, System.StringComparison.OrdinalIgnoreCase)) {
109 // Match found.
110 mat = new Material(materialPropertySource);
111 mat.mainTexture = textures[j];
112 if (renameMaterial)
113 mat.name = pageName;
114 break;
115 }
116 }
117
118 if (mat != null)
119 materials[i] = mat;
120 else
121 throw new ArgumentException("Could not find matching atlas page in the texture array.");
122 }
123
124 // Create AtlasAsset normally
125 return CreateRuntimeInstance(atlasText, materials, initialize, newCustomTextureLoader);
126 }

157 public override Atlas GetAtlas (bool onlyMetaData = false) {
158 if (atlasFile == null) {
159 Debug.LogError("Atlas file not set for atlas asset: " + name, this);
160 Clear();
161 return null;
162 }
163
164 if (!onlyMetaData && (materials == null || materials.Length == 0)) {
165 Debug.LogError("Materials not set for atlas asset: " + name, this);
166 Clear();
167 return null;
168 }
169
170 if (atlas != null) return atlas;
171
172 try {
173 TextureLoader loader;
174 if (!onlyMetaData)
175 loader = customTextureLoader == null ? new MaterialsTextureLoader(this) : customTextureLoader;
176 else
177 loader = new NoOpTextureLoader();
178 atlas = new Atlas(new StringReader(atlasFile.text), "", loader);
179 atlas.FlipV();
180 return atlas;
181 } catch (Exception ex) {
182 Debug.LogError("Error reading atlas file for atlas asset: " + name + "\n" + ex.Message + "\n" + ex.StackTrace, this);
183 return null;
184 }

187 public Mesh GenerateMesh (string name, Mesh mesh, out Material material, float scale = 0.01f) {
188 AtlasRegion region = atlas.FindRegion(name);
189 material = null;
190 if (region != null) {
191 if (mesh == null) {
192 mesh = new Mesh();
193 mesh.name = name;
194 }
195
196 Vector3[] verts = new Vector3[4];
197 Vector2[] uvs = new Vector2[4];
198 Color[] colors = { Color.white, Color.white, Color.white, Color.white };
199 int[] triangles = { 0, 1, 2, 2, 3, 0 };
200
201 float left, right, top, bottom;
202 left = region.width / -2f;
203 right = left * -1f;
204 top = region.height / 2f;
205 bottom = top * -1;
206
207 verts[0] = new Vector3(left, bottom, 0) * scale;
208 verts[1] = new Vector3(left, top, 0) * scale;
209 verts[2] = new Vector3(right, top, 0) * scale;
210 verts[3] = new Vector3(right, bottom, 0) * scale;
211 float u, v, u2, v2;
212 u = region.u;
213 v = region.v;
214 u2 = region.u2;
215 v2 = region.v2;
216
217 if (region.degrees == 90) {
218 uvs[0] = new Vector2(u2, v2);
219 uvs[1] = new Vector2(u, v2);
220 uvs[2] = new Vector2(u, v);
221 uvs[3] = new Vector2(u2, v);
222 } else {
223 uvs[0] = new Vector2(u, v2);
224 uvs[1] = new Vector2(u, v);
225 uvs[2] = new Vector2(u2, v);
226 uvs[3] = new Vector2(u2, v2);
227 }
228
229 mesh.triangles = new int[0];
230 mesh.vertices = verts;
231 mesh.uv = uvs;
232 mesh.colors = colors;
233 mesh.triangles = triangles;
234 mesh.RecalculateNormals();
235 mesh.RecalculateBounds();
236
237 material = (Material)region.page.rendererObject;
238 } else {
239 mesh = null;
240 }
241
242 return mesh;
243 }

258 public void Load (AtlasPage page, string path) {
259#if UNITY_EDITOR
260 if (BuildUtilities.IsInSkeletonAssetBuildPreProcessing ||
261 BuildUtilities.IsInSkeletonAssetBuildPostProcessing)
262 return;
263#endif
264 String name = Path.GetFileNameWithoutExtension(path);
265 Material material = null;
266 foreach (Material other in atlasAsset.materials) {
267 if (other.mainTexture == null) {
268 Debug.LogError("Material is missing texture: " + other.name, other);
269 return;
270 }
271 string textureName = other.mainTexture.name;
272 if (textureName == name ||
273 (atlasAsset.OnDemandTextureLoader != null &&
274 textureName == atlasAsset.OnDemandTextureLoader.GetPlaceholderTextureName(name))) {
275 material = other;
276 break;
277 }
278 }
279 if (material == null) {
280 Debug.LogError("Material with texture name \"" + name + "\" not found for atlas asset: " + atlasAsset.name, atlasAsset);
281 return;
282 }
283 page.rendererObject = material;
284
285 // Very old atlas files expected the texture's actual size to be used at runtime.
286 if (page.width == 0 || page.height == 0) {
287 page.width = material.mainTexture.width;
288 page.height = material.mainTexture.height;
289 }
290 }
