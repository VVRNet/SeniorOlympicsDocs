65 public Slot (Slot slot, Bone bone) {
66 if (slot == null) throw new ArgumentNullException("slot", "slot cannot be null.");
67 if (bone == null) throw new ArgumentNullException("bone", "bone cannot be null.");
68 data = slot.data;
69 this.bone = bone;
70 r = slot.r;
71 g = slot.g;
72 b = slot.b;
73 a = slot.a;
74
75 // darkColor = slot.darkColor == null ? null : new Color(slot.darkColor);
76 if (slot.hasSecondColor) {
77 r2 = slot.r2;
78 g2 = slot.g2;
79 b2 = slot.b2;
80 } else {
81 r2 = g2 = b2 = 0;
82 }
83 hasSecondColor = slot.hasSecondColor;
84
85 attachment = slot.attachment;
86 sequenceIndex = slot.sequenceIndex;
87 deform.AddRange(slot.deform);
88 }
