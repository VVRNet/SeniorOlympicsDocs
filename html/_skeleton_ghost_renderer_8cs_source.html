32using System.Collections;
33using UnityEngine;
34
35namespace Spine.Unity.Examples {
36 public class SkeletonGhostRenderer : MonoBehaviour {
37 static readonly Color32 TransparentBlack = new Color32(0, 0, 0, 0);
38 const string colorPropertyName = "_Color";
39
40 float fadeSpeed = 10;
41 Color32 startColor;
42 MeshFilter meshFilter;
43 MeshRenderer meshRenderer;
44
45 MaterialPropertyBlock mpb;
46 int colorId;
47
48 void Awake () {
49 meshRenderer = gameObject.AddComponent<MeshRenderer>();
50 meshFilter = gameObject.AddComponent<MeshFilter>();
51
52 colorId = Shader.PropertyToID(colorPropertyName);
53 mpb = new MaterialPropertyBlock();
54 }
55
56 public void Initialize (Mesh mesh, Material[] materials, Color32 color, bool additive, float speed, int sortingLayerID, int sortingOrder) {
57 StopAllCoroutines();
58
59 gameObject.SetActive(true);
60 meshRenderer.sharedMaterials = materials;
61 meshRenderer.sortingLayerID = sortingLayerID;
62 meshRenderer.sortingOrder = sortingOrder;
63 meshFilter.sharedMesh = Instantiate(mesh);
64 startColor = color;
65 mpb.SetColor(colorId, color);
66 meshRenderer.SetPropertyBlock(mpb);
67
68 fadeSpeed = speed;
69
70 if (additive)
71 StartCoroutine(FadeAdditive());
72 else
73 StartCoroutine(Fade());
74 }
75
76 IEnumerator Fade () {
77 Color32 c = startColor;
78 Color32 black = SkeletonGhostRenderer.TransparentBlack;
79
80 float t = 1f;
81 for (float hardTimeLimit = 5f; hardTimeLimit > 0; hardTimeLimit -= Time.deltaTime) {
82 c = Color32.Lerp(black, startColor, t);
83 mpb.SetColor(colorId, c);
84 meshRenderer.SetPropertyBlock(mpb);
85
86 t = Mathf.Lerp(t, 0, Time.deltaTime * fadeSpeed);
87 if (t <= 0)
88 break;
89
90 yield return null;
91 }
92
93 Destroy(meshFilter.sharedMesh);
94 gameObject.SetActive(false);
95 }
96
97 IEnumerator FadeAdditive () {
98 Color32 c = startColor;
99 Color32 black = SkeletonGhostRenderer.TransparentBlack;
100
101 float t = 1f;
102
103 for (float hardTimeLimit = 5f; hardTimeLimit > 0; hardTimeLimit -= Time.deltaTime) {
104 c = Color32.Lerp(black, startColor, t);
105 mpb.SetColor(colorId, c);
106 meshRenderer.SetPropertyBlock(mpb);
107
108 t = Mathf.Lerp(t, 0, Time.deltaTime * fadeSpeed);
109 if (t <= 0)
110 break;
111
112 yield return null;
113 }
114
115 Destroy(meshFilter.sharedMesh);
116
117 gameObject.SetActive(false);
118 }
119
120 public void Cleanup () {
121 if (meshFilter != null && meshFilter.sharedMesh != null)
122 Destroy(meshFilter.sharedMesh);
123
124 Destroy(gameObject);
125 }
126 }
127
128}
