107 protected virtual void RenderStencilProperties () {
108 if (_StencilComp == null)
109 return; // not a shader supporting custom stencil operations
110
111 // Use default labelWidth
112 EditorGUIUtility.labelWidth = 0f;
113 _showStencilSettings = EditorGUILayout.Foldout(_showStencilSettings, _ShowStencilText);

115 using (new SpineInspectorUtility.IndentScope()) {

122 protected virtual void RenderOutlineProperties () {
123
124 if (_OutlineWidth == null)
125 return; // not an outline shader
126
127 // Use default labelWidth
128 EditorGUIUtility.labelWidth = 0f;
129
130 bool mixedValue;
131 bool hasOutlineVariant = !IsShaderWithoutStandardVariantShader(_materialEditor, out mixedValue);
132 bool isOutlineEnabled = true;
133 if (hasOutlineVariant) {
134 isOutlineEnabled = IsOutlineEnabled(_materialEditor, out mixedValue);
135 EditorGUI.showMixedValue = mixedValue;
136 EditorGUI.BeginChangeCheck();
137
138 FontStyle origFontStyle = EditorStyles.label.fontStyle;
139 EditorStyles.label.fontStyle = FontStyle.Bold;
140 isOutlineEnabled = EditorGUILayout.Toggle(_EnableOutlineText, isOutlineEnabled);
141 EditorStyles.label.fontStyle = origFontStyle;
142 EditorGUI.showMixedValue = false;
143 if (EditorGUI.EndChangeCheck()) {
144 foreach (Material material in _materialEditor.targets) {
145 SwitchShaderToOutlineSettings(material, isOutlineEnabled);
146 }
147 }
148 } else {
149 FontStyle origFontStyle = EditorStyles.label.fontStyle;
150 EditorStyles.label.fontStyle = FontStyle.Bold;
151 EditorGUILayout.LabelField(_EnableOutlineText);
152 EditorStyles.label.fontStyle = origFontStyle;
153 }
154
155 if (isOutlineEnabled) {
156 _materialEditor.ShaderProperty(_OutlineWidth, _OutlineWidthText);
157 if (_UseScreenSpaceOutlineWidth != null)
158 _materialEditor.ShaderProperty(_UseScreenSpaceOutlineWidth, _UseScreenSpaceOutlineWidthText);
159 _materialEditor.ShaderProperty(_OutlineColor, _OutlineColorText);
160
161 _showAdvancedOutlineSettings = EditorGUILayout.Foldout(_showAdvancedOutlineSettings, _OutlineAdvancedText);
162 if (_showAdvancedOutlineSettings) {
163 using (new SpineInspectorUtility.IndentScope()) {
164 _materialEditor.ShaderProperty(_OutlineReferenceTexWidth, _OutlineReferenceTexWidthText);
165 _materialEditor.ShaderProperty(_ThresholdEnd, _ThresholdEndText);
166 _materialEditor.ShaderProperty(_OutlineSmoothness, _OutlineSmoothnessText);
167 _materialEditor.ShaderProperty(_Use8Neighbourhood, _Use8NeighbourhoodText);
168 _materialEditor.ShaderProperty(_OutlineOpaqueAlpha, _OutlineOpaqueAlphaText);
169 _materialEditor.ShaderProperty(_OutlineMipLevel, _OutlineMipLevelText);
170 }
171 }
172 }

175 protected virtual void SwitchShaderToOutlineSettings (Material material, bool enableOutline) {
176
177 string shaderName = material.shader.name;
178 bool isSetToOutlineShader = shaderName.Contains(ShaderOutlineNamePrefix);
179 if (isSetToOutlineShader && !enableOutline) {
180 shaderName = shaderName.Replace(ShaderOutlineNamePrefix, ShaderNormalNamePrefix);
181 _materialEditor.SetShader(Shader.Find(shaderName), false);
182 return;
183 } else if (!isSetToOutlineShader && enableOutline) {
184 shaderName = shaderName.Replace(ShaderNormalNamePrefix, ShaderOutlineNamePrefix);
185 _materialEditor.SetShader(Shader.Find(shaderName), false);
186 return;
187 }
188 }
189
190 protected static bool IsOutlineEnabled (MaterialEditor editor, out bool mixedValue) {
191 mixedValue = false;
192 bool isAnyEnabled = false;
193 foreach (Material material in editor.targets) {
194 if (material.shader.name.Contains(ShaderOutlineNamePrefix)) {
195 isAnyEnabled = true;
196 } else if (isAnyEnabled) {
197 mixedValue = true;
198 }
199 }
200 return isAnyEnabled;
201 }
202
203 protected static bool IsShaderWithoutStandardVariantShader (MaterialEditor editor, out bool mixedValue) {
204 mixedValue = false;
205 bool isAnyShaderWithoutVariant = false;
206 foreach (Material material in editor.targets) {
207 if (material.shader.name.Contains(ShaderWithoutStandardVariantSuffix)) {
208 isAnyShaderWithoutVariant = true;
209 } else if (isAnyShaderWithoutVariant) {
210 mixedValue = true;
211 }
212 }
213 return isAnyShaderWithoutVariant;
214 }

218 EditorStyles.label.fontStyle = FontStyle.Bold;
219 value = EditorGUILayout.Toggle(label, value, EditorStyles.toggle);
220 EditorStyles.label.fontStyle = origFontStyle;
221 return value;
