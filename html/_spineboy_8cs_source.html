30using Spine;
31using Spine.Unity;
32using UnityEngine;

34namespace Spine.Unity.Examples {
35 public class Spineboy : MonoBehaviour {
36 SkeletonAnimation skeletonAnimation;
37
38 public void Start () {
39 skeletonAnimation = GetComponent<SkeletonAnimation>(); // Get the SkeletonAnimation component for the GameObject this script is attached to.
40 AnimationState animationState = skeletonAnimation.AnimationState;
41
42 animationState.Event += HandleEvent; ; // Call our method any time an animation fires an event.
43 animationState.End += (entry) => Debug.Log("start: " + entry.TrackIndex); // A lambda can be used for the callback instead of a method.
44
45 animationState.AddAnimation(0, "jump", false, 2); // Queue jump to be played on track 0 two seconds after the starting animation.
46 animationState.AddAnimation(0, "run", true, 0); // Queue walk to be looped on track 0 after the jump animation.
47 }
48
49 void HandleEvent (TrackEntry trackEntry, Spine.Event e) {
50 Debug.Log(trackEntry.TrackIndex + " " + trackEntry.Animation.Name + ": event " + e + ", " + e.Int);
51 }
52
53 public void OnMouseDown () {
54 skeletonAnimation.AnimationState.SetAnimation(0, "jump", false); // Set jump to be played on track 0 immediately.
55 skeletonAnimation.AnimationState.AddAnimation(0, "run", true, 0); // Queue walk to be looped on track 0 after the jump animation.
56 }
57 }
