123 #region SerializedProperty Helpers
124 public static SerializedProperty FindBaseOrSiblingProperty (this SerializedProperty property, string propertyName) {
125 if (string.IsNullOrEmpty(propertyName)) return null;
126
127 SerializedProperty relativeProperty = property.serializedObject.FindProperty(propertyName); // baseProperty
128
129 // If base property is not found, look for the sibling property.
130 if (relativeProperty == null) {
131 string propertyPath = property.propertyPath;
132 int localPathLength = property.name.Length;
133
134 string newPropertyPath = propertyPath.Remove(propertyPath.Length - localPathLength, localPathLength) + propertyName;
135 relativeProperty = property.serializedObject.FindProperty(newPropertyPath);
136
137 // If a direct sibling property was not found, try to find the sibling of the array.
138 if (relativeProperty == null && property.isArray) {
139 int propertyPathLength = propertyPath.Length;
140
141 int dotCount = 0;
142 const int SiblingOfListDotCount = 3;
143 for (int i = 1; i < propertyPathLength; i++) {
144 if (propertyPath[propertyPathLength - i] == '.') {
145 dotCount++;
146 if (dotCount >= SiblingOfListDotCount) {
147 localPathLength = i - 1;
148 break;
149 }
150 }
151 }
152
153 newPropertyPath = propertyPath.Remove(propertyPath.Length - localPathLength, localPathLength) + propertyName;
154 relativeProperty = property.serializedObject.FindProperty(newPropertyPath);
155 }
156 }
157
158 return relativeProperty;
159 }

282 public static bool TargetsUseSameData (SerializedObject so) {
283 if (so.isEditingMultipleObjects) {
284 int n = so.targetObjects.Length;
285 IHasSkeletonDataAsset first = so.targetObjects[0] as IHasSkeletonDataAsset;
286 for (int i = 1; i < n; i++) {
287 IHasSkeletonDataAsset sr = so.targetObjects[i] as IHasSkeletonDataAsset;
288 if (sr != null && sr.SkeletonDataAsset != first.SkeletonDataAsset)
289 return false;
290 }
291 }
292 return true;
293 }

295 public static SerializedObject GetRenderersSerializedObject (SerializedObject serializedObject) {
296 if (serializedObject.isEditingMultipleObjects) {
297 List<Object> renderers = new List<Object>();
298 foreach (UnityEngine.Object o in serializedObject.targetObjects) {
299 Component component = o as Component;
300 if (component != null) {
301 Renderer renderer = component.GetComponent<Renderer>();
302 if (renderer != null)
303 renderers.Add(renderer);
304 }
305 }
306 return new SerializedObject(renderers.ToArray());
307 } else {
308 Component component = serializedObject.targetObject as Component;
309 if (component != null) {
310 Renderer renderer = component.GetComponent<Renderer>();
311 if (renderer != null)
312 return new SerializedObject(renderer);
313 }
314 }
315
316 return null;
317 }
