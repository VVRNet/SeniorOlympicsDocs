253 public static Spine.Attachment GetAttachment (string attachmentPath, Spine.SkeletonData skeletonData) {
254 SpineAttachment.Hierarchy hierarchy = SpineAttachment.GetHierarchy(attachmentPath);
255 if (string.IsNullOrEmpty(hierarchy.name)) return null;
256
257 SlotData slot = skeletonData.FindSlot(hierarchy.slot);
258 if (slot == null) return null;
259 return skeletonData.FindSkin(hierarchy.skin).GetAttachment(slot.Index, hierarchy.name);
260 }

262 public static Spine.Attachment GetAttachment (string attachmentPath, SkeletonDataAsset skeletonDataAsset) {
263 return GetAttachment(attachmentPath, skeletonDataAsset.GetSkeletonData(true));
264 }

268 public struct Hierarchy {
269 public string skin;
270 public string slot;
271 public string name;
272
273 public Hierarchy (string fullPath) {
274 string[] chunks = fullPath.Split(new char[] { '/' }, System.StringSplitOptions.RemoveEmptyEntries);
275 if (chunks.Length == 0) {
276 skin = "";
277 slot = "";
278 name = "";
279 return;
280 } else if (chunks.Length < 2) {
281 throw new System.Exception("Cannot generate Attachment Hierarchy from string! Not enough components! [" + fullPath + "]");
282 }
283 skin = chunks[0];
284 slot = chunks[1];
285 name = "";
286 for (int i = 2; i < chunks.Length; i++) {
287 name += chunks[i];
288 }
289 }
290 }
291 }
