101 public static SkeletonRenderSeparator AddToSkeletonRenderer (SkeletonRenderer skeletonRenderer, int sortingLayerID = 0, int extraPartsRenderers = 0, int sortingOrderIncrement = DefaultSortingOrderIncrement, int baseSortingOrder = 0, bool addMinimumPartsRenderers = true) {
102 if (skeletonRenderer == null) {
103 Debug.Log("Tried to add SkeletonRenderSeparator to a null SkeletonRenderer reference.");
104 return null;
105 }
106
107 SkeletonRenderSeparator srs = skeletonRenderer.gameObject.AddComponent<SkeletonRenderSeparator>();
108 srs.skeletonRenderer = skeletonRenderer;
109
110 skeletonRenderer.Initialize(false);
111 int count = extraPartsRenderers;
112 if (addMinimumPartsRenderers)
113 count = extraPartsRenderers + skeletonRenderer.separatorSlots.Count + 1;
114
115 Transform skeletonRendererTransform = skeletonRenderer.transform;
116 List<SkeletonPartsRenderer> componentRenderers = srs.partsRenderers;
117
118 for (int i = 0; i < count; i++) {
119 SkeletonPartsRenderer spr = SkeletonPartsRenderer.NewPartsRendererGameObject(skeletonRendererTransform, i.ToString());
120 MeshRenderer mr = spr.MeshRenderer;
121 mr.sortingLayerID = sortingLayerID;
122 mr.sortingOrder = baseSortingOrder + (i * sortingOrderIncrement);
123 componentRenderers.Add(spr);
124 }
125
126 srs.OnEnable();
127
128#if UNITY_EDITOR
129 // Make sure editor updates properly in edit mode.
130 if (!Application.isPlaying) {
131 skeletonRenderer.enabled = false;
132 skeletonRenderer.enabled = true;
133 skeletonRenderer.LateUpdateMesh();
134 }
135#endif
136
137 return srs;
138 }

142 int sortingLayerID = 0;

144 if (partsRenderers.Count > 0) {
145 SkeletonPartsRenderer previous = partsRenderers[partsRenderers.Count - 1];
146 MeshRenderer previousMeshRenderer = previous.MeshRenderer;
147 sortingLayerID = previousMeshRenderer.sortingLayerID;
148 sortingOrder = previousMeshRenderer.sortingOrder + sortingOrderIncrement;

157 MeshRenderer mr = spr.MeshRenderer;
158 mr.sortingLayerID = sortingLayerID;
159 mr.sortingOrder = sortingOrder;
160
161 return spr;

234 void HandleRender (SkeletonRendererInstruction instruction) {
235 int rendererCount = partsRenderers.Count;
236 if (rendererCount <= 0) return;
237
238#if HAS_PROPERTY_BLOCK_QUERY
239 bool assignPropertyBlock = this.copyPropertyBlock && mainMeshRenderer.HasPropertyBlock();
240#else
241 bool assignPropertyBlock = this.copyPropertyBlock;
242#endif
243 if (assignPropertyBlock)
244 mainMeshRenderer.GetPropertyBlock(copiedBlock);
245
246 MeshGenerator.Settings settings = new MeshGenerator.Settings {
247 addNormals = skeletonRenderer.addNormals,
248 calculateTangents = skeletonRenderer.calculateTangents,
249 immutableTriangles = false, // parts cannot do immutable triangles.
250 pmaVertexColors = skeletonRenderer.pmaVertexColors,
251 tintBlack = skeletonRenderer.tintBlack,
252 useClipping = true,
253 zSpacing = skeletonRenderer.zSpacing
254 };
255
256 ExposedList<SubmeshInstruction> submeshInstructions = instruction.submeshInstructions;
257 SubmeshInstruction[] submeshInstructionsItems = submeshInstructions.Items;
258 int lastSubmeshInstruction = submeshInstructions.Count - 1;
259
260 int rendererIndex = 0;
261 SkeletonPartsRenderer currentRenderer = partsRenderers[rendererIndex];
262 for (int si = 0, start = 0; si <= lastSubmeshInstruction; si++) {
263 if (currentRenderer == null)
264 continue;
265 if (submeshInstructionsItems[si].forceSeparate || si == lastSubmeshInstruction) {
266 // Apply properties
267 MeshGenerator meshGenerator = currentRenderer.MeshGenerator;
268 meshGenerator.settings = settings;
269
270 if (assignPropertyBlock)
271 currentRenderer.SetPropertyBlock(copiedBlock);
272 // Render
273 currentRenderer.RenderParts(instruction.submeshInstructions, start, si + 1);
274
275 start = si + 1;
276 rendererIndex++;
277 if (rendererIndex < rendererCount) {
278 currentRenderer = partsRenderers[rendererIndex];
279 } else {
280 // Not enough renderers. Skip the rest of the instructions.
281 break;
282 }
283 }
284 }
285
286 if (OnMeshAndMaterialsUpdated != null)
287 OnMeshAndMaterialsUpdated(this.skeletonRenderer);
288
289 // Clear extra renderers if they exist.
290 for (; rendererIndex < rendererCount; rendererIndex++) {
291 currentRenderer = partsRenderers[rendererIndex];
292 if (currentRenderer != null)
293 partsRenderers[rendererIndex].ClearMesh();
294 }
295 }

297 protected void ClearPartsRendererMeshes () {
298 foreach (SkeletonPartsRenderer partsRenderer in partsRenderers) {
299 if (partsRenderer != null)
300 partsRenderer.ClearMesh();
301 }
302 }
