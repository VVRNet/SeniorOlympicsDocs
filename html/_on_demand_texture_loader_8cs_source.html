60 public virtual bool HasNullMainTexturesAssigned (out List<Material> nullTextureMaterials) {
61 nullTextureMaterials = null;
62 if (!atlasAsset) return false;
63
64 bool anyNullTexture = false;
65 foreach (Material material in atlasAsset.Materials) {
66 if (material.mainTexture == null) {
67 anyNullTexture = true;
68 if (nullTextureMaterials == null) nullTextureMaterials = new List<Material>();
69 nullTextureMaterials.Add(material);
70 }
71 }
72 return anyNullTexture;
73 }
