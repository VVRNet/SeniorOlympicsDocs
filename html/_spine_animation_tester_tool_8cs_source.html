56 [System.Serializable]

74 [Space]

77 [Header("UI")]
78 public UnityEngine.UI.Text boundAnimationsText;
79 public UnityEngine.UI.Text skeletonNameText;
80
81 void OnValidate () {
82 // Fill in the SkeletonData asset name
83 if (skeletonNameText != null) {
84 if (skeletonAnimation != null && skeletonAnimation.skeletonDataAsset != null) {
85 skeletonNameText.text = SkeletonDataAsset.name.Replace("_SkeletonData", "");
86 }
87 }
88
89 // Fill in the control list.
90 if (boundAnimationsText != null) {
91 StringBuilder boundAnimationsStringBuilder = new StringBuilder();
92 boundAnimationsStringBuilder.AppendLine("Animation Controls:");
93
94 for (int trackIndex = 0; trackIndex < trackControls.Count; trackIndex++) {
95
96 if (trackIndex > 0)
97 boundAnimationsStringBuilder.AppendLine();
98
99 boundAnimationsStringBuilder.AppendFormat("---- Track {0} ---- \n", trackIndex);
100 foreach (AnimationControl ba in trackControls[trackIndex].controls) {
101 string animationName = ba.animationName;
102 if (string.IsNullOrEmpty(animationName))
103 animationName = "SetEmptyAnimation";
104
105 boundAnimationsStringBuilder.AppendFormat("[{0}] {1}\n", ba.key.ToString(), animationName);
106 }
107
108 }
109
110 boundAnimationsText.text = boundAnimationsStringBuilder.ToString();
111
112 }
113
114 }
115
116 void Start () {
117 if (useOverrideMixDuration) {
118 skeletonAnimation.AnimationState.Data.DefaultMix = overrideMixDuration;
119 }
120 }
121
122 void Update () {
123 AnimationState animationState = skeletonAnimation.AnimationState;
124
125 // For each track
126 for (int trackIndex = 0; trackIndex < trackControls.Count; trackIndex++) {
127
128 // For each control in the track
129 foreach (AnimationControl control in trackControls[trackIndex].controls) {
130
131 // Check each control, and play the appropriate animation.
132 if (Input.GetKeyDown(control.key)) {
133 TrackEntry trackEntry;
134 if (!string.IsNullOrEmpty(control.animationName)) {
135 trackEntry = animationState.SetAnimation(trackIndex, control.animationName, control.loop);
136
137 } else {
138 float mix = control.useCustomMixDuration ? control.mixDuration : animationState.Data.DefaultMix;
139 trackEntry = animationState.SetEmptyAnimation(trackIndex, mix);
140 }
141
142 if (trackEntry != null) {
143 if (control.useCustomMixDuration)
144 trackEntry.SetMixDuration(control.mixDuration, 0f); // use SetMixDuration(mixDuration, delay) to update delay correctly
145
146 if (useOverrideAttachmentThreshold)
147 trackEntry.MixAttachmentThreshold = attachmentThreshold;
148
149 if (useOverrideDrawOrderThreshold)
150 trackEntry.MixDrawOrderThreshold = drawOrderThreshold;
151 }
152
153 // Don't parse more than one animation per track.
154 break;
155 }
156 }
157 }
158
159 }
160
161 }
