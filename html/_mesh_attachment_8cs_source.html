128 public void UpdateRegion () {
129 float[] regionUVs = this.regionUVs;
130 if (this.uvs == null || this.uvs.Length != regionUVs.Length) this.uvs = new float[regionUVs.Length];
131 float[] uvs = this.uvs;
132 int n = uvs.Length;
133 float u, v, width, height;
134
135 if (region is AtlasRegion) {
136 u = this.region.u;
137 v = this.region.v;
138 AtlasRegion region = (AtlasRegion)this.region;
139 // Note: difference from reference implementation.
140 // Covers rotation since region.width and height are already setup accordingly.
141 float textureWidth = this.region.width / (region.u2 - region.u);
142 float textureHeight = this.region.height / (region.v2 - region.v);
143 switch (region.degrees) {
144 case 90:
145 u -= (region.originalHeight - region.offsetY - region.packedWidth) / textureWidth;
146 v -= (region.originalWidth - region.offsetX - region.packedHeight) / textureHeight;
147 width = region.originalHeight / textureWidth;
148 height = region.originalWidth / textureHeight;
149 for (int i = 0; i < n; i += 2) {
150 uvs[i] = u + regionUVs[i + 1] * width;
151 uvs[i + 1] = v + (1 - regionUVs[i]) * height;
152 }
153 return;
154 case 180:
155 u -= (region.originalWidth - region.offsetX - region.packedWidth) / textureWidth;
156 v -= region.offsetY / textureHeight;
157 width = region.originalWidth / textureWidth;
158 height = region.originalHeight / textureHeight;
159 for (int i = 0; i < n; i += 2) {
160 uvs[i] = u + (1 - regionUVs[i]) * width;
161 uvs[i + 1] = v + (1 - regionUVs[i + 1]) * height;
162 }
163 return;
164 case 270:
165 u -= region.offsetY / textureWidth;
166 v -= region.offsetX / textureHeight;
167 width = region.originalHeight / textureWidth;
168 height = region.originalWidth / textureHeight;
169 for (int i = 0; i < n; i += 2) {
170 uvs[i] = u + (1 - regionUVs[i + 1]) * width;
171 uvs[i + 1] = v + regionUVs[i] * height;
172 }
173 return;
174 }
175 u -= region.offsetX / textureWidth;
176 v -= (region.originalHeight - region.offsetY - region.packedHeight) / textureHeight;
177 width = region.originalWidth / textureWidth;
178 height = region.originalHeight / textureHeight;
179 } else if (region == null) {
180 u = v = 0;
181 width = height = 1;
182 } else {
183 u = region.u;
184 v = region.v;
185 width = region.u2 - u;
186 height = region.v2 - v;
187 }
188 for (int i = 0; i < n; i += 2) {
189 uvs[i] = u + regionUVs[i] * width;
190 uvs[i + 1] = v + regionUVs[i + 1] * height;
191 }
192 }
