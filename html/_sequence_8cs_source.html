50 public Sequence (int count) {
51 lock (Sequence.nextIdLock) {
52 id = Sequence.nextID++;

54 regions = new TextureRegion[count];

58 public Sequence (Sequence other) {
59 lock (Sequence.nextIdLock) {
60 id = Sequence.nextID++;
61 }
62 regions = new TextureRegion[other.regions.Length];
63 Array.Copy(other.regions, 0, regions, 0, regions.Length);
64
65 start = other.start;
66 digits = other.digits;
67 setupIndex = other.setupIndex;
68 }

70 public void Apply (Slot slot, IHasTextureRegion attachment) {
71 int index = slot.SequenceIndex;
72 if (index == -1) index = setupIndex;
73 if (index >= regions.Length) index = regions.Length - 1;
74 TextureRegion region = regions[index];
75 if (attachment.Region != region) {
76 attachment.Region = region;
77 attachment.UpdateRegion();
78 }
79 }

81 public string GetPath (string basePath, int index) {
82 StringBuilder buffer = new StringBuilder(basePath.Length + digits);
83 buffer.Append(basePath);
84 string frame = (start + index).ToString();
85 for (int i = digits - frame.Length; i > 0; i--)
86 buffer.Append('0');
87 buffer.Append(frame);
88 return buffer.ToString();
89 }
