78 YAxis

116 public virtual void Initialize () {
117 bone = null;
118 valid = skeletonRenderer != null && skeletonRenderer.valid;
119 if (!valid) return;
120
121 skeletonTransform = skeletonRenderer.transform;
122 skeletonRenderer.OnRebuild -= HandleRebuildRenderer;
123 skeletonRenderer.OnRebuild += HandleRebuildRenderer;
124 skeletonTransformIsParent = Transform.ReferenceEquals(skeletonTransform, transform.parent);
125
126 if (!string.IsNullOrEmpty(boneName))
127 bone = skeletonRenderer.skeleton.FindBone(boneName);
128
129#if UNITY_EDITOR
130 if (Application.isEditor)
131 LateUpdate();
132#endif
133 }

135 void OnDestroy () {
136 if (skeletonRenderer != null)
137 skeletonRenderer.OnRebuild -= HandleRebuildRenderer;
138 }
139
140 public virtual void LateUpdate () {
141 if (!valid) {
142 Initialize();
143 return;
144 }
145
146#if UNITY_EDITOR
147 if (!Application.isPlaying)
148 skeletonTransformIsParent = Transform.ReferenceEquals(skeletonTransform, transform.parent);
149#endif
150
151 if (bone == null) {
152 if (string.IsNullOrEmpty(boneName)) return;
153 bone = skeletonRenderer.skeleton.FindBone(boneName);
154 if (!SetBone(boneName)) return;
155 }
156
157 Transform thisTransform = this.transform;
158 float additionalFlipScale = 1;
159 if (skeletonTransformIsParent) {
160 // Recommended setup: Use local transform properties if Spine GameObject is the immediate parent
161 thisTransform.localPosition = new Vector3(followXYPosition ? bone.WorldX : thisTransform.localPosition.x,
162 followXYPosition ? bone.WorldY : thisTransform.localPosition.y,
163 followZPosition ? 0f : thisTransform.localPosition.z);
164 if (followBoneRotation) {
165 float halfRotation = Mathf.Atan2(bone.C, bone.A) * 0.5f;
166 if (followLocalScale && bone.ScaleX < 0) // Negate rotation from negative scaleX. Don't use negative determinant. local scaleY doesn't factor into used rotation.
167 halfRotation += Mathf.PI * 0.5f;
168
169 Quaternion q = default(Quaternion);
170 q.z = Mathf.Sin(halfRotation);
171 q.w = Mathf.Cos(halfRotation);
172 thisTransform.localRotation = q;
173 }
174 } else {
175 // For special cases: Use transform world properties if transform relationship is complicated
176 Vector3 targetWorldPosition = skeletonTransform.TransformPoint(new Vector3(bone.WorldX, bone.WorldY, 0f));
177 if (!followZPosition) targetWorldPosition.z = thisTransform.position.z;
178 if (!followXYPosition) {
179 targetWorldPosition.x = thisTransform.position.x;
180 targetWorldPosition.y = thisTransform.position.y;
181 }
182
183 Vector3 skeletonLossyScale = skeletonTransform.lossyScale;
184 Transform transformParent = thisTransform.parent;
185 Vector3 parentLossyScale = transformParent != null ? transformParent.lossyScale : Vector3.one;
186 if (followBoneRotation) {
187 float boneWorldRotation = bone.WorldRotationX;
188
189 if ((skeletonLossyScale.x * skeletonLossyScale.y) < 0)
190 boneWorldRotation = -boneWorldRotation;
191
192 if (followSkeletonFlip || maintainedAxisOrientation == AxisOrientation.XAxis) {
193 if ((skeletonLossyScale.x * parentLossyScale.x < 0))
194 boneWorldRotation += 180f;
195 } else {
196 if ((skeletonLossyScale.y * parentLossyScale.y < 0))
197 boneWorldRotation += 180f;
198 }
199
200 Vector3 worldRotation = skeletonTransform.rotation.eulerAngles;
201 if (followLocalScale && bone.ScaleX < 0) boneWorldRotation += 180f;
202 thisTransform.SetPositionAndRotation(targetWorldPosition, Quaternion.Euler(worldRotation.x, worldRotation.y, worldRotation.z + boneWorldRotation));
203 } else {
204 thisTransform.position = targetWorldPosition;
205 }
206
207 additionalFlipScale = Mathf.Sign(skeletonLossyScale.x * parentLossyScale.x
208 * skeletonLossyScale.y * parentLossyScale.y);
209 }
210
211 Bone parentBone = bone.Parent;
212 if (followParentWorldScale || followLocalScale || followSkeletonFlip) {
213 Vector3 localScale = new Vector3(1f, 1f, 1f);
214 if (followParentWorldScale && parentBone != null)
215 localScale = new Vector3(parentBone.WorldScaleX, parentBone.WorldScaleY, 1f);
216 if (followLocalScale)
217 localScale.Scale(new Vector3(bone.ScaleX, bone.ScaleY, 1f));
218 if (followSkeletonFlip)
219 localScale.y *= Mathf.Sign(bone.Skeleton.ScaleX * bone.Skeleton.ScaleY) * additionalFlipScale;
220 thisTransform.localScale = localScale;
221 }
222 }
223 }
