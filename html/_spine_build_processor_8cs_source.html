99#if HAS_ON_POSTPROCESS_PREFAB
100 internal static void PreprocessSpinePrefabMeshes () {
101 BuildUtilities.IsInSkeletonAssetBuildPreProcessing = true;
102 try {
103 AssetDatabase.StartAssetEditing();
104 prefabsToRestore.Clear();
105 string[] prefabAssets = AssetDatabase.FindAssets("t:Prefab");
106 foreach (string asset in prefabAssets) {
107 string assetPath = AssetDatabase.GUIDToAssetPath(asset);
108 if (!AssetUtility.AssetCanBeModified(assetPath)) continue;
109
110 GameObject prefabGameObject = AssetDatabase.LoadAssetAtPath<GameObject>(assetPath);
111 if (SpineEditorUtilities.CleanupSpinePrefabMesh(prefabGameObject)) {
112#if HAS_SAVE_ASSET_IF_DIRTY
113 AssetDatabase.SaveAssetIfDirty(prefabGameObject);
114#endif
115 prefabsToRestore.Add(assetPath);
116 }
117 }
118 EditorUtility.UnloadUnusedAssetsImmediate();
119 AssetDatabase.StopAssetEditing();
120#if !HAS_SAVE_ASSET_IF_DIRTY
121 if (prefabAssets.Length > 0)
122 AssetDatabase.SaveAssets();
123#endif
124 } finally {
125 BuildUtilities.IsInSkeletonAssetBuildPreProcessing = false;
126 }
127 }
128
129 internal static void PostprocessSpinePrefabMeshes () {
130 BuildUtilities.IsInSkeletonAssetBuildPostProcessing = true;
131 try {
132 foreach (string assetPath in prefabsToRestore) {
133 GameObject g = AssetDatabase.LoadAssetAtPath<GameObject>(assetPath);
134 SpineEditorUtilities.SetupSpinePrefabMesh(g, null);
135#if HAS_SAVE_ASSET_IF_DIRTY
136 AssetDatabase.SaveAssetIfDirty(g);
137#endif
138 }
139#if !HAS_SAVE_ASSET_IF_DIRTY
140 if (prefabsToRestore.Count > 0)
141 AssetDatabase.SaveAssets();
142#endif
143 prefabsToRestore.Clear();
144
145 } finally {
146 BuildUtilities.IsInSkeletonAssetBuildPostProcessing = false;
147 }
148 }
149#endif

151#if SPINE_OPTIONAL_ON_DEMAND_LOADING
152 internal static void PreprocessOnDemandTextureLoaders () {
153 BuildUtilities.IsInSkeletonAssetBuildPreProcessing = true;
154 try {
155 AssetDatabase.StartAssetEditing();
156 textureLoadersToRestore.Clear();
157 string[] loaderAssets = AssetDatabase.FindAssets("t:OnDemandTextureLoader");
158 foreach (string loaderAsset in loaderAssets) {
159 string assetPath = AssetDatabase.GUIDToAssetPath(loaderAsset);
160 OnDemandTextureLoader loader = AssetDatabase.LoadAssetAtPath<OnDemandTextureLoader>(assetPath);
161 bool isLoaderUsed = loader.atlasAsset && loader.atlasAsset.OnDemandTextureLoader == loader &&
162 loader.atlasAsset.TextureLoadingMode == AtlasAssetBase.LoadingMode.OnDemand;
163 if (isLoaderUsed) {
164 IEnumerable<Material> modifiedMaterials;
165 textureLoadersToRestore.Add(assetPath);
166 loader.AssignPlaceholderTextures(out modifiedMaterials);
167
168#if HAS_SAVE_ASSET_IF_DIRTY
169 foreach (Material material in modifiedMaterials) {
170 AssetDatabase.SaveAssetIfDirty(material);
171 }
172#endif
173 }
174 }
175 EditorUtility.UnloadUnusedAssetsImmediate();
176 AssetDatabase.StopAssetEditing();
177#if !HAS_SAVE_ASSET_IF_DIRTY
178 if (textureLoadersToRestore.Count > 0)
179 AssetDatabase.SaveAssets();
180#endif
181 } finally {
182 BuildUtilities.IsInSkeletonAssetBuildPreProcessing = false;
183 }
184 }
185
186 internal static void PostprocessOnDemandTextureLoaders () {
187 BuildUtilities.IsInSkeletonAssetBuildPostProcessing = true;
188 try {
189 foreach (string assetPath in textureLoadersToRestore) {
190 OnDemandTextureLoader loader = AssetDatabase.LoadAssetAtPath<OnDemandTextureLoader>(assetPath);
191 IEnumerable<Material> modifiedMaterials;
192 loader.AssignTargetTextures(out modifiedMaterials);
193#if HAS_SAVE_ASSET_IF_DIRTY
194 foreach (Material material in modifiedMaterials) {
195 AssetDatabase.SaveAssetIfDirty(material);
196 }
197#endif
198 }
199#if !HAS_SAVE_ASSET_IF_DIRTY
200 if (textureLoadersToRestore.Count > 0)
201 AssetDatabase.SaveAssets();
202#endif
203 textureLoadersToRestore.Clear();
204
205 } finally {
206 BuildUtilities.IsInSkeletonAssetBuildPostProcessing = false;
207 }
208 }
209#endif
210 internal static void PreprocessSpriteAtlases () {
211 BuildUtilities.IsInSpriteAtlasBuildPreProcessing = true;
212 try {
213 AssetDatabase.StartAssetEditing();
214 spriteAtlasTexturesToRestore.Clear();
215 string[] spriteAtlasAssets = AssetDatabase.FindAssets("t:SpineSpriteAtlasAsset");
216 foreach (string asset in spriteAtlasAssets) {
217 string assetPath = AssetDatabase.GUIDToAssetPath(asset);
218 if (!AssetUtility.AssetCanBeModified(assetPath)) continue;
219
220 SpineSpriteAtlasAsset atlasAsset = AssetDatabase.LoadAssetAtPath<SpineSpriteAtlasAsset>(assetPath);
221 if (atlasAsset && atlasAsset.materials.Length > 0) {
222 spriteAtlasTexturesToRestore[assetPath] = AssetDatabase.GetAssetPath(atlasAsset.materials[0].mainTexture);
223 atlasAsset.materials[0].mainTexture = null;
224 }
225#if HAS_SAVE_ASSET_IF_DIRTY
226 AssetDatabase.SaveAssetIfDirty(atlasAsset);
227#endif
228 }
229 EditorUtility.UnloadUnusedAssetsImmediate();
230 AssetDatabase.StopAssetEditing();
231#if !HAS_SAVE_ASSET_IF_DIRTY
232 if (spriteAtlasAssets.Length > 0)
233 AssetDatabase.SaveAssets();
234#endif
235 } finally {
236 BuildUtilities.IsInSpriteAtlasBuildPreProcessing = false;
237 }
238 }
239
240 internal static void PostprocessSpriteAtlases () {
241 BuildUtilities.IsInSpriteAtlasBuildPostProcessing = true;
242 try {
243 foreach (KeyValuePair<string, string> pair in spriteAtlasTexturesToRestore) {
244 string assetPath = pair.Key;
245 SpineSpriteAtlasAsset atlasAsset = AssetDatabase.LoadAssetAtPath<SpineSpriteAtlasAsset>(assetPath);
246 if (atlasAsset && atlasAsset.materials.Length > 0) {
247 Texture atlasTexture = AssetDatabase.LoadAssetAtPath<Texture>(pair.Value);
248 atlasAsset.materials[0].mainTexture = atlasTexture;
249 }
250#if HAS_SAVE_ASSET_IF_DIRTY
251 AssetDatabase.SaveAssetIfDirty(atlasAsset);
252#endif
253 }
254#if !HAS_SAVE_ASSET_IF_DIRTY
255 if (spriteAtlasTexturesToRestore.Count > 0)
256 AssetDatabase.SaveAssets();
257#endif
258 spriteAtlasTexturesToRestore.Clear();
259 } finally {
260 BuildUtilities.IsInSpriteAtlasBuildPostProcessing = false;
261 }
262 }
