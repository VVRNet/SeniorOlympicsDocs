59 void OnEnable () {
60 mode = this.serializedObject.FindProperty("mode");
61 boneName = this.serializedObject.FindProperty("boneName");
62 zPosition = this.serializedObject.FindProperty("zPosition");
63 position = this.serializedObject.FindProperty("position");
64 rotation = this.serializedObject.FindProperty("rotation");
65 scale = this.serializedObject.FindProperty("scale");
66 overrideAlpha = this.serializedObject.FindProperty("overrideAlpha");
67 hierarchy = this.serializedObject.FindProperty("hierarchy");
68 hierarchyLabel = new GUIContent("Skeleton Utility Parent");
69 parentReference = this.serializedObject.FindProperty("parentReference");
70
71 utilityBone = (SkeletonUtilityBone)target;
72 skeletonUtility = utilityBone.hierarchy;
73 EvaluateFlags();
74
75 if (!utilityBone.valid && skeletonUtility != null) {
76 if (skeletonUtility.skeletonRenderer != null)
77 skeletonUtility.skeletonRenderer.Initialize(false);
78 if (skeletonUtility.skeletonGraphic != null)
79 skeletonUtility.skeletonGraphic.Initialize(false);
80 }
81
82 canCreateHingeChain = CanCreateHingeChain();
83 boundingBoxTable.Clear();
84
85 if (multiObject) return;
86 if (utilityBone.bone == null) return;
87
88 Skeleton skeleton = utilityBone.bone.Skeleton;
89 int slotCount = skeleton.Slots.Count;
90 Skin skin = skeleton.Skin;
91 if (skeleton.Skin == null)
92 skin = skeleton.Data.DefaultSkin;
93
94 for (int i = 0; i < slotCount; i++) {
95 Slot slot = skeletonUtility.Skeleton.Slots.Items[i];
96 if (slot.Bone == utilityBone.bone) {
97 List<Skin.SkinEntry> slotAttachments = new List<Skin.SkinEntry>();
98 int slotIndex = skeleton.Data.FindSlot(slot.Data.Name).Index;
99 skin.GetAttachments(slotIndex, slotAttachments);
100
101 List<BoundingBoxAttachment> boundingBoxes = new List<BoundingBoxAttachment>();
102 foreach (Skin.SkinEntry entry in slotAttachments) {
103 BoundingBoxAttachment boundingBoxAttachment = entry.Attachment as BoundingBoxAttachment;
104 if (boundingBoxAttachment != null)
105 boundingBoxes.Add(boundingBoxAttachment);
106 }
107
108 if (boundingBoxes.Count > 0)
109 boundingBoxTable.Add(slot, boundingBoxes);
110 }
111 }

114 void EvaluateFlags () {
115 if (Selection.objects.Length == 1) {
116 containsFollows = utilityBone.mode == SkeletonUtilityBone.Mode.Follow;
117 containsOverrides = utilityBone.mode == SkeletonUtilityBone.Mode.Override;
118 } else {
119 int boneCount = 0;
120 foreach (Object o in Selection.objects) {
121 GameObject go = o as GameObject;
122 if (go != null) {
123 SkeletonUtilityBone sub = go.GetComponent<SkeletonUtilityBone>();
124 if (sub != null) {
125 boneCount++;
126 containsFollows |= (sub.mode == SkeletonUtilityBone.Mode.Follow);
127 containsOverrides |= (sub.mode == SkeletonUtilityBone.Mode.Override);
128 }
129 }
130 }
131
132 multiObject |= (boneCount > 1);
133 }

294 bool CanCreateHingeChain () {
295 if (utilityBone == null)
296 return false;
297 if (utilityBone.GetComponent<Rigidbody>() != null || utilityBone.GetComponent<Rigidbody2D>() != null)
298 return false;
299 if (utilityBone.bone != null && utilityBone.bone.Children.Count == 0)
300 return false;
301
302 Rigidbody[] rigidbodies = utilityBone.GetComponentsInChildren<Rigidbody>();
303 Rigidbody2D[] rigidbodies2D = utilityBone.GetComponentsInChildren<Rigidbody2D>();
304 return rigidbodies.Length <= 0 && rigidbodies2D.Length <= 0;

307 void CreateHingeChain2D () {
308 SkeletonUtilityBone kinematicParentUtilityBone = utilityBone.transform.parent.GetComponent<SkeletonUtilityBone>();
309 if (kinematicParentUtilityBone == null) {
310 UnityEditor.EditorUtility.DisplayDialog("No parent SkeletonUtilityBone found!", "Please select the first physically moving chain node, having a parent GameObject with a SkeletonUtilityBone component attached.", "OK");
311 return;
312 }
313
314 float mass = 10;
315 const float rotationLimit = 20.0f;
316
317 SetSkeletonUtilityToFlipByRotation();
318
319 kinematicParentUtilityBone.mode = SkeletonUtilityBone.Mode.Follow;
320 kinematicParentUtilityBone.position = kinematicParentUtilityBone.rotation = kinematicParentUtilityBone.scale = kinematicParentUtilityBone.zPosition = true;
321
322 GameObject commonParentObject = new GameObject(skeletonUtility.name + " HingeChain Parent " + utilityBone.name);
323 ActivateBasedOnFlipDirection commonParentActivateOnFlip = commonParentObject.AddComponent<ActivateBasedOnFlipDirection>();
324 commonParentActivateOnFlip.skeletonRenderer = skeletonUtility.skeletonRenderer;
325 commonParentActivateOnFlip.skeletonGraphic = skeletonUtility.skeletonGraphic;
326
327 // HingeChain Parent
328 // Needs to be on top hierarchy level (not attached to the moving skeleton at least) for physics to apply proper momentum.
329 GameObject normalChainParentObject = new GameObject("HingeChain");
330 normalChainParentObject.transform.SetParent(commonParentObject.transform);
331 commonParentActivateOnFlip.activeOnNormalX = normalChainParentObject;
332
333 //FollowSkeletonUtilityRootRotation followRotationComponent = normalChainParentObject.AddComponent<FollowSkeletonUtilityRootRotation>();
334 //followRotationComponent.reference = skeletonUtility.boneRoot;
335
336 // Follower Kinematic Rigidbody
337 GameObject followerKinematicObject = new GameObject(kinematicParentUtilityBone.name + " Follower");
338 followerKinematicObject.transform.parent = normalChainParentObject.transform;
339 Rigidbody2D followerRigidbody = followerKinematicObject.AddComponent<Rigidbody2D>();
340 followerRigidbody.mass = mass;
341 followerRigidbody.isKinematic = true;
342 followerKinematicObject.AddComponent<FollowLocationRigidbody2D>().reference = kinematicParentUtilityBone.transform;
343 followerKinematicObject.transform.position = kinematicParentUtilityBone.transform.position;
344 followerKinematicObject.transform.rotation = kinematicParentUtilityBone.transform.rotation;
345
346 // Child Bones
347 SkeletonUtilityBone[] utilityBones = utilityBone.GetComponentsInChildren<SkeletonUtilityBone>();
348 Transform childBoneParentReference = followerKinematicObject.transform;
349 for (int i = 0; i < utilityBones.Length; ++i) {
350 SkeletonUtilityBone childBone = utilityBones[i];
351 mass *= 0.75f;
352 childBone.parentReference = (i == 0) ? kinematicParentUtilityBone.transform : childBoneParentReference;
353 childBone.transform.SetParent(normalChainParentObject.transform, true); // we need a flat hierarchy of all Joint objects in Unity.
354 AttachRigidbodyAndCollider2D(childBone);
355 childBone.mode = SkeletonUtilityBone.Mode.Override;
356 childBone.scale = childBone.position = childBone.zPosition = false;
357
358 HingeJoint2D joint = childBone.gameObject.AddComponent<HingeJoint2D>();
359 joint.connectedBody = childBoneParentReference.GetComponent<Rigidbody2D>();
360 joint.useLimits = true;
361 ApplyJoint2DAngleLimits(joint, rotationLimit, childBoneParentReference, childBone.transform);
362
363 childBone.GetComponent<Rigidbody2D>().mass = mass;
364 childBoneParentReference = childBone.transform;
365 }
366
367 Duplicate2DHierarchyForFlippedChains(normalChainParentObject, commonParentActivateOnFlip, skeletonUtility.transform, rotationLimit);
368 UnityEditor.Selection.activeGameObject = commonParentObject;
369 }

371 void ApplyJoint2DAngleLimits (HingeJoint2D joint, float rotationLimit, Transform parentBone, Transform bone) {
372#if HINGE_JOINT_NEW_BEHAVIOUR
373 float referenceAngle = (parentBone.eulerAngles.z - bone.eulerAngles.z + 360f) % 360f;
374 float minAngle = referenceAngle - rotationLimit;
375 float maxAngle = referenceAngle + rotationLimit;
376 if (maxAngle > 270f) {
377 minAngle -= 360f;
378 maxAngle -= 360f;
379 }
380 if (minAngle < -90f) {
381 minAngle += 360f;
382 maxAngle += 360f;
383 }
384#else
385 float minAngle = -rotationLimit;
386 float maxAngle = rotationLimit;
387#endif
388 joint.limits = new JointAngleLimits2D {
389 min = minAngle,
390 max = maxAngle
391 };
392 }

394 void Duplicate2DHierarchyForFlippedChains (GameObject normalChainParentObject, ActivateBasedOnFlipDirection commonParentActivateOnFlip,
395 Transform skeletonUtilityRoot, float rotationLimit) {
396
397 GameObject mirroredChain = GameObject.Instantiate(normalChainParentObject, normalChainParentObject.transform.position,
398 normalChainParentObject.transform.rotation, commonParentActivateOnFlip.transform);
399 mirroredChain.name = normalChainParentObject.name + " FlippedX";
400
401 commonParentActivateOnFlip.activeOnFlippedX = mirroredChain;
402
403 FollowLocationRigidbody2D followerKinematicObject = mirroredChain.GetComponentInChildren<FollowLocationRigidbody2D>();
404 followerKinematicObject.followFlippedX = true;
405 FlipBone2DHorizontal(followerKinematicObject.transform, skeletonUtilityRoot);
406
407 HingeJoint2D[] childBoneJoints = mirroredChain.GetComponentsInChildren<HingeJoint2D>();
408 Transform prevRotatedChild = null;
409 Transform parentTransformForAngles = followerKinematicObject.transform;
410 for (int i = 0; i < childBoneJoints.Length; ++i) {
411 HingeJoint2D joint = childBoneJoints[i];
412 FlipBone2DHorizontal(joint.transform, skeletonUtilityRoot);
413 ApplyJoint2DAngleLimits(joint, rotationLimit, parentTransformForAngles, joint.transform);
414
415 GameObject rotatedChild = GameObject.Instantiate(joint.gameObject, joint.transform, true);
416 rotatedChild.name = joint.name + " rotated";
417 Vector3 rotationEulerAngles = rotatedChild.transform.localEulerAngles;
418 rotationEulerAngles.x = 180;
419 rotatedChild.transform.localEulerAngles = rotationEulerAngles;
420 DestroyImmediate(rotatedChild.GetComponent<HingeJoint2D>());
421 DestroyImmediate(rotatedChild.GetComponent<BoxCollider2D>());
422 DestroyImmediate(rotatedChild.GetComponent<Rigidbody2D>());
423
424 DestroyImmediate(joint.gameObject.GetComponent<SkeletonUtilityBone>());
425
426 if (i > 0) {
427 SkeletonUtilityBone utilityBone = rotatedChild.GetComponent<SkeletonUtilityBone>();
428 utilityBone.parentReference = prevRotatedChild;
429 }
430 prevRotatedChild = rotatedChild.transform;
431 parentTransformForAngles = joint.transform;
432 }
433
434 mirroredChain.SetActive(false);
435 }

437 void FlipBone2DHorizontal (Transform bone, Transform mirrorPosition) {
438 Vector3 position = bone.position;
439 position.x = 2 * mirrorPosition.position.x - position.x; // = mirrorPosition + (mirrorPosition - bone.position)
440 bone.position = position;
441
442 Vector3 boneZ = bone.forward;
443 Vector3 boneX = bone.right;
444 boneX.x *= -1;
445
446 bone.rotation = Quaternion.LookRotation(boneZ, Vector3.Cross(boneZ, boneX));
447 }

449 void CreateHingeChain () {
450 SkeletonUtilityBone kinematicParentUtilityBone = utilityBone.transform.parent.GetComponent<SkeletonUtilityBone>();
451 if (kinematicParentUtilityBone == null) {
452 UnityEditor.EditorUtility.DisplayDialog("No parent SkeletonUtilityBone found!", "Please select the first physically moving chain node, having a parent GameObject with a SkeletonUtilityBone component attached.", "OK");
453 return;
454 }
455
456 SetSkeletonUtilityToFlipByRotation();
457
458 kinematicParentUtilityBone.mode = SkeletonUtilityBone.Mode.Follow;
459 kinematicParentUtilityBone.position = kinematicParentUtilityBone.rotation = kinematicParentUtilityBone.scale = kinematicParentUtilityBone.zPosition = true;
460
461 // HingeChain Parent
462 // Needs to be on top hierarchy level (not attached to the moving skeleton at least) for physics to apply proper momentum.
463 GameObject chainParentObject = new GameObject(skeletonUtility.name + " HingeChain Parent " + utilityBone.name);
464 FollowSkeletonUtilityRootRotation followRotationComponent = chainParentObject.AddComponent<FollowSkeletonUtilityRootRotation>();
465 followRotationComponent.reference = skeletonUtility.boneRoot;
466
467 // Follower Kinematic Rigidbody
468 GameObject followerKinematicObject = new GameObject(kinematicParentUtilityBone.name + " Follower");
469 followerKinematicObject.transform.parent = chainParentObject.transform;
470 Rigidbody followerRigidbody = followerKinematicObject.AddComponent<Rigidbody>();
471 followerRigidbody.mass = 10;
472 followerRigidbody.isKinematic = true;
473 followerKinematicObject.AddComponent<FollowLocationRigidbody>().reference = kinematicParentUtilityBone.transform;
474 followerKinematicObject.transform.position = kinematicParentUtilityBone.transform.position;
475 followerKinematicObject.transform.rotation = kinematicParentUtilityBone.transform.rotation;
476
477 // Child Bones
478 SkeletonUtilityBone[] utilityBones = utilityBone.GetComponentsInChildren<SkeletonUtilityBone>();
479 Transform childBoneParentReference = followerKinematicObject.transform;
480 foreach (SkeletonUtilityBone childBone in utilityBones) {
481 childBone.parentReference = childBoneParentReference;
482 childBone.transform.SetParent(chainParentObject.transform, true); // we need a flat hierarchy of all Joint objects in Unity.
483 AttachRigidbodyAndCollider(childBone);
484 childBone.mode = SkeletonUtilityBone.Mode.Override;
485
486 HingeJoint joint = childBone.gameObject.AddComponent<HingeJoint>();
487 joint.axis = Vector3.forward;
488 joint.connectedBody = childBoneParentReference.GetComponent<Rigidbody>();
489 joint.useLimits = true;
490 joint.limits = new JointLimits {
491 min = -20,
492 max = 20
493 };
494 childBone.GetComponent<Rigidbody>().mass = childBoneParentReference.transform.GetComponent<Rigidbody>().mass * 0.75f;
495
496 childBoneParentReference = childBone.transform;
497 }
498 UnityEditor.Selection.activeGameObject = chainParentObject;
499 }

508 static void AttachRigidbodyAndCollider (SkeletonUtilityBone utilBone, bool enableCollider = false) {
509 if (utilBone.GetComponent<Collider>() == null) {
510 if (utilBone.bone.Data.Length == 0) {
511 SphereCollider sphere = utilBone.gameObject.AddComponent<SphereCollider>();
512 sphere.radius = 0.1f;
513 sphere.enabled = enableCollider;
514 } else {
515 float length = utilBone.bone.Data.Length;
516 BoxCollider box = utilBone.gameObject.AddComponent<BoxCollider>();
517 box.size = new Vector3(length, length / 3f, 0.2f);
518 box.center = new Vector3(length / 2f, 0, 0);
519 box.enabled = enableCollider;
520 }
521 }
522 utilBone.gameObject.AddComponent<Rigidbody>();
523 }

528 CircleCollider2D sphere = utilBone.gameObject.AddComponent<CircleCollider2D>();
529 sphere.radius = 0.1f;
530 sphere.enabled = enableCollider;
531 } else {
532 float length = utilBone.bone.Data.Length;
533 BoxCollider2D box = utilBone.gameObject.AddComponent<BoxCollider2D>();
534 box.size = new Vector3(length, length / 3f, 0.2f);
535 box.offset = new Vector3(length / 2f, 0, 0);
536 box.enabled = enableCollider;
537 }

539 utilBone.gameObject.AddComponent<Rigidbody2D>();
