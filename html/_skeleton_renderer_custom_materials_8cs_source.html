55#if UNITY_EDITOR
56 void Reset () {
57 skeletonRenderer = GetComponent<SkeletonRenderer>();
58
59 // Populate atlas list
60 if (skeletonRenderer != null && skeletonRenderer.skeletonDataAsset != null) {
61 AtlasAssetBase[] atlasAssets = skeletonRenderer.skeletonDataAsset.atlasAssets;
62
63 List<AtlasMaterialOverride> initialAtlasMaterialOverrides = new List<AtlasMaterialOverride>();
64 foreach (AtlasAssetBase atlasAsset in atlasAssets) {
65 foreach (Material atlasMaterial in atlasAsset.Materials) {
66 AtlasMaterialOverride atlasMaterialOverride = new AtlasMaterialOverride {
67 overrideDisabled = true,
68 originalMaterial = atlasMaterial
69 };
70
71 initialAtlasMaterialOverrides.Add(atlasMaterialOverride);
72 }
73 }
74
75 customMaterialOverrides = initialAtlasMaterialOverrides;
76 }
77 }
78#endif
79 #endregion
80
81 void SetCustomSlotMaterials () {
82 if (skeletonRenderer == null) {
83 Debug.LogError("skeletonRenderer == null");
84 return;
85 }
86
87 for (int i = 0; i < customSlotMaterials.Count; i++) {
88 SlotMaterialOverride slotMaterialOverride = customSlotMaterials[i];
89 if (slotMaterialOverride.overrideDisabled || string.IsNullOrEmpty(slotMaterialOverride.slotName))
90 continue;
91
92 Slot slotObject = skeletonRenderer.skeleton.FindSlot(slotMaterialOverride.slotName);
93 if (slotObject != null)
94 skeletonRenderer.CustomSlotMaterials[slotObject] = slotMaterialOverride.material;
95 }
96 }
97
98 void RemoveCustomSlotMaterials () {
99 if (skeletonRenderer == null) {
100 Debug.LogError("skeletonRenderer == null");
101 return;
102 }
103
104 for (int i = 0; i < customSlotMaterials.Count; i++) {
105 SlotMaterialOverride slotMaterialOverride = customSlotMaterials[i];
106 if (string.IsNullOrEmpty(slotMaterialOverride.slotName))
107 continue;
108
109 Slot slotObject = skeletonRenderer.skeleton.FindSlot(slotMaterialOverride.slotName);
110 if (slotObject == null)
111 continue;
112 Material currentMaterial;
113 if (!skeletonRenderer.CustomSlotMaterials.TryGetValue(slotObject, out currentMaterial))
114 continue;
115
116 // Do not revert the material if it was changed by something else
117 if (currentMaterial != slotMaterialOverride.material)
118 continue;
119
120 skeletonRenderer.CustomSlotMaterials.Remove(slotObject);
121 }
122 }
123
124 void SetCustomMaterialOverrides () {
125 if (skeletonRenderer == null) {
126 Debug.LogError("skeletonRenderer == null");
127 return;
128 }
129
130#if SPINE_OPTIONAL_MATERIALOVERRIDE
131 for (int i = 0; i < customMaterialOverrides.Count; i++) {
132 AtlasMaterialOverride atlasMaterialOverride = customMaterialOverrides[i];
133 if (atlasMaterialOverride.overrideDisabled)
134 continue;
135
136 skeletonRenderer.CustomMaterialOverride[atlasMaterialOverride.originalMaterial] = atlasMaterialOverride.replacementMaterial;
137 }
138#endif
139 }
140
141 void RemoveCustomMaterialOverrides () {
142 if (skeletonRenderer == null) {
143 Debug.LogError("skeletonRenderer == null");
144 return;
145 }
146
147#if SPINE_OPTIONAL_MATERIALOVERRIDE
148 for (int i = 0; i < customMaterialOverrides.Count; i++) {
149 AtlasMaterialOverride atlasMaterialOverride = customMaterialOverrides[i];
150 Material currentMaterial;
151
152 if (!skeletonRenderer.CustomMaterialOverride.TryGetValue(atlasMaterialOverride.originalMaterial, out currentMaterial))
153 continue;
154
155 // Do not revert the material if it was changed by something else
156 if (currentMaterial != atlasMaterialOverride.replacementMaterial)
157 continue;
158
159 skeletonRenderer.CustomMaterialOverride.Remove(atlasMaterialOverride.originalMaterial);
160 }
161#endif
162 }
163
164 // OnEnable applies the overrides at runtime, and when the editor loads.
165 void OnEnable () {
166 if (skeletonRenderer == null)
167 skeletonRenderer = GetComponent<SkeletonRenderer>();
168
169 if (skeletonRenderer == null) {
170 Debug.LogError("skeletonRenderer == null");
171 return;
172 }
173
174 skeletonRenderer.Initialize(false);
175 SetCustomMaterialOverrides();
176 SetCustomSlotMaterials();
177 }
178
179 // OnDisable removes the overrides at runtime, and in the editor when the component is disabled or destroyed.
180 void OnDisable () {
181 if (skeletonRenderer == null) {
182 Debug.LogError("skeletonRenderer == null");
183 return;
184 }
185
186 RemoveCustomMaterialOverrides();
187 RemoveCustomSlotMaterials();
188 }

190 [Serializable]
191 public struct SlotMaterialOverride : IEquatable<SlotMaterialOverride> {
192 public bool overrideDisabled;
193
194 [SpineSlot]
195 public string slotName;
196 public Material material;
197
198 public bool Equals (SlotMaterialOverride other) {
199 return overrideDisabled == other.overrideDisabled && slotName == other.slotName && material == other.material;
200 }
201 }
202
203 [Serializable]
204 public struct AtlasMaterialOverride : IEquatable<AtlasMaterialOverride> {
205 public bool overrideDisabled;
206 public Material originalMaterial;
207 public Material replacementMaterial;
208
209 public bool Equals (AtlasMaterialOverride other) {
210 return overrideDisabled == other.overrideDisabled && originalMaterial == other.originalMaterial && replacementMaterial == other.replacementMaterial;
211 }
212 }
