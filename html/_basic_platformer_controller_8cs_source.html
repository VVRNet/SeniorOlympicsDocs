30using Spine.Unity;
31using UnityEngine;
32using UnityEngine.Events;
33
34namespace Spine.Unity.Examples {
35
36 [RequireComponent(typeof(CharacterController))]
37 public class BasicPlatformerController : MonoBehaviour {
38
39 public enum CharacterState {
40 None,
41 Idle,
42 Walk,
43 Run,
44 Crouch,
45 Rise,
46 Fall,
47 Attack
48 }
49
50 [Header("Components")]
51 public CharacterController controller;
52
53 [Header("Controls")]
54 public string XAxis = "Horizontal";
55 public string YAxis = "Vertical";
56 public string JumpButton = "Jump";
57
58 [Header("Moving")]
59 public float walkSpeed = 1.5f;
60 public float runSpeed = 7f;
61 public float gravityScale = 6.6f;
62
63 [Header("Jumping")]
64 public float jumpSpeed = 25;
65 public float minimumJumpDuration = 0.5f;
66 public float jumpInterruptFactor = 0.5f;
67 public float forceCrouchVelocity = 25;
68 public float forceCrouchDuration = 0.5f;
69
70 [Header("Animation")]
71 public SkeletonAnimationHandleExample animationHandle;
72
73 // Events
74 public event UnityAction OnJump, OnLand, OnHardLand;
75
76 Vector2 input = default(Vector2);
77 Vector3 velocity = default(Vector3);
78 float minimumJumpEndTime = 0;
79 float forceCrouchEndTime;
80 bool wasGrounded = false;
81
82 CharacterState previousState, currentState;
83
84 void Update () {
85 float dt = Time.deltaTime;
86 bool isGrounded = controller.isGrounded;
87 bool landed = !wasGrounded && isGrounded;
88
89 // Dummy input.
90 input.x = Input.GetAxis(XAxis);
91 input.y = Input.GetAxis(YAxis);
92 bool inputJumpStop = Input.GetButtonUp(JumpButton);
93 bool inputJumpStart = Input.GetButtonDown(JumpButton);
94 bool doCrouch = (isGrounded && input.y < -0.5f) || (forceCrouchEndTime > Time.time);
95 bool doJumpInterrupt = false;
96 bool doJump = false;
97 bool hardLand = false;
98
99 if (landed) {
100 if (-velocity.y > forceCrouchVelocity) {
101 hardLand = true;
102 doCrouch = true;
103 forceCrouchEndTime = Time.time + forceCrouchDuration;
104 }
105 }
106
107 if (!doCrouch) {
108 if (isGrounded) {
109 if (inputJumpStart) {
110 doJump = true;
111 }
112 } else {
113 doJumpInterrupt = inputJumpStop && Time.time < minimumJumpEndTime;
114 }
115 }
116
117 // Dummy physics and controller using UnityEngine.CharacterController.
118 Vector3 gravityDeltaVelocity = Physics.gravity * gravityScale * dt;
119
120 if (doJump) {
121 velocity.y = jumpSpeed;
122 minimumJumpEndTime = Time.time + minimumJumpDuration;
123 } else if (doJumpInterrupt) {
124 if (velocity.y > 0)
125 velocity.y *= jumpInterruptFactor;
126 }
127
128 velocity.x = 0;
129 if (!doCrouch) {
130 if (input.x != 0) {
131 velocity.x = Mathf.Abs(input.x) > 0.6f ? runSpeed : walkSpeed;
132 velocity.x *= Mathf.Sign(input.x);
133 }
134 }
135
136
137 if (!isGrounded) {
138 if (wasGrounded) {
139 if (velocity.y < 0)
140 velocity.y = 0;
141 } else {
142 velocity += gravityDeltaVelocity;
143 }
144 }
145 controller.Move(velocity * dt);
146 wasGrounded = isGrounded;
147
148 // Determine and store character state
149 if (isGrounded) {
150 if (doCrouch) {
151 currentState = CharacterState.Crouch;
152 } else {
153 if (input.x == 0)
154 currentState = CharacterState.Idle;
155 else
156 currentState = Mathf.Abs(input.x) > 0.6f ? CharacterState.Run : CharacterState.Walk;
157 }
158 } else {
159 currentState = velocity.y > 0 ? CharacterState.Rise : CharacterState.Fall;
160 }
161
162 bool stateChanged = previousState != currentState;
163 previousState = currentState;
164
165 // Animation
166 // Do not modify character parameters or state in this phase. Just read them.
167 // Detect changes in state, and communicate with animation handle if it changes.
168 if (stateChanged)
169 HandleStateChanged();
170
171 if (input.x != 0)
172 animationHandle.SetFlip(input.x);
173
174 // Fire events.
175 if (doJump) {
176 OnJump.Invoke();
177 }
178 if (landed) {
179 if (hardLand) {
180 OnHardLand.Invoke();
181 } else {
182 OnLand.Invoke();
183 }
184 }
185 }
186
187 void HandleStateChanged () {
188 // When the state changes, notify the animation handle of the new state.
189 string stateName = null;
190 switch (currentState) {
191 case CharacterState.Idle:
192 stateName = "idle";
193 break;
194 case CharacterState.Walk:
195 stateName = "walk";
196 break;
197 case CharacterState.Run:
198 stateName = "run";
199 break;
200 case CharacterState.Crouch:
201 stateName = "crouch";
202 break;
203 case CharacterState.Rise:
204 stateName = "rise";
205 break;
206 case CharacterState.Fall:
207 stateName = "fall";
208 break;
209 case CharacterState.Attack:
210 stateName = "attack";
211 break;
212 default:
213 break;
214 }
215
216 animationHandle.PlayAnimationForState(stateName, 0);
217 }
218
219 }
220}
