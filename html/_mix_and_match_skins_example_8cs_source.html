136 public void OptimizeSkin () {
137 // Create a repacked skin.
138 Skin previousSkin = skeletonAnimation.Skeleton.Skin;
139 // Note: materials and textures returned by GetRepackedSkin() behave like 'new Texture2D()' and need to be destroyed
140 if (runtimeMaterial)
141 Destroy(runtimeMaterial);
142 if (runtimeAtlas)
143 Destroy(runtimeAtlas);
144 Skin repackedSkin = previousSkin.GetRepackedSkin("Repacked skin", skeletonAnimation.SkeletonDataAsset.atlasAssets[0].PrimaryMaterial, out runtimeMaterial, out runtimeAtlas);
145 previousSkin.Clear();
146
147 // Use the repacked skin.
148 skeletonAnimation.Skeleton.Skin = repackedSkin;
149 skeletonAnimation.Skeleton.SetSlotsToSetupPose();
150 skeletonAnimation.AnimationState.Apply(skeletonAnimation.Skeleton);
151
152 // `GetRepackedSkin()` and each call to `GetRemappedClone()` with parameter `premultiplyAlpha` set to `true`
153 // cache necessarily created Texture copies which can be cleared by calling AtlasUtilities.ClearCache().
154 // You can optionally clear the textures cache after multiple repack operations.
155 // Just be aware that while this cleanup frees up memory, it is also a costly operation
156 // and will likely cause a spike in the framerate.
157 AtlasUtilities.ClearCache();
158 Resources.UnloadUnusedAssets();
159 }
160
161 void UpdateCharacterSkin () {
162 Skeleton skeleton = skeletonAnimation.Skeleton;
163 SkeletonData skeletonData = skeleton.Data;
164 characterSkin = new Skin("character-base");
165 // Note that the result Skin returned by calls to skeletonData.FindSkin()
166 // could be cached once in Start() instead of searching for the same skin
167 // every time. For demonstration purposes we keep it simple here.
168 characterSkin.AddSkin(skeletonData.FindSkin(baseSkin));
169 characterSkin.AddSkin(skeletonData.FindSkin(noseSkins[activeNoseIndex]));
170 characterSkin.AddSkin(skeletonData.FindSkin(eyelidsSkin));
171 characterSkin.AddSkin(skeletonData.FindSkin(eyesSkins[activeEyesIndex]));
172 characterSkin.AddSkin(skeletonData.FindSkin(hairSkins[activeHairIndex]));
173 }
174
175 void AddEquipmentSkinsTo (Skin combinedSkin) {
176 Skeleton skeleton = skeletonAnimation.Skeleton;
177 SkeletonData skeletonData = skeleton.Data;
178 combinedSkin.AddSkin(skeletonData.FindSkin(clothesSkin));
179 combinedSkin.AddSkin(skeletonData.FindSkin(pantsSkin));
180 if (!string.IsNullOrEmpty(bagSkin)) combinedSkin.AddSkin(skeletonData.FindSkin(bagSkin));
181 if (!string.IsNullOrEmpty(hatSkin)) combinedSkin.AddSkin(skeletonData.FindSkin(hatSkin));
182 }
183
184 void UpdateCombinedSkin () {
185 Skeleton skeleton = skeletonAnimation.Skeleton;
186 Skin resultCombinedSkin = new Skin("character-combined");
187
188 resultCombinedSkin.AddSkin(characterSkin);
189 AddEquipmentSkinsTo(resultCombinedSkin);
190
191 skeleton.SetSkin(resultCombinedSkin);
192 skeleton.SetSlotsToSetupPose();
193 }
194 }
