38 [System.Serializable]
39 public class EventPair {
40 [SpineEvent] public string spineEvent;
41 public UnityEvent unityHandler;
42 public AnimationState.TrackEntryEventDelegate eventDelegate;
43 }
44
45 public List<EventPair> events = new List<EventPair>();
46
47 ISkeletonComponent skeletonComponent;
48 IAnimationStateComponent animationStateComponent;
49
50 void Start () {
51 if (skeletonComponent == null)
52 skeletonComponent = GetComponent<ISkeletonComponent>();
53 if (skeletonComponent == null) return;
54 if (animationStateComponent == null)
55 animationStateComponent = skeletonComponent as IAnimationStateComponent;
56 if (animationStateComponent == null) return;
57 Skeleton skeleton = skeletonComponent.Skeleton;
58 if (skeleton == null) return;
59
60
61 SkeletonData skeletonData = skeleton.Data;
62 AnimationState state = animationStateComponent.AnimationState;
63 foreach (EventPair ep in events) {
64 EventData eventData = skeletonData.FindEvent(ep.spineEvent);
65 ep.eventDelegate = ep.eventDelegate ?? delegate (TrackEntry trackEntry, Event e) { if (e.Data == eventData) ep.unityHandler.Invoke(); };
66 state.Event += ep.eventDelegate;
67 }
68 }
69
70 void OnDestroy () {
71 if (animationStateComponent == null) animationStateComponent = GetComponent<IAnimationStateComponent>();
72 if (animationStateComponent.IsNullOrDestroyed()) return;
73
74 AnimationState state = animationStateComponent.AnimationState;
75 foreach (EventPair ep in events) {
76 if (ep.eventDelegate != null) state.Event -= ep.eventDelegate;
77 ep.eventDelegate = null;
78 }
79 }

81 }
