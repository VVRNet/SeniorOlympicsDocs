34namespace Spine.Unity.Examples {
35 public class SpineboyBeginnerView : MonoBehaviour {
36
37 #region Inspector
38 [Header("Components")]
39 public SpineboyBeginnerModel model;
40 public SkeletonAnimation skeletonAnimation;
41
42 public AnimationReferenceAsset run, idle, aim, shoot, jump;
43 public EventDataReferenceAsset footstepEvent;
44
45 [Header("Audio")]
46 public float footstepPitchOffset = 0.2f;
47 public float gunsoundPitchOffset = 0.13f;
48 public AudioSource footstepSource, gunSource, jumpSource;
49
50 [Header("Effects")]
51 public ParticleSystem gunParticles;
52 #endregion
53
54 SpineBeginnerBodyState previousViewState;
55
56 void Start () {
57 if (skeletonAnimation == null) return;
58 model.ShootEvent += PlayShoot;
59 model.StartAimEvent += StartPlayingAim;
60 model.StopAimEvent += StopPlayingAim;
61 skeletonAnimation.AnimationState.Event += HandleEvent;
62 }
63
64 void HandleEvent (Spine.TrackEntry trackEntry, Spine.Event e) {
65 if (e.Data == footstepEvent.EventData)
66 PlayFootstepSound();
67 }
68
69 void Update () {
70 if (skeletonAnimation == null) return;
71 if (model == null) return;
72
73 if ((skeletonAnimation.skeleton.ScaleX < 0) != model.facingLeft) { // Detect changes in model.facingLeft
74 Turn(model.facingLeft);
75 }
76
77 // Detect changes in model.state
78 SpineBeginnerBodyState currentModelState = model.state;
79
80 if (previousViewState != currentModelState) {
81 PlayNewStableAnimation();
82 }
83
84 previousViewState = currentModelState;
85 }
86
87 void PlayNewStableAnimation () {
88 SpineBeginnerBodyState newModelState = model.state;
89 Animation nextAnimation;
90
91 // Add conditionals to not interrupt transient animations.
92
93 if (previousViewState == SpineBeginnerBodyState.Jumping && newModelState != SpineBeginnerBodyState.Jumping) {
94 PlayFootstepSound();
95 }
96
97 if (newModelState == SpineBeginnerBodyState.Jumping) {
98 jumpSource.Play();
99 nextAnimation = jump;
100 } else {
101 if (newModelState == SpineBeginnerBodyState.Running) {
102 nextAnimation = run;
103 } else {
104 nextAnimation = idle;
105 }
106 }
107
108 skeletonAnimation.AnimationState.SetAnimation(0, nextAnimation, true);
109 }
110
111 void PlayFootstepSound () {
112 footstepSource.Play();
113 footstepSource.pitch = GetRandomPitch(footstepPitchOffset);
114 }
115
116 [ContextMenu("Check Tracks")]
117 void CheckTracks () {
118 AnimationState state = skeletonAnimation.AnimationState;
119 Debug.Log(state.GetCurrent(0));
120 Debug.Log(state.GetCurrent(1));
121 }
122
123 #region Transient Actions
124 public void PlayShoot () {
125 // Play the shoot animation on track 1.
126 TrackEntry shootTrack = skeletonAnimation.AnimationState.SetAnimation(1, shoot, false);
127 shootTrack.MixAttachmentThreshold = 1f;
128 shootTrack.SetMixDuration(0f, 0f);
129 skeletonAnimation.state.AddEmptyAnimation(1, 0.5f, 0.1f);
130
131 // Play the aim animation on track 2 to aim at the mouse target.
132 TrackEntry aimTrack = skeletonAnimation.AnimationState.SetAnimation(2, aim, false);
133 aimTrack.MixAttachmentThreshold = 1f;
134 aimTrack.SetMixDuration(0f, 0f);
135 skeletonAnimation.state.AddEmptyAnimation(2, 0.5f, 0.1f);
136
137 gunSource.pitch = GetRandomPitch(gunsoundPitchOffset);
138 gunSource.Play();
139 //gunParticles.randomSeed = (uint)Random.Range(0, 100);
140 gunParticles.Play();
141 }
142
143 public void StartPlayingAim () {
144 // Play the aim animation on track 2 to aim at the mouse target.
145 TrackEntry aimTrack = skeletonAnimation.AnimationState.SetAnimation(2, aim, true);
146 aimTrack.MixAttachmentThreshold = 1f;
147 aimTrack.SetMixDuration(0f, 0f); // use SetMixDuration(mixDuration, delay) to update delay correctly
148 }
149
150 public void StopPlayingAim () {
151 skeletonAnimation.state.AddEmptyAnimation(2, 0.5f, 0.1f);
152 }
153
154 public void Turn (bool facingLeft) {
155 skeletonAnimation.Skeleton.ScaleX = facingLeft ? -1f : 1f;
156 // Maybe play a transient turning animation too, then call ChangeStableAnimation.
157 }
158 #endregion
159
160 #region Utility
161 public float GetRandomPitch (float maxPitchOffset) {
162 return 1f + Random.Range(-maxPitchOffset, maxPitchOffset);
163 }
164 #endregion
165 }
