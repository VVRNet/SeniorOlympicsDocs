204 internal static SpinePreferences GetOrCreateSettings () {
205 if (settings != null)
206 return settings;
207
208 settings = AssetDatabase.LoadAssetAtPath<SpinePreferences>(SPINE_SETTINGS_ASSET_PATH);
209 if (settings == null)
210 settings = FindSpinePreferences();
211 if (settings == null) {
212 settings = ScriptableObject.CreateInstance<SpinePreferences>();
213 SpineEditorUtilities.OldPreferences.CopyOldToNewPreferences(ref settings);
214 // Multiple threads may be calling this method during import, creating the folder
215 // multiple times with ascending number suffix. Atomic wasPreferencesDirCreated int
216 // variable is used to prevent any redundant create operations.
217 if (!AssetDatabase.IsValidFolder("Assets/Editor") && Interlocked.Exchange(ref wasPreferencesDirCreated, 1) == 0)
218 AssetDatabase.CreateFolder("Assets", "Editor");
219 if (Interlocked.Exchange(ref wasPreferencesAssetCreated, 1) == 0)
220 AssetDatabase.CreateAsset(settings, SPINE_SETTINGS_ASSET_PATH);
221 }
222
223#if HAS_ON_POSTPROCESS_PREFAB
224 SkeletonRenderer.fixPrefabOverrideViaMeshFilterGlobal = settings.fixPrefabOverrideViaMeshFilter;
225#endif
226 SkeletonDataAsset.errorIfSkeletonFileNullGlobal = settings.skeletonDataAssetNoFileError;
227 return settings;
228 }
229
230 static SpinePreferences FindSpinePreferences () {
231 string typeSearchString = " t:SpinePreferences";
232 string[] guids = AssetDatabase.FindAssets(typeSearchString);
233 foreach (string guid in guids) {
234 string path = AssetDatabase.GUIDToAssetPath(guid);
235 SpinePreferences preferences = AssetDatabase.LoadAssetAtPath<SpinePreferences>(path);
236 if (preferences != null)
237 return preferences;
238 }
239 return null;
240 }

242 private static void ShowBlendModeMaterialProperty (SerializedProperty blendModeMaterialProperty,
243 string blendType, bool isTexturePresetPMA) {
244
245 EditorGUILayout.PropertyField(blendModeMaterialProperty, new GUIContent(blendType + " Material", blendType + " blend mode Material template."));
246 Material material = blendModeMaterialProperty.objectReferenceValue as Material;
247 if (material == null)
248 return;
249
250 bool isMaterialPMA = MaterialChecks.IsPMATextureMaterial(material);
251 if (!isTexturePresetPMA && isMaterialPMA) {
252 EditorGUILayout.HelpBox(string.Format("'{0} Material' uses PMA but 'Atlas Texture Settings' uses Straight Alpha. " +
253 "You might want to assign 'SkeletonStraight{0}' instead.", blendType), MessageType.Warning);
254 } else if (isTexturePresetPMA && !isMaterialPMA) {
255 EditorGUILayout.HelpBox(string.Format("'{0} Material' uses Straight Alpha but 'Atlas Texture Settings' uses PMA. " +
256 "You might want to assign 'SkeletonPMA{0}' instead.", blendType), MessageType.Warning);
257 }
