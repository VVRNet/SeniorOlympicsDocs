100 public void Update (Physics physics) {
101 PathAttachment attachment = target.Attachment as PathAttachment;
102 if (attachment == null) return;
103
104 float mixRotate = this.mixRotate, mixX = this.mixX, mixY = this.mixY;
105 if (mixRotate == 0 && mixX == 0 && mixY == 0) return;
106
107 PathConstraintData data = this.data;
108 bool tangents = data.rotateMode == RotateMode.Tangent, scale = data.rotateMode == RotateMode.ChainScale;
109 int boneCount = this.bones.Count, spacesCount = tangents ? boneCount : boneCount + 1;
110 Bone[] bonesItems = this.bones.Items;
111 float[] spaces = this.spaces.Resize(spacesCount).Items, lengths = scale ? this.lengths.Resize(boneCount).Items : null;
112 float spacing = this.spacing;
113 switch (data.spacingMode) {
114 case SpacingMode.Percent:
115 if (scale) {
116 for (int i = 0, n = spacesCount - 1; i < n; i++) {
117 Bone bone = bonesItems[i];
118 float setupLength = bone.data.length;
119 float x = setupLength * bone.a, y = setupLength * bone.c;
120 lengths[i] = (float)Math.Sqrt(x * x + y * y);
121 }
122 }
123 ArraysFill(spaces, 1, spacesCount, spacing);
124 break;
125 case SpacingMode.Proportional: {
126 float sum = 0;
127 for (int i = 0, n = spacesCount - 1; i < n;) {
128 Bone bone = bonesItems[i];
129 float setupLength = bone.data.length;
130 if (setupLength < PathConstraint.Epsilon) {
131 if (scale) lengths[i] = 0;
132 spaces[++i] = spacing;
133 } else {
134 float x = setupLength * bone.a, y = setupLength * bone.c;
135 float length = (float)Math.Sqrt(x * x + y * y);
136 if (scale) lengths[i] = length;
137 spaces[++i] = length;
138 sum += length;
139 }
140 }
141 if (sum > 0) {
142 sum = spacesCount / sum * spacing;
143 for (int i = 1; i < spacesCount; i++)
144 spaces[i] *= sum;
145 }
146 break;
147 }
148 default: {
149 bool lengthSpacing = data.spacingMode == SpacingMode.Length;
150 for (int i = 0, n = spacesCount - 1; i < n;) {
151 Bone bone = bonesItems[i];
152 float setupLength = bone.data.length;
153 if (setupLength < PathConstraint.Epsilon) {
154 if (scale) lengths[i] = 0;
155 spaces[++i] = spacing;
156 } else {
157 float x = setupLength * bone.a, y = setupLength * bone.c;
158 float length = (float)Math.Sqrt(x * x + y * y);
159 if (scale) lengths[i] = length;
160 spaces[++i] = (lengthSpacing ? setupLength + spacing : spacing) * length / setupLength;
161 }
162 }
163 break;
164 }
165 }
166
167 float[] positions = ComputeWorldPositions(attachment, spacesCount, tangents);
168 float boneX = positions[0], boneY = positions[1], offsetRotation = data.offsetRotation;
169 bool tip;
170 if (offsetRotation == 0) {
171 tip = data.rotateMode == RotateMode.Chain;
172 } else {
173 tip = false;
174 Bone p = target.bone;
175 offsetRotation *= p.a * p.d - p.b * p.c > 0 ? MathUtils.DegRad : -MathUtils.DegRad;
176 }
177 for (int i = 0, p = 3; i < boneCount; i++, p += 3) {
178 Bone bone = bonesItems[i];
179 bone.worldX += (boneX - bone.worldX) * mixX;
180 bone.worldY += (boneY - bone.worldY) * mixY;
181 float x = positions[p], y = positions[p + 1], dx = x - boneX, dy = y - boneY;
182 if (scale) {
183 float length = lengths[i];
184 if (length >= PathConstraint.Epsilon) {
185 float s = ((float)Math.Sqrt(dx * dx + dy * dy) / length - 1) * mixRotate + 1;
186 bone.a *= s;
187 bone.c *= s;
188 }
189 }
190 boneX = x;
191 boneY = y;
192 if (mixRotate > 0) {
193 float a = bone.a, b = bone.b, c = bone.c, d = bone.d, r, cos, sin;
194 if (tangents)
195 r = positions[p - 1];
196 else if (spaces[i + 1] < PathConstraint.Epsilon)
197 r = positions[p + 2];
198 else
199 r = MathUtils.Atan2(dy, dx);
200 r -= MathUtils.Atan2(c, a);
201 if (tip) {
202 cos = MathUtils.Cos(r);
203 sin = MathUtils.Sin(r);
204 float length = bone.data.length;
205 boneX += (length * (cos * a - sin * c) - dx) * mixRotate;
206 boneY += (length * (sin * a + cos * c) - dy) * mixRotate;
207 } else
208 r += offsetRotation;
209 if (r > MathUtils.PI)
210 r -= MathUtils.PI2;
211 else if (r < -MathUtils.PI) //
212 r += MathUtils.PI2;
213 r *= mixRotate;
214 cos = MathUtils.Cos(r);
215 sin = MathUtils.Sin(r);
216 bone.a = cos * a - sin * c;
217 bone.b = cos * b - sin * d;
218 bone.c = sin * a + cos * c;
219 bone.d = sin * b + cos * d;
220 }
221 bone.UpdateAppliedTransform();
222 }
223 }

225 float[] ComputeWorldPositions (PathAttachment path, int spacesCount, bool tangents) {
226 Slot target = this.target;
227 float position = this.position;
228 float[] spaces = this.spaces.Items, output = this.positions.Resize(spacesCount * 3 + 2).Items, world;
229 bool closed = path.Closed;
230 int verticesLength = path.WorldVerticesLength, curveCount = verticesLength / 6, prevCurve = NONE;
231
232 float pathLength, multiplier;
233 if (!path.ConstantSpeed) {
234 float[] lengths = path.Lengths;
235 curveCount -= closed ? 1 : 2;
236 pathLength = lengths[curveCount];
237
238 if (data.positionMode == PositionMode.Percent) position *= pathLength;
239
240 switch (data.spacingMode) {
241 case SpacingMode.Percent:
242 multiplier = pathLength;
243 break;
244 case SpacingMode.Proportional:
245 multiplier = pathLength / spacesCount;
246 break;
247 default:
248 multiplier = 1;
249 break;
250 }
251
252 world = this.world.Resize(8).Items;
253 for (int i = 0, o = 0, curve = 0; i < spacesCount; i++, o += 3) {
254 float space = spaces[i] * multiplier;
255 position += space;
256 float p = position;
257
258 if (closed) {
259 p %= pathLength;
260 if (p < 0) p += pathLength;
261 curve = 0;
262 } else if (p < 0) {
263 if (prevCurve != BEFORE) {
264 prevCurve = BEFORE;
265 path.ComputeWorldVertices(target, 2, 4, world, 0, 2);
266 }
267 AddBeforePosition(p, world, 0, output, o);
268 continue;
269 } else if (p > pathLength) {
270 if (prevCurve != AFTER) {
271 prevCurve = AFTER;
272 path.ComputeWorldVertices(target, verticesLength - 6, 4, world, 0, 2);
273 }
274 AddAfterPosition(p - pathLength, world, 0, output, o);
275 continue;
276 }
277
278 // Determine curve containing position.
279 for (; ; curve++) {
280 float length = lengths[curve];
281 if (p > length) continue;
282 if (curve == 0)
283 p /= length;
284 else {
285 float prev = lengths[curve - 1];
286 p = (p - prev) / (length - prev);
287 }
288 break;
289 }
290 if (curve != prevCurve) {
291 prevCurve = curve;
292 if (closed && curve == curveCount) {
293 path.ComputeWorldVertices(target, verticesLength - 4, 4, world, 0, 2);
294 path.ComputeWorldVertices(target, 0, 4, world, 4, 2);
295 } else
296 path.ComputeWorldVertices(target, curve * 6 + 2, 8, world, 0, 2);
297 }
298 AddCurvePosition(p, world[0], world[1], world[2], world[3], world[4], world[5], world[6], world[7], output, o,
299 tangents || (i > 0 && space < PathConstraint.Epsilon));
300 }
301 return output;
302 }
303
304 // World vertices.
305 if (closed) {
306 verticesLength += 2;
307 world = this.world.Resize(verticesLength).Items;
308 path.ComputeWorldVertices(target, 2, verticesLength - 4, world, 0, 2);
309 path.ComputeWorldVertices(target, 0, 2, world, verticesLength - 4, 2);
310 world[verticesLength - 2] = world[0];
311 world[verticesLength - 1] = world[1];
312 } else {
313 curveCount--;
314 verticesLength -= 4;
315 world = this.world.Resize(verticesLength).Items;
316 path.ComputeWorldVertices(target, 2, verticesLength, world, 0, 2);
317 }
318
319 // Curve lengths.
320 float[] curves = this.curves.Resize(curveCount).Items;
321 pathLength = 0;
322 float x1 = world[0], y1 = world[1], cx1 = 0, cy1 = 0, cx2 = 0, cy2 = 0, x2 = 0, y2 = 0;
323 float tmpx, tmpy, dddfx, dddfy, ddfx, ddfy, dfx, dfy;
324 for (int i = 0, w = 2; i < curveCount; i++, w += 6) {
325 cx1 = world[w];
326 cy1 = world[w + 1];
327 cx2 = world[w + 2];
328 cy2 = world[w + 3];
329 x2 = world[w + 4];
330 y2 = world[w + 5];
331 tmpx = (x1 - cx1 * 2 + cx2) * 0.1875f;
332 tmpy = (y1 - cy1 * 2 + cy2) * 0.1875f;
333 dddfx = ((cx1 - cx2) * 3 - x1 + x2) * 0.09375f;
334 dddfy = ((cy1 - cy2) * 3 - y1 + y2) * 0.09375f;
335 ddfx = tmpx * 2 + dddfx;
336 ddfy = tmpy * 2 + dddfy;
337 dfx = (cx1 - x1) * 0.75f + tmpx + dddfx * 0.16666667f;
338 dfy = (cy1 - y1) * 0.75f + tmpy + dddfy * 0.16666667f;
339 pathLength += (float)Math.Sqrt(dfx * dfx + dfy * dfy);
340 dfx += ddfx;
341 dfy += ddfy;
342 ddfx += dddfx;
343 ddfy += dddfy;
344 pathLength += (float)Math.Sqrt(dfx * dfx + dfy * dfy);
345 dfx += ddfx;
346 dfy += ddfy;
347 pathLength += (float)Math.Sqrt(dfx * dfx + dfy * dfy);
348 dfx += ddfx + dddfx;
349 dfy += ddfy + dddfy;
350 pathLength += (float)Math.Sqrt(dfx * dfx + dfy * dfy);
351 curves[i] = pathLength;
352 x1 = x2;
353 y1 = y2;
354 }
355
356 if (data.positionMode == PositionMode.Percent) position *= pathLength;
357
358 switch (data.spacingMode) {
359 case SpacingMode.Percent:
360 multiplier = pathLength;
361 break;
362 case SpacingMode.Proportional:
363 multiplier = pathLength / spacesCount;
364 break;
365 default:
366 multiplier = 1;
367 break;
368 }
369
370 float[] segments = this.segments;
371 float curveLength = 0;
372 for (int i = 0, o = 0, curve = 0, segment = 0; i < spacesCount; i++, o += 3) {
373 float space = spaces[i] * multiplier;
374 position += space;
375 float p = position;
376
377 if (closed) {
378 p %= pathLength;
379 if (p < 0) p += pathLength;
380 curve = 0;
381 } else if (p < 0) {
382 AddBeforePosition(p, world, 0, output, o);
383 continue;
384 } else if (p > pathLength) {
385 AddAfterPosition(p - pathLength, world, verticesLength - 4, output, o);
386 continue;
387 }
388
389 // Determine curve containing position.
390 for (; ; curve++) {
391 float length = curves[curve];
392 if (p > length) continue;
393 if (curve == 0)
394 p /= length;
395 else {
396 float prev = curves[curve - 1];
397 p = (p - prev) / (length - prev);
398 }
399 break;
400 }
401
402 // Curve segment lengths.
403 if (curve != prevCurve) {
404 prevCurve = curve;
405 int ii = curve * 6;
406 x1 = world[ii];
407 y1 = world[ii + 1];
408 cx1 = world[ii + 2];
409 cy1 = world[ii + 3];
410 cx2 = world[ii + 4];
411 cy2 = world[ii + 5];
412 x2 = world[ii + 6];
413 y2 = world[ii + 7];
414 tmpx = (x1 - cx1 * 2 + cx2) * 0.03f;
415 tmpy = (y1 - cy1 * 2 + cy2) * 0.03f;
416 dddfx = ((cx1 - cx2) * 3 - x1 + x2) * 0.006f;
417 dddfy = ((cy1 - cy2) * 3 - y1 + y2) * 0.006f;
418 ddfx = tmpx * 2 + dddfx;
419 ddfy = tmpy * 2 + dddfy;
420 dfx = (cx1 - x1) * 0.3f + tmpx + dddfx * 0.16666667f;
421 dfy = (cy1 - y1) * 0.3f + tmpy + dddfy * 0.16666667f;
422 curveLength = (float)Math.Sqrt(dfx * dfx + dfy * dfy);
423 segments[0] = curveLength;
424 for (ii = 1; ii < 8; ii++) {
425 dfx += ddfx;
426 dfy += ddfy;
427 ddfx += dddfx;
428 ddfy += dddfy;
429 curveLength += (float)Math.Sqrt(dfx * dfx + dfy * dfy);
430 segments[ii] = curveLength;
431 }
432 dfx += ddfx;
433 dfy += ddfy;
434 curveLength += (float)Math.Sqrt(dfx * dfx + dfy * dfy);
435 segments[8] = curveLength;
436 dfx += ddfx + dddfx;
437 dfy += ddfy + dddfy;
438 curveLength += (float)Math.Sqrt(dfx * dfx + dfy * dfy);
439 segments[9] = curveLength;
440 segment = 0;
441 }
442
443 // Weight by segment length.
444 p *= curveLength;
445 for (; ; segment++) {
446 float length = segments[segment];
447 if (p > length) continue;
448 if (segment == 0)
449 p /= length;
450 else {
451 float prev = segments[segment - 1];
452 p = segment + (p - prev) / (length - prev);
453 }
454 break;
455 }
456 AddCurvePosition(p * 0.1f, x1, y1, cx1, cy1, cx2, cy2, x2, y2, output, o, tangents || (i > 0 && space < PathConstraint.Epsilon));
457 }
458 return output;
459 }

462 float x1 = temp[i], y1 = temp[i + 1], dx = temp[i + 2] - x1, dy = temp[i + 3] - y1, r = MathUtils.Atan2(dy, dx);
463 output[o] = x1 + p * MathUtils.Cos(r);
464 output[o + 1] = y1 + p * MathUtils.Sin(r);
465 output[o + 2] = r;

471 output[o + 1] = y1 + p * MathUtils.Sin(r);

475 static void AddCurvePosition (float p, float x1, float y1, float cx1, float cy1, float cx2, float cy2, float x2, float y2,
476 float[] output, int o, bool tangents) {
477 if (p < PathConstraint.Epsilon || float.IsNaN(p)) {
478 output[o] = x1;
479 output[o + 1] = y1;
480 output[o + 2] = (float)Math.Atan2(cy1 - y1, cx1 - x1);
481 return;
482 }
483 float tt = p * p, ttt = tt * p, u = 1 - p, uu = u * u, uuu = uu * u;
484 float ut = u * p, ut3 = ut * 3, uut3 = u * ut3, utt3 = ut3 * p;
485 float x = x1 * uuu + cx1 * uut3 + cx2 * utt3 + x2 * ttt, y = y1 * uuu + cy1 * uut3 + cy2 * utt3 + y2 * ttt;
486 output[o] = x;
487 output[o + 1] = y;
488 if (tangents) {
489 if (p < 0.001f)
490 output[o + 2] = (float)Math.Atan2(cy1 - y1, cx1 - x1);
491 else
492 output[o + 2] = (float)Math.Atan2(y - (y1 * uu + cy1 * ut * 2 + cy2 * tt), x - (x1 * uu + cx1 * ut * 2 + cx2 * tt));
493 }
