163 particles.transform.position = new Vector3(0,particles.transform.position.y,0);
164 #if UNITY_3_5

173
174#if UNITY_5_5_OR_NEWER

181 ps.gameObject.AddComponent<CFX_AutoDestructShuriken>();

187 ps.gameObject.AddComponent<CFX_AutoStopLoopedEffect>();

197 IEnumerator CheckForDeletedParticles()
198 {

201 yield return new WaitForSeconds(5.0f);
202 for(int i = onScreenParticles.Count - 1; i >= 0; i--)
