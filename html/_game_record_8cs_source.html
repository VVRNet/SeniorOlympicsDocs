294 private float EstimateAgeFromSpeed(float speed, Dictionary<int, float> speedAverages)
295 {
296 var keys = speedAverages.Keys.ToList();
297 for (int i = 0; i < keys.Count - 1; i++)
298 {
299 if (speed >= speedAverages[keys[i + 1]] && speed <= speedAverages[keys[i]])
300 {
301 // 선형 보간법 적용
302 float ageRange = keys[i] - keys[i + 1];
303 float speedRange = speedAverages[keys[i]] - speedAverages[keys[i + 1]];
304 float ratio = (speed - speedAverages[keys[i + 1]]) / speedRange;
305 return keys[i + 1] + ratio * ageRange;
306 }
307 }
308 return keys.Last(); // 최대 연령 반환
309 }
