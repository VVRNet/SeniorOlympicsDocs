46 #region Context Menu Item
47 [MenuItem("CONTEXT/SkeletonGraphic/Add BoneFollower GameObject")]
48 static void AddBoneFollowerGameObject (MenuCommand cmd) {
49 SkeletonGraphic skeletonGraphic = cmd.context as SkeletonGraphic;
50 GameObject go = EditorInstantiation.NewGameObject("BoneFollower", true, typeof(RectTransform));
51 Transform t = go.transform;
52 t.SetParent(skeletonGraphic.transform);
53 t.localPosition = Vector3.zero;
54
55 BoneFollowerGraphic f = go.AddComponent<BoneFollowerGraphic>();
56 f.skeletonGraphic = skeletonGraphic;
57 f.SetBone(skeletonGraphic.Skeleton.RootBone.Data.Name);
58
59 EditorGUIUtility.PingObject(t);
60
61 Undo.RegisterCreatedObjectUndo(go, "Add BoneFollowerGraphic");
62 }

64 // Validate

83 targetBoneFollower = (BoneFollowerGraphic)target;
84 if (targetBoneFollower.SkeletonGraphic != null)

95 public void OnSceneGUI () {
96 BoneFollowerGraphic tbf = target as BoneFollowerGraphic;
97 SkeletonGraphic skeletonGraphicComponent = tbf.SkeletonGraphic;
98 if (skeletonGraphicComponent == null) return;
99
100 Transform transform = skeletonGraphicComponent.transform;
101 Skeleton skeleton = skeletonGraphicComponent.Skeleton;
102 float positionScale = skeletonGraphicComponent.MeshScale;
103
104 if (string.IsNullOrEmpty(boneName.stringValue)) {
105 SpineHandles.DrawBones(transform, skeleton, positionScale);
106 SpineHandles.DrawBoneNames(transform, skeleton, positionScale);
107 Handles.Label(tbf.transform.position, "No bone selected", EditorStyles.helpBox);
108 } else {
109 Bone targetBone = tbf.bone;
110 if (targetBone == null) return;
111
112 SpineHandles.DrawBoneWireframe(transform, targetBone, SpineHandles.TransformContraintColor, positionScale);
113 Handles.Label(targetBone.GetWorldPosition(transform, positionScale), targetBone.Data.Name, SpineHandles.BoneNameStyle);
114 }
115 }

117 override public void OnInspectorGUI () {
118 if (serializedObject.isEditingMultipleObjects) {
119 if (needsReset) {
120 needsReset = false;
121 foreach (Object o in targets) {
122 BoneFollower bf = (BoneFollower)o;
123 bf.Initialize();
124 bf.LateUpdate();
125 }
126 SceneView.RepaintAll();
127 }
128
129 EditorGUI.BeginChangeCheck();
130 DrawDefaultInspector();
131 needsReset |= EditorGUI.EndChangeCheck();
132 return;
133 }
134
135 if (needsReset && Event.current.type == EventType.Layout) {
136 targetBoneFollower.Initialize();
137 targetBoneFollower.LateUpdate();
138 needsReset = false;
139 SceneView.RepaintAll();
140 }
141 serializedObject.Update();
142
143 // Find Renderer
144 if (skeletonGraphic.objectReferenceValue == null) {
145 SkeletonGraphic parentRenderer = targetBoneFollower.GetComponentInParent<SkeletonGraphic>();
146 if (parentRenderer != null && parentRenderer.gameObject != targetBoneFollower.gameObject) {
147 skeletonGraphic.objectReferenceValue = parentRenderer;
148 Debug.Log("Inspector automatically assigned BoneFollowerGraphic.SkeletonGraphic");
149 }
150 }
151
152 EditorGUILayout.PropertyField(skeletonGraphic);
153 SkeletonGraphic skeletonGraphicComponent = skeletonGraphic.objectReferenceValue as SkeletonGraphic;
154 if (skeletonGraphicComponent != null) {
155 if (skeletonGraphicComponent.gameObject == targetBoneFollower.gameObject) {
156 skeletonGraphic.objectReferenceValue = null;
157 EditorUtility.DisplayDialog("Invalid assignment.", "BoneFollowerGraphic can only follow a skeleton on a separate GameObject.\n\nCreate a new GameObject for your BoneFollower, or choose a SkeletonGraphic from a different GameObject.", "Ok");
158 }
159 }
160
161 if (!targetBoneFollower.valid) {
162 needsReset = true;
163 }
164
165 if (targetBoneFollower.valid) {
166 EditorGUI.BeginChangeCheck();
167 EditorGUILayout.PropertyField(boneName);
168 needsReset |= EditorGUI.EndChangeCheck();
169
170 EditorGUILayout.PropertyField(followBoneRotation);
171 EditorGUILayout.PropertyField(followXYPosition);
172 EditorGUILayout.PropertyField(followZPosition);
173 EditorGUILayout.PropertyField(followLocalScale);
174 EditorGUILayout.PropertyField(followParentWorldScale);
175 EditorGUILayout.PropertyField(followSkeletonFlip);
176 if ((followSkeletonFlip.hasMultipleDifferentValues || followSkeletonFlip.boolValue == false) &&
177 (followBoneRotation.hasMultipleDifferentValues || followBoneRotation.boolValue == true)) {
178 using (new SpineInspectorUtility.IndentScope())
179 EditorGUILayout.PropertyField(maintainedAxisOrientation);
180 }
181
182 //BoneFollowerInspector.RecommendRigidbodyButton(targetBoneFollower);
183 } else {
184 SkeletonGraphic boneFollowerSkeletonGraphic = targetBoneFollower.skeletonGraphic;
185 if (boneFollowerSkeletonGraphic == null) {
186 EditorGUILayout.HelpBox("SkeletonGraphic is unassigned. Please assign a SkeletonRenderer (SkeletonAnimation or SkeletonMecanim).", MessageType.Warning);
187 } else {
188 boneFollowerSkeletonGraphic.Initialize(false);
189
190 if (boneFollowerSkeletonGraphic.skeletonDataAsset == null)
191 EditorGUILayout.HelpBox("Assigned SkeletonGraphic does not have SkeletonData assigned to it.", MessageType.Warning);
192
193 if (!boneFollowerSkeletonGraphic.IsValid)
194 EditorGUILayout.HelpBox("Assigned SkeletonGraphic is invalid. Check target SkeletonGraphic, its SkeletonData asset or the console for other errors.", MessageType.Warning);
195 }
196 }
197
198 Event current = Event.current;
199 bool wasUndo = (current.type == EventType.ValidateCommand && current.commandName == "UndoRedoPerformed");
200 if (wasUndo)
201 targetBoneFollower.Initialize();
202
203 serializedObject.ApplyModifiedProperties();
204 }
205
206 }
