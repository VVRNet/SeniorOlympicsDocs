167 static public void Apply (Bone bone, float targetX, float targetY, bool compress, bool stretch, bool uniform,
168 float alpha) {
169 if (bone == null) throw new ArgumentNullException("bone", "bone cannot be null.");
170 Bone p = bone.parent;
171
172 float pa = p.a, pb = p.b, pc = p.c, pd = p.d;
173 float rotationIK = -bone.ashearX - bone.arotation;
174 float tx = 0, ty = 0;
175
176 switch (bone.inherit) {
177 case Inherit.OnlyTranslation:
178 tx = (targetX - bone.worldX) * Math.Sign(bone.skeleton.ScaleX);
179 ty = (targetY - bone.worldY) * Math.Sign(bone.skeleton.ScaleY);
180 break;
181 case Inherit.NoRotationOrReflection: {
182 float s = Math.Abs(pa * pd - pb * pc) / Math.Max(0.0001f, pa * pa + pc * pc);
183 float sa = pa / bone.skeleton.scaleX;
184 float sc = pc / bone.skeleton.ScaleY;
185 pb = -sc * s * bone.skeleton.scaleX;
186 pd = sa * s * bone.skeleton.ScaleY;
187 rotationIK += MathUtils.Atan2Deg(sc, sa);
188 goto default; // Fall through.
189 }
190 default: {
191 float x = targetX - p.worldX, y = targetY - p.worldY;
192 float d = pa * pd - pb * pc;
193 if (Math.Abs(d) <= 0.0001f) {
194 tx = 0;
195 ty = 0;
196 } else {
197 tx = (x * pd - y * pb) / d - bone.ax;
198 ty = (y * pa - x * pc) / d - bone.ay;
199 }
200 break;
201 }
202 }
203
204 rotationIK += MathUtils.Atan2Deg(ty, tx);
205 if (bone.ascaleX < 0) rotationIK += 180;
206 if (rotationIK > 180)
207 rotationIK -= 360;
208 else if (rotationIK < -180) //
209 rotationIK += 360;
210
211 float sx = bone.ascaleX, sy = bone.ascaleY;
212 if (compress || stretch) {
213 switch (bone.inherit) {
214 case Inherit.NoScale:
215 case Inherit.NoScaleOrReflection:
216 tx = targetX - bone.worldX;
217 ty = targetY - bone.worldY;
218 break;
219 }
220 float b = bone.data.length * sx;
221 if (b > 0.0001f) {
222 float dd = tx * tx + ty * ty;
223 if ((compress && dd < b * b) || (stretch && dd > b * b)) {
224 float s = ((float)Math.Sqrt(dd) / b - 1) * alpha + 1;
225 sx *= s;
226 if (uniform) sy *= s;
227 }
228 }
229 }
230 bone.UpdateWorldTransform(bone.ax, bone.ay, bone.arotation + rotationIK * alpha, sx, sy, bone.ashearX, bone.ashearY);
231 }

235 static public void Apply (Bone parent, Bone child, float targetX, float targetY, int bendDir, bool stretch, bool uniform,
236 float softness, float alpha) {
237 if (parent == null) throw new ArgumentNullException("parent", "parent cannot be null.");
238 if (child == null) throw new ArgumentNullException("child", "child cannot be null.");
239 if (parent.inherit != Inherit.Normal || child.inherit != Inherit.Normal) return;
240 float px = parent.ax, py = parent.ay, psx = parent.ascaleX, psy = parent.ascaleY, sx = psx, sy = psy, csx = child.ascaleX;
241 int os1, os2, s2;
242 if (psx < 0) {
243 psx = -psx;
244 os1 = 180;
245 s2 = -1;
246 } else {
247 os1 = 0;
248 s2 = 1;
249 }
250 if (psy < 0) {
251 psy = -psy;
252 s2 = -s2;
253 }
254 if (csx < 0) {
255 csx = -csx;
256 os2 = 180;
257 } else
258 os2 = 0;
259 float cx = child.ax, cy, cwx, cwy, a = parent.a, b = parent.b, c = parent.c, d = parent.d;
260 bool u = Math.Abs(psx - psy) <= 0.0001f;
261 if (!u || stretch) {
262 cy = 0;
263 cwx = a * cx + parent.worldX;
264 cwy = c * cx + parent.worldY;
265 } else {
266 cy = child.ay;
267 cwx = a * cx + b * cy + parent.worldX;
268 cwy = c * cx + d * cy + parent.worldY;
269 }
270 Bone pp = parent.parent;
271 a = pp.a;
272 b = pp.b;
273 c = pp.c;
274 d = pp.d;
275 float id = a * d - b * c, x = cwx - pp.worldX, y = cwy - pp.worldY;
276 id = Math.Abs(id) <= 0.0001f ? 0 : 1 / id;
277 float dx = (x * d - y * b) * id - px, dy = (y * a - x * c) * id - py;
278 float l1 = (float)Math.Sqrt(dx * dx + dy * dy), l2 = child.data.length * csx, a1, a2;
279 if (l1 < 0.0001f) {
280 Apply(parent, targetX, targetY, false, stretch, false, alpha);
281 child.UpdateWorldTransform(cx, cy, 0, child.ascaleX, child.ascaleY, child.ashearX, child.ashearY);
282 return;
283 }
284 x = targetX - pp.worldX;
285 y = targetY - pp.worldY;
286 float tx = (x * d - y * b) * id - px, ty = (y * a - x * c) * id - py;
287 float dd = tx * tx + ty * ty;
288 if (softness != 0) {
289 softness *= psx * (csx + 1) * 0.5f;
290 float td = (float)Math.Sqrt(dd), sd = td - l1 - l2 * psx + softness;
291 if (sd > 0) {
292 float p = Math.Min(1, sd / (softness * 2)) - 1;
293 p = (sd - softness * (1 - p * p)) / td;
294 tx -= p * tx;
295 ty -= p * ty;
296 dd = tx * tx + ty * ty;
297 }
298 }
299 if (u) {
300 l2 *= psx;
301 float cos = (dd - l1 * l1 - l2 * l2) / (2 * l1 * l2);
302 if (cos < -1) {
303 cos = -1;
304 a2 = MathUtils.PI * bendDir;
305 } else if (cos > 1) {
306 cos = 1;
307 a2 = 0;
308 if (stretch) {
309 a = ((float)Math.Sqrt(dd) / (l1 + l2) - 1) * alpha + 1;
310 sx *= a;
311 if (uniform) sy *= a;
312 }
313 } else
314 a2 = (float)Math.Acos(cos) * bendDir;
315 a = l1 + l2 * cos;
316 b = l2 * (float)Math.Sin(a2);
317 a1 = (float)Math.Atan2(ty * a - tx * b, tx * a + ty * b);
318 } else {
319 a = psx * l2;
320 b = psy * l2;
321 float aa = a * a, bb = b * b, ta = (float)Math.Atan2(ty, tx);
322 c = bb * l1 * l1 + aa * dd - aa * bb;
323 float c1 = -2 * bb * l1, c2 = bb - aa;
324 d = c1 * c1 - 4 * c2 * c;
325 if (d >= 0) {
326 float q = (float)Math.Sqrt(d);
327 if (c1 < 0) q = -q;
328 q = -(c1 + q) * 0.5f;
329 float r0 = q / c2, r1 = c / q;
330 float r = Math.Abs(r0) < Math.Abs(r1) ? r0 : r1;
331 if (r * r <= dd) {
332 y = (float)Math.Sqrt(dd - r * r) * bendDir;
333 a1 = ta - (float)Math.Atan2(y, r);
334 a2 = (float)Math.Atan2(y / psy, (r - l1) / psx);
335 goto break_outer; // break outer;
336 }
337 }
338 float minAngle = MathUtils.PI, minX = l1 - a, minDist = minX * minX, minY = 0;
339 float maxAngle = 0, maxX = l1 + a, maxDist = maxX * maxX, maxY = 0;
340 c = -a * l1 / (aa - bb);
341 if (c >= -1 && c <= 1) {
342 c = (float)Math.Acos(c);
343 x = a * (float)Math.Cos(c) + l1;
344 y = b * (float)Math.Sin(c);
345 d = x * x + y * y;
346 if (d < minDist) {
347 minAngle = c;
348 minDist = d;
349 minX = x;
350 minY = y;
351 }
352 if (d > maxDist) {
353 maxAngle = c;
354 maxDist = d;
355 maxX = x;
356 maxY = y;
357 }
358 }
359 if (dd <= (minDist + maxDist) * 0.5f) {
360 a1 = ta - (float)Math.Atan2(minY * bendDir, minX);
361 a2 = minAngle * bendDir;
362 } else {
363 a1 = ta - (float)Math.Atan2(maxY * bendDir, maxX);
364 a2 = maxAngle * bendDir;
365 }
366 }
367 break_outer:
368 float os = (float)Math.Atan2(cy, cx) * s2;
369 float rotation = parent.arotation;
370 a1 = (a1 - os) * MathUtils.RadDeg + os1 - rotation;
371 if (a1 > 180)
372 a1 -= 360;
373 else if (a1 < -180)
374 a1 += 360;
375 parent.UpdateWorldTransform(px, py, rotation + a1 * alpha, sx, sy, 0, 0);
376 rotation = child.arotation;
377 a2 = ((a2 + os) * MathUtils.RadDeg - child.ashearX) * s2 + os2 - rotation;
378 if (a2 > 180)
379 a2 -= 360;
380 else if (a2 < -180)
381 a2 += 360;
382 child.UpdateWorldTransform(cx, cy, rotation + a2 * alpha, child.ascaleX, child.ascaleY, child.ashearX, child.ashearY);
383 }
384 }
