398 public virtual void ApplyTransformMovementToPhysics () {
399 if (Application.isPlaying) {
400 if (physicsPositionInheritanceFactor != Vector2.zero) {
401 Vector2 position = GetPhysicsTransformPosition();
402 Vector2 positionDelta = (position - lastPosition) / meshScale;
403
404 positionDelta = transform.InverseTransformVector(positionDelta);
405 if (physicsMovementRelativeTo != null) {
406 positionDelta = physicsMovementRelativeTo.TransformVector(positionDelta);
407 }
408 positionDelta.x *= physicsPositionInheritanceFactor.x;
409 positionDelta.y *= physicsPositionInheritanceFactor.y;
410
411 skeleton.PhysicsTranslate(positionDelta.x, positionDelta.y);
412 lastPosition = position;
413 }
414 if (physicsRotationInheritanceFactor != 0f) {
415 float rotation = GetPhysicsTransformRotation();
416 skeleton.PhysicsRotate(0, 0, physicsRotationInheritanceFactor * (rotation - lastRotation));
417 lastRotation = rotation;
418 }
419 }
420 }

422 protected Vector2 GetPhysicsTransformPosition () {
423 if (physicsMovementRelativeTo == null) {
424 return transform.position;
425 } else {
426 if (physicsMovementRelativeTo == transform.parent)
427 return transform.localPosition;
428 else
429 return physicsMovementRelativeTo.InverseTransformPoint(transform.position);
430 }
431 }
432
433 protected float GetPhysicsTransformRotation () {
434 if (physicsMovementRelativeTo == null) {
435 return this.transform.rotation.eulerAngles.z;
436 } else {
437 if (physicsMovementRelativeTo == this.transform.parent)
438 return this.transform.localRotation.eulerAngles.z;
439 else {
440 Quaternion relative = Quaternion.Inverse(physicsMovementRelativeTo.rotation) * this.transform.rotation;
441 return relative.eulerAngles.z;
442 }
443 }

672 protected bool MatchRectTransformSingleRenderer () {
673 Mesh mesh = this.GetLastMesh();
674 if (mesh == null) {
675 return false;
676 }
677 if (mesh.vertexCount == 0 || mesh.bounds.size == Vector3.zero) {
678 this.rectTransform.sizeDelta = new Vector2(50f, 50f);
679 this.rectTransform.pivot = new Vector2(0.5f, 0.5f);
680 return false;
681 }
682 mesh.RecalculateBounds();
683 SetRectTransformBounds(mesh.bounds);
684 return true;
685 }
686
687 protected bool MatchRectTransformMultipleRenderers () {
688 bool anyBoundsAdded = false;
689 Bounds combinedBounds = new Bounds();
690 for (int i = 0; i < canvasRenderers.Count; ++i) {
691 CanvasRenderer canvasRenderer = canvasRenderers[i];
692 if (!canvasRenderer.gameObject.activeSelf)
693 continue;
694
695 Mesh mesh = meshes.Items[i];
696 if (mesh == null || mesh.vertexCount == 0)
697 continue;
698
699 mesh.RecalculateBounds();
700 Bounds bounds = mesh.bounds;
701 if (anyBoundsAdded)
702 combinedBounds.Encapsulate(bounds);
703 else {
704 anyBoundsAdded = true;
705 combinedBounds = bounds;
706 }
707 }
708
709 if (!anyBoundsAdded || combinedBounds.size == Vector3.zero) {
710 this.rectTransform.sizeDelta = new Vector2(50f, 50f);
711 this.rectTransform.pivot = new Vector2(0.5f, 0.5f);
712 return false;
713 }
714
715 SetRectTransformBounds(combinedBounds);
716 return true;
717 }

719 private void SetRectTransformBounds (Bounds combinedBounds) {
720 Vector3 size = combinedBounds.size;
721 Vector3 center = combinedBounds.center;
722 Vector2 p = new Vector2(
723 0.5f - (center.x / size.x),
724 0.5f - (center.y / size.y)
725 );
726
727 SetRectTransformSize(this, size);
728 this.rectTransform.pivot = p;
729
730 foreach (Transform separatorPart in separatorParts) {
731 RectTransform separatorTransform = separatorPart.GetComponent<RectTransform>();
732 if (separatorTransform) {
733 SetRectTransformSize(separatorTransform, size);
734 separatorTransform.pivot = p;
735 }
736 }
737 foreach (SkeletonSubmeshGraphic submeshGraphic in submeshGraphics) {
738 SetRectTransformSize(submeshGraphic, size);
739 submeshGraphic.rectTransform.pivot = p;
740 }
741
742 this.referenceSize = size;
743 }
