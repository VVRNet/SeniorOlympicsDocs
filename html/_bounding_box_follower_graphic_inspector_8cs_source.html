70 public override void OnInspectorGUI () {
71
72#if !NEW_PREFAB_SYSTEM
73 bool isInspectingPrefab = (PrefabUtility.GetPrefabType(target) == PrefabType.Prefab);
74#else
75 bool isInspectingPrefab = false;
76#endif
77
78 // Note: when calling InitializeEditor() in OnEnable, it throws exception
79 // "SerializedObjectNotCreatableException: Object at index 0 is null".
80 InitializeEditor();
81
82 // Try to auto-assign SkeletonGraphic field.
83 if (skeletonGraphic.objectReferenceValue == null) {
84 SkeletonGraphic foundSkeletonGraphic = follower.GetComponentInParent<SkeletonGraphic>();
85 if (foundSkeletonGraphic != null)
86 Debug.Log("BoundingBoxFollowerGraphic automatically assigned: " + foundSkeletonGraphic.gameObject.name);
87 else if (Event.current.type == EventType.Repaint)
88 Debug.Log("No Spine GameObject detected. Make sure to set this GameObject as a child of the Spine GameObject; or set BoundingBoxFollowerGraphic's 'Skeleton Graphic' field in the inspector.");
89
90 skeletonGraphic.objectReferenceValue = foundSkeletonGraphic;
91 serializedObject.ApplyModifiedProperties();
92 InitializeEditor();
93 }
94
95 SkeletonGraphic skeletonGraphicValue = skeletonGraphic.objectReferenceValue as SkeletonGraphic;
96 if (skeletonGraphicValue != null && skeletonGraphicValue.gameObject == follower.gameObject) {
97 using (new EditorGUILayout.VerticalScope(EditorStyles.helpBox)) {
98 EditorGUILayout.HelpBox("It's ideal to add BoundingBoxFollowerGraphic to a separate child GameObject of the Spine GameObject.", MessageType.Warning);
99
100 if (GUILayout.Button(new GUIContent("Move BoundingBoxFollowerGraphic to new GameObject", Icons.boundingBox), GUILayout.Height(30f))) {
101 AddBoundingBoxFollowerGraphicChild(skeletonGraphicValue, follower);
102 DestroyImmediate(follower);
103 return;
104 }
105 }
106 EditorGUILayout.Space();
107 }
108
109 EditorGUI.BeginChangeCheck();
110 EditorGUILayout.PropertyField(skeletonGraphic);
111 EditorGUILayout.PropertyField(slotName, new GUIContent("Slot"));
112 if (EditorGUI.EndChangeCheck()) {
113 serializedObject.ApplyModifiedProperties();
114 InitializeEditor();
115#if !NEW_PREFAB_SYSTEM
116 if (!isInspectingPrefab)
117 rebuildRequired = true;
118#endif
119 }
120
121 using (new SpineInspectorUtility.LabelWidthScope(150f)) {
122 EditorGUI.BeginChangeCheck();
123 EditorGUILayout.PropertyField(isTrigger);
124 EditorGUILayout.PropertyField(usedByEffector);
125 EditorGUILayout.PropertyField(usedByComposite);
126 bool colliderParamChanged = EditorGUI.EndChangeCheck();
127
128 EditorGUI.BeginChangeCheck();
129 EditorGUILayout.PropertyField(clearStateOnDisable, new GUIContent(clearStateOnDisable.displayName, "Enable this if you are pooling your Spine GameObject"));
130 bool clearStateChanged = EditorGUI.EndChangeCheck();
131
132 if (clearStateChanged || colliderParamChanged) {
133 serializedObject.ApplyModifiedProperties();
134 InitializeEditor();
135 if (colliderParamChanged)
136 foreach (PolygonCollider2D col in follower.colliderTable.Values) {
137 col.isTrigger = isTrigger.boolValue;
138 col.usedByEffector = usedByEffector.boolValue;
139 col.usedByComposite = usedByComposite.boolValue;
140 }
141 }
142 }
143
144 if (isInspectingPrefab) {
145 follower.colliderTable.Clear();
146 follower.nameTable.Clear();
147 EditorGUILayout.HelpBox("BoundingBoxAttachments cannot be previewed in prefabs.", MessageType.Info);
148
149 // How do you prevent components from being saved into the prefab? No such HideFlag. DontSaveInEditor | DontSaveInBuild does not work. DestroyImmediate does not work.
150 PolygonCollider2D collider = follower.GetComponent<PolygonCollider2D>();
151 if (collider != null) Debug.LogWarning("Found BoundingBoxFollowerGraphic collider components in prefab. These are disposed and regenerated at runtime.");
152
153 } else {
154 using (new SpineInspectorUtility.BoxScope()) {
155 if (debugIsExpanded = EditorGUILayout.Foldout(debugIsExpanded, "Debug Colliders")) {
156 EditorGUI.indentLevel++;
157 EditorGUILayout.LabelField(string.Format("Attachment Names ({0} PolygonCollider2D)", follower.colliderTable.Count));
158 EditorGUI.BeginChangeCheck();
159 foreach (KeyValuePair<BoundingBoxAttachment, string> kp in follower.nameTable) {
160 string attachmentName = kp.Value;
161 PolygonCollider2D collider = follower.colliderTable[kp.Key];
162 bool isPlaceholder = attachmentName != kp.Key.Name;
163 collider.enabled = EditorGUILayout.ToggleLeft(new GUIContent(!isPlaceholder ? attachmentName : string.Format("{0} [{1}]", attachmentName, kp.Key.Name), isPlaceholder ? Icons.skinPlaceholder : Icons.boundingBox), collider.enabled);
164 }
165 sceneRepaintRequired |= EditorGUI.EndChangeCheck();
166 EditorGUI.indentLevel--;
167 }
168 }
169
170 }
171
172 if (follower.Slot == null)
173 follower.Initialize(false);
174 bool hasBoneFollower = follower.GetComponent<BoneFollowerGraphic>() != null;
175 if (!hasBoneFollower) {
176 bool buttonDisabled = follower.Slot == null;
177 using (new EditorGUI.DisabledGroupScope(buttonDisabled)) {
178 addBoneFollower |= SpineInspectorUtility.LargeCenteredButton(AddBoneFollowerLabel, true);
179 EditorGUILayout.Space();
180 }
181 }
182
183
184 if (Event.current.type == EventType.Repaint) {
185 if (addBoneFollower) {
186 BoneFollowerGraphic boneFollower = follower.gameObject.AddComponent<BoneFollowerGraphic>();
187 boneFollower.skeletonGraphic = skeletonGraphicValue;
188 boneFollower.SetBone(follower.Slot.Data.BoneData.Name);
189 addBoneFollower = false;
190 }
191
192 if (sceneRepaintRequired) {
193 SceneView.RepaintAll();
194 sceneRepaintRequired = false;
195 }
196
197 if (rebuildRequired) {
198 follower.Initialize();
199 rebuildRequired = false;
200 }
201 }
202 }

219 public static GameObject AddBoundingBoxFollowerGraphicChild (SkeletonGraphic skeletonGraphic,
220 BoundingBoxFollowerGraphic original = null, string name = "BoundingBoxFollowerGraphic",
221 string slotName = null) {
222
223 GameObject go = EditorInstantiation.NewGameObject(name, true);
224 go.transform.SetParent(skeletonGraphic.transform, false);
225 go.AddComponent<RectTransform>();
226 BoundingBoxFollowerGraphic newFollower = go.AddComponent<BoundingBoxFollowerGraphic>();
227
228 if (original != null) {
229 newFollower.slotName = original.slotName;
230 newFollower.isTrigger = original.isTrigger;
231 newFollower.usedByEffector = original.usedByEffector;
232 newFollower.usedByComposite = original.usedByComposite;
233 newFollower.clearStateOnDisable = original.clearStateOnDisable;
234 }
235 if (slotName != null)
236 newFollower.slotName = slotName;
237
238 newFollower.skeletonGraphic = skeletonGraphic;
239 newFollower.Initialize();
240
241 Selection.activeGameObject = go;
242 EditorGUIUtility.PingObject(go);
243 return go;
244 }
245
246 public static List<GameObject> AddAllBoundingBoxFollowerGraphicChildren (
247 SkeletonGraphic skeletonGraphic, BoundingBoxFollowerGraphic original = null) {
248
249 List<GameObject> createdGameObjects = new List<GameObject>();
250 foreach (Skin skin in skeletonGraphic.Skeleton.Data.Skins) {
251 ICollection<Skin.SkinEntry> attachments = skin.Attachments;
252 foreach (Skin.SkinEntry entry in attachments) {
253 BoundingBoxAttachment boundingBoxAttachment = entry.Attachment as BoundingBoxAttachment;
254 if (boundingBoxAttachment == null)
255 continue;
256 int slotIndex = entry.SlotIndex;
257 Slot slot = skeletonGraphic.Skeleton.Slots.Items[slotIndex];
258 string slotName = slot.Data.Name;
259 GameObject go = AddBoundingBoxFollowerGraphicChild(skeletonGraphic,
260 original, boundingBoxAttachment.Name, slotName);
261 BoneFollowerGraphic boneFollower = go.AddComponent<BoneFollowerGraphic>();
262 boneFollower.skeletonGraphic = skeletonGraphic;
263 boneFollower.SetBone(slot.Data.BoneData.Name);
264 createdGameObjects.Add(go);
265 }
266 }
267 return createdGameObjects;
268 }
