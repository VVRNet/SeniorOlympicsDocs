81 protected bool IsValueValid (SerializedProperty property) {
82 if (skeletonDataAsset != null) {
83 SkeletonData skeletonData = skeletonDataAsset.GetSkeletonData(true);
84 if (skeletonData != null && !string.IsNullOrEmpty(property.stringValue))
85 return IsValueValid(skeletonData, property);

92 public override void OnGUI (Rect position, SerializedProperty property, GUIContent label) {
93 SerializedProperty = property;
94
95 if (property.propertyType != SerializedPropertyType.String) {
96 EditorGUI.LabelField(position, "ERROR:", "May only apply to type string");
97 return;
98 }
99
100 // Handle multi-editing when instances don't use the same SkeletonDataAsset.
101 if (!SpineInspectorUtility.TargetsUseSameData(property.serializedObject)) {
102 EditorGUI.DelayedTextField(position, property, label);
103 return;
104 }
105
106 SerializedProperty dataField = property.FindBaseOrSiblingProperty(TargetAttribute.dataField);
107
108 if (dataField != null) {
109 UnityEngine.Object objectReferenceValue = dataField.objectReferenceValue;
110 if (objectReferenceValue is SkeletonDataAsset) {
111 skeletonDataAsset = (SkeletonDataAsset)objectReferenceValue;
112 } else if (objectReferenceValue is IHasSkeletonDataAsset) {
113 IHasSkeletonDataAsset hasSkeletonDataAsset = (IHasSkeletonDataAsset)objectReferenceValue;
114 if (hasSkeletonDataAsset != null)
115 skeletonDataAsset = hasSkeletonDataAsset.SkeletonDataAsset;
116 } else if (objectReferenceValue != null) {
117 EditorGUI.LabelField(position, "ERROR:", "Invalid reference type");
118 return;
119 }
120
121 } else {
122 UnityEngine.Object targetObject = property.serializedObject.targetObject;
123
124 IHasSkeletonDataAsset hasSkeletonDataAsset = targetObject as IHasSkeletonDataAsset;
125 if (hasSkeletonDataAsset == null) {
126 Component component = targetObject as Component;
127 if (component != null)
128 hasSkeletonDataAsset = component.GetComponentInChildren(typeof(IHasSkeletonDataAsset)) as IHasSkeletonDataAsset;
129 }
130
131 if (hasSkeletonDataAsset != null)
132 skeletonDataAsset = hasSkeletonDataAsset.SkeletonDataAsset;
133 }
134
135 if (skeletonDataAsset == null) {
136 if (TargetAttribute.fallbackToTextField) {
137 EditorGUI.PropertyField(position, property); //EditorGUI.TextField(position, label, property.stringValue);
138 } else {
139 EditorGUI.LabelField(position, "ERROR:", "Must have reference to a SkeletonDataAsset");
140 }
141
142 skeletonDataAsset = property.serializedObject.targetObject as SkeletonDataAsset;
143 if (skeletonDataAsset == null) return;
144 }
145
146 position = EditorGUI.PrefixLabel(position, label);
147
148 Texture2D image = Icon;
149 GUIStyle usedStyle = IsValueValid(property) ? EditorStyles.popup : ErrorPopupStyle;
150 string propertyStringValue = (property.hasMultipleDifferentValues) ? SpineInspectorUtility.EmDash : property.stringValue;
151 if (GUI.Button(position, string.IsNullOrEmpty(propertyStringValue) ? NoneLabel(image) :
152 SpineInspectorUtility.TempContent(propertyStringValue, image), usedStyle))
153 Selector(property);
154 }
155
156 public ISkeletonComponent GetTargetSkeletonComponent (SerializedProperty property) {
157 SerializedProperty dataField = property.FindBaseOrSiblingProperty(TargetAttribute.dataField);
158
159 if (dataField != null) {
160 ISkeletonComponent skeletonComponent = dataField.objectReferenceValue as ISkeletonComponent;
161 if (dataField.objectReferenceValue != null && skeletonComponent != null) // note the overloaded UnityEngine.Object == null check. Do not simplify.
162 return skeletonComponent;
163 } else {
164 Component component = property.serializedObject.targetObject as Component;
165 if (component != null)
166 return component.GetComponentInChildren(typeof(ISkeletonComponent)) as ISkeletonComponent;
167 }
168
169 return null;
170 }
171
172 protected virtual void Selector (SerializedProperty property) {
173 SkeletonData data = skeletonDataAsset.GetSkeletonData(true);
174 if (data == null) return;
175
176 GenericMenu menu = new GenericMenu();
177 PopulateMenu(menu, property, this.TargetAttribute, data);
178 menu.ShowAsContext();
179 }

184 SpineDrawerValuePair clickedItem = (SpineDrawerValuePair)menuItemObject;
185 SerializedProperty serializedProperty = clickedItem.property;
186 if (serializedProperty.serializedObject.isEditingMultipleObjects) serializedProperty.stringValue = "oaifnoiasf��123526"; // HACK: to trigger change on multi-editing.
187 serializedProperty.stringValue = clickedItem.stringValue;
188 serializedProperty.serializedObject.ApplyModifiedProperties();
189 }

207 if (TargetAttribute.includeNone)
208 menu.AddItem(new GUIContent(NoneString), !property.hasMultipleDifferentValues && string.IsNullOrEmpty(property.stringValue), HandleSelect, new SpineDrawerValuePair(string.Empty, property));
209
210 IEnumerable<SlotData> orderedSlots = data.Slots.Items.OrderBy(slotData => slotData.Name);
211 foreach (SlotData slotData in orderedSlots) {
212 int slotIndex = slotData.Index;
213 string name = slotData.Name;
214 if (name.StartsWith(targetAttribute.startsWith, StringComparison.Ordinal)) {
215
216 if (targetAttribute.containsBoundingBoxes) {
217 List<Skin.SkinEntry> skinEntries = new List<Skin.SkinEntry>();
218 foreach (Skin skin in data.Skins) {
219 skin.GetAttachments(slotIndex, skinEntries);
220 }
221
222 bool hasBoundingBox = false;
223 foreach (Skin.SkinEntry entry in skinEntries) {
224 BoundingBoxAttachment bbAttachment = entry.Attachment as BoundingBoxAttachment;
225 if (bbAttachment != null) {
226 string menuLabel = bbAttachment.IsWeighted() ? name + " (!)" : name;
227 menu.AddItem(new GUIContent(menuLabel), !property.hasMultipleDifferentValues && name == property.stringValue, HandleSelect, new SpineDrawerValuePair(name, property));
228 hasBoundingBox = true;
229 break;
230 }
231 }
232
233 if (!hasBoundingBox)
234 menu.AddDisabledItem(new GUIContent(name));
235
236 } else {
237 menu.AddItem(new GUIContent(name), !property.hasMultipleDifferentValues && name == property.stringValue, HandleSelect, new SpineDrawerValuePair(name, property));
238 }

240 }

472
473 protected override void PopulateMenu (GenericMenu menu, SerializedProperty property, SpineAttachment targetAttribute, SkeletonData data) {
474 ISkeletonComponent skeletonComponent = GetTargetSkeletonComponent(property);
475 List<Skin> validSkins = new List<Skin>();
476
477 if (skeletonComponent != null && targetAttribute.currentSkinOnly) {
478 Skin currentSkin = null;
479
480 SerializedProperty skinProperty = property.FindBaseOrSiblingProperty(targetAttribute.skinField);
481 if (skinProperty != null) currentSkin = skeletonComponent.Skeleton.Data.FindSkin(skinProperty.stringValue);
482
483 currentSkin = currentSkin ?? skeletonComponent.Skeleton.Skin;
484 if (currentSkin != null)
485 validSkins.Add(currentSkin);
486 else
487 validSkins.Add(data.Skins.Items[0]);
488
489 } else {
490 foreach (Skin skin in data.Skins)
491 if (skin != null) validSkins.Add(skin);
492 }
493
494 List<string> attachmentNames = new List<string>();
495 List<string> placeholderNames = new List<string>();
496 string prefix = "";
497
498 if (skeletonComponent != null && targetAttribute.currentSkinOnly)
499 menu.AddDisabledItem(new GUIContent((skeletonComponent as Component).gameObject.name + " (Skeleton)"));
500 else
501 menu.AddDisabledItem(new GUIContent(skeletonDataAsset.name));
502
503 menu.AddSeparator("");
504 if (TargetAttribute.includeNone) {
505 const string NullAttachmentName = "";
506 menu.AddItem(new GUIContent("Null"), !property.hasMultipleDifferentValues && property.stringValue == NullAttachmentName, HandleSelect, new SpineDrawerValuePair(NullAttachmentName, property));
507 menu.AddSeparator("");
508 }
509
510 Skin defaultSkin = data.Skins.Items[0];
511 SerializedProperty slotProperty = property.FindBaseOrSiblingProperty(TargetAttribute.slotField);
512
513 string slotMatch = "";
514 if (slotProperty != null) {
515 if (slotProperty.propertyType == SerializedPropertyType.String)
516 slotMatch = slotProperty.stringValue.ToLower();
517 }
518
519 foreach (Skin skin in validSkins) {
520 string skinPrefix = skin.Name + "/";
521
522 if (validSkins.Count > 1)
523 prefix = skinPrefix;
524
525 for (int i = 0; i < data.Slots.Count; i++) {
526 if (slotMatch.Length > 0 && !(data.Slots.Items[i].Name.Equals(slotMatch, StringComparison.OrdinalIgnoreCase)))
527 continue;
528
529 attachmentNames.Clear();
530 placeholderNames.Clear();
531
532 List<Skin.SkinEntry> skinEntries = new List<Skin.SkinEntry>();
533 skin.GetAttachments(i, skinEntries);
534 foreach (Skin.SkinEntry entry in skinEntries) {
535 attachmentNames.Add(entry.Name);
536 }
537
538 if (skin != defaultSkin) {
539 foreach (Skin.SkinEntry entry in skinEntries) {
540 placeholderNames.Add(entry.Name);
541 }
542 skinEntries.Clear();
543 defaultSkin.GetAttachments(i, skinEntries);
544 foreach (Skin.SkinEntry entry in skinEntries) {
545 attachmentNames.Add(entry.Name);
546 }
547 }
548
549 for (int a = 0; a < attachmentNames.Count; a++) {
550 string attachmentPath = attachmentNames[a];
551 string menuPath = prefix + data.Slots.Items[i].Name + "/" + attachmentPath;
552 string name = attachmentNames[a];
553
554 if (targetAttribute.returnAttachmentPath)
555 name = skin.Name + "/" + data.Slots.Items[i].Name + "/" + attachmentPath;
556
557 if (targetAttribute.placeholdersOnly && !placeholderNames.Contains(attachmentPath)) {
558 menu.AddDisabledItem(new GUIContent(menuPath));
559 } else {
560 menu.AddItem(new GUIContent(menuPath), !property.hasMultipleDifferentValues && name == property.stringValue, HandleSelect, new SpineDrawerValuePair(name, property));
561 }
562 }
563
564 }
