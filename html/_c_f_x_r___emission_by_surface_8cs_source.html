53 void CalculateAndUpdateEmission()
54 {
55 if (!active) return;
56 if (this == null) return;
57 if (ps == null) ps = this.GetComponent<ParticleSystem>();
58 density = CalculateShapeDensity(ps.shape, ps.main.scalingMode == ParticleSystemScalingMode.Shape, this.transform);
59 if (density == 0) return;
60 float emissionOverTime = density * particlesPerUnit;
61 ParticleSystem.EmissionModule emission = ps.emission;
62 if (Math.Abs(emission.rateOverTime.constant - emissionOverTime) > 0.1f)
63 {
64 emission.rateOverTime = Mathf.Min(maxEmissionRate, emissionOverTime);
65 }
66 }

68 float CalculateShapeDensity(ParticleSystem.ShapeModule shapeModule, bool isShapeScaling, Transform transform)
69 {
70 float arcPercentage = Mathf.Max(0.01f, shapeModule.arc / 360f);
71 float thicknessPercentage = Mathf.Max(0.01f, 1.0f - shapeModule.radiusThickness);
72
73 float scaleX = shapeModule.scale.x;
74 float scaleY = shapeModule.scale.y;
75 float scaleZ = shapeModule.scale.z;
76 if (isShapeScaling)
77 {
78 Vector3 localScale = Quaternion.Euler(shapeModule.rotation) * transform.localScale;
79 scaleX = scaleX * localScale.x;
80 scaleY = scaleY * localScale.y;
81 scaleZ = scaleZ * localScale.z;
82 }
83 scaleX = Mathf.Abs(scaleX);
84 scaleY = Mathf.Abs(scaleY);
85 scaleZ = Mathf.Abs(scaleZ);
86
87 switch (shapeModule.shapeType)
88 {
89 case ParticleSystemShapeType.Hemisphere:
90 case ParticleSystemShapeType.Sphere:
91 {
92 float rX = shapeModule.radius * scaleX;
93 float rY = shapeModule.radius * scaleY;
94 float rZ = shapeModule.radius * scaleZ;
95 float rmX = rX * thicknessPercentage;
96 float rmY = rY * thicknessPercentage;
97 float rmZ = rZ * thicknessPercentage;
98 float volume = (rX * rY * rZ - rmX * rmY * rmZ) * Mathf.PI;
99 if (shapeModule.shapeType == ParticleSystemShapeType.Hemisphere)
100 {
101 volume /= 2.0f;
102 }
103 return volume * arcPercentage;
104 }
105 case ParticleSystemShapeType.Cone:
106 {
107 float innerDisk = shapeModule.radius * scaleX * thicknessPercentage * shapeModule.radius * scaleY * thicknessPercentage * Mathf.PI;
108 float outerDisk = shapeModule.radius *scaleX * shapeModule.radius * scaleY * Mathf.PI;
109 return outerDisk - innerDisk;
110 }
111 case ParticleSystemShapeType.ConeVolume:
112 {
113 // cylinder volume, changing the angle doesn't actually extend the area from where the particles are emitted
114 float innerCylinder = shapeModule.radius * scaleX * thicknessPercentage * shapeModule.radius * scaleY * thicknessPercentage * Mathf.PI * shapeModule.length * scaleZ;
115 float outerCylinder = shapeModule.radius * scaleX * shapeModule.radius * scaleY * Mathf.PI * shapeModule.length * scaleZ;
116 return outerCylinder - innerCylinder;
117 }
118 case ParticleSystemShapeType.BoxEdge:
119 case ParticleSystemShapeType.BoxShell:
120 case ParticleSystemShapeType.Box:
121 {
122 return scaleX * scaleY * scaleZ;
123 }
124 case ParticleSystemShapeType.Circle:
125 {
126 float radiusX = shapeModule.radius * scaleX;
127 float radiusY = shapeModule.radius * scaleY;
128
129 float radiusMinX = radiusX * thicknessPercentage;
130 float radiusMinY = radiusY * thicknessPercentage;
131 float area = (radiusX * radiusY - radiusMinX * radiusMinY) * Mathf.PI;
132 return area * arcPercentage;
133 }
134 case ParticleSystemShapeType.SingleSidedEdge:
135 {
136 return shapeModule.radius * scaleX;
137 }
138 case ParticleSystemShapeType.Donut:
139 {
140 float outerDonutVolume = 2 * Mathf.PI * Mathf.PI * shapeModule.donutRadius * shapeModule.donutRadius * shapeModule.radius * arcPercentage;
141 float innerDonutVolume = 2 * Mathf.PI * Mathf.PI * shapeModule.donutRadius * thicknessPercentage * thicknessPercentage * shapeModule.donutRadius * shapeModule.radius * arcPercentage;
142 return (outerDonutVolume - innerDonutVolume) * scaleX * scaleY * scaleZ;
143 }
144 case ParticleSystemShapeType.Rectangle:
145 {
146 return scaleX * scaleY;
147 }
148 case ParticleSystemShapeType.Mesh:
149 case ParticleSystemShapeType.SkinnedMeshRenderer:
150 case ParticleSystemShapeType.MeshRenderer:
151 {
152 Debug.LogWarning( string.Format("[{0}] Calculating volume for a mesh is unsupported.", nameof(CFXR_EmissionBySurface)));
153 this.active = false;
154 return 0;
155 }
156 case ParticleSystemShapeType.Sprite:
157 case ParticleSystemShapeType.SpriteRenderer:
158 {
159 Debug.LogWarning( string.Format("[{0}] Calculating volume for a sprite is unsupported.", nameof(CFXR_EmissionBySurface)));
160 this.active = false;
161 return 0;
162 }
163 }
164
165 return 0;
166 }
