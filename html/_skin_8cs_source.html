66 public void AddSkin (Skin skin) {
67 foreach (BoneData data in skin.bones)
68 if (!bones.Contains(data)) bones.Add(data);
69
70 foreach (ConstraintData data in skin.constraints)
71 if (!constraints.Contains(data)) constraints.Add(data);
72
73 foreach (KeyValuePair<SkinKey, SkinEntry> item in skin.attachments) {
74 SkinEntry entry = item.Value;
75 SetAttachment(entry.slotIndex, entry.name, entry.attachment);
76 }
77 }

80 public void CopySkin (Skin skin) {
81 foreach (BoneData data in skin.bones)
82 if (!bones.Contains(data)) bones.Add(data);
83
84 foreach (ConstraintData data in skin.constraints)
85 if (!constraints.Contains(data)) constraints.Add(data);
86
87 foreach (KeyValuePair<SkinKey, SkinEntry> item in skin.attachments) {
88 SkinEntry entry = item.Value;
89 if (entry.attachment is MeshAttachment) {
90 SetAttachment(entry.slotIndex, entry.name,
91 entry.attachment != null ? ((MeshAttachment)entry.attachment).NewLinkedMesh() : null);
92 } else
93 SetAttachment(entry.slotIndex, entry.name, entry.attachment != null ? entry.attachment.Copy() : null);
94 }

133 internal void AttachAll (Skeleton skeleton, Skin oldSkin) {
134 Slot[] slots = skeleton.slots.Items;
135 foreach (KeyValuePair<SkinKey, SkinEntry> item in oldSkin.attachments) {
136 SkinEntry entry = item.Value;
137 int slotIndex = entry.slotIndex;
138 Slot slot = slots[slotIndex];
139 if (slot.Attachment == entry.attachment) {
140 Attachment attachment = GetAttachment(slotIndex, entry.name);
141 if (attachment != null) slot.Attachment = attachment;
142 }
143 }
