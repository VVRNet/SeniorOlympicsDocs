48 public float GetValue(float delta)
49 {
50 var value = 0.0f;
51 var fingers = GetFingers(true, true);
52 var scale = 1.0f;
53
54 value -= CwInput.GetKeyIsHeld(KeyNegative) == true ? KeySensitivity * delta : 0.0f;
55 value += CwInput.GetKeyIsHeld(KeyPositive) == true ? KeySensitivity * delta : 0.0f;
56
57 value -= CwInput.GetKeyIsHeld(KeyNegativeAlt) == true ? KeySensitivity * delta : 0.0f;
58 value += CwInput.GetKeyIsHeld(KeyPositiveAlt) == true ? KeySensitivity * delta : 0.0f;
59
60 if (FingerCount > 0 && fingers.Count == FingerCount)
61 {
62 if (FingerInvert == true && fingers[0].Index >= 0)
63 {
64 scale = -1.0f;
65 }
66
67 switch (FingerGesture)
68 {
69 case AxisGesture.HorizontalDrag: value += GetAverageDeltaScaled(fingers).x * FingerSensitivity * scale; break;
70 case AxisGesture.VerticalDrag: value += GetAverageDeltaScaled(fingers).y * FingerSensitivity * scale; break;
71 case AxisGesture.Twist: value += GetAverageTwistRadians(fingers) * FingerSensitivity; break;
72 case AxisGesture.HorizontalPull: value += GetAveragePullScaled(fingers).x * FingerSensitivity * delta * scale; break;
73 case AxisGesture.VerticalPull: value += GetAveragePullScaled(fingers).y * FingerSensitivity * delta * scale; break;
74 }
75 }
76
77 return value;

359 {
360 tempRaycastResults.Clear();
361
362 var currentEventSystem = EventSystem.current;
363
364 if (currentEventSystem != null)
365 {
366 // Create point event data for this event system?
367 if (currentEventSystem != tempEventSystem)
368 {
369 tempEventSystem = currentEventSystem;
370
371 if (tempPointerEventData == null)
372 {
373 tempPointerEventData = new PointerEventData(tempEventSystem);
374 }
375 else
376 {
377 tempPointerEventData.Reset();
378 }
379 }
380
381 // Raycast event system at the specified point
382 tempPointerEventData.position = screenPosition;
383
384 currentEventSystem.RaycastAll(tempPointerEventData, tempRaycastResults);
385
386 // Loop through all results and remove any that don't match the layer mask
387 if (tempRaycastResults.Count > 0)
388 {
389 for (var i = tempRaycastResults.Count - 1; i >= 0; i--)
390 {
391 var raycastResult = tempRaycastResults[i];
392 var raycastLayer = 1 << raycastResult.gameObject.layer;
393
394 if ((raycastLayer & guiLayers) == 0)
395 {
396 tempRaycastResults.RemoveAt(i);
397 }
398 }
399 }
400 }

496 for (var i = 0; i < CwInput.GetTouchCount(); i++)
497 {
498 int id; Vector2 position; float pressure; bool set;

500 CwInput.GetTouch(i, out id, out position, out pressure, out set);

502 AddFinger(id, position, pressure, set);
503 }
504 }
505 // If there are no real touches, simulate some from the mouse?
506 else if (CwInput.GetMouseExists() == true)
507 {
508 var mouseSet = false;
509 var mouseUp = false;

582 private static Vector2 Hermite(Vector2 a, Vector2 b, Vector2 c, Vector2 d, float t)
583 {
584 var mu2 = t * t;
585 var mu3 = mu2 * t;
586 var x = HermiteInterpolate(a.x, b.x, c.x, d.x, t, mu2, mu3);
587 var y = HermiteInterpolate(a.y, b.y, c.y, d.y, t, mu2, mu3);
588
589 return new Vector2(x, y);
590 }
591
592 private static float HermiteInterpolate(float y0,float y1, float y2,float y3, float mu, float mu2, float mu3)
593 {
594 var m0 = (y1 - y0) * 0.5f + (y2 - y1) * 0.5f;
595 var m1 = (y2 - y1) * 0.5f + (y3 - y2) * 0.5f;
596 var a0 = 2.0f * mu3 - 3.0f * mu2 + 1.0f;
597 var a1 = mu3 - 2.0f * mu2 + mu;
598 var a2 = mu3 - mu2;
599 var a3 = -2.0f * mu3 + 3.0f * mu2;
600
601 return(a0*y1+a1*m0+a2*m1+a3*y2);
602 }

606 return Mathf.Atan2(screenPosition.x - referencePoint.x, screenPosition.y - referencePoint.y);

609 private static float GetDeltaRadians(Finger finger, Vector2 referencePoint, Vector2 lastReferencePoint)
610 {
611 var a = GetRadians(finger.ScreenPositionOld, lastReferencePoint);
612 var b = GetRadians(finger.ScreenPosition, referencePoint);
613 var d = Mathf.Repeat(a - b, Mathf.PI * 2.0f);
614
615 if (d > Mathf.PI)
616 {
617 d -= Mathf.PI * 2.0f;
618 }
619
620 return d;
