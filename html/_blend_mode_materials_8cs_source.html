59 public BlendMode BlendModeForMaterial (Material material) {
60 foreach (ReplacementMaterial pair in multiplyMaterials)
61 if (pair.material == material) return BlendMode.Multiply;
62 foreach (ReplacementMaterial pair in additiveMaterials)
63 if (pair.material == material) return BlendMode.Additive;
64 foreach (ReplacementMaterial pair in screenMaterials)
65 if (pair.material == material) return BlendMode.Screen;
66 return BlendMode.Normal;

75 public bool UpdateBlendmodeMaterialsRequiredState (SkeletonData skeletonData) {
76 requiresBlendModeMaterials = false;
77
78 if (skeletonData == null) return false;
79
80 List<Skin.SkinEntry> skinEntries = new List<Skin.SkinEntry>();
81 SlotData[] slotsItems = skeletonData.Slots.Items;
82 for (int slotIndex = 0, slotCount = skeletonData.Slots.Count; slotIndex < slotCount; slotIndex++) {
83 SlotData slot = slotsItems[slotIndex];
84 if (slot.BlendMode == BlendMode.Normal) continue;
85 if (!applyAdditiveMaterial && slot.BlendMode == BlendMode.Additive) continue;
86
87 skinEntries.Clear();
88 foreach (Skin skin in skeletonData.Skins)
89 skin.GetAttachments(slotIndex, skinEntries);
90
91 foreach (Skin.SkinEntry entry in skinEntries) {
92 if (entry.Attachment is IHasTextureRegion) {
93 requiresBlendModeMaterials = true;
94 return true;
95 }
96 }
97 }
98 return false;
99 }

121 public static bool CreateAndAssignMaterials (SkeletonDataAsset skeletonDataAsset,
122 TemplateMaterials templateMaterials, ref bool anyReplacementMaterialsChanged,
123 System.Action<SkeletonDataAsset> clearSkeletonDataAssetFunc,
124 System.Action<SkeletonDataAsset> afterAssetModifiedFunc,
125 CreateForRegionDelegate createForRegionFunc) {
126
127 bool anyCreationFailed = false;
128 BlendModeMaterials blendModeMaterials = skeletonDataAsset.blendModeMaterials;
129 bool applyAdditiveMaterial = blendModeMaterials.applyAdditiveMaterial;
130
131 List<Skin.SkinEntry> skinEntries = new List<Skin.SkinEntry>();
132
133 clearSkeletonDataAssetFunc(skeletonDataAsset);
134 skeletonDataAsset.isUpgradingBlendModeMaterials = true;
135 SkeletonData skeletonData = skeletonDataAsset.GetSkeletonData(true);
136
137 SlotData[] slotsItems = skeletonData.Slots.Items;
138 for (int slotIndex = 0, slotCount = skeletonData.Slots.Count; slotIndex < slotCount; slotIndex++) {
139 SlotData slot = slotsItems[slotIndex];
140 if (slot.BlendMode == BlendMode.Normal) continue;
141 if (!applyAdditiveMaterial && slot.BlendMode == BlendMode.Additive) continue;
142
143 List<BlendModeMaterials.ReplacementMaterial> replacementMaterials = null;
144 Material materialTemplate = null;
145 string materialSuffix = null;
146 switch (slot.BlendMode) {
147 case BlendMode.Multiply:
148 replacementMaterials = blendModeMaterials.multiplyMaterials;
149 materialTemplate = templateMaterials.multiplyTemplate;
150 materialSuffix = MATERIAL_SUFFIX_MULTIPLY;
151 break;
152 case BlendMode.Screen:
153 replacementMaterials = blendModeMaterials.screenMaterials;
154 materialTemplate = templateMaterials.screenTemplate;
155 materialSuffix = MATERIAL_SUFFIX_SCREEN;
156 break;
157 case BlendMode.Additive:
158 replacementMaterials = blendModeMaterials.additiveMaterials;
159 materialTemplate = templateMaterials.additiveTemplate;
160 materialSuffix = MATERIAL_SUFFIX_ADDITIVE;
161 break;
162 }
163
164 skinEntries.Clear();
165 foreach (Skin skin in skeletonData.Skins)
166 skin.GetAttachments(slotIndex, skinEntries);
167
168 foreach (Skin.SkinEntry entry in skinEntries) {
169 IHasTextureRegion renderableAttachment = entry.Attachment as IHasTextureRegion;
170 if (renderableAttachment != null) {
171 AtlasRegion originalRegion = (AtlasRegion)renderableAttachment.Region;
172 if (originalRegion != null) {
173 anyCreationFailed |= createForRegionFunc(
174 ref replacementMaterials, ref anyReplacementMaterialsChanged,
175 originalRegion, materialTemplate, materialSuffix, skeletonDataAsset);
176 } else {
177 Sequence sequence = renderableAttachment.Sequence;
178 if (sequence != null && sequence.Regions != null) {
179 for (int i = 0, count = sequence.Regions.Length; i < count; ++i) {
180 originalRegion = (AtlasRegion)sequence.Regions[i];
181 anyCreationFailed |= createForRegionFunc(
182 ref replacementMaterials, ref anyReplacementMaterialsChanged,
183 originalRegion, materialTemplate, materialSuffix, skeletonDataAsset);
184 }
185 }
186 }
187 }
188 }
189 }
190 skeletonDataAsset.isUpgradingBlendModeMaterials = false;
191 if (afterAssetModifiedFunc != null) afterAssetModifiedFunc(skeletonDataAsset);
192 return !anyCreationFailed;
193 }

195 protected static bool CreateForRegion (ref List<BlendModeMaterials.ReplacementMaterial> replacementMaterials,
196 ref bool anyReplacementMaterialsChanged,
197 AtlasRegion originalRegion, Material materialTemplate, string materialSuffix,
198 SkeletonDataAsset skeletonDataAsset) {
199
200 bool anyCreationFailed = false;
201 bool replacementExists = replacementMaterials.Exists(
202 replacement => replacement.pageName == originalRegion.page.name);
203 if (!replacementExists) {
204 BlendModeMaterials.ReplacementMaterial replacement = CreateReplacementMaterial(originalRegion, materialTemplate, materialSuffix);
205 if (replacement != null) {
206 replacementMaterials.Add(replacement);
207 anyReplacementMaterialsChanged = true;
208 } else {
209 Debug.LogError(string.Format("Failed creating blend mode Material for SkeletonData asset '{0}'," +
210 " atlas page '{1}', template '{2}'.",
211 skeletonDataAsset.name, originalRegion.page.name, materialTemplate.name),
212 skeletonDataAsset);
213 anyCreationFailed = true;
214 }
215 }
216 return anyCreationFailed;
217 }

219 protected static BlendModeMaterials.ReplacementMaterial CreateReplacementMaterial (
220 AtlasRegion originalRegion, Material materialTemplate, string materialSuffix) {
221
222 BlendModeMaterials.ReplacementMaterial newReplacement = new BlendModeMaterials.ReplacementMaterial();
223 AtlasPage originalPage = originalRegion.page;
224 Material originalMaterial = originalPage.rendererObject as Material;
225
226 newReplacement.pageName = originalPage.name;
227
228 Material blendModeMaterial = new Material(materialTemplate) {
229 name = originalMaterial.name + " " + materialTemplate.name,
230 mainTexture = originalMaterial.mainTexture
231 };
232 newReplacement.material = blendModeMaterial;
233
234 if (newReplacement.material)
235 return newReplacement;
236 else
237 return null;

240 public void ApplyMaterials (SkeletonData skeletonData) {
241 if (skeletonData == null) throw new ArgumentNullException("skeletonData");
242 if (!requiresBlendModeMaterials)
243 return;
244
245 List<Skin.SkinEntry> skinEntries = new List<Skin.SkinEntry>();
246 SlotData[] slotsItems = skeletonData.Slots.Items;
247 for (int slotIndex = 0, slotCount = skeletonData.Slots.Count; slotIndex < slotCount; slotIndex++) {
248 SlotData slot = slotsItems[slotIndex];
249 if (slot.BlendMode == BlendMode.Normal) continue;
250 if (!applyAdditiveMaterial && slot.BlendMode == BlendMode.Additive) continue;
251
252 List<ReplacementMaterial> replacementMaterials = null;
253 switch (slot.BlendMode) {
254 case BlendMode.Multiply:
255 replacementMaterials = multiplyMaterials;
256 break;
257 case BlendMode.Screen:
258 replacementMaterials = screenMaterials;
259 break;
260 case BlendMode.Additive:
261 replacementMaterials = additiveMaterials;
262 break;
263 }
264 if (replacementMaterials == null)
265 continue;
266
267 skinEntries.Clear();
268 foreach (Skin skin in skeletonData.Skins)
269 skin.GetAttachments(slotIndex, skinEntries);
270
271 foreach (Skin.SkinEntry entry in skinEntries) {
272 IHasTextureRegion renderableAttachment = entry.Attachment as IHasTextureRegion;
273 if (renderableAttachment != null) {
274 if (renderableAttachment.Region != null) {
275 renderableAttachment.Region = CloneAtlasRegionWithMaterial(
276 (AtlasRegion)renderableAttachment.Region, replacementMaterials);
277 } else {
278 if (renderableAttachment.Sequence != null) {
279 TextureRegion[] regions = renderableAttachment.Sequence.Regions;
280 for (int i = 0; i < regions.Length; ++i) {
281 regions[i] = CloneAtlasRegionWithMaterial(
282 (AtlasRegion)regions[i], replacementMaterials);
283 }
284 }
285 }
286 }
287 }
288 }
289 }

291 protected AtlasRegion CloneAtlasRegionWithMaterial (AtlasRegion originalRegion, List<ReplacementMaterial> replacementMaterials) {
292 AtlasRegion newRegion = originalRegion.Clone();
293 Material material = null;
294 foreach (ReplacementMaterial replacement in replacementMaterials) {
295 if (replacement.pageName == originalRegion.page.name) {
296 material = replacement.material;
297 break;
298 }
299 }
300
301 AtlasPage originalPage = originalRegion.page;
302 AtlasPage newPage = originalPage.Clone();
303 newPage.rendererObject = material;
304 newRegion.page = newPage;
305 return newRegion;
306 }
