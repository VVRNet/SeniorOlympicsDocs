35 // This is an example of an animation handle. This is implemented with strings as state names.
36 // Strings can serve as the identifier when Mecanim is used as the state machine and state source.
37 // If you don't use Mecanim, using custom ScriptableObjects may be a more efficient way to store information about the state and its connection with specific Spine animations.
38
39 // This animation handle implementation also comes with a dummy implementation of transition-handling.
40 public class SkeletonAnimationHandleExample : MonoBehaviour {
41 public SkeletonAnimation skeletonAnimation;
42 public List<StateNameToAnimationReference> statesAndAnimations = new List<StateNameToAnimationReference>();
43 public List<AnimationTransition> transitions = new List<AnimationTransition>(); // Alternately, an AnimationPair-Animation Dictionary (commented out) can be used for more efficient lookups.

106 public Spine.Animation GetAnimationForState (int shortNameHash) {
107 StateNameToAnimationReference foundState = statesAndAnimations.Find(entry => StringToHash(entry.stateName) == shortNameHash);
108 return (foundState == null) ? null : foundState.animation;
109 }

112 public void PlayNewAnimation (Spine.Animation target, int layerIndex) {
113 Spine.Animation transition = null;
114 Spine.Animation current = null;
115
116 current = GetCurrentAnimation(layerIndex);
117 if (current != null)
118 transition = TryGetTransition(current, target);
119
120 if (transition != null) {
121 skeletonAnimation.AnimationState.SetAnimation(layerIndex, transition, false);
122 skeletonAnimation.AnimationState.AddAnimation(layerIndex, target, true, 0f);
123 } else {
124 skeletonAnimation.AnimationState.SetAnimation(layerIndex, target, true);
125 }
126
127 this.TargetAnimation = target;
128 }
129
131 public void PlayOneShot (Spine.Animation oneShot, int layerIndex) {
132 AnimationState state = skeletonAnimation.AnimationState;
133 state.SetAnimation(0, oneShot, false);
134
135 Animation transition = TryGetTransition(oneShot, TargetAnimation);
136 if (transition != null)
137 state.AddAnimation(0, transition, false, 0f);
138
139 state.AddAnimation(0, this.TargetAnimation, true, 0f);
140 }
141
142 Spine.Animation TryGetTransition (Spine.Animation from, Spine.Animation to) {
143 foreach (AnimationTransition transition in transitions) {
144 if (transition.from.Animation == from && transition.to.Animation == to) {
145 return transition.transition.Animation;
146 }
147 }
148 return null;

155 Spine.Animation GetCurrentAnimation (int layerIndex) {
156 TrackEntry currentTrackEntry = skeletonAnimation.AnimationState.GetCurrent(layerIndex);
157 return (currentTrackEntry != null) ? currentTrackEntry.Animation : null;
158 }
