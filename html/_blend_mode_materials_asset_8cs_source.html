50 public static void ApplyMaterials (SkeletonData skeletonData, Material multiplyTemplate, Material screenTemplate, Material additiveTemplate, bool includeAdditiveSlots) {
51 if (skeletonData == null) throw new ArgumentNullException("skeletonData");
52
53 using (AtlasMaterialCache materialCache = new AtlasMaterialCache()) {
54 List<Skin.SkinEntry> entryBuffer = new List<Skin.SkinEntry>();
55 SlotData[] slotsItems = skeletonData.Slots.Items;
56 for (int slotIndex = 0, slotCount = skeletonData.Slots.Count; slotIndex < slotCount; slotIndex++) {
57 SlotData slot = slotsItems[slotIndex];
58 if (slot.BlendMode == BlendMode.Normal) continue;
59 if (!includeAdditiveSlots && slot.BlendMode == BlendMode.Additive) continue;
60
61 entryBuffer.Clear();
62 foreach (Skin skin in skeletonData.Skins)
63 skin.GetAttachments(slotIndex, entryBuffer);
64
65 Material templateMaterial = null;
66 switch (slot.BlendMode) {
67 case BlendMode.Multiply:
68 templateMaterial = multiplyTemplate;
69 break;
70 case BlendMode.Screen:
71 templateMaterial = screenTemplate;
72 break;
73 case BlendMode.Additive:
74 templateMaterial = additiveTemplate;
75 break;
76 }
77 if (templateMaterial == null) continue;
78
79 foreach (Skin.SkinEntry entry in entryBuffer) {
80 IHasTextureRegion renderableAttachment = entry.Attachment as IHasTextureRegion;
81 if (renderableAttachment != null) {
82 renderableAttachment.Region = materialCache.CloneAtlasRegionWithMaterial(
83 (AtlasRegion)renderableAttachment.Region, templateMaterial);
84 }
85 }
86 }
87
88 }
89 //attachmentBuffer.Clear();
90 }

92 class AtlasMaterialCache : IDisposable {
93 readonly Dictionary<KeyValuePair<AtlasPage, Material>, AtlasPage> cache = new Dictionary<KeyValuePair<AtlasPage, Material>, AtlasPage>();
94
96 public AtlasRegion CloneAtlasRegionWithMaterial (AtlasRegion originalRegion, Material materialTemplate) {
97 AtlasRegion newRegion = originalRegion.Clone();
98 newRegion.page = GetAtlasPageWithMaterial(originalRegion.page, materialTemplate);
99 return newRegion;
100 }
101
102 AtlasPage GetAtlasPageWithMaterial (AtlasPage originalPage, Material materialTemplate) {
103 if (originalPage == null) throw new ArgumentNullException("originalPage");
104
105 AtlasPage newPage = null;
106 var key = new KeyValuePair<AtlasPage, Material>(originalPage, materialTemplate);
107 cache.TryGetValue(key, out newPage);
108
109 if (newPage == null) {
110 newPage = originalPage.Clone();
111 Material originalMaterial = originalPage.rendererObject as Material;
112 newPage.rendererObject = new Material(materialTemplate) {
113 name = originalMaterial.name + " " + materialTemplate.name,
114 mainTexture = originalMaterial.mainTexture
115 };
116 cache.Add(key, newPage);
117 }
118
119 return newPage;
120 }
121
122 public void Dispose () {
123 cache.Clear();
124 }
125 }
