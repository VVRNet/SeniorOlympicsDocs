61 public void Initialize () {
62 if (skeletonDataAsset == null) return;
63 SkeletonData skeletonData = skeletonDataAsset.GetSkeletonData(AnimationReferenceAsset.QuietSkeletonData);
64 this.animation = skeletonData != null ? skeletonData.FindAnimation(animationName) : null;
65 if (this.animation == null) Debug.LogWarningFormat("Animation '{0}' not found in SkeletonData : {1}.", animationName, skeletonDataAsset.name);
66 }
