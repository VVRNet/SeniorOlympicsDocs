32using UnityEngine;
33
34namespace Spine.Unity.Examples {
35 public class ConstrainedCamera : MonoBehaviour {
36 public Transform target;
37 public Vector3 offset;
38 public Vector3 min;
39 public Vector3 max;
40 public float smoothing = 5f;
41
42 // Update is called once per frame
43 void LateUpdate () {
44 Vector3 goalPoint = target.position + offset;
45 goalPoint.x = Mathf.Clamp(goalPoint.x, min.x, max.x);
46 goalPoint.y = Mathf.Clamp(goalPoint.y, min.y, max.y);
47 goalPoint.z = Mathf.Clamp(goalPoint.z, min.z, max.z);
48
49 transform.position = Vector3.Lerp(transform.position, goalPoint, smoothing * Time.deltaTime);
50 }
51 }
52}
