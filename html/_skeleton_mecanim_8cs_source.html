121 wasUpdatedAfterInit = true;
122
123 // animation status is kept by Mecanim Animator component

134 public virtual void ApplyAnimation () {
135 if (_BeforeApply != null)
136 _BeforeApply(this);
137
138#if UNITY_EDITOR
139 Animator translatorAnimator = translator.Animator;
140 if (translatorAnimator != null && !translatorAnimator.isInitialized)
141 translatorAnimator.Rebind();
142
143 if (Application.isPlaying) {
144 translator.Apply(skeleton);
145 } else {
146 if (translatorAnimator != null && translatorAnimator.isInitialized &&
147 translatorAnimator.isActiveAndEnabled && translatorAnimator.runtimeAnimatorController != null) {
148 // Note: Rebind is required to prevent warning "Animator is not playing an AnimatorController" with prefabs
149 translatorAnimator.Rebind();
150 translator.Apply(skeleton);
151 }
152 }
153#else
154 translator.Apply(skeleton);
155#endif
156 AfterAnimationApplied();
157 }

160 if (_UpdateLocal != null)
161 _UpdateLocal(this);
162
163 if (_UpdateWorld == null) {
164 UpdateWorldTransform(Skeleton.Physics.Update);

166 UpdateWorldTransform(Skeleton.Physics.Pose);
167 _UpdateWorld(this);
168 UpdateWorldTransform(Skeleton.Physics.Update);

170
171 if (_UpdateComplete != null)
172 _UpdateComplete(this);
173 }

176 if (updateTiming == UpdateTiming.InLateUpdate && valid && translator != null && translator.Animator != null)
177 UpdateAnimation(Time.deltaTime);
178 // instantiation can happen from Update() after this component, leading to a missing Update() call.
179 if (!wasUpdatedAfterInit) Update();
180 base.LateUpdate();
181 }

192 LateUpdate();

275 private bool ApplyAnimation (Skeleton skeleton, AnimatorClipInfo info, AnimatorStateInfo stateInfo,
276 int layerIndex, float layerWeight, MixBlend layerBlendMode, bool useClipWeight1 = false) {
277 float weight = info.weight * layerWeight;
278 if (weight < WeightEpsilon)
279 return false;
280
281 Animation clip = GetAnimation(info.clip);
282 if (clip == null)
283 return false;
284 float time = AnimationTime(stateInfo.normalizedTime, info.clip.length,
285 info.clip.isLooping, stateInfo.speed < 0);
286 weight = useClipWeight1 ? layerWeight : weight;
287 clip.Apply(skeleton, 0, time, info.clip.isLooping, null,
288 weight, layerBlendMode, MixDirection.In);
289 if (_OnClipApplied != null)
290 OnClipAppliedCallback(clip, stateInfo, layerIndex, time, info.clip.isLooping, weight);
291 return true;
292 }
293
294 private bool ApplyInterruptionAnimation (Skeleton skeleton,
295 bool interpolateWeightTo1, AnimatorClipInfo info, AnimatorStateInfo stateInfo,
296 int layerIndex, float layerWeight, MixBlend layerBlendMode, float interruptingClipTimeAddition,
297 bool useClipWeight1 = false) {
298
299 float clipWeight = interpolateWeightTo1 ? (info.weight + 1.0f) * 0.5f : info.weight;
300 float weight = clipWeight * layerWeight;
301 if (weight < WeightEpsilon)
302 return false;
303
304 Animation clip = GetAnimation(info.clip);
305 if (clip == null)
306 return false;
307
308 float time = AnimationTime(stateInfo.normalizedTime + interruptingClipTimeAddition,
309 info.clip.length, info.clip.isLooping, stateInfo.speed < 0);
310 weight = useClipWeight1 ? layerWeight : weight;
311 clip.Apply(skeleton, 0, time, info.clip.isLooping, null,
312 weight, layerBlendMode, MixDirection.In);
313 if (_OnClipApplied != null) {
314 OnClipAppliedCallback(clip, stateInfo, layerIndex, time, info.clip.isLooping, weight);
315 }
316 return true;
317 }

319 private void OnClipAppliedCallback (Spine.Animation clip, AnimatorStateInfo stateInfo,
320 int layerIndex, float time, bool isLooping, float weight) {
321
322 float speedFactor = stateInfo.speedMultiplier * stateInfo.speed;
323 float lastTime = time - (Time.deltaTime * speedFactor);
324 float clipDuration = clip.Duration;
325 if (isLooping && clipDuration != 0) {
326 time %= clipDuration;
327 lastTime %= clipDuration;
328 }
329 _OnClipApplied(clip, layerIndex, weight, time, lastTime, speedFactor < 0);
330 }

332 public void Apply (Skeleton skeleton) {
333#if UNITY_EDITOR
334 if (!Application.isPlaying) {
335 GetLayerBlendModes();
336 }
337#endif
338
339 if (layerMixModes.Length < animator.layerCount) {
340 int oldSize = layerMixModes.Length;
341 System.Array.Resize<MixMode>(ref layerMixModes, animator.layerCount);
342 for (int layer = oldSize; layer < animator.layerCount; ++layer) {
343 bool isAdditiveLayer = false;
344 if (layer < layerBlendModes.Length)
345 isAdditiveLayer = layerBlendModes[layer] == MixBlend.Add;
346 layerMixModes[layer] = isAdditiveLayer ? MixMode.AlwaysMix : MixMode.MixNext;
347 }
348 }
349
350 InitClipInfosForLayers();
351 for (int layer = 0, n = animator.layerCount; layer < n; layer++) {
352 GetStateUpdatesFromAnimator(layer);
353 }
354
355 // Clear Previous
356 if (autoReset) {
357 List<Animation> previousAnimations = this.previousAnimations;
358 for (int i = 0, n = previousAnimations.Count; i < n; i++)
359 previousAnimations[i].Apply(skeleton, 0, 0, false, null, 0, MixBlend.Setup, MixDirection.Out); // SetKeyedItemsToSetupPose
360
361 previousAnimations.Clear();
362 for (int layer = 0, n = animator.layerCount; layer < n; layer++) {
363 float layerWeight = (layer == 0) ? 1 : animator.GetLayerWeight(layer); // Animator.GetLayerWeight always returns 0 on the first layer. Should be interpreted as 1.
364 if (layerWeight <= 0) continue;
365
366 AnimatorStateInfo nextStateInfo = animator.GetNextAnimatorStateInfo(layer);
367
368 bool hasNext = nextStateInfo.fullPathHash != 0;
369
370 int clipInfoCount, nextClipInfoCount, interruptingClipInfoCount;
371 IList<AnimatorClipInfo> clipInfo, nextClipInfo, interruptingClipInfo;
372 bool isInterruptionActive, shallInterpolateWeightTo1;
373 GetAnimatorClipInfos(layer, out isInterruptionActive, out clipInfoCount, out nextClipInfoCount, out interruptingClipInfoCount,
374 out clipInfo, out nextClipInfo, out interruptingClipInfo, out shallInterpolateWeightTo1);
375
376 for (int c = 0; c < clipInfoCount; c++) {
377 AnimatorClipInfo info = clipInfo[c];
378 float weight = info.weight * layerWeight; if (weight < WeightEpsilon) continue;
379 Spine.Animation clip = GetAnimation(info.clip);
380 if (clip != null)
381 previousAnimations.Add(clip);
382 }
383
384 if (hasNext) {
385 for (int c = 0; c < nextClipInfoCount; c++) {
386 AnimatorClipInfo info = nextClipInfo[c];
387 float weight = info.weight * layerWeight; if (weight < WeightEpsilon) continue;
388 Spine.Animation clip = GetAnimation(info.clip);
389 if (clip != null)
390 previousAnimations.Add(clip);
391 }
392 }
393
394 if (isInterruptionActive) {
395 for (int c = 0; c < interruptingClipInfoCount; c++) {
396 AnimatorClipInfo info = interruptingClipInfo[c];
397 float clipWeight = shallInterpolateWeightTo1 ? (info.weight + 1.0f) * 0.5f : info.weight;
398 float weight = clipWeight * layerWeight; if (weight < WeightEpsilon) continue;
399 Spine.Animation clip = GetAnimation(info.clip);
400 if (clip != null)
401 previousAnimations.Add(clip);
402 }
403 }
404 }
405 }
406
407 // Apply
408 for (int layer = 0, n = animator.layerCount; layer < n; layer++) {
409 float layerWeight = (layer == 0) ? 1 : animator.GetLayerWeight(layer); // Animator.GetLayerWeight always returns 0 on the first layer. Should be interpreted as 1.
410
411 bool isInterruptionActive;
412 AnimatorStateInfo stateInfo;
413 AnimatorStateInfo nextStateInfo;
414 AnimatorStateInfo interruptingStateInfo;
415 float interruptingClipTimeAddition;
416 GetAnimatorStateInfos(layer, out isInterruptionActive, out stateInfo, out nextStateInfo, out interruptingStateInfo, out interruptingClipTimeAddition);
417
418 bool hasNext = nextStateInfo.fullPathHash != 0;
419
420 int clipInfoCount, nextClipInfoCount, interruptingClipInfoCount;
421 IList<AnimatorClipInfo> clipInfo, nextClipInfo, interruptingClipInfo;
422 bool interpolateWeightTo1;
423 GetAnimatorClipInfos(layer, out isInterruptionActive, out clipInfoCount, out nextClipInfoCount, out interruptingClipInfoCount,
424 out clipInfo, out nextClipInfo, out interruptingClipInfo, out interpolateWeightTo1);
425
426 MixBlend layerBlendMode = (layer < layerBlendModes.Length) ? layerBlendModes[layer] : MixBlend.Replace;
427 MixMode mode = GetMixMode(layer, layerBlendMode);
428 if (mode == MixMode.AlwaysMix) {
429 // Always use Mix instead of Applying the first non-zero weighted clip.
430 for (int c = 0; c < clipInfoCount; c++) {
431 ApplyAnimation(skeleton, clipInfo[c], stateInfo, layer, layerWeight, layerBlendMode);
432 }
433 if (hasNext) {
434 for (int c = 0; c < nextClipInfoCount; c++) {
435 ApplyAnimation(skeleton, nextClipInfo[c], nextStateInfo, layer, layerWeight, layerBlendMode);
436 }
437 }
438 if (isInterruptionActive) {
439 for (int c = 0; c < interruptingClipInfoCount; c++) {
440 ApplyInterruptionAnimation(skeleton, interpolateWeightTo1,
441 interruptingClipInfo[c], interruptingStateInfo,
442 layer, layerWeight, layerBlendMode, interruptingClipTimeAddition);
443 }
444 }
445 } else { // case MixNext || Hard
446 // Apply first non-zero weighted clip
447 int c = 0;
448 for (; c < clipInfoCount; c++) {
449 if (!ApplyAnimation(skeleton, clipInfo[c], stateInfo, layer, layerWeight, layerBlendMode, useClipWeight1: true))
450 continue;
451 ++c; break;
452 }
453 // Mix the rest
454 for (; c < clipInfoCount; c++) {
455 ApplyAnimation(skeleton, clipInfo[c], stateInfo, layer, layerWeight, layerBlendMode);
456 }
457
458 c = 0;
459 if (hasNext) {
460 // Apply next clip directly instead of mixing (ie: no crossfade, ignores mecanim transition weights)
461 if (mode == MixMode.Hard) {
462 for (; c < nextClipInfoCount; c++) {
463 if (!ApplyAnimation(skeleton, nextClipInfo[c], nextStateInfo, layer, layerWeight, layerBlendMode, useClipWeight1: true))
464 continue;
465 ++c; break;
466 }
467 }
468 // Mix the rest
469 for (; c < nextClipInfoCount; c++) {
470 if (!ApplyAnimation(skeleton, nextClipInfo[c], nextStateInfo, layer, layerWeight, layerBlendMode))
471 continue;
472 }
473 }
474
475 c = 0;
476 if (isInterruptionActive) {
477 // Apply next clip directly instead of mixing (ie: no crossfade, ignores mecanim transition weights)
478 if (mode == MixMode.Hard) {
479 for (; c < interruptingClipInfoCount; c++) {
480 if (ApplyInterruptionAnimation(skeleton, interpolateWeightTo1,
481 interruptingClipInfo[c], interruptingStateInfo,
482 layer, layerWeight, layerBlendMode, interruptingClipTimeAddition, useClipWeight1: true)) {
483
484 ++c; break;
485 }
486 }
487 }
488 // Mix the rest
489 for (; c < interruptingClipInfoCount; c++) {
490 ApplyInterruptionAnimation(skeleton, interpolateWeightTo1,
491 interruptingClipInfo[c], interruptingStateInfo,
492 layer, layerWeight, layerBlendMode, interruptingClipTimeAddition);
493 }
494 }
495 }
496 }

499 public KeyValuePair<Spine.Animation, float> GetActiveAnimationAndTime (int layer) {
500 if (layer >= layerClipInfos.Length)
501 return new KeyValuePair<Spine.Animation, float>(null, 0);
502
503 ClipInfos layerInfos = layerClipInfos[layer];
504 bool isInterruptionActive = layerInfos.isInterruptionActive;
505 AnimationClip clip = null;
506 Spine.Animation animation = null;
507 AnimatorStateInfo stateInfo;
508 if (isInterruptionActive && layerInfos.interruptingClipInfoCount > 0) {
509 clip = layerInfos.interruptingClipInfos[0].clip;
510 stateInfo = layerInfos.interruptingStateInfo;
511 } else {
512 clip = layerInfos.clipInfos[0].clip;
513 stateInfo = layerInfos.stateInfo;
514 }
515 animation = GetAnimation(clip);
516 float time = AnimationTime(stateInfo.normalizedTime, clip.length,
517 clip.isLooping, stateInfo.speed < 0);
518 return new KeyValuePair<Animation, float>(animation, time);
519 }

521 static float AnimationTime (float normalizedTime, float clipLength, bool loop, bool reversed) {
522 float time = ToSpineAnimationTime(normalizedTime, clipLength, loop, reversed);
523 if (loop) return time;
524 const float EndSnapEpsilon = 1f / 30f; // Workaround for end-duration keys not being applied.
525 return (clipLength - time < EndSnapEpsilon) ? clipLength : time; // return a time snapped to clipLength;
526 }
527
528 static float ToSpineAnimationTime (float normalizedTime, float clipLength, bool loop, bool reversed) {
529 if (reversed)
530 normalizedTime = (1 - normalizedTime);
531 if (normalizedTime < 0.0f)
532 normalizedTime = loop ? (normalizedTime % 1.0f) + 1.0f : 0.0f;
533 return normalizedTime * clipLength;
534 }

560 private MixMode GetMixMode (int layer, MixBlend layerBlendMode) {
561 if (useCustomMixMode) {
562 MixMode mode = layerMixModes[layer];
563 // Note: at additive blending it makes no sense to use constant weight 1 at a fadeout anim add1 as
564 // with override layers, so we use AlwaysMix instead to use the proper weights.
565 // AlwaysMix leads to the expected result = lower_layer + lerp(add1, add2, transition_weight).
566 if (layerBlendMode == MixBlend.Add && mode == MixMode.MixNext) {
567 mode = MixMode.AlwaysMix;
568 layerMixModes[layer] = mode;
569 }
570 return mode;
571 } else {
572 return layerBlendMode == MixBlend.Add ? MixMode.AlwaysMix : MixMode.MixNext;
573 }
574 }

594 void GetStateUpdatesFromAnimator (int layer) {
595
596 ClipInfos layerInfos = layerClipInfos[layer];
597 int clipInfoCount = animator.GetCurrentAnimatorClipInfoCount(layer);
598 int nextClipInfoCount = animator.GetNextAnimatorClipInfoCount(layer);
599
600 List<AnimatorClipInfo> clipInfos = layerInfos.clipInfos;
601 List<AnimatorClipInfo> nextClipInfos = layerInfos.nextClipInfos;
602 List<AnimatorClipInfo> interruptingClipInfos = layerInfos.interruptingClipInfos;
603
604 layerInfos.isInterruptionActive = (clipInfoCount == 0 && clipInfos.Count != 0 &&
605 nextClipInfoCount == 0 && nextClipInfos.Count != 0);
606
607 // Note: during interruption, GetCurrentAnimatorClipInfoCount and GetNextAnimatorClipInfoCount
608 // are returning 0 in calls above. Therefore we keep previous clipInfos and nextClipInfos
609 // until the interruption is over.
610 if (layerInfos.isInterruptionActive) {
611
612 // Note: The last frame of a transition interruption
613 // will have fullPathHash set to 0, therefore we have to use previous
614 // frame's infos about interruption clips and correct some values
615 // accordingly (normalizedTime and weight).
616 AnimatorStateInfo interruptingStateInfo = animator.GetNextAnimatorStateInfo(layer);
617 layerInfos.isLastFrameOfInterruption = interruptingStateInfo.fullPathHash == 0;
618 if (!layerInfos.isLastFrameOfInterruption) {
619 animator.GetNextAnimatorClipInfo(layer, interruptingClipInfos);
620 layerInfos.interruptingClipInfoCount = interruptingClipInfos.Count;
621 float oldTime = layerInfos.interruptingStateInfo.normalizedTime;
622 float newTime = interruptingStateInfo.normalizedTime;
623 layerInfos.interruptingClipTimeAddition = newTime - oldTime;
624 layerInfos.interruptingStateInfo = interruptingStateInfo;
625 }
626 } else {
627 layerInfos.clipInfoCount = clipInfoCount;
628 layerInfos.nextClipInfoCount = nextClipInfoCount;
629 layerInfos.interruptingClipInfoCount = 0;
630 layerInfos.isLastFrameOfInterruption = false;
631
632 if (clipInfos.Capacity < clipInfoCount) clipInfos.Capacity = clipInfoCount;
633 if (nextClipInfos.Capacity < nextClipInfoCount) nextClipInfos.Capacity = nextClipInfoCount;
634
635 animator.GetCurrentAnimatorClipInfo(layer, clipInfos);
636 animator.GetNextAnimatorClipInfo(layer, nextClipInfos);
637
638 layerInfos.stateInfo = animator.GetCurrentAnimatorStateInfo(layer);
639 layerInfos.nextStateInfo = animator.GetNextAnimatorStateInfo(layer);
640 }
641 }

643 void GetAnimatorClipInfos (
644 int layer,
645 out bool isInterruptionActive,
646 out int clipInfoCount,
647 out int nextClipInfoCount,
648 out int interruptingClipInfoCount,
649 out IList<AnimatorClipInfo> clipInfo,
650 out IList<AnimatorClipInfo> nextClipInfo,
651 out IList<AnimatorClipInfo> interruptingClipInfo,
652 out bool shallInterpolateWeightTo1) {
653
654 ClipInfos layerInfos = layerClipInfos[layer];
655 isInterruptionActive = layerInfos.isInterruptionActive;
