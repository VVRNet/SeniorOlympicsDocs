36 // This is a sample component for C# vertex effects for Spine rendering components.
37 // Using shaders and materials to control vertex properties is still more performant
38 // than using this API, but in cases where your vertex effect logic cannot be
39 // expressed as shader code, these vertex effects can be useful.
40 public class TwoByTwoTransformEffectExample : MonoBehaviour {
41
42 public Vector2 xAxis = new Vector2(1, 0);
43 public Vector2 yAxis = new Vector2(0, 1);
44
45 SkeletonRenderer skeletonRenderer;
46
47 void OnEnable () {
48 skeletonRenderer = GetComponent<SkeletonRenderer>();
49 if (skeletonRenderer == null) return;
50
51 // Use the OnPostProcessVertices callback to modify the vertices at the correct time.
52 skeletonRenderer.OnPostProcessVertices -= ProcessVertices;
53 skeletonRenderer.OnPostProcessVertices += ProcessVertices;
54
55 Debug.Log("2x2 Transform Effect Enabled.");
56 }
57
58 void ProcessVertices (MeshGeneratorBuffers buffers) {
59 if (!this.enabled)
60 return;
61
62 int vertexCount = buffers.vertexCount; // For efficiency, limit your effect to the actual mesh vertex count using vertexCount
63
64 // Modify vertex positions by accessing Vector3[] vertexBuffer
65 Vector3[] vertices = buffers.vertexBuffer;
66 Vector3 transformedPos = default(Vector3);
67 for (int i = 0; i < vertexCount; i++) {
68 Vector3 originalPos = vertices[i];
69 transformedPos.x = (xAxis.x * originalPos.x) + (yAxis.x * originalPos.y);
70 transformedPos.y = (xAxis.y * originalPos.x) + (yAxis.y * originalPos.y);
71 vertices[i] = transformedPos;
72 }
73
74 }
75
76 void OnDisable () {
77 if (skeletonRenderer == null) return;
78 skeletonRenderer.OnPostProcessVertices -= ProcessVertices;
79 Debug.Log("2x2 Transform Effect Disabled.");
80 }
81 }
82
83}
