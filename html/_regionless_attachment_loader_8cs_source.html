40 if (emptyRegion == null) {
41 Shader hiddenShader = Shader.Find("Spine/Special/HiddenPass");
42 if (hiddenShader == null) {
43 Debug.LogError("Shader \"Spine/Special/HiddenPass\" not found while loading SkeletonDataAsset" +
44 " with 0 Atlas Assets. Please add this shader to Project Settings - Graphics - Always" +
45 " Included Shaders, or make sure your SkeletonDataAssets all have an AtlasAsset assigned.");
46 }
47 emptyRegion = new AtlasRegion {
48 name = "Empty AtlasRegion",
49 page = new AtlasPage {
50 name = "Empty AtlasPage",
51 rendererObject = new Material(hiddenShader) { name = "NoRender Material" }
52 }
53 };
54 }
55 return emptyRegion;
56 }
