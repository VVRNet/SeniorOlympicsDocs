30using System.Collections;
31using System.Collections.Generic;
32using UnityEngine;
33
34namespace Spine.Unity.Examples {
35 public class DummyMecanimControllerExample : MonoBehaviour {
36
37 public Animator logicAnimator;
38 public SkeletonAnimationHandleExample animationHandle;
39
40 [Header("Controls")]
41 public KeyCode walkButton = KeyCode.LeftShift;
42 public KeyCode jumpButton = KeyCode.Space;
43
44 [Header("Animator Properties")]
45 public string horizontalSpeedProperty = "Speed";
46 public string verticalSpeedProperty = "VerticalSpeed";
47 public string groundedProperty = "Grounded";
48
49 [Header("Fake Physics")]
50 public float jumpDuration = 1.5f;
51 public Vector2 speed;
52 public bool isGrounded;
53
54 void Awake () {
55 isGrounded = true;
56 }
57
58 void Update () {
59 float x = Input.GetAxisRaw("Horizontal");
60 if (Input.GetKey(walkButton)) {
61 x *= 0.4f;
62 }
63
64 speed.x = x;
65
66 // Flip skeleton.
67 if (x != 0) {
68 animationHandle.SetFlip(x);
69 }
70
71 if (Input.GetKeyDown(jumpButton)) {
72 if (isGrounded)
73 StartCoroutine(FakeJump());
74 }
75
76 logicAnimator.SetFloat(horizontalSpeedProperty, Mathf.Abs(speed.x));
77 logicAnimator.SetFloat(verticalSpeedProperty, speed.y);
78 logicAnimator.SetBool(groundedProperty, isGrounded);
79 }
80
81 IEnumerator FakeJump () {
82 // Rise
83 isGrounded = false;
84 speed.y = 10f;
85 float durationLeft = jumpDuration * 0.5f;
86 while (durationLeft > 0) {
87 durationLeft -= Time.deltaTime;
88 if (!Input.GetKey(jumpButton)) break;
89 yield return null;
90 }
91
92 // Fall
93 speed.y = -10f;
94 float fallDuration = (jumpDuration * 0.5f) - durationLeft;
95 yield return new WaitForSeconds(fallDuration);
96
97 // Land
98 speed.y = 0f;
99 isGrounded = true;
100 yield return null;
101 }
102 }
103
104}
