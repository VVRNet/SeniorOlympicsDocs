89 protected void TrySetAnimation (SkeletonAnimation skeletonAnimation) {
90 if (skeletonAnimation == null) return;
91 if (!skeletonAnimation.valid || skeletonAnimation.AnimationState == null)
92 return;
93
94 TrackEntry current = skeletonAnimation.AnimationState.GetCurrent(0);
95 if (!isInspectingPrefab) {
96 string activeAnimation = (current != null) ? current.Animation.Name : "";
97 bool activeLoop = (current != null) ? current.Loop : false;
98 bool animationParameterChanged = this.wasAnimationParameterChanged &&
99 ((activeAnimation != animationName.stringValue) || (activeLoop != loop.boolValue));
100 if (animationParameterChanged) {
101 this.wasAnimationParameterChanged = false;
102 Skeleton skeleton = skeletonAnimation.Skeleton;
103 AnimationState state = skeletonAnimation.AnimationState;
104
105 if (!Application.isPlaying) {
106 if (state != null) state.ClearTrack(0);
107 skeleton.SetToSetupPose();
108 }
109
110 Spine.Animation animationToUse = skeleton.Data.FindAnimation(animationName.stringValue);
111
112 if (!Application.isPlaying) {
113 if (animationToUse != null) {
114 skeletonAnimation.AnimationState.SetAnimation(0, animationToUse, loop.boolValue);
115 }
116 skeletonAnimation.Update(0);
117 skeletonAnimation.LateUpdate();
118 requireRepaint = true;
119 } else {
120 if (animationToUse != null)
121 state.SetAnimation(0, animationToUse, loop.boolValue);
122 else
123 state.ClearTrack(0);
124 }
125 }
126
127 // Reflect animationName serialized property in the inspector even if SetAnimation API was used.
128 if (Application.isPlaying) {
129 if (current != null && current.Animation != null) {
130 if (skeletonAnimation.AnimationName != animationName.stringValue)
131 animationName.stringValue = current.Animation.Name;
132 }
133 }
134 }
135 }
136 }
