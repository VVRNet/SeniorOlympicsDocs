130 public static Texture2D GetAttachmentIcon (Attachment attachment) {
131 // Analysis disable once CanBeReplacedWithTryCastAndCheckForNull
132 if (attachment is RegionAttachment)
133 return Icons.image;
134 else if (attachment is MeshAttachment)
135 return ((MeshAttachment)attachment).IsWeighted() ? Icons.weights : Icons.mesh;
136 else if (attachment is BoundingBoxAttachment)
137 return Icons.boundingBox;
138 else if (attachment is PointAttachment)
139 return Icons.point;
140 else if (attachment is PathAttachment)
141 return Icons.path;
142 else if (attachment is ClippingAttachment)
143 return Icons.clipping;
144 else
145 return Icons.warning;
146 }
147 }
