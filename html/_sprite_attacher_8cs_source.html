49 void OnValidate () {
50 ISkeletonComponent skeletonComponent = GetComponent<ISkeletonComponent>();
51 SkeletonRenderer skeletonRenderer = skeletonComponent as SkeletonRenderer;
52 bool applyPMA;
53
54 if (skeletonRenderer != null) {
55 applyPMA = skeletonRenderer.pmaVertexColors;
56 } else {
57 SkeletonGraphic skeletonGraphic = skeletonComponent as SkeletonGraphic;
58 applyPMA = skeletonGraphic != null && skeletonGraphic.MeshGenerator.settings.pmaVertexColors;
59 }
60
61 if (applyPMA) {
62 try {
63 if (sprite == null)
64 return;
65 sprite.texture.GetPixel(0, 0);
66 } catch (UnityException e) {
67 Debug.LogFormat("Texture of {0} ({1}) is not read/write enabled. SpriteAttacher requires this in order to work with a SkeletonRenderer that renders premultiplied alpha. Please check the texture settings.", sprite.name, sprite.texture.name);
68 UnityEditor.EditorGUIUtility.PingObject(sprite.texture);
69 throw e;
70 }
71 }

79 static Dictionary<Texture, AtlasPage> atlasPageCache;
80 static AtlasPage GetPageFor (Texture texture, Shader shader) {
81 if (atlasPageCache == null) atlasPageCache = new Dictionary<Texture, AtlasPage>();
82 AtlasPage atlasPage;
83 atlasPageCache.TryGetValue(texture, out atlasPage);
84 if (atlasPage == null) {
85 Material newMaterial = new Material(shader);
86 atlasPage = newMaterial.ToSpineAtlasPage();
87 atlasPageCache[texture] = atlasPage;
88 }
89 return atlasPage;
90 }

105 public void Initialize (bool overwrite = true) {
106 if (overwrite || attachment == null) {
107 // Get the applyPMA value.
108 ISkeletonComponent skeletonComponent = GetComponent<ISkeletonComponent>();
109 SkeletonRenderer skeletonRenderer = skeletonComponent as SkeletonRenderer;
110 if (skeletonRenderer != null)
111 this.applyPMA = skeletonRenderer.pmaVertexColors;
112 else {
113 SkeletonGraphic skeletonGraphic = skeletonComponent as SkeletonGraphic;
114 if (skeletonGraphic != null)
115 this.applyPMA = skeletonGraphic.MeshGenerator.settings.pmaVertexColors;
116 }
117
118 // Subscribe to UpdateComplete to override animation keys.
119 if (overrideAnimation) {
120 ISkeletonAnimation animatedSkeleton = skeletonComponent as ISkeletonAnimation;
121 if (animatedSkeleton != null) {
122 animatedSkeleton.UpdateComplete -= AnimationOverrideSpriteAttach;
123 animatedSkeleton.UpdateComplete += AnimationOverrideSpriteAttach;
124 }
125 }
126
127 spineSlot = spineSlot ?? skeletonComponent.Skeleton.FindSlot(slot);
128 Shader attachmentShader = applyPMA ? Shader.Find(DefaultPMAShader) : Shader.Find(DefaultStraightAlphaShader);
129 if (sprite == null)
130 attachment = null;
131 else
132 attachment = applyPMA ? sprite.ToRegionAttachmentPMAClone(attachmentShader) : sprite.ToRegionAttachment(SpriteAttacher.GetPageFor(sprite.texture, attachmentShader));
133 }
134 }
135
136 void OnDestroy () {
137 ISkeletonAnimation animatedSkeleton = GetComponent<ISkeletonAnimation>();
138 if (animatedSkeleton != null)
139 animatedSkeleton.UpdateComplete -= AnimationOverrideSpriteAttach;
140 }
141
143 public void Attach () {
144 if (spineSlot != null)
145 spineSlot.Attachment = attachment;
146 }
147
148 }
