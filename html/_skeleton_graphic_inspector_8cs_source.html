686 static void DetectMaterial (SkeletonGraphic skeletonGraphic) {
687 MeshGenerator.Settings settings = skeletonGraphic.MeshGenerator.settings;
688
689 bool detectionSucceeded;
690 bool usesPMATexture = IsSkeletonTexturePMA(skeletonGraphic, out detectionSucceeded);
691 if (!detectionSucceeded) {
692 Debug.LogWarning(string.Format("Unable to assign Material for skeleton '{0}'.", skeletonGraphic), skeletonGraphic);
693 return;
694 }
695
696 Material newMaterial = null;
697 if (usesPMATexture) {
698 if (settings.tintBlack) {
699 if (settings.canvasGroupCompatible)
700 newMaterial = MaterialWithName("SkeletonGraphicTintBlack-CanvasGroup");
701 else
702 newMaterial = MaterialWithName("SkeletonGraphicTintBlack");
703 } else { // not tintBlack
704 if (settings.canvasGroupCompatible)
705 newMaterial = MaterialWithName("SkeletonGraphicDefault-CanvasGroup");
706 else
707 newMaterial = MaterialWithName("SkeletonGraphicDefault");
708 }
709 } else { // straight alpha texture
710 if (settings.tintBlack) {
711 if (settings.canvasGroupCompatible)
712 newMaterial = MaterialWithName("SkeletonGraphicTintBlack-CanvasGroupStraight");
713 else
714 newMaterial = MaterialWithName("SkeletonGraphicTintBlack-Straight");
715 } else { // not tintBlack
716 if (settings.canvasGroupCompatible)
717 newMaterial = MaterialWithName("SkeletonGraphicDefault-CanvasGroupStraight");
718 else
719 newMaterial = MaterialWithName("SkeletonGraphicDefault-Straight");
720 }
721 }
722 if (newMaterial != null) {
723 Debug.Log(string.Format("Assigning material '{0}' at skeleton '{1}'",
724 newMaterial, skeletonGraphic), newMaterial);
725 skeletonGraphic.material = newMaterial;
726 }

868 protected static Material MaterialWithName (string name) {
869 string[] guids = AssetDatabase.FindAssets(name + " t:material");
870 if (guids.Length <= 0) return null;
871
872 int closestNameDistance = int.MaxValue;
873 int closestNameIndex = 0;
874 for (int i = 0; i < guids.Length; ++i) {
875 string assetPath = AssetDatabase.GUIDToAssetPath(guids[i]);
876 string assetName = System.IO.Path.GetFileNameWithoutExtension(assetPath);
877 int distance = string.CompareOrdinal(assetName, name);
878 if (distance < closestNameDistance) {
879 closestNameDistance = distance;
880 closestNameIndex = i;
881 }
882 }
883
884 string foundAssetPath = AssetDatabase.GUIDToAssetPath(guids[closestNameIndex]);
885 if (string.IsNullOrEmpty(foundAssetPath)) return null;
886
887 Material firstMaterial = AssetDatabase.LoadAssetAtPath<Material>(foundAssetPath);
888 return firstMaterial;
889 }
