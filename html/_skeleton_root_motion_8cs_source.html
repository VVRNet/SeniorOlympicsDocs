99 protected override Vector2 CalculateAnimationsMovementDelta () {
100 Vector2 localDelta = Vector2.zero;
101 int trackCount = animationState.Tracks.Count;
102
103 for (int trackIndex = 0; trackIndex < trackCount; ++trackIndex) {
104 // note: animationTrackFlags != -1 below covers trackIndex >= 32,
105 // with -1 corresponding to entry "everything" of the dropdown list.
106 if (animationTrackFlags != -1 && (animationTrackFlags & 1 << trackIndex) == 0)
107 continue;
108
109 TrackEntry track = animationState.GetCurrent(trackIndex);
110 TrackEntry next = null;
111 while (track != null) {
112 Animation animation = track.Animation;
113 float start = track.AnimationLast;
114 float end = track.AnimationTime;
115 Vector2 currentDelta = GetAnimationRootMotion(start, end, animation);
116 if (currentDelta != Vector2.zero) {
117 ApplyMixAlphaToDelta(ref currentDelta, next, track);
118 localDelta += currentDelta;
119 }
120
121 // Traverse mixingFrom chain.
122 next = track;
123 track = track.MixingFrom;
124 }
125 }
126 return localDelta;
127 }

159 void ApplyMixAlphaToDelta (ref Vector2 currentDelta, TrackEntry next, TrackEntry track) {

171 void GetMixAlpha (ref float cumulatedMixAlpha, TrackEntry next, TrackEntry track) {
172 // code below based on AnimationState.cs
173 float mix;
174 if (next != null) {
175 if (next.MixDuration == 0) { // Single frame mix to undo mixingFrom changes.
176 mix = 1;
177 } else {
178 mix = next.MixTime / next.MixDuration;
179 if (mix > 1) mix = 1;
180 }
181 float mixAndAlpha = track.Alpha * next.InterruptAlpha * (1 - mix);
182 cumulatedMixAlpha *= mixAndAlpha;
183 } else {
184 if (track.MixDuration == 0) {
185 mix = 1;
186 } else {
187 mix = track.Alpha * (track.MixTime / track.MixDuration);
188 if (mix > 1) mix = 1;
189 }
190 cumulatedMixAlpha *= mix;
191 }
192 }
