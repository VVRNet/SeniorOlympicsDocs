162 private static UnityEngine.InputSystem.Controls.ButtonControl GetMouseButtonControl(int index)
163 {
164 if (UnityEngine.InputSystem.Mouse.current != null)
165 {
166 switch (index)
167 {
168 case 0: return UnityEngine.InputSystem.Mouse.current.leftButton;
169 case 1: return UnityEngine.InputSystem.Mouse.current.rightButton;
170 case 2: return UnityEngine.InputSystem.Mouse.current.middleButton;
171 case 3: return UnityEngine.InputSystem.Mouse.current.forwardButton;
172 case 4: return UnityEngine.InputSystem.Mouse.current.backButton;
173 }
174 }
175
176 return null;
177 }

179 private static UnityEngine.InputSystem.Controls.ButtonControl GetButtonControl(KeyCode oldKey)
180 {
181 if (UnityEngine.InputSystem.Mouse.current != null)
182 {
183 switch (oldKey)
184 {
185 case KeyCode.Mouse0: return UnityEngine.InputSystem.Mouse.current.leftButton;
186 case KeyCode.Mouse1: return UnityEngine.InputSystem.Mouse.current.rightButton;
187 case KeyCode.Mouse2: return UnityEngine.InputSystem.Mouse.current.middleButton;
188 case KeyCode.Mouse3: return UnityEngine.InputSystem.Mouse.current.forwardButton;
189 case KeyCode.Mouse4: return UnityEngine.InputSystem.Mouse.current.backButton;
190 }
191 }
192
193 if (UnityEngine.InputSystem.Keyboard.current != null)
194 {
195 var newKey = default(UnityEngine.InputSystem.Key);
196
197 if (keyMapping.TryGetValue(oldKey, out newKey) == true)
198 {
199 return UnityEngine.InputSystem.Keyboard.current[newKey];
200 }
201 }
202
203 return null;
204 }

221 public static void GetTouch(int index, out int id, out UnityEngine.Vector2 position, out float pressure, out bool set)
222 {
223#if USE_NEW_INPUT_SYSTEM
224 var touch = UnityEngine.InputSystem.EnhancedTouch.Touch.activeTouches[index];
225
226 id = touch.finger.index;
227 position = touch.screenPosition;
228 pressure = touch.pressure;
229 set =
230 touch.phase == UnityEngine.InputSystem.TouchPhase.Began ||
231 touch.phase == UnityEngine.InputSystem.TouchPhase.Stationary ||
232 touch.phase == UnityEngine.InputSystem.TouchPhase.Moved;
233#else
234 var touch = UnityEngine.Input.GetTouch(index);
235
236 id = touch.fingerId;
237 position = touch.position;
238 pressure = touch.pressure;
239 set =
240 touch.phase == UnityEngine.TouchPhase.Began ||
241 touch.phase == UnityEngine.TouchPhase.Stationary ||
242 touch.phase == UnityEngine.TouchPhase.Moved;
243#endif
244 }
