64 public void Update (Skeleton skeleton, bool updateAabb) {
65 ExposedList<BoundingBoxAttachment> boundingBoxes = BoundingBoxes;
66 ExposedList<Polygon> polygons = Polygons;
67 Slot[] slots = skeleton.slots.Items;
68 int slotCount = skeleton.slots.Count;
69
70 boundingBoxes.Clear();
71 for (int i = 0, n = polygons.Count; i < n; i++)
72 polygonPool.Add(polygons.Items[i]);
73 polygons.Clear();
74
75 for (int i = 0; i < slotCount; i++) {
76 Slot slot = slots[i];
77 if (!slot.bone.active) continue;
78 BoundingBoxAttachment boundingBox = slot.attachment as BoundingBoxAttachment;
79 if (boundingBox == null) continue;
80 boundingBoxes.Add(boundingBox);
81
82 Polygon polygon = null;
83 int poolCount = polygonPool.Count;
84 if (poolCount > 0) {
85 polygon = polygonPool.Items[poolCount - 1];
86 polygonPool.RemoveAt(poolCount - 1);
87 } else
88 polygon = new Polygon();
89 polygons.Add(polygon);
90
91 int count = boundingBox.worldVerticesLength;
92 polygon.Count = count;
93 if (polygon.Vertices.Length < count) polygon.Vertices = new float[count];
94 boundingBox.ComputeWorldVertices(slot, polygon.Vertices);
95 }
96
97 if (updateAabb) {
98 AabbCompute();
99 } else {
100 minX = int.MinValue;
101 minY = int.MinValue;
102 maxX = int.MaxValue;
103 maxY = int.MaxValue;
104 }
105 }

107 private void AabbCompute () {
108 float minX = int.MaxValue, minY = int.MaxValue, maxX = int.MinValue, maxY = int.MinValue;
109 Polygon[] polygons = Polygons.Items;
110 for (int i = 0, n = Polygons.Count; i < n; i++) {
111 Polygon polygon = polygons[i];
112 float[] vertices = polygon.Vertices;
113 for (int ii = 0, nn = polygon.Count; ii < nn; ii += 2) {
114 float x = vertices[ii];
115 float y = vertices[ii + 1];
116 minX = Math.Min(minX, x);
117 minY = Math.Min(minY, y);
118 maxX = Math.Max(maxX, x);
119 maxY = Math.Max(maxY, y);
120 }
121 }
122 this.minX = minX;
123 this.minY = minY;
124 this.maxX = maxX;
125 this.maxY = maxY;
126 }

134 public bool AabbIntersectsSegment (float x1, float y1, float x2, float y2) {
135 float minX = this.minX;
136 float minY = this.minY;
137 float maxX = this.maxX;
138 float maxY = this.maxY;
139 if ((x1 <= minX && x2 <= minX) || (y1 <= minY && y2 <= minY) || (x1 >= maxX && x2 >= maxX) || (y1 >= maxY && y2 >= maxY))
140 return false;
141 float m = (y2 - y1) / (x2 - x1);
142 float y = m * (minX - x1) + y1;
143 if (y > minY && y < maxY) return true;
144 y = m * (maxX - x1) + y1;
145 if (y > minY && y < maxY) return true;
146 float x = (minY - y1) / m + x1;
147 if (x > minX && x < maxX) return true;
148 x = (maxY - y1) / m + x1;
149 if (x > minX && x < maxX) return true;
150 return false;
151 }

159 public bool ContainsPoint (Polygon polygon, float x, float y) {
160 float[] vertices = polygon.Vertices;
161 int nn = polygon.Count;
162
163 int prevIndex = nn - 2;
164 bool inside = false;
165 for (int ii = 0; ii < nn; ii += 2) {
166 float vertexY = vertices[ii + 1];
167 float prevY = vertices[prevIndex + 1];
168 if ((vertexY < y && prevY >= y) || (prevY < y && vertexY >= y)) {
169 float vertexX = vertices[ii];
170 if (vertexX + (y - vertexY) / (prevY - vertexY) * (vertices[prevIndex] - vertexX) < x) inside = !inside;
171 }
172 prevIndex = ii;
173 }
174 return inside;
175 }

196 public bool IntersectsSegment (Polygon polygon, float x1, float y1, float x2, float y2) {
197 float[] vertices = polygon.Vertices;
198 int nn = polygon.Count;
199
200 float width12 = x1 - x2, height12 = y1 - y2;
201 float det1 = x1 * y2 - y1 * x2;
202 float x3 = vertices[nn - 2], y3 = vertices[nn - 1];
203 for (int ii = 0; ii < nn; ii += 2) {
204 float x4 = vertices[ii], y4 = vertices[ii + 1];
205 float det2 = x3 * y4 - y3 * x4;
206 float width34 = x3 - x4, height34 = y3 - y4;
207 float det3 = width12 * height34 - height12 * width34;
208 float x = (det1 * width34 - width12 * det2) / det3;
209 if (((x >= x3 && x <= x4) || (x >= x4 && x <= x3)) && ((x >= x1 && x <= x2) || (x >= x2 && x <= x1))) {
210 float y = (det1 * height34 - height12 * det2) / det3;
211 if (((y >= y3 && y <= y4) || (y >= y4 && y <= y3)) && ((y >= y1 && y <= y2) || (y >= y2 && y <= y1))) return true;
212 }
213 x3 = x4;
214 y3 = y4;
215 }
216 return false;
217 }
