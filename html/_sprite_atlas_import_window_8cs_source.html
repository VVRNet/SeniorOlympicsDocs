150 SpineSpriteAtlasAsset FindSpineSpriteAtlasAsset (UnityEngine.U2D.SpriteAtlas spriteAtlasAsset) {
151 string path = AssetDatabase.GetAssetPath(spriteAtlasAsset).Replace(".spriteatlas", AssetUtility.SpriteAtlasSuffix + ".asset");
152 if (System.IO.File.Exists(path)) {
153 return AssetDatabase.LoadAssetAtPath<SpineSpriteAtlasAsset>(path);
154 }
155 return null;
156 }
157
158 SkeletonDataAsset FindSkeletonDataAsset (TextAsset skeletonDataFile) {
159 string path = AssetDatabase.GetAssetPath(skeletonDataFile);
160 path = path.Replace(".json", AssetUtility.SkeletonDataSuffix + ".asset");
161 path = path.Replace(".skel.bytes", AssetUtility.SkeletonDataSuffix + ".asset");
162 if (System.IO.File.Exists(path)) {
163 return AssetDatabase.LoadAssetAtPath<SkeletonDataAsset>(path);
164 }
165 return null;
166 }
