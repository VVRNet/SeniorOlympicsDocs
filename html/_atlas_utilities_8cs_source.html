62 public static AtlasRegion ToAtlasRegion (this Texture2D t, Shader shader, float scale = DefaultScale, Material materialPropertySource = null) {
63 Material material = new Material(shader);
64 if (materialPropertySource != null) {
65 material.CopyPropertiesFromMaterial(materialPropertySource);
66 material.shaderKeywords = materialPropertySource.shaderKeywords;
67 }
68
69 material.mainTexture = t;
70 AtlasPage page = material.ToSpineAtlasPage();
71
72 float width = t.width;
73 float height = t.height;
74
75 AtlasRegion region = new AtlasRegion();
76 region.name = t.name;
77
78 // World space units
79 Vector2 boundsMin = Vector2.zero, boundsMax = new Vector2(width, height) * scale;
80
81 // Texture space/pixel units
82 region.width = (int)width;
83 region.originalWidth = (int)width;
84 region.height = (int)height;
85 region.originalHeight = (int)height;
86 region.offsetX = width * (0.5f - InverseLerp(boundsMin.x, boundsMax.x, 0));
87 region.offsetY = height * (0.5f - InverseLerp(boundsMin.y, boundsMax.y, 0));
88
89 // Use the full area of the texture.
90 region.u = 0;
91 region.v = 1;
92 region.u2 = 1;
93 region.v2 = 0;
94 region.x = 0;
95 region.y = 0;
96
97 region.page = page;
98
99 return region;
100 }

190 internal static AtlasRegion ToAtlasRegion (this Sprite s, bool isolatedTexture = false) {
191 AtlasRegion region = new AtlasRegion();
192 region.name = s.name;
193 region.index = -1;
194 region.degrees = s.packed && s.packingRotation != SpritePackingRotation.None ? 90 : 0;
195
196 // World space units
197 Bounds bounds = s.bounds;
198 Vector2 boundsMin = bounds.min, boundsMax = bounds.max;
199
200 // Texture space/pixel units
201 Rect spineRect = s.textureRect.SpineUnityFlipRect(s.texture.height);
202 Rect originalRect = s.rect;
203 region.width = (int)spineRect.width;
204 region.originalWidth = (int)originalRect.width;
205 region.height = (int)spineRect.height;
206 region.originalHeight = (int)originalRect.height;
207 region.offsetX = s.textureRectOffset.x + spineRect.width * (0.5f - InverseLerp(boundsMin.x, boundsMax.x, 0));
208 region.offsetY = s.textureRectOffset.y + spineRect.height * (0.5f - InverseLerp(boundsMin.y, boundsMax.y, 0));
209
210 if (isolatedTexture) {
211 region.u = 0;
212 region.v = 1;
213 region.u2 = 1;
214 region.v2 = 0;
215 region.x = 0;
216 region.y = 0;
217 } else {
218 Texture2D tex = s.texture;
219 Rect uvRect = TextureRectToUVRect(s.textureRect, tex.width, tex.height);
220 region.u = uvRect.xMin;
221 region.v = uvRect.yMax;
222 region.u2 = uvRect.xMax;
223 region.v2 = uvRect.yMin;
224 region.x = (int)spineRect.x;
225 region.y = (int)spineRect.y;
226 }
227
228 return region;
229 }

311 public static void GetRepackedAttachments (List<Attachment> sourceAttachments, List<Attachment> outputAttachments, Shader shader,
312 out Material outputMaterial, out Texture2D outputTexture,
313 int maxAtlasSize = 1024, int padding = 2, TextureFormat textureFormat = SpineTextureFormat, bool mipmaps = UseMipMaps,
314 string newAssetName = "Repacked Attachments",
315 Material materialPropertySource = null, bool clearCache = false, bool useOriginalNonrenderables = true,
316 int[] additionalTexturePropertyIDsToCopy = null, Texture2D[] additionalOutputTextures = null,
317 TextureFormat[] additionalTextureFormats = null, bool[] additionalTextureIsLinear = null) {
318
319 if (sourceAttachments == null) throw new System.ArgumentNullException("sourceAttachments");
320 if (outputAttachments == null) throw new System.ArgumentNullException("outputAttachments");
321 outputTexture = null;
322 if (additionalTexturePropertyIDsToCopy != null && additionalTextureIsLinear == null) {
323 additionalTextureIsLinear = new bool[additionalTexturePropertyIDsToCopy.Length];
324 for (int i = 0; i < additionalTextureIsLinear.Length; ++i) {
325 additionalTextureIsLinear[i] = true;
326 }
327 }
328
329 // Use these to detect and use shared regions.
330 existingRegions.Clear();
331 regionIndices.Clear();
332
333 // Collect all textures from original attachments.
334 int numTextureParamsToRepack = 1 + (additionalTexturePropertyIDsToCopy == null ? 0 : additionalTexturePropertyIDsToCopy.Length);
335 additionalOutputTextures = (additionalTexturePropertyIDsToCopy == null ? null : new Texture2D[additionalTexturePropertyIDsToCopy.Length]);
336 if (texturesToPackAtParam.Length < numTextureParamsToRepack)
337 Array.Resize(ref texturesToPackAtParam, numTextureParamsToRepack);
338 for (int i = 0; i < numTextureParamsToRepack; ++i) {
339 if (texturesToPackAtParam[i] != null)
340 texturesToPackAtParam[i].Clear();
341 else
342 texturesToPackAtParam[i] = new List<Texture2D>();
343 }
344 originalRegions.Clear();
345
346 if (!object.ReferenceEquals(sourceAttachments, outputAttachments)) {
347 outputAttachments.Clear();
348 outputAttachments.AddRange(sourceAttachments);
349 }
350
351 int newRegionIndex = 0;
352 for (int attachmentIndex = 0, n = sourceAttachments.Count; attachmentIndex < n; attachmentIndex++) {
353 Attachment originalAttachment = sourceAttachments[attachmentIndex];
354
355 if (originalAttachment is IHasTextureRegion) {
356 MeshAttachment originalMeshAttachment = originalAttachment as MeshAttachment;
357 IHasTextureRegion originalTextureAttachment = (IHasTextureRegion)originalAttachment;
358
359 Attachment newAttachment = (originalTextureAttachment.Sequence != null) ? originalAttachment :
360 (originalMeshAttachment != null) ? originalMeshAttachment.NewLinkedMesh() :
361 originalAttachment.Copy();
362 IHasTextureRegion newTextureAttachment = (IHasTextureRegion)newAttachment;
363 AtlasRegion region = newTextureAttachment.Region as AtlasRegion;
364 if (region == null && originalTextureAttachment.Sequence != null)
365 region = (AtlasRegion)originalTextureAttachment.Sequence.Regions[0];
366
367 int existingIndex;
368 if (existingRegions.TryGetValue(region, out existingIndex)) {
369 regionIndices.Add(existingIndex);
370 } else {
371 existingRegions.Add(region, newRegionIndex);
372 Sequence originalSequence = originalTextureAttachment.Sequence;
373 if (originalSequence != null) {
374 newTextureAttachment.Sequence = new Sequence(originalSequence);
375 for (int i = 0, regionCount = originalSequence.Regions.Length; i < regionCount; ++i) {
376 AtlasRegion sequenceRegion = (AtlasRegion)originalSequence.Regions[i];
377 AddRegionTexturesToPack(numTextureParamsToRepack, sequenceRegion, textureFormat, mipmaps,
378 additionalTextureFormats, additionalTexturePropertyIDsToCopy, additionalTextureIsLinear);
379 originalRegions.Add(sequenceRegion);
380 regionIndices.Add(newRegionIndex);
381 newRegionIndex++;
382 }
383 } else {
384 AddRegionTexturesToPack(numTextureParamsToRepack, region, textureFormat, mipmaps,
385 additionalTextureFormats, additionalTexturePropertyIDsToCopy, additionalTextureIsLinear);
386 originalRegions.Add(region);
387 regionIndices.Add(newRegionIndex);
388 newRegionIndex++;
389 }
390 }
391 outputAttachments[attachmentIndex] = newAttachment;
392 } else {
393 outputAttachments[attachmentIndex] = useOriginalNonrenderables ? originalAttachment : originalAttachment.Copy();
394 regionIndices.Add(NonrenderingRegion); // Output attachments pairs with regionIndices list 1:1. Pad with a sentinel if the attachment doesn't have a region.
395 }
396 }
397
398 // Rehydrate the repacked textures as a Material, Spine atlas and Spine.AtlasAttachments
399 Material newMaterial = new Material(shader);
400 if (materialPropertySource != null) {
401 newMaterial.CopyPropertiesFromMaterial(materialPropertySource);
402 newMaterial.shaderKeywords = materialPropertySource.shaderKeywords;
403 }
404 newMaterial.name = newAssetName;
405
406 Rect[] rects = null;
407 for (int i = 0; i < numTextureParamsToRepack; ++i) {
408 // Fill a new texture with the collected attachment textures.
409 Texture2D newTexture = new Texture2D(maxAtlasSize, maxAtlasSize,
410 (i > 0 && additionalTextureFormats != null && i - 1 < additionalTextureFormats.Length) ?
411 additionalTextureFormats[i - 1] : textureFormat,
412 mipmaps,
413 (i > 0) ? additionalTextureIsLinear[i - 1] : false);
414 newTexture.mipMapBias = AtlasUtilities.DefaultMipmapBias;
415
416 List<Texture2D> texturesToPack = texturesToPackAtParam[i];
417 if (texturesToPack.Count > 0) {
418 Texture2D sourceTexture = texturesToPack[0];
419 newTexture.CopyTextureAttributesFrom(sourceTexture);
420 }
421 newTexture.name = newAssetName;
422 Rect[] rectsForTexParam = newTexture.PackTextures(texturesToPack.ToArray(), padding, maxAtlasSize);
423 if (i == 0) {
424 rects = rectsForTexParam;
425 newMaterial.mainTexture = newTexture;
426 outputTexture = newTexture;
427 } else {
428 newMaterial.SetTexture(additionalTexturePropertyIDsToCopy[i - 1], newTexture);
429 additionalOutputTextures[i - 1] = newTexture;
430 }
431 }
432
433 AtlasPage page = newMaterial.ToSpineAtlasPage();
434 page.name = newAssetName;
435
436 repackedRegions.Clear();
437 for (int i = 0, n = originalRegions.Count; i < n; i++) {
438 AtlasRegion oldRegion = originalRegions[i];
439 AtlasRegion newRegion = UVRectToAtlasRegion(rects[i], oldRegion, page);
440 repackedRegions.Add(newRegion);
441 }
442
443 // Map the cloned attachments to the repacked atlas.
444 for (int attachmentIndex = 0, repackedIndex = 0, n = outputAttachments.Count;
445 attachmentIndex < n;
446 ++attachmentIndex, ++repackedIndex) {
447
448 Attachment attachment = outputAttachments[attachmentIndex];
449 IHasTextureRegion textureAttachment = attachment as IHasTextureRegion;
450 if (textureAttachment != null) {
451 if (textureAttachment.Sequence != null) {
452 TextureRegion[] regions = textureAttachment.Sequence.Regions;
453 textureAttachment.Region = repackedRegions[regionIndices[repackedIndex]];
454 for (int r = 0, regionCount = regions.Length; r < regionCount; ++r) {
455 regions[r] = repackedRegions[regionIndices[repackedIndex++]];
456 }
457 --repackedIndex;
458 } else {
459 textureAttachment.Region = repackedRegions[regionIndices[repackedIndex]];
460 }
461 textureAttachment.UpdateRegion();
462 }
463 }
464
465 // Clean up.
466 if (clearCache)
467 AtlasUtilities.ClearCache();
468
469 outputMaterial = newMaterial;
470 }

472 private static void AddRegionTexturesToPack (int numTextureParamsToRepack, AtlasRegion region,
473 TextureFormat textureFormat, bool mipmaps, TextureFormat[] additionalTextureFormats,
474 int[] additionalTexturePropertyIDsToCopy, bool[] additionalTextureIsLinear) {
475
476 for (int i = 0; i < numTextureParamsToRepack; ++i) {
477 Texture2D regionTexture = (i == 0 ?
478 region.ToTexture(textureFormat, mipmaps) :
479 region.ToTexture((additionalTextureFormats != null && i - 1 < additionalTextureFormats.Length) ?
480 additionalTextureFormats[i - 1] : textureFormat,
481 mipmaps, additionalTexturePropertyIDsToCopy[i - 1], additionalTextureIsLinear[i - 1]));
482 texturesToPackAtParam[i].Add(regionTexture);
483 }
484 }

532 int maxAtlasSize = 1024, int padding = 2, TextureFormat textureFormat = SpineTextureFormat, bool mipmaps = UseMipMaps,

603 public static Texture2D ToTexture (this AtlasRegion ar, TextureFormat textureFormat = SpineTextureFormat, bool mipmaps = UseMipMaps,
604 int texturePropertyId = 0, bool linear = false, bool applyPMA = false) {
605
606 Texture2D output;
607
608 IntAndAtlasRegionKey cacheKey = new IntAndAtlasRegionKey(texturePropertyId, ar);
609 CachedRegionTextures.TryGetValue(cacheKey, out output);
610 if (output == null) {
611 Texture2D sourceTexture = texturePropertyId == 0 ? ar.GetMainTexture() : ar.GetTexture(texturePropertyId);
612 Rect r = ar.GetUnityRect();
613 int width = (int)r.width;
614 int height = (int)r.height;
615 output = new Texture2D(width, height, textureFormat, mipmaps, linear) { name = ar.name };
616 output.CopyTextureAttributesFrom(sourceTexture);
617 if (applyPMA)
618 AtlasUtilities.CopyTextureApplyPMA(sourceTexture, r, output);
619 else
620 AtlasUtilities.CopyTexture(sourceTexture, r, output);
621 CachedRegionTextures.Add(cacheKey, output);
622 CachedRegionTexturesList.Add(output);
623 }
624
625 return output;
626 }

628 static Texture2D ToTexture (this Sprite s, TextureFormat textureFormat = SpineTextureFormat,
629 bool mipmaps = UseMipMaps, bool linear = false, bool applyPMA = false) {
630
631 Texture2D spriteTexture = s.texture;
632 Rect r;
633 if (!s.packed || s.packingMode == SpritePackingMode.Rectangle) {
634 r = s.textureRect;
635 } else {
636 r = new Rect();
637 r.xMin = Math.Min(s.uv[0].x, s.uv[1].x) * spriteTexture.width;
638 r.xMax = Math.Max(s.uv[0].x, s.uv[1].x) * spriteTexture.width;
639 r.yMin = Math.Min(s.uv[0].y, s.uv[2].y) * spriteTexture.height;
640 r.yMax = Math.Max(s.uv[0].y, s.uv[2].y) * spriteTexture.height;
641#if UNITY_EDITOR
642 if (s.uv.Length > 4) {
643 Debug.LogError("When using a tightly packed SpriteAtlas with Spine, you may only access Sprites that are packed as 'FullRect' from it! " +
644 "You can either disable 'Tight Packing' at the whole SpriteAtlas, or change the single Sprite's TextureImporter Setting 'MeshType' to 'Full Rect'." +
645 "Sprite Asset: " + s.name, s);
646 }
647#endif
648 }
649 Texture2D newTexture = new Texture2D((int)r.width, (int)r.height, textureFormat, mipmaps, linear);
650 newTexture.CopyTextureAttributesFrom(spriteTexture);
651 if (applyPMA)
652 AtlasUtilities.CopyTextureApplyPMA(spriteTexture, r, newTexture);
653 else
654 AtlasUtilities.CopyTexture(spriteTexture, r, newTexture);
655 return newTexture;
656 }

660

662 newTexture.CopyTextureAttributesFrom(t);
663 if (applyPMA)

667 return newTexture;

681 static void CopyTextureApplyPMA (Texture2D source, Rect sourceRect, Texture2D destination) {
682 Color[] pixelBuffer = source.GetPixels((int)sourceRect.x, (int)sourceRect.y, (int)sourceRect.width, (int)sourceRect.height);

753 static AtlasRegion UVRectToAtlasRegion (Rect uvRect, AtlasRegion referenceRegion, AtlasPage page) {
754 Rect tr = UVRectToTextureRect(uvRect, page.width, page.height);
755 Rect rr = tr.SpineUnityFlipRect(page.height);
756
757 int x = (int)rr.x;
758 int y = (int)rr.y;
759 int w = (int)rr.width;
760 int h = (int)rr.height;
761
762 if (referenceRegion.degrees == 270) {
763 int tempW = w;
764 w = h;
765 h = tempW;
766 }
767
768 // Note: originalW and originalH need to be scaled according to the
769 // repacked width and height, repacking can mess with aspect ratio, etc.
770 int originalW = Mathf.RoundToInt((float)w * ((float)referenceRegion.originalWidth / (float)referenceRegion.width));
771 int originalH = Mathf.RoundToInt((float)h * ((float)referenceRegion.originalHeight / (float)referenceRegion.height));
772
773 int offsetX = Mathf.RoundToInt((float)referenceRegion.offsetX * ((float)w / (float)referenceRegion.width));
774 int offsetY = Mathf.RoundToInt((float)referenceRegion.offsetY * ((float)h / (float)referenceRegion.height));
775
776 float u = uvRect.xMin;
777 float u2 = uvRect.xMax;
778 float v = uvRect.yMax;
779 float v2 = uvRect.yMin;
780
781 if (referenceRegion.degrees == 270) {
782 // at a 270 degree region, u2/v2 deltas and atlas width/height are swapped, and delta-v is negative.
783 float du = uvRect.width; // u2 - u;
784 float dv = uvRect.height; // v - v2;
785 float atlasAspectRatio = (float)page.width / (float)page.height;
786 u2 = u + (dv / atlasAspectRatio);
787 v2 = v - (du * atlasAspectRatio);
788 }
789
790 return new AtlasRegion {
791 page = page,
792 name = referenceRegion.name,
793
794 u = u,
795 u2 = u2,
796 v = v,
797 v2 = v2,
798
799 index = -1,
800
801 width = w,
802 originalWidth = originalW,
803 height = h,
804 originalHeight = originalH,
805 offsetX = offsetX,
806 offsetY = offsetY,
807 x = x,
808 y = y,
809
810 rotate = referenceRegion.rotate,
811 degrees = referenceRegion.degrees
812 };
813 }
