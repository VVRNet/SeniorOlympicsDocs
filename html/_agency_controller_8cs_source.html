299 public IEnumerator LoadUrl(string url, Image image)
300 {
301 // URL 유효성 검사
302 if (!Uri.IsWellFormedUriString(url, UriKind.Absolute))
303 {
304 yield break;
305 }
306
307 using (UnityWebRequest uwr = UnityWebRequestTexture.GetTexture(url))
308 {
309 yield return uwr.SendWebRequest();
310
311 if (uwr.result == UnityWebRequest.Result.ConnectionError || uwr.result == UnityWebRequest.Result.ProtocolError)
312 {
313 Debug.Log(uwr.error);
314 }
315 else
316 {
317 Texture2D texture = DownloadHandlerTexture.GetContent(uwr);
318 Sprite sprite = Sprite.Create(texture, new Rect(0, 0, texture.width, texture.height), new Vector2(0.5f, 0.5f));
319 image.sprite = sprite;
320 }
321 }
