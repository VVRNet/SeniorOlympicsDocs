20 static public bool EvaluateExpression(string expression, EvaluateFunction evalFunction)
21 {
22 //Remove white spaces and double && ||
23 string cleanExpr = "";
24 for(int i = 0; i < expression.Length; i++)
25 {
26 switch(expression[i])
27 {
28 case ' ': break;
29 case '&': cleanExpr += expression[i]; i++; break;
30 case '|': cleanExpr += expression[i]; i++; break;
31 default: cleanExpr += expression[i]; break;
32 }
33 }
34
35 List<Token> tokens = new List<Token>();
36 StringReader reader = new StringReader(cleanExpr);
37 Token t = null;
38 do
39 {
40 t = new Token(reader);
41 tokens.Add(t);
42 } while(t.type != Token.TokenType.EXPR_END);
43
44 List<Token> polishNotation = Token.TransformToPolishNotation(tokens);
45
46 var enumerator = polishNotation.GetEnumerator();
47 enumerator.MoveNext();
48 Expression root = MakeExpression(ref enumerator, evalFunction);
49
50 return root.Evaluate();
51 }

56 public class Token
57 {
58 static Dictionary<char, KeyValuePair<TokenType, string>> typesDict = new Dictionary<char, KeyValuePair<TokenType, string>>()
59 {
60 {'(', new KeyValuePair<TokenType, string>(TokenType.OPEN_PAREN, "(")},
61 {')', new KeyValuePair<TokenType, string>(TokenType.CLOSE_PAREN, ")")},
62 {'!', new KeyValuePair<TokenType, string>(TokenType.UNARY_OP, "NOT")},
63 {'&', new KeyValuePair<TokenType, string>(TokenType.BINARY_OP, "AND")},
64 {'|', new KeyValuePair<TokenType, string>(TokenType.BINARY_OP, "OR")}
65 };
66
67 public enum TokenType
68 {
69 OPEN_PAREN,
70 CLOSE_PAREN,
71 UNARY_OP,
72 BINARY_OP,
73 LITERAL,
74 EXPR_END
75 }
76
77 public TokenType type;
78 public string value;
79
80 public Token(StringReader s)
81 {
82 int c = s.Read();
83 if(c == -1)
84 {
85 type = TokenType.EXPR_END;
86 value = "";
87 return;
88 }
89
90 char ch = (char)c;
91
92 //Special case: solve bug where !COND_FALSE_1 && COND_FALSE_2 would return True
93 bool embeddedNot = (ch == '!' && s.Peek() != '(');
94
95 if(typesDict.ContainsKey(ch) && !embeddedNot)
96 {
97 type = typesDict[ch].Key;
98 value = typesDict[ch].Value;
99 }
100 else
101 {
102 string str = "";
103 str += ch;
104 while(s.Peek() != -1 && !typesDict.ContainsKey((char)s.Peek()))
105 {
106 str += (char)s.Read();
107 }
108 type = TokenType.LITERAL;
109 value = str;
110 }
111 }

113 static public List<Token> TransformToPolishNotation(List<Token> infixTokenList)
114 {
115 Queue<Token> outputQueue = new Queue<Token>();
116 Stack<Token> stack = new Stack<Token>();
117
118 int index = 0;
119 while(infixTokenList.Count > index)
120 {
121 Token t = infixTokenList[index];
122
123 switch(t.type)
124 {
125 case Token.TokenType.LITERAL:
126 outputQueue.Enqueue(t);
127 break;
128 case Token.TokenType.BINARY_OP:
129 case Token.TokenType.UNARY_OP:
130 case Token.TokenType.OPEN_PAREN:
131 stack.Push(t);
132 break;
133 case Token.TokenType.CLOSE_PAREN:
134 while(stack.Peek().type != Token.TokenType.OPEN_PAREN)
135 {
136 outputQueue.Enqueue(stack.Pop());
137 }
138 stack.Pop();
139 if(stack.Count > 0 && stack.Peek().type == Token.TokenType.UNARY_OP)
140 {
141 outputQueue.Enqueue(stack.Pop());
142 }
143 break;
144 default:
145 break;
146 }
147
148 index++;
149 }
150 while(stack.Count > 0)
151 {
152 outputQueue.Enqueue(stack.Pop());
153 }
154
155 var list = new List<Token>(outputQueue);
156 list.Reverse();
157 return list;
158 }

241 static public Expression MakeExpression(ref List<Token>.Enumerator polishNotationTokensEnumerator, EvaluateFunction _evalFunction)
242 {
243 if(polishNotationTokensEnumerator.Current.type == Token.TokenType.LITERAL)
244 {
245 Expression lit = new ExpressionLeaf(_evalFunction, polishNotationTokensEnumerator.Current.value);
246 polishNotationTokensEnumerator.MoveNext();
247 return lit;
248 }
249 else
250 {
251 if(polishNotationTokensEnumerator.Current.value == "NOT")
252 {
253 polishNotationTokensEnumerator.MoveNext();
254 Expression operand = MakeExpression(ref polishNotationTokensEnumerator, _evalFunction);
255 return new ExpressionNot(operand);
256 }
257 else if(polishNotationTokensEnumerator.Current.value == "AND")
258 {
259 polishNotationTokensEnumerator.MoveNext();
260 Expression left = MakeExpression(ref polishNotationTokensEnumerator, _evalFunction);
261 Expression right = MakeExpression(ref polishNotationTokensEnumerator, _evalFunction);
262 return new ExpressionAnd(left, right);
263 }
264 else if(polishNotationTokensEnumerator.Current.value == "OR")
265 {
266 polishNotationTokensEnumerator.MoveNext();
267 Expression left = MakeExpression(ref polishNotationTokensEnumerator, _evalFunction);
268 Expression right = MakeExpression(ref polishNotationTokensEnumerator, _evalFunction);
269 return new ExpressionOr(left, right);
270 }
271 }
272 return null;
273 }
