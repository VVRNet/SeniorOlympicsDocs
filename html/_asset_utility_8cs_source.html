95 public static void HandleOnPostprocessAllAssets (string[] imported, List<string> texturesWithoutMetaFile) {
96 // In case user used "Assets -> Reimport All", during the import process,
97 // asset database is not initialized until some point. During that period,
98 // all attempts to load any assets using API (i.e. AssetDatabase.LoadAssetAtPath)
99 // will return null, and as result, assets won't be loaded even if they actually exists,
100 // which may lead to numerous importing errors.
101 // This situation also happens if Library folder is deleted from the project, which is a pretty
102 // common case, since when using version control systems, the Library folder must be excluded.
103 //
104 // So to avoid this, in case asset database is not available, we delay loading the assets
105 // until next time.
106 //
107 // Unity *always* reimports some internal assets after the process is done, so this method
108 // is always called once again in a state when asset database is available.
109 //
110 // Checking whether AssetDatabase is initialized is done by attempting to load
111 // a known "marker" asset that should always be available. Failing to load this asset
112 // means that AssetDatabase is not initialized.
113 AssetUtility.assetsImportedInWrongState.UnionWith(imported);
114 if (AssetDatabaseAvailabilityDetector.IsAssetDatabaseAvailable()) {
115 string[] combinedAssets = AssetUtility.assetsImportedInWrongState.ToArray();
116 AssetUtility.assetsImportedInWrongState.Clear();
117 AssetUtility.ImportSpineContent(combinedAssets, texturesWithoutMetaFile);
118 }

135 public static List<string> GetRequiredAtlasRegions (string skeletonDataPath) {
136 HashSet<string> requiredPaths = new HashSet<string>();
137
138 if (skeletonDataPath.Contains(".skel")) {
139 List<string> requiredPathsResult = new List<string>();
140 AddRequiredAtlasRegionsFromBinary(skeletonDataPath, requiredPathsResult);
141 return requiredPathsResult;
142 }
143
144 TextReader reader = null;
145 TextAsset spineJson = AssetDatabase.LoadAssetAtPath<TextAsset>(skeletonDataPath);
146 Dictionary<string, object> root = null;
147 try {
148 if (spineJson != null) {
149 reader = new StringReader(spineJson.text);
150 } else {
151 // On a "Reimport All" the order of imports can be wrong, thus LoadAssetAtPath() above could return null.
152 // as a workaround, we provide a fallback reader.
153 reader = new StreamReader(skeletonDataPath);
154 }
155 root = Json.Deserialize(reader) as Dictionary<string, object>;
156 } finally {
157 if (reader != null)
158 reader.Dispose();
159 }
160
161 if (root == null || !root.ContainsKey("skins"))
162 return new List<string>();
163
164 List<object> skinsList = root["skins"] as List<object>;
165 if (skinsList == null)
166 return new List<string>();
167
168 foreach (Dictionary<string, object> skinMap in skinsList) {
169 if (!skinMap.ContainsKey("attachments"))
170 continue;
171 foreach (var slot in (Dictionary<string, object>)skinMap["attachments"]) {
172 foreach (var attachment in ((Dictionary<string, object>)slot.Value)) {
173 var data = ((Dictionary<string, object>)attachment.Value);
174
175 // Ignore non-atlas-requiring types.
176 if (data.ContainsKey("type")) {
177 AttachmentType attachmentType;
178 string typeString = (string)data["type"];
179 try {
180 attachmentType = (AttachmentType)System.Enum.Parse(typeof(AttachmentType), typeString, true);
181 } catch (System.ArgumentException e) {
182 // For more info, visit: http://esotericsoftware.com/forum/Spine-editor-and-runtime-version-management-6534
183 Debug.LogWarning(string.Format("Unidentified Attachment type: \"{0}\". Skeleton may have been exported from an incompatible Spine version.", typeString), spineJson);
184 throw e;
185 }
186
187 if (!AtlasTypes.Contains(attachmentType))
188 continue;
189 }
190
191 if (data.ContainsKey("path"))
192 requiredPaths.Add((string)data["path"]);
193 else if (data.ContainsKey("name"))
194 requiredPaths.Add((string)data["name"]);
195 else if (data.ContainsKey("sequence")) {
196 Sequence sequence = SkeletonJson.ReadSequence(data["sequence"]);
197 if (sequence != null)
198 for (int index = 0; index < sequence.Regions.Length; ++index)
199 requiredPaths.Add(sequence.GetPath(attachment.Key, index));
200 else
201 requiredPaths.Add(attachment.Key);
202 } else
203 requiredPaths.Add(attachment.Key);
204 }
205 }
206 }
207
208 return requiredPaths.ToList();
209 }

211 internal static void AddRequiredAtlasRegionsFromBinary (string skeletonDataPath, List<string> requiredPaths) {
212 SkeletonBinary binary = new SkeletonBinary(new AtlasRequirementLoader(requiredPaths));
213 Stream input = null;
214 TextAsset data = AssetDatabase.LoadAssetAtPath<TextAsset>(skeletonDataPath);
215 try {
216 if (data != null) {
217 input = new MemoryStream(data.bytes);
218 } else {
219 // On a "Reimport All" the order of imports can be wrong, thus LoadAssetAtPath() above could return null.
220 // as a workaround, we provide a fallback reader.
221 input = File.Open(skeletonDataPath, FileMode.Open, FileAccess.Read);
222 }
223 binary.ReadSkeletonData(input);
224 } finally {
225 if (input != null)
226 input.Dispose();
227 }
228 binary = null;
229 }

239 internal static AtlasRegion FindRegionIgnoringNumberSuffix (this Atlas atlas, string regionPath) {

242 return region;
243 return atlas.FindRegionWithNumberSuffix(regionPath);

246 internal static AtlasRegion FindRegionWithNumberSuffix (this Atlas atlas, string regionPath) {
247 int pathLength = regionPath.Length;
248 foreach (AtlasRegion region in atlas.Regions) {
249 string name = region.name;
250 if (name.StartsWith(regionPath)) {
251 string suffix = name.Substring(pathLength);
252 if (suffix.All(c => c >= '0' && c <= '9'))
253 return region;
254 }
255 }
256 return null;
257 }

259 internal static AtlasAssetBase GetMatchingAtlas (List<string> requiredPaths, List<AtlasAssetBase> atlasAssets) {
260 AtlasAssetBase atlasAssetMatch = null;
261
262 foreach (AtlasAssetBase a in atlasAssets) {
263 Atlas atlas = a.GetAtlas();
264 bool failed = false;
265 foreach (string regionPath in requiredPaths) {
266 if (atlas.FindRegionIgnoringNumberSuffix(regionPath) == null) {
267 failed = true;
268 break;
269 }
270 }
271
272 if (!failed) {
273 atlasAssetMatch = a;
274 break;
275 }
276 }
277
278 return atlasAssetMatch;

340 public static void ImportSpineContent (string[] imported, List<string> texturesWithoutMetaFile,
341 bool reimport = false) {
342
343 List<string> atlasPaths = new List<string>();
344 List<string> imagePaths = new List<string>();
345 List<PathAndProblemInfo> skeletonPaths = new List<PathAndProblemInfo>();
346 CompatibilityProblemInfo compatibilityProblemInfo = null;
347
348 foreach (string str in imported) {
349 string extension = Path.GetExtension(str).ToLower();
350 switch (extension) {
351 case ".atlas":
352 if (SpineEditorUtilities.Preferences.atlasTxtImportWarning) {
353 Debug.LogWarningFormat("`{0}` : If this file is a Spine atlas, please change its extension to `.atlas.txt`. This is to allow Unity to recognize it and avoid filename collisions. You can also set this file extension when exporting from the Spine editor.", str);
354 }
355 break;
356 case ".skel":
357 if (SpineEditorUtilities.Preferences.atlasTxtImportWarning) {
358 Debug.LogWarningFormat("`{0}` : If this file is a Spine skeleton, please change its extension to `.skel.bytes`. This is to allow Unity to recognize it and avoid filename collisions. You can also set this file extension when exporting from the Spine editor.", str);
359 }
360 break;
361 case ".txt":
362 if (str.EndsWith(".atlas.txt", System.StringComparison.Ordinal))
363 atlasPaths.Add(str);
364 break;
365 case ".png":
366 case ".jpg":
367 imagePaths.Add(str);
368 break;
369 case ".json": {
370 TextAsset jsonAsset = AssetDatabase.LoadAssetAtPath<TextAsset>(str);
371 string problemDescription = null;
372 if (jsonAsset != null && IsSpineData(jsonAsset, out compatibilityProblemInfo, ref problemDescription))
373 skeletonPaths.Add(new PathAndProblemInfo(str, compatibilityProblemInfo, problemDescription));
374 if (problemDescription != null)
375 Debug.LogError(problemDescription, jsonAsset);
376 break;
377 }
378 case ".bytes": {
379 if (str.ToLower().EndsWith(".skel.bytes", System.StringComparison.Ordinal)) {
380 TextAsset binaryAsset = AssetDatabase.LoadAssetAtPath<TextAsset>(str);
381 string problemDescription = null;
382 if (IsSpineData(binaryAsset, out compatibilityProblemInfo, ref problemDescription))
383 skeletonPaths.Add(new PathAndProblemInfo(str, compatibilityProblemInfo, problemDescription));
384 if (problemDescription != null)
385 Debug.LogError(problemDescription, binaryAsset);
386 }
387 break;
388 }
389 }
390 }
391
392 AddDependentAtlasIfImageChanged(atlasPaths, imagePaths);
393
394 // Import atlases first.
395 List<AtlasAssetBase> newAtlases = new List<AtlasAssetBase>();
396 foreach (string ap in atlasPaths) {
397#if PROBLEMATIC_PACKAGE_ASSET_MODIFICATION
398 if (ap.StartsWith("Packages"))
399 continue;
400#endif
401 TextAsset atlasText = AssetDatabase.LoadAssetAtPath<TextAsset>(ap);
402 AtlasAssetBase atlas = IngestSpineAtlas(atlasText, texturesWithoutMetaFile);
403 newAtlases.Add(atlas);
404 }
405 AddDependentSkeletonIfAtlasChanged(skeletonPaths, atlasPaths);
406
407 // Import skeletons and match them with atlases.
408 bool abortSkeletonImport = false;
409 foreach (PathAndProblemInfo skeletonPathEntry in skeletonPaths) {
410 string skeletonPath = skeletonPathEntry.path;
411 CompatibilityProblemInfo compatibilityProblems = skeletonPathEntry.compatibilityProblems;
412 string otherProblemDescription = skeletonPathEntry.otherProblemDescription;
413#if PROBLEMATIC_PACKAGE_ASSET_MODIFICATION
414 if (skeletonPath.StartsWith("Packages"))
415 continue;
416#endif
417 if (!reimport && CheckForValidSkeletonData(skeletonPath)) {
418 ReloadSkeletonData(skeletonPath, compatibilityProblems);
419 continue;
420 }
421
422 TextAsset loadedAsset = AssetDatabase.LoadAssetAtPath<TextAsset>(skeletonPath);
423 if (compatibilityProblems != null) {
424 IngestIncompatibleSpineProject(loadedAsset, compatibilityProblems);
425 continue;
426 }
427 if (otherProblemDescription != null) {
428 continue;
429 }
430
431 string dir = Path.GetDirectoryName(skeletonPath).Replace('\\', '/');
432
433#if SPINE_TK2D
434 IngestSpineProject(loadedAsset, null);
435#else
436 string skeletonName = Path.GetFileNameWithoutExtension(skeletonPath);
437 List<string> requiredPaths = GetRequiredAtlasRegions(skeletonPath);
438
439 List<AtlasAssetBase> atlasesForSkeleton = FindAtlasesAtPath(dir);
440 atlasesForSkeleton = atlasesForSkeleton.Union(newAtlases).ToList();
441 List<AtlasAssetBase> atlasesInSameDir = atlasesForSkeleton.Where(
442 atlas => AssetDatabase.GetAssetPath(atlas).Contains(dir)).ToList();
443
444 AtlasAssetBase atlasMatch = GetMatchingAtlas(requiredPaths, skeletonName, atlasesInSameDir);
445 if (atlasMatch == null && atlasesInSameDir.Count > 0) {
446 AtlasAssetBase firstAtlas = atlasesInSameDir[0];
447 Debug.LogWarning(string.Format(
448 "'{0}' atlas found in skeleton directory does not contain all required attachments",
449 firstAtlas.name), firstAtlas);
450
451 List<AtlasAssetBase> atlasesInOtherDir = atlasesForSkeleton.Except(atlasesInSameDir).ToList();
452 atlasMatch = GetMatchingAtlas(requiredPaths, skeletonName, atlasesInOtherDir);
453 if (atlasMatch != null) {
454 Debug.Log(string.Format(
455 "Using suitable atlas '{0}' of other imported directory. If this is the " +
456 "wrong atlas asset, please assign the correct one at the SkeletonData asset.",
457 atlasMatch.name), atlasMatch);
458 }
459 }
460
461 if (atlasMatch != null || requiredPaths.Count == 0) {
462 IngestSpineProject(loadedAsset, atlasMatch);
463 } else {
464 SkeletonImportDialog(skeletonPath, atlasesForSkeleton, requiredPaths, ref abortSkeletonImport);
465 }
466
467 if (abortSkeletonImport)
468 break;
469#endif
470 }
471
472 if (atlasPaths.Count > 0 || imagePaths.Count > 0 || skeletonPaths.Count > 0) {
473 SkeletonDataAssetInspector[] skeletonDataInspectors = Resources.FindObjectsOfTypeAll<SkeletonDataAssetInspector>();
474 foreach (SkeletonDataAssetInspector inspector in skeletonDataInspectors) {
475 inspector.UpdateSkeletonData();
476 }
477 }
478
479 // Any post processing of images
480
481 // Under some circumstances (e.g. on first import) SkeletonGraphic objects
482 // have their skeletonGraphic.skeletonDataAsset reference corrupted
483 // by the instance of the ScriptableObject being destroyed but still assigned.
484 // Here we restore broken skeletonGraphic.skeletonDataAsset references.
485 SkeletonGraphic[] skeletonGraphicObjects = Resources.FindObjectsOfTypeAll(typeof(SkeletonGraphic)) as SkeletonGraphic[];
486 foreach (SkeletonGraphic skeletonGraphic in skeletonGraphicObjects) {
487
488 if (skeletonGraphic.skeletonDataAsset == null) {
489 int skeletonGraphicID = skeletonGraphic.GetInstanceID();
490 if (SpineEditorUtilities.DataReloadHandler.savedSkeletonDataAssetAtSKeletonGraphicID.ContainsKey(skeletonGraphicID)) {
491 string assetPath = SpineEditorUtilities.DataReloadHandler.savedSkeletonDataAssetAtSKeletonGraphicID[skeletonGraphicID];
492 skeletonGraphic.skeletonDataAsset = (SkeletonDataAsset)AssetDatabase.LoadAssetAtPath<SkeletonDataAsset>(assetPath);
493 }
494 }
495 }

529 static void ReloadSkeletonData (string skeletonJSONPath, CompatibilityProblemInfo compatibilityProblemInfo) {
530 string dir = Path.GetDirectoryName(skeletonJSONPath).Replace('\\', '/');
531 TextAsset textAsset = AssetDatabase.LoadAssetAtPath<TextAsset>(skeletonJSONPath);
532 DirectoryInfo dirInfo = new DirectoryInfo(dir);
533 FileInfo[] files = dirInfo.GetFiles("*.asset");
534
535 foreach (FileInfo f in files) {
536 string localPath = dir + "/" + f.Name;
537 UnityEngine.Object obj = AssetDatabase.LoadAssetAtPath(localPath, typeof(Object));
538 SkeletonDataAsset skeletonDataAsset = obj as SkeletonDataAsset;
539 if (skeletonDataAsset != null) {
540 if (skeletonDataAsset.skeletonJSON == textAsset) {
541 if (Selection.activeObject == skeletonDataAsset)
542 Selection.activeObject = null;
543
544 if (compatibilityProblemInfo != null) {
545 SkeletonDataCompatibility.DisplayCompatibilityProblem(compatibilityProblemInfo.DescriptionString(), textAsset);
546 return;
547 }
548
549 Debug.LogFormat("Changes to '{0}' or atlas detected. Clearing SkeletonDataAsset: {1}", skeletonJSONPath, localPath);
550 SpineEditorUtilities.ClearSkeletonDataAsset(skeletonDataAsset);
551
552 string guid = AssetDatabase.AssetPathToGUID(AssetDatabase.GetAssetPath(skeletonDataAsset));
553 string lastHash = EditorPrefs.GetString(guid + "_hash");
554
555 // For some weird reason sometimes Unity loses the internal Object pointer,
556 // and as a result, all comparisons with null returns true.
557 // But the C# wrapper is still alive, so we can "restore" the object
558 // by reloading it from its Instance ID.
559 AtlasAssetBase[] skeletonDataAtlasAssets = skeletonDataAsset.atlasAssets;
560 if (skeletonDataAtlasAssets != null) {
561 for (int i = 0; i < skeletonDataAtlasAssets.Length; i++) {
562 if (!ReferenceEquals(null, skeletonDataAtlasAssets[i]) &&
563 skeletonDataAtlasAssets[i].Equals(null) &&
564 skeletonDataAtlasAssets[i].GetInstanceID() != 0
565 ) {
566 skeletonDataAtlasAssets[i] = EditorUtility.InstanceIDToObject(skeletonDataAtlasAssets[i].GetInstanceID()) as AtlasAssetBase;
567 }
568 }
569 }
570
571 SkeletonData skeletonData = skeletonDataAsset.GetSkeletonData(true);
572 if (skeletonData != null)
573 BlendModeMaterialsUtility.UpdateBlendModeMaterials(skeletonDataAsset, ref skeletonData);
574
575 string currentHash = skeletonData != null ? skeletonData.Hash : null;
576
577#if SPINE_SKELETONMECANIM
578 if (currentHash == null || lastHash != currentHash)
579 SkeletonBaker.UpdateMecanimClips(skeletonDataAsset);
580#endif
581
582 // if (currentHash == null || lastHash != currentHash)
583 // Do any upkeep on synchronized assets
584
585 if (currentHash != null)
586 EditorPrefs.SetString(guid + "_hash", currentHash);
587 }
588 SpineEditorUtilities.DataReloadHandler.ReloadSceneSkeletonComponents(skeletonDataAsset);
589 SpineEditorUtilities.DataReloadHandler.ReloadAnimationReferenceAssets(skeletonDataAsset);
590 }
591 }

611 static AtlasAssetBase IngestSpineAtlas (TextAsset atlasText, List<string> texturesWithoutMetaFile) {
612 if (atlasText == null) {
613 Debug.LogWarning("Atlas source cannot be null!");
614 return null;
615 }
616
617 string primaryName = Path.GetFileNameWithoutExtension(atlasText.name).Replace(".atlas", "");
618 string assetPath = Path.GetDirectoryName(AssetDatabase.GetAssetPath(atlasText)).Replace('\\', '/');
619
620 string atlasPath = assetPath + "/" + primaryName + AtlasSuffix + ".asset";
621
622 SpineAtlasAsset atlasAsset = (SpineAtlasAsset)AssetDatabase.LoadAssetAtPath(atlasPath, typeof(SpineAtlasAsset));
623
624 List<Material> vestigialMaterials = new List<Material>();
625
626 if (atlasAsset == null)
627 atlasAsset = SpineAtlasAsset.CreateInstance<SpineAtlasAsset>();
628 else {
629 foreach (Material m in atlasAsset.materials)
630 vestigialMaterials.Add(m);
631 }
632
633 protectFromStackGarbageCollection.Add(atlasAsset);
634 atlasAsset.atlasFile = atlasText;
635
636 List<string> pageFiles = new List<string>();
637 atlasAsset.Clear(); // force reload
638 Atlas atlas = atlasAsset.GetAtlas(onlyMetaData: true);
639 if (atlas != null) {
640 foreach (AtlasPage page in atlas.Pages)
641 pageFiles.Add(page.name);
642 }
643
644 List<Material> populatingMaterials = new List<Material>(pageFiles.Count);
645
646 for (int i = 0; i < pageFiles.Count; i++) {

655 }

657 string pageName = Path.GetFileNameWithoutExtension(pageFiles[i]);
658
659 //because this looks silly
660 if (pageName == primaryName && pageFiles.Count == 1)
661 pageName = "Material";

674 }
675 } else {

678 material.mainTexture = texture;

681 }

684 populatingMaterials.Add(material);
685 }
686 }

691 AssetDatabase.DeleteAsset(AssetDatabase.GetAssetPath(vestigialMaterials[i]));

693 if (AssetDatabase.GetAssetPath(atlasAsset) == "")
694 AssetDatabase.CreateAsset(atlasAsset, atlasPath);

697

703 else
704 Debug.Log(string.Format("{0} :: Imported with {1} material", atlasAsset.name, atlasAsset.materials.Length), atlasAsset);
705

707 atlasAsset.Clear();
708 atlas = atlasAsset.GetAtlas(onlyMetaData: false);
709 if (atlas != null) {
710 FieldInfo field = typeof(Atlas).GetField("regions", BindingFlags.Instance | BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.NonPublic);
711 List<AtlasRegion> regions = (List<AtlasRegion>)field.GetValue(atlas);
712 string atlasAssetPath = AssetDatabase.GetAssetPath(atlasAsset);
713 string atlasAssetDirPath = Path.GetDirectoryName(atlasAssetPath).Replace('\\', '/');
714 string bakedDirPath = Path.Combine(atlasAssetDirPath, atlasAsset.name);
715
716 bool hasBakedRegions = false;
717 for (int i = 0; i < regions.Count; i++) {
718 AtlasRegion region = regions[i];
719 string bakedPrefabPath = Path.Combine(bakedDirPath, AssetUtility.GetPathSafeName(region.name) + ".prefab").Replace("\\", "/");
720 GameObject prefab = (GameObject)AssetDatabase.LoadAssetAtPath(bakedPrefabPath, typeof(GameObject));
721 if (prefab != null) {
722 SkeletonBaker.BakeRegion(atlasAsset, region, false);
723 hasBakedRegions = true;
724 }
725 }
726
727 if (hasBakedRegions) {
728 AssetDatabase.SaveAssets();
729 AssetDatabase.Refresh();
730 }
731 }

734 // note: at Asset Pipeline V2 this LoadAssetAtPath of the just created
735 // asset returns null, regardless of refresh calls.
736 AtlasAssetBase loadedAtlas = (AtlasAssetBase)AssetDatabase.LoadAssetAtPath(atlasPath, typeof(AtlasAssetBase));
737 return loadedAtlas != null ? loadedAtlas : atlasAsset;
